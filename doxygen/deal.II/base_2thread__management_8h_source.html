196  template <typename ForwardIterator>
197  std::vector<std::pair<ForwardIterator, ForwardIterator>>
198  split_range(const ForwardIterator &begin,
199  const ForwardIterator &end,
200  const unsigned int n_intervals)
201  {
202  using IteratorPair = std::pair<ForwardIterator, ForwardIterator>;
203 
204  // in non-multithreaded mode, we often have the case that this
205  // function is called with n_intervals==1, so have a shortcut here
206  // to handle that case efficiently
207 
208  if (n_intervals == 1)
209  return (std::vector<IteratorPair>(1, IteratorPair(begin, end)));
210 
211  // if more than one interval requested, do the full work
212  const unsigned int n_elements = std::distance(begin, end);
213  const unsigned int n_elements_per_interval = n_elements / n_intervals;
214  const unsigned int residual = n_elements % n_intervals;
215 
216  std::vector<IteratorPair> return_values(n_intervals);
217 
218  return_values[0].first = begin;
219  for (unsigned int i = 0; i < n_intervals; ++i)
220  {
221  if (i != n_intervals - 1)
222  {
223  return_values[i].second = return_values[i].first;
224  // note: the cast is performed to avoid a warning of gcc
225  // that in the library `dist>=0' is checked (dist has a
226  // template type, which here is unsigned if no cast is
227  // performed)
228  std::advance(return_values[i].second,
229  static_cast<signed int>(n_elements_per_interval));
230  // distribute residual in division equally among the first
231  // few subintervals
232  if (i < residual)
233  ++return_values[i].second;
234 
235  return_values[i + 1].first = return_values[i].second;
236  }
237  else
238  return_values[i].second = end;
239  }
240  return return_values;
241  }

1202  void
1203  wait()
1204  {
1205  // If we have previously already moved the result, then we don't
1206  // need a lock and can just return.
1207  if (task_has_finished)
1208  return;
1209 
1210  // Else, we need to go under a lock and try again. A different thread
1211  // may have waited and finished the task since then, so we have to try
1212  // a second time. (This is Schmidt's double-checking pattern.)
1213  std::lock_guard<std::mutex> lock(mutex);
1214  if (task_has_finished)
1215  return;
1216  else
1217  {
1218  // Wait for the task to finish and then move its
1219  // result. (We could have made the set_from() function
1220  // that we call here wait for the future to be ready --
1221  // which happens implicitly when it calls future.get() --
1222  // but that would have required putting an explicit
1223  // future.wait() into the implementation of
1224  // internal::return_value<void>::set_from(), which is a
1225  // bit awkward: that class doesn't actually need to set
1226  // anything, and so it looks odd to have the explicit call
1227  // to future.wait() in the set_from() function. Avoid the
1228  // issue by just explicitly calling future.wait() here.)
1229  future.wait();

1231 
1232  // Now we can safely set the flag and return.
