395  template <int dim>
396  void EnergyGroup<dim>::assemble_system_matrix()
397  {
398  const QGauss<dim> quadrature_formula(fe.degree + 1);
399 
400  FEValues<dim> fe_values(fe,
401  quadrature_formula,
402  update_values | update_gradients |
403  update_JxW_values);
404 
405  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
406  const unsigned int n_q_points = quadrature_formula.size();
407 
408  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
409  Vector<double> cell_rhs(dofs_per_cell);
410 
411  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
412 
413  for (const auto &cell : dof_handler.active_cell_iterators())
414  {
415  cell_matrix = 0;
416 
417  fe_values.reinit(cell);
418 
419  const double diffusion_coefficient =
420  material_data.get_diffusion_coefficient(group, cell->material_id());
421  const double removal_XS =
422  material_data.get_removal_XS(group, cell->material_id());
423 
424  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
425  for (unsigned int i = 0; i < dofs_per_cell; ++i)
426  for (unsigned int j = 0; j < dofs_per_cell; ++j)
427  cell_matrix(i, j) +=
428  ((diffusion_coefficient * fe_values.shape_grad(i, q_point) *
429  fe_values.shape_grad(j, q_point) +
430  removal_XS * fe_values.shape_value(i, q_point) *
431  fe_values.shape_value(j, q_point)) *
432  fe_values.JxW(q_point));
433 
434  cell->get_dof_indices(local_dof_indices);
435 
436  for (unsigned int i = 0; i < dofs_per_cell; ++i)
437  for (unsigned int j = 0; j < dofs_per_cell; ++j)
438  system_matrix.add(local_dof_indices[i],
439  local_dof_indices[j],
440  cell_matrix(i, j));
441  }
442 
443  hanging_node_constraints.condense(system_matrix);
444  }

450  EnergyGroup<dim>::assemble_ingroup_rhs(const Function<dim> &extraneous_source)

458

460  quadrature_formula,
461  update_values | update_quadrature_points |
462  update_JxW_values);
463 
464  Vector<double> cell_rhs(dofs_per_cell);
465  std::vector<double> extraneous_source_values(n_q_points);
466  std::vector<double> solution_old_values(n_q_points);
467 
468  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
469 
470  for (const auto &cell : dof_handler.active_cell_iterators())
471  {
472  cell_rhs = 0;
473 
474  fe_values.reinit(cell);
475 
476  const double fission_dist_XS =
477  material_data.get_fission_dist_XS(group, group, cell->material_id());
478 
479  extraneous_source.value_list(fe_values.get_quadrature_points(),
480  extraneous_source_values);
481 
482  fe_values.get_function_values(solution_old, solution_old_values);
483 
484  cell->get_dof_indices(local_dof_indices);
485 
486  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
487  for (unsigned int i = 0; i < dofs_per_cell; ++i)
488  cell_rhs(i) +=
489  ((extraneous_source_values[q_point] +
490  fission_dist_XS * solution_old_values[q_point]) *
491  fe_values.shape_value(i, q_point) * fe_values.JxW(q_point));
492 
493  for (unsigned int i = 0; i < dofs_per_cell; ++i)
494  system_rhs(local_dof_indices[i]) += cell_rhs(i);
495  }

526  template <int dim>
527  void EnergyGroup<dim>::assemble_cross_group_rhs_recursive(
528  const EnergyGroup<dim> & g_prime,
529  const typename DoFHandler<dim>::cell_iterator &cell_g,
530  const typename DoFHandler<dim>::cell_iterator &cell_g_prime,
531  const FullMatrix<double> & prolongation_matrix)
532  {
533  if (!cell_g->has_children() && !cell_g_prime->has_children())
534  {
535  const QGauss<dim> quadrature_formula(fe.degree + 1);
536  const unsigned int n_q_points = quadrature_formula.size();
537 
538  FEValues<dim> fe_values(fe,
539  quadrature_formula,
540  update_values | update_JxW_values);
541 
542  if (cell_g->level() > cell_g_prime->level())
543  fe_values.reinit(cell_g);
544  else
545  fe_values.reinit(cell_g_prime);
546 
547  const double fission_dist_XS =
548  material_data.get_fission_dist_XS(group,
549  g_prime.group,
550  cell_g_prime->material_id());
551 
552  const double scattering_XS =
553  material_data.get_scattering_XS(g_prime.group,
554  group,
555  cell_g_prime->material_id());
556 
557  FullMatrix<double> local_mass_matrix_f(fe.n_dofs_per_cell(),
558  fe.n_dofs_per_cell());
559  FullMatrix<double> local_mass_matrix_g(fe.n_dofs_per_cell(),
560  fe.n_dofs_per_cell());
561 
562  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
563  for (unsigned int i = 0; i < fe.n_dofs_per_cell(); ++i)
564  for (unsigned int j = 0; j < fe.n_dofs_per_cell(); ++j)
565  {
566  local_mass_matrix_f(i, j) +=
567  (fission_dist_XS * fe_values.shape_value(i, q_point) *
568  fe_values.shape_value(j, q_point) * fe_values.JxW(q_point));
569  local_mass_matrix_g(i, j) +=
570  (scattering_XS * fe_values.shape_value(i, q_point) *
571  fe_values.shape_value(j, q_point) * fe_values.JxW(q_point));
572  }
573 
574  Vector<double> g_prime_new_values(fe.n_dofs_per_cell());
575  Vector<double> g_prime_old_values(fe.n_dofs_per_cell());
576  cell_g_prime->get_dof_values(g_prime.solution_old, g_prime_old_values);
577  cell_g_prime->get_dof_values(g_prime.solution, g_prime_new_values);
578 
579  Vector<double> cell_rhs(fe.n_dofs_per_cell());
580  Vector<double> tmp(fe.n_dofs_per_cell());
581 
582  if (cell_g->level() > cell_g_prime->level())
583  {
584  prolongation_matrix.vmult(tmp, g_prime_old_values);
585  local_mass_matrix_f.vmult(cell_rhs, tmp);
586 
587  prolongation_matrix.vmult(tmp, g_prime_new_values);
588  local_mass_matrix_g.vmult_add(cell_rhs, tmp);
589  }
590  else
591  {
592  local_mass_matrix_f.vmult(tmp, g_prime_old_values);
593  prolongation_matrix.Tvmult(cell_rhs, tmp);
594 
595  local_mass_matrix_g.vmult(tmp, g_prime_new_values);
596  prolongation_matrix.Tvmult_add(cell_rhs, tmp);
597  }
598 
599  std::vector<types::global_dof_index> local_dof_indices(
600  fe.n_dofs_per_cell());
601  cell_g->get_dof_indices(local_dof_indices);
602 
603  for (unsigned int i = 0; i < fe.n_dofs_per_cell(); ++i)
604  system_rhs(local_dof_indices[i]) += cell_rhs(i);
605  }
606 
607  else
608  for (unsigned int child = 0;
609  child < GeometryInfo<dim>::max_children_per_cell;
610  ++child)
611  {
612  FullMatrix<double> new_matrix(fe.n_dofs_per_cell(),
613  fe.n_dofs_per_cell());
614  fe.get_prolongation_matrix(child).mmult(new_matrix,
615  prolongation_matrix);
616 
617  if (cell_g->has_children())
618  assemble_cross_group_rhs_recursive(g_prime,
619  cell_g->child(child),
620  cell_g_prime,
621  new_matrix);
622  else
623  assemble_cross_group_rhs_recursive(g_prime,
624  cell_g,
625  cell_g_prime->child(child),
626  new_matrix);
627  }
