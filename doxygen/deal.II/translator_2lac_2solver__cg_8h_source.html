278 template <typename VectorType>
279 inline void
280 SolverCG<VectorType>::compute_eigs_and_cond(
281  const std::vector<typename VectorType::value_type> &diagonal,
282  const std::vector<typename VectorType::value_type> &offdiagonal,
283  const boost::signals2::signal<void(const std::vector<double> &)>
284  & eigenvalues_signal,
285  const boost::signals2::signal<void(double)> &cond_signal)
286 {
287  // Avoid computing eigenvalues unless they are needed.
288  if (!cond_signal.empty() || !eigenvalues_signal.empty())
289  {
290  TridiagonalMatrix<typename VectorType::value_type> T(diagonal.size(),
291  true);
292  for (size_type i = 0; i < diagonal.size(); ++i)
293  {
294  T(i, i) = diagonal[i];
295  if (i < diagonal.size() - 1)
296  T(i, i + 1) = offdiagonal[i];
297  }
298  T.compute_eigenvalues();
299  // Need two eigenvalues to estimate the condition number.
300  if (diagonal.size() > 1)
301  {
302  auto condition_number = T.eigenvalue(T.n() - 1) / T.eigenvalue(0);
303  // Condition number is real valued and nonnegative; simply take
304  // the absolute value:
305  cond_signal(std::abs(condition_number));
306  }
307  // Avoid copying the eigenvalues of T to a vector unless a signal is
308  // connected.
309  if (!eigenvalues_signal.empty())
310  {
311  std::vector<double> eigenvalues(T.n());
312  for (unsigned int j = 0; j < T.n(); ++j)
313  {
314  // for a hermitian matrix, all eigenvalues are real-valued
315  // and non-negative, simply return the absolute value:
316  eigenvalues[j] = std::abs(T.eigenvalue(j));
317  }
318  eigenvalues_signal(eigenvalues);
319  }
320  }
321 }

326 template <typename MatrixType, typename PreconditionerType>
327 void
328 SolverCG<VectorType>::solve(const MatrixType & A,
329  VectorType & x,
330  const VectorType & b,
331  const PreconditionerType &preconditioner)
332 {
333  using number = typename VectorType::value_type;
334 
335  SolverControl::State conv = SolverControl::iterate;
336 
337  LogStream::Prefix prefix("cg");
338 
339  // Memory allocation
340  typename VectorMemory<VectorType>::Pointer g_pointer(this->memory);
341  typename VectorMemory<VectorType>::Pointer d_pointer(this->memory);
342  typename VectorMemory<VectorType>::Pointer h_pointer(this->memory);
343 
344  // define some aliases for simpler access
345  VectorType &g = *g_pointer;
346  VectorType &d = *d_pointer;
347  VectorType &h = *h_pointer;
348 
349  // Should we build the matrix for eigenvalue computations?
350  const bool do_eigenvalues =
351  !condition_number_signal.empty() || !all_condition_numbers_signal.empty() ||
352  !eigenvalues_signal.empty() || !all_eigenvalues_signal.empty();
353 
354  // vectors used for eigenvalue computations
355  std::vector<typename VectorType::value_type> diagonal;
356  std::vector<typename VectorType::value_type> offdiagonal;
357 
358  typename VectorType::value_type eigen_beta_alpha = 0;
359 
360  // resize the vectors, but do not set the values since they'd be overwritten
361  // soon anyway.
362  g.reinit(x, true);
363  d.reinit(x, true);
364  h.reinit(x, true);
365 
366  int it = 0;
367  number gh = number();
368  number beta = number();
369  number alpha = number();
370  number old_alpha = number();
371 
372  // compute residual. if vector is zero, then short-circuit the full
373  // computation
374  if (!x.all_zero())
375  {
376  A.vmult(g, x);
377  g.add(-1., b);
378  }
379  else
380  g.equ(-1., b);
381 
382  double res = g.l2_norm();
383  conv = this->iteration_status(0, res, x);
384  if (conv != SolverControl::iterate)
385  return;
386 
387  while (conv == SolverControl::iterate)
388  {
389  it++;
390  old_alpha = alpha;
391 
392  if (it > 1)
393  {
394  if (std::is_same<PreconditionerType, PreconditionIdentity>::value ==
395  false)
396  {
397  preconditioner.vmult(h, g);
398  beta = gh;
399  Assert(std::abs(beta) != 0., ExcDivideByZero());
400  gh = g * h;
401  beta = gh / beta;
402  d.sadd(beta, -1., h);
403  }
404  else
405  {
406  beta = gh;
407  gh = res * res;
408  beta = gh / beta;
409  d.sadd(beta, -1., g);
410  }
411  }
412  else
413  {
414  if (std::is_same<PreconditionerType, PreconditionIdentity>::value ==
415  false)
416  {
417  preconditioner.vmult(h, g);
418  d.equ(-1., h);
419  gh = g * h;
420  }
421  else
422  {
423  d.equ(-1., g);
424  gh = res * res;
425  }
426  }
427 
428  A.vmult(h, d);
429 
430  alpha = d * h;
431  Assert(std::abs(alpha) != 0., ExcDivideByZero());
432  alpha = gh / alpha;
433 
434  x.add(alpha, d);
435  res = std::sqrt(std::abs(g.add_and_dot(alpha, h, g)));
436 
437  print_vectors(it, x, g, d);
438 
439  if (it > 1)
440  {
441  this->coefficients_signal(old_alpha, beta);
442  // set up the vectors containing the diagonal and the off diagonal of
443  // the projected matrix.
444  if (do_eigenvalues)
445  {
446  diagonal.push_back(number(1.) / old_alpha + eigen_beta_alpha);
447  eigen_beta_alpha = beta / old_alpha;
448  offdiagonal.push_back(std::sqrt(beta) / old_alpha);
449  }
450  compute_eigs_and_cond(diagonal,
451  offdiagonal,
452  all_eigenvalues_signal,
453  all_condition_numbers_signal);
454  }
455 
456  conv = this->iteration_status(it, res, x);
457  }
458 
459  compute_eigs_and_cond(diagonal,
460  offdiagonal,
461  eigenvalues_signal,
462  condition_number_signal);
463 
464  // in case of failure: throw exception
465  if (conv != SolverControl::success)
466  AssertThrow(false, SolverControl::NoConvergence(it, res));
467  // otherwise exit as normal
468 }
