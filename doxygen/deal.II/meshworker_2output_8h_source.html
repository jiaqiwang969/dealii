214  // If patches are 1D, end the

218  (patch_dim == 1) ? row_length : (row_length * row_length);
219 
220  // AssertDimension(np, Utilities::fixed_power<dim>(n_points));
221  AssertDimension(nv, n_vectors + dim);
222 
223 
224  for (unsigned int k = 0; k < np; ++k)
225  {
226  if (k % row_length == 0)
227  write_endl();
228  if (k % row_length2 == 0)
