95  template <int dim>
96  void LaplaceProblem<dim>::setup_system()
97  {
98  dof_handler.distribute_dofs(fe);
99  solution.reinit(dof_handler.n_dofs());
100  system_rhs.reinit(dof_handler.n_dofs());
101 
102  const IndexSet boundary_dofs = DoFTools::extract_boundary_dofs(dof_handler);
103 
104  const types::global_dof_index first_boundary_dof =
105  boundary_dofs.nth_index_in_set(0);
106 
107  mean_value_constraints.clear();
108  mean_value_constraints.add_line(first_boundary_dof);
109  for (types::global_dof_index i : boundary_dofs)
110  if (i != first_boundary_dof)
111  mean_value_constraints.add_entry(first_boundary_dof, i, -1);
112  mean_value_constraints.close();
113 
114  DynamicSparsityPattern dsp(dof_handler.n_dofs(), dof_handler.n_dofs());
115  DoFTools::make_sparsity_pattern(dof_handler, dsp);
116  mean_value_constraints.condense(dsp);
117 
118  sparsity_pattern.copy_from(dsp);
119  system_matrix.reinit(sparsity_pattern);
120  }

124  template <int dim>
125  void LaplaceProblem<dim>::assemble_and_solve()
126  {
127  const unsigned int gauss_degree =
128  std::max(static_cast<unsigned int>(
129  std::ceil(1. * (mapping.get_degree() + 1) / 2)),
130  2U);
131  MatrixTools::create_laplace_matrix(mapping,
132  dof_handler,
133  QGauss<dim>(gauss_degree),
134  system_matrix);
135  VectorTools::create_right_hand_side(mapping,
136  dof_handler,
137  QGauss<dim>(gauss_degree),
138  Functions::ConstantFunction<dim>(-2),
139  system_rhs);
140  Vector<double> tmp(system_rhs.size());
141  VectorTools::create_boundary_right_hand_side(
142  mapping,
143  dof_handler,
144  QGauss<dim - 1>(gauss_degree),
145  Functions::ConstantFunction<dim>(1),
146  tmp);
147  system_rhs += tmp;
148 
149  mean_value_constraints.condense(system_matrix);
150  mean_value_constraints.condense(system_rhs);
151 
152  solve();
153  mean_value_constraints.distribute(solution);
154 
155  Vector<float> norm_per_cell(triangulation.n_active_cells());
156  VectorTools::integrate_difference(mapping,
157  dof_handler,
158  solution,
159  Functions::ZeroFunction<dim>(),
160  norm_per_cell,
161  QGauss<dim>(gauss_degree + 1),
162  VectorTools::H1_seminorm);
163  const double norm =
164  VectorTools::compute_global_error(triangulation,
165  norm_per_cell,
166  VectorTools::H1_seminorm);
167 
168  output_table.add_value("cells", triangulation.n_active_cells());
169  output_table.add_value("|u|_1", norm);
170  output_table.add_value("error",
171  std::fabs(norm - std::sqrt(3.14159265358 / 2)));
172  }
