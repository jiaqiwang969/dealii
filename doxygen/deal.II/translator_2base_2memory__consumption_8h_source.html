303  template <typename T>
304  std::size_t
305  memory_consumption(const std::vector<T> &v)
306  {
307  // shortcut for types that do not allocate memory themselves
308  if (std::is_fundamental<T>::value || std::is_pointer<T>::value)
309  {
310  return v.capacity() * sizeof(T) + sizeof(v);
311  }
312  else
313  {
314  std::size_t mem = sizeof(std::vector<T>);
315  for (unsigned int i = 0; i < v.size(); ++i)
316  {
317  mem += memory_consumption(v[i]);
318  }
319  mem += (v.capacity() - v.size()) * sizeof(T);
320  return mem;
321  }
322  }
