161  template <>
162  double RightHandSide<3>::value(const Point<3> &p,
163  const unsigned int /*component*/) const
164  {
165  using numbers::PI;
166 
167  Tensor<2, 3> hessian;
168 
169  hessian[0][0] = -PI * PI * sin(PI * p(0)) * cos(PI * p(1)) * exp(p(2));
170  hessian[1][1] = -PI * PI * sin(PI * p(0)) * cos(PI * p(1)) * exp(p(2));
171  hessian[2][2] = sin(PI * p(0)) * cos(PI * p(1)) * exp(p(2));
172 
173  hessian[0][1] = -PI * PI * cos(PI * p(0)) * sin(PI * p(1)) * exp(p(2));
174  hessian[1][0] = -PI * PI * cos(PI * p(0)) * sin(PI * p(1)) * exp(p(2));
175 
176  hessian[0][2] = PI * cos(PI * p(0)) * cos(PI * p(1)) * exp(p(2));
177  hessian[2][0] = PI * cos(PI * p(0)) * cos(PI * p(1)) * exp(p(2));
178 
179  hessian[1][2] = -PI * sin(PI * p(0)) * sin(PI * p(1)) * exp(p(2));
180  hessian[2][1] = -PI * sin(PI * p(0)) * sin(PI * p(1)) * exp(p(2));
181 
182  Tensor<1, 3> gradient;
183  gradient[0] = PI * cos(PI * p(0)) * cos(PI * p(1)) * exp(p(2));
184  gradient[1] = -PI * sin(PI * p(0)) * sin(PI * p(1)) * exp(p(2));
185  gradient[2] = sin(PI * p(0)) * cos(PI * p(1)) * exp(p(2));
186 
187  Point<3> normal = p;
188  normal /= p.norm();
189 
190  return (-trace(hessian) + 2 * (gradient * normal) +
191  (hessian * normal) * normal);
192  }

245  template <int spacedim>
246  void LaplaceBeltramiProblem<spacedim>::assemble_system()
247  {
248  system_matrix = 0;
249  system_rhs = 0;
250 
251  const QGauss<dim> quadrature_formula(2 * fe.degree);
252  FEValues<dim, spacedim> fe_values(mapping,
253  fe,
254  quadrature_formula,
255  update_values | update_gradients |
256  update_quadrature_points |
257  update_JxW_values);
258 
259  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
260  const unsigned int n_q_points = quadrature_formula.size();
261 
262  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
263  Vector<double> cell_rhs(dofs_per_cell);
264 
265  std::vector<double> rhs_values(n_q_points);
266  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
267 
268  RightHandSide<spacedim> rhs;
269 
270  for (const auto &cell : dof_handler.active_cell_iterators())
271  {
272  cell_matrix = 0;
273  cell_rhs = 0;
274 
275  fe_values.reinit(cell);
276 
277  rhs.value_list(fe_values.get_quadrature_points(), rhs_values);
278 
279  for (unsigned int i = 0; i < dofs_per_cell; ++i)
280  for (unsigned int j = 0; j < dofs_per_cell; ++j)
281  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
282  cell_matrix(i, j) += fe_values.shape_grad(i, q_point) *
283  fe_values.shape_grad(j, q_point) *
284  fe_values.JxW(q_point);
285 
286  for (unsigned int i = 0; i < dofs_per_cell; ++i)
287  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
288  cell_rhs(i) += fe_values.shape_value(i, q_point) *
289  rhs_values[q_point] * fe_values.JxW(q_point);
290 
291  cell->get_dof_indices(local_dof_indices);
292  for (unsigned int i = 0; i < dofs_per_cell; ++i)
293  {
294  for (unsigned int j = 0; j < dofs_per_cell; ++j)
295  system_matrix.add(local_dof_indices[i],
296  local_dof_indices[j],
297  cell_matrix(i, j));
298 
299  system_rhs(local_dof_indices[i]) += cell_rhs(i);
300  }
301  }
302 
303  std::map<types::global_dof_index, double> boundary_values;
304  VectorTools::interpolate_boundary_values(
305  mapping, dof_handler, 0, Solution<spacedim>(), boundary_values);
306 
307  MatrixTools::apply_boundary_values(
308  boundary_values, system_matrix, solution, system_rhs, false);
309  }
