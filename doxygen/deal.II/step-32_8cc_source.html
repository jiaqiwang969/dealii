139  template <int dim>
140  double TemperatureInitialValues<dim>::value(const Point<dim> &p,
141  const unsigned int) const
142  {
143  const double r = p.norm();
144  const double h = R1 - R0;
145 
146  const double s = (r - R0) / h;
147  const double q =
148  (dim == 3) ? std::max(0.0, cos(numbers::PI * abs(p(2) / R1))) : 1.0;
149  const double phi = std::atan2(p(0), p(1));
150  const double tau = s + 0.2 * s * (1 - s) * std::sin(6 * phi) * q;
151 
152  return T0 * (1.0 - tau) + T1 * tau;
153  }

194  const TrilinosWrappers::MPI::BlockVector &src) const
195  {
196  TrilinosWrappers::MPI::Vector utmp(src.block(0));
197 
198  {
199  SolverControl solver_control(5000, 1e-6 * src.block(1).l2_norm());
200 
201  SolverCG<TrilinosWrappers::MPI::Vector> solver(solver_control);
202 
203  solver.solve(stokes_preconditioner_matrix->block(1, 1),
204  dst.block(1),
205  src.block(1),
206  mp_preconditioner);
207 
208  dst.block(1) *= -1.0;
209  }
210 
211  {
212  stokes_matrix->block(0, 1).vmult(utmp, dst.block(1));
213  utmp *= -1.0;
214  utmp.add(src.block(0));
215  }

219  SolverControl solver_control(5000, utmp.l2_norm() * 1e-2);
220  TrilinosWrappers::SolverCG solver(solver_control);
221  solver.solve(stokes_matrix->block(0, 0),
222  dst.block(0),
223  utmp,
224  a_preconditioner);
225  }
226  else
227  a_preconditioner.vmult(dst.block(0), utmp);
228  }

1042  template <int dim>
1043  double BoussinesqFlowProblem<dim>::get_entropy_variation(
1044  const double average_temperature) const
1045  {
1046  if (parameters.stabilization_alpha != 2)
1047  return 1.;
1048 
1049  const QGauss<dim> quadrature_formula(parameters.temperature_degree + 1);
1050  const unsigned int n_q_points = quadrature_formula.size();
1051 
1052  FEValues<dim> fe_values(temperature_fe,
1053  quadrature_formula,
1054  update_values | update_JxW_values);
1055  std::vector<double> old_temperature_values(n_q_points);
1056  std::vector<double> old_old_temperature_values(n_q_points);
1057 
1058  double min_entropy = std::numeric_limits<double>::max(),
1059  max_entropy = -std::numeric_limits<double>::max(), area = 0,
1060  entropy_integrated = 0;
1061 
1062  for (const auto &cell : temperature_dof_handler.active_cell_iterators())
1063  if (cell->is_locally_owned())
1064  {
1065  fe_values.reinit(cell);
1066  fe_values.get_function_values(old_temperature_solution,
1067  old_temperature_values);
1068  fe_values.get_function_values(old_old_temperature_solution,
1069  old_old_temperature_values);
1070  for (unsigned int q = 0; q < n_q_points; ++q)
1071  {
1072  const double T =
1073  (old_temperature_values[q] + old_old_temperature_values[q]) / 2;
1074  const double entropy =
1075  ((T - average_temperature) * (T - average_temperature));
1076 
1077  min_entropy = std::min(min_entropy, entropy);
1078  max_entropy = std::max(max_entropy, entropy);
1079  area += fe_values.JxW(q);
1080  entropy_integrated += fe_values.JxW(q) * entropy;
1081  }
1082  }
1083 
1084  const double local_sums[2] = {entropy_integrated, area},
1085  local_maxima[2] = {-min_entropy, max_entropy};
1086  double global_sums[2], global_maxima[2];
1087 
1088  Utilities::MPI::sum(local_sums, MPI_COMM_WORLD, global_sums);
1089  Utilities::MPI::max(local_maxima, MPI_COMM_WORLD, global_maxima);
1090 
1091  const double average_entropy = global_sums[0] / global_sums[1];
1092  const double entropy_diff = std::max(global_maxima[1] - average_entropy,
1093  average_entropy - (-global_maxima[0]));
1094  return entropy_diff;
1095  }

1173  template <int dim>
1174  double BoussinesqFlowProblem<dim>::compute_viscosity(
1175  const std::vector<double> & old_temperature,
1176  const std::vector<double> & old_old_temperature,
1177  const std::vector<Tensor<1, dim>> & old_temperature_grads,
1178  const std::vector<Tensor<1, dim>> & old_old_temperature_grads,
1179  const std::vector<double> & old_temperature_laplacians,
1180  const std::vector<double> & old_old_temperature_laplacians,
1181  const std::vector<Tensor<1, dim>> & old_velocity_values,
1182  const std::vector<Tensor<1, dim>> & old_old_velocity_values,
1183  const std::vector<SymmetricTensor<2, dim>> &old_strain_rates,
1184  const std::vector<SymmetricTensor<2, dim>> &old_old_strain_rates,
1185  const double global_u_infty,
1186  const double global_T_variation,
1187  const double average_temperature,
1188  const double global_entropy_variation,
1189  const double cell_diameter) const
1190  {
1191  if (global_u_infty == 0)
1192  return 5e-3 * cell_diameter;
1193 
1194  const unsigned int n_q_points = old_temperature.size();
1195 
1196  double max_residual = 0;
1197  double max_velocity = 0;
1198 
1199  for (unsigned int q = 0; q < n_q_points; ++q)
1200  {
1201  const Tensor<1, dim> u =
1202  (old_velocity_values[q] + old_old_velocity_values[q]) / 2;
1203 
1204  const SymmetricTensor<2, dim> strain_rate =
1205  (old_strain_rates[q] + old_old_strain_rates[q]) / 2;
1206 
1207  const double T = (old_temperature[q] + old_old_temperature[q]) / 2;
1208  const double dT_dt =
1209  (old_temperature[q] - old_old_temperature[q]) / old_time_step;
1210  const double u_grad_T =
1211  u * (old_temperature_grads[q] + old_old_temperature_grads[q]) / 2;
1212 
1213  const double kappa_Delta_T =
1214  EquationData::kappa *
1215  (old_temperature_laplacians[q] + old_old_temperature_laplacians[q]) /
1216  2;
1217  const double gamma =
1218  ((EquationData::radiogenic_heating * EquationData::density(T) +
1219  2 * EquationData::eta * strain_rate * strain_rate) /
1220  (EquationData::density(T) * EquationData::specific_heat));
1221 
1222  double residual = std::abs(dT_dt + u_grad_T - kappa_Delta_T - gamma);
1223  if (parameters.stabilization_alpha == 2)
1224  residual *= std::abs(T - average_temperature);
1225 
1226  max_residual = std::max(residual, max_residual);
1227  max_velocity = std::max(std::sqrt(u * u), max_velocity);
1228  }
1229 
1230  const double max_viscosity =
1231  (parameters.stabilization_beta * max_velocity * cell_diameter);
1232  if (timestep_number == 0)
1233  return max_viscosity;
1234  else
1235  {
1236  Assert(old_time_step > 0, ExcInternalError());
1237 
1238  double entropy_viscosity;
1239  if (parameters.stabilization_alpha == 2)
1240  entropy_viscosity =
1241  (parameters.stabilization_c_R * cell_diameter * cell_diameter *
1242  max_residual / global_entropy_variation);
1243  else
1244  entropy_viscosity =
1245  (parameters.stabilization_c_R * cell_diameter *
1246  global_Omega_diameter * max_velocity * max_residual /
1247  (global_u_infty * global_T_variation));
1248 
1249  return std::min(max_viscosity, entropy_viscosity);
1250  }
1251  }
