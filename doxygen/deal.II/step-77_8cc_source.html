160  template <int dim>
161  void MinimalSurfaceProblem<dim>::compute_and_factorize_jacobian(
162  const Vector<double> &evaluation_point)
163  {
164  {
165  TimerOutput::Scope t(computing_timer, "assembling the Jacobian");
166 
167  std::cout << " Computing Jacobian matrix" << std::endl;
168 
169  const QGauss<dim> quadrature_formula(fe.degree + 1);
170 
171  jacobian_matrix = 0;
172 
173  FEValues<dim> fe_values(fe,
174  quadrature_formula,
175  update_gradients | update_quadrature_points |
176  update_JxW_values);
177 
178  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
179  const unsigned int n_q_points = quadrature_formula.size();
180 
181  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
182 
183  std::vector<Tensor<1, dim>> evaluation_point_gradients(n_q_points);
184 
185  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
186 
187  for (const auto &cell : dof_handler.active_cell_iterators())
188  {
189  cell_matrix = 0;
190 
191  fe_values.reinit(cell);
192 
193  fe_values.get_function_gradients(evaluation_point,
194  evaluation_point_gradients);
195 
196  for (unsigned int q = 0; q < n_q_points; ++q)
197  {
198  const double coeff =
199  1.0 / std::sqrt(1 + evaluation_point_gradients[q] *
200  evaluation_point_gradients[q]);
201 
202  for (unsigned int i = 0; i < dofs_per_cell; ++i)
203  {
204  for (unsigned int j = 0; j < dofs_per_cell; ++j)
205  cell_matrix(i, j) +=
206  (((fe_values.shape_grad(i, q) // ((\nabla \phi_i
207  * coeff // * a_n
208  * fe_values.shape_grad(j, q)) // * \nabla \phi_j)
209  - // -
210  (fe_values.shape_grad(i, q) // (\nabla \phi_i
211  * coeff * coeff * coeff // * a_n^3
212  *
213  (fe_values.shape_grad(j, q) // * (\nabla \phi_j
214  * evaluation_point_gradients[q]) // * \nabla u_n)
215  * evaluation_point_gradients[q])) // * \nabla u_n)))
216  * fe_values.JxW(q)); // * dx
217  }
218  }
219 
220  cell->get_dof_indices(local_dof_indices);
221  hanging_node_constraints.distribute_local_to_global(cell_matrix,
222  local_dof_indices,
223  jacobian_matrix);
224  }
225 
226  std::map<types::global_dof_index, double> boundary_values;
227  VectorTools::interpolate_boundary_values(dof_handler,
228  0,
229  Functions::ZeroFunction<dim>(),
230  boundary_values);
231  Vector<double> dummy_solution(dof_handler.n_dofs());
232  Vector<double> dummy_rhs(dof_handler.n_dofs());
233  MatrixTools::apply_boundary_values(boundary_values,
234  jacobian_matrix,
235  dummy_solution,
236  dummy_rhs);
237  }
238 
239  {
240  TimerOutput::Scope t(computing_timer, "factorizing the Jacobian");
241 
242  std::cout << " Factorizing Jacobian matrix" << std::endl;
243 
244  jacobian_matrix_factorization = std::make_unique<SparseDirectUMFPACK>();
245  jacobian_matrix_factorization->factorize(jacobian_matrix);
246  }

257  TimerOutput::Scope t(computing_timer, "assembling the residual");

266 
267  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
268  const unsigned int n_q_points = quadrature_formula.size();
269 
270  Vector<double> cell_residual(dofs_per_cell);
271  std::vector<Tensor<1, dim>> evaluation_point_gradients(n_q_points);
272 
273  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
274 
275  for (const auto &cell : dof_handler.active_cell_iterators())
276  {
277  cell_residual = 0;
278  fe_values.reinit(cell);
279 
280  fe_values.get_function_gradients(evaluation_point,
281  evaluation_point_gradients);
282 
283 
284  for (unsigned int q = 0; q < n_q_points; ++q)
285  {
286  const double coeff =
287  1.0 / std::sqrt(1 + evaluation_point_gradients[q] *
288  evaluation_point_gradients[q]);
289 
290  for (unsigned int i = 0; i < dofs_per_cell; ++i)
291  cell_residual(i) = (fe_values.shape_grad(i, q) // \nabla \phi_i
292  * coeff // * a_n
293  * evaluation_point_gradients[q] // * u_n
294  * fe_values.JxW(q)); // * dx
295  }
296

303 
304  for (const types::global_dof_index i :

306  residual(i) = 0;
307

309  DoFTools::extract_hanging_node_dofs(dof_handler))
310  residual(i) = 0;
