278  template <int dim>
279  void WGDarcyEquation<dim>::assemble_system()
280  {
281  const QGauss<dim> quadrature_formula(fe_dgrt.degree + 1);
282  const QGauss<dim - 1> face_quadrature_formula(fe_dgrt.degree + 1);
283 
284  FEValues<dim> fe_values(fe,
285  quadrature_formula,
286  update_values | update_quadrature_points |
287  update_JxW_values);
288  FEFaceValues<dim> fe_face_values(fe,
289  face_quadrature_formula,
290  update_values | update_normal_vectors |
291  update_quadrature_points |
292  update_JxW_values);
293 
294  FEValues<dim> fe_values_dgrt(fe_dgrt,
295  quadrature_formula,
296  update_values | update_gradients |
297  update_quadrature_points |
298  update_JxW_values);
299  FEFaceValues<dim> fe_face_values_dgrt(fe_dgrt,
300  face_quadrature_formula,
301  update_values |
302  update_normal_vectors |
303  update_quadrature_points |
304  update_JxW_values);
305 
306  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
307  const unsigned int dofs_per_cell_dgrt = fe_dgrt.n_dofs_per_cell();
308 
309  const unsigned int n_q_points = fe_values.get_quadrature().size();
310  const unsigned int n_q_points_dgrt = fe_values_dgrt.get_quadrature().size();
311 
312  const unsigned int n_face_q_points = fe_face_values.get_quadrature().size();
313 
314  RightHandSide<dim> right_hand_side;
315  std::vector<double> right_hand_side_values(n_q_points);
316 
317  const Coefficient<dim> coefficient;
318  std::vector<Tensor<2, dim>> coefficient_values(n_q_points);
319 
320  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
321 
322 
323  FullMatrix<double> cell_matrix_M(dofs_per_cell_dgrt, dofs_per_cell_dgrt);
324  FullMatrix<double> cell_matrix_G(dofs_per_cell_dgrt, dofs_per_cell);
325  FullMatrix<double> cell_matrix_C(dofs_per_cell, dofs_per_cell_dgrt);
326  FullMatrix<double> local_matrix(dofs_per_cell, dofs_per_cell);
327  Vector<double> cell_rhs(dofs_per_cell);
328  Vector<double> cell_solution(dofs_per_cell);
329 
330  const FEValuesExtractors::Vector velocities(0);
331  const FEValuesExtractors::Scalar pressure_interior(0);
332  const FEValuesExtractors::Scalar pressure_face(1);
333 
334  for (const auto &cell : dof_handler.active_cell_iterators())
335  {
336  fe_values.reinit(cell);
337 
338  const typename Triangulation<dim>::active_cell_iterator cell_dgrt =
339  cell;
340  fe_values_dgrt.reinit(cell_dgrt);
341 
342  right_hand_side.value_list(fe_values.get_quadrature_points(),
343  right_hand_side_values);
344  coefficient.value_list(fe_values.get_quadrature_points(),
345  coefficient_values);
346 
347  cell_matrix_M = 0;
348  for (unsigned int q = 0; q < n_q_points_dgrt; ++q)
349  for (unsigned int i = 0; i < dofs_per_cell_dgrt; ++i)
350  {
351  const Tensor<1, dim> v_i = fe_values_dgrt[velocities].value(i, q);
352  for (unsigned int k = 0; k < dofs_per_cell_dgrt; ++k)
353  {
354  const Tensor<1, dim> v_k =
355  fe_values_dgrt[velocities].value(k, q);
356  cell_matrix_M(i, k) += (v_i * v_k * fe_values_dgrt.JxW(q));
357  }
358  }
359  cell_matrix_M.gauss_jordan();
360 
361  cell_matrix_G = 0;
362  for (unsigned int q = 0; q < n_q_points; ++q)
363  for (unsigned int i = 0; i < dofs_per_cell_dgrt; ++i)
364  {
365  const double div_v_i =
366  fe_values_dgrt[velocities].divergence(i, q);
367  for (unsigned int j = 0; j < dofs_per_cell; ++j)
368  {
369  const double phi_j_interior =
370  fe_values[pressure_interior].value(j, q);
371 
372  cell_matrix_G(i, j) -=
373  (div_v_i * phi_j_interior * fe_values.JxW(q));
374  }
375  }
376 
377 
378  for (const auto &face : cell->face_iterators())
379  {
380  fe_face_values.reinit(cell, face);
381  fe_face_values_dgrt.reinit(cell_dgrt, face);
382 
383  for (unsigned int q = 0; q < n_face_q_points; ++q)
384  {
385  const Tensor<1, dim> &normal = fe_face_values.normal_vector(q);
386 
387  for (unsigned int i = 0; i < dofs_per_cell_dgrt; ++i)
388  {
389  const Tensor<1, dim> v_i =
390  fe_face_values_dgrt[velocities].value(i, q);
391  for (unsigned int j = 0; j < dofs_per_cell; ++j)
392  {
393  const double phi_j_face =
394  fe_face_values[pressure_face].value(j, q);
395 
396  cell_matrix_G(i, j) +=
397  ((v_i * normal) * phi_j_face * fe_face_values.JxW(q));
398  }
399  }
400  }
401  }
402 
403  cell_matrix_G.Tmmult(cell_matrix_C, cell_matrix_M);
404 
405  local_matrix = 0;
406  for (unsigned int q = 0; q < n_q_points_dgrt; ++q)
407  {
408  for (unsigned int k = 0; k < dofs_per_cell_dgrt; ++k)
409  {
410  const Tensor<1, dim> v_k =
411  fe_values_dgrt[velocities].value(k, q);
412  for (unsigned int l = 0; l < dofs_per_cell_dgrt; ++l)
413  {
414  const Tensor<1, dim> v_l =
415  fe_values_dgrt[velocities].value(l, q);
416 
417  for (unsigned int i = 0; i < dofs_per_cell; ++i)
418  for (unsigned int j = 0; j < dofs_per_cell; ++j)
419  local_matrix(i, j) +=
420  (coefficient_values[q] * cell_matrix_C[i][k] * v_k) *
421  cell_matrix_C[j][l] * v_l * fe_values_dgrt.JxW(q);
422  }
423  }
424  }
425 
426  cell_rhs = 0;
427  for (unsigned int q = 0; q < n_q_points; ++q)
428  for (unsigned int i = 0; i < dofs_per_cell; ++i)
429  {
430  cell_rhs(i) += (fe_values[pressure_interior].value(i, q) *
431  right_hand_side_values[q] * fe_values.JxW(q));
432  }
433 
434  cell->get_dof_indices(local_dof_indices);
435  constraints.distribute_local_to_global(
436  local_matrix, cell_rhs, local_dof_indices, system_matrix, system_rhs);
437  }

532  for (unsigned int k = 0; k < dofs_per_cell_dgrt; ++k)
533  {

535  fe_values_dgrt[velocities].value(k, q);
536

538  (coefficient_values[q] * v_i * v_k * fe_values_dgrt.JxW(q));

560  }
561  }

567

569  {

571

576  for (unsigned int j = 0; j < dofs_per_cell; ++j)
577  {

579  fe_face_values[pressure_face].value(j, q);
580

583  }
584  }
585  }
586  }

588 
589  cell->get_dof_values(solution, cell_solution);
590

611  void WGDarcyEquation<dim>::compute_pressure_error()
612  {
613  Vector<float> difference_per_cell(triangulation.n_active_cells());
614  const ComponentSelectFunction<dim> select_interior_pressure(0, 2);
615  VectorTools::integrate_difference(dof_handler,
616  solution,
617  ExactPressure<dim>(),
618  difference_per_cell,
619  QGauss<dim>(fe.degree + 2),
620  VectorTools::L2_norm,
621  &select_interior_pressure);
