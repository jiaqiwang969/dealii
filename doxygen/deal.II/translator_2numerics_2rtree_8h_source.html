559 inline std::vector<BoundingBox<
560  boost::geometry::dimension<typename Rtree::indexable_type>::value>>
561 extract_rtree_level(const Rtree &tree, const unsigned int level)
562 {
563  constexpr unsigned int dim =
564  boost::geometry::dimension<typename Rtree::indexable_type>::value;
565 
566  using RtreeView =
567  boost::geometry::index::detail::rtree::utilities::view<Rtree>;
568  RtreeView rtv(tree);
569 
570  std::vector<BoundingBox<dim>> boxes;
571 
572  if (rtv.depth() == 0)
573  {
574  // The below algorithm does not work for `rtv.depth()==0`, which might
575  // happen if the number entries in the tree is too small.
576  // In this case, simply return a single bounding box.
577  boxes.resize(1);
578  boost::geometry::convert(tree.bounds(), boxes[0]);
579  }
580  else
581  {
582  const unsigned int target_level =
583  std::min<unsigned int>(level, rtv.depth() - 1);
584 
585  ExtractLevelVisitor<typename RtreeView::value_type,
586  typename RtreeView::options_type,
587  typename RtreeView::translator_type,
588  typename RtreeView::box_type,
589  typename RtreeView::allocators_type>
590  extract_level_visitor(rtv.translator(), target_level, boxes);
591  rtv.apply_visitor(extract_level_visitor);
592  }
593 
594  return boxes;
595 }
