465 BarycentricPolynomial<dim, Number>::print(std::ostream &out) const
466 {
467  const auto &coeffs = this->coefficients;
468  auto first = index_to_indices(0, coeffs.size());
469  bool print_plus = false;
470  if (coeffs(first) != Number())
471  {
472  out << coeffs(first);
473  print_plus = true;

475  for (std::size_t i = 1; i < coeffs.n_elements(); ++i)
476  {
477  const auto indices = index_to_indices(i, coeffs.size());
478  if (coeffs(indices) == Number())

483  for (unsigned int d = 0; d < dim + 1; ++d)
484  {
485  if (indices[d] != 0)
486  out << " * t" << d << '^' << indices[d];
487  }
488  print_plus = true;
489  }
490 
491  if (!print_plus)
492  out << Number();

647 template <int dim, typename Number>
648 BarycentricPolynomial<dim, Number>
649 BarycentricPolynomial<dim, Number>::barycentric_derivative(
650  const unsigned int coordinate) const
651 {
652  AssertIndexRange(coordinate, dim + 1);
653 
654  if (degrees()[coordinate] == 0)
655  return BarycentricPolynomial<dim, Number>();
656 
657  auto deg = degrees();
658  deg[coordinate] -= 1;
659  BarycentricPolynomial<dim, Number> result(deg,
660  std::numeric_limits<Number>::max());
661  const auto & coeffs_in = coefficients;
662  auto & coeffs_out = result.coefficients;
663  for (std::size_t i = 0; i < coeffs_out.n_elements(); ++i)
664  {
665  const auto out_index = index_to_indices(i, coeffs_out.size());
666  auto input_index = out_index;
667  input_index[coordinate] += 1;
668 
669  coeffs_out(out_index) = coeffs_in(input_index) * input_index[coordinate];
670  }
671 
672  return result;
673 }

688 template <int dim, typename Number>
689 Number
690 BarycentricPolynomial<dim, Number>::value(const Point<dim> &point) const
691 {
692  // TODO: this is probably not numerically stable for higher order.
693  // We really need some version of Horner's method.
694  Number result = {};
695 
696  // Begin by converting point (which is in Cartesian coordinates) to
697  // barycentric coordinates:
698  std::array<Number, dim + 1> b_point;
699  b_point[0] = 1.0;
700  for (unsigned int d = 0; d < dim; ++d)
701  {
702  b_point[0] -= point[d];
703  b_point[d + 1] = point[d];
704  }
705 
706  // Now evaluate the polynomial at the computed barycentric point:
707  for (std::size_t i = 0; i < coefficients.n_elements(); ++i)
708  {
709  const auto indices = index_to_indices(i, coefficients.size());
710  const auto coef = coefficients(indices);
711  if (coef == Number())
712  continue;
713 
714  auto temp = Number(1);
715  for (unsigned int d = 0; d < dim + 1; ++d)
716  temp *= std::pow(b_point[d], indices[d]);
717  result += coef * temp;
718  }
719 
720  return result;
721 }

732 BarycentricPolynomial<dim, Number>::index_to_indices(
