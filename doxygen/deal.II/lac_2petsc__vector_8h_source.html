432  template <typename number>
433  inline Vector &
434  Vector::operator=(const ::Vector<number> &v)
435  {
436  Assert(size() == v.size(), ExcDimensionMismatch(size(), v.size()));
437 
438  // FIXME: the following isn't necessarily fast, but this is due to
439  // the fact that PETSc doesn't offer an inlined access operator.
440  //
441  // if someone wants to contribute some code: to make this code
442  // faster, one could either first convert all values to PetscScalar,
443  // and then set them all at once using VecSetValues. This has the
444  // drawback that it could take quite some memory, if the vector is
445  // large, and it would in addition allocate memory on the heap, which
446  // is expensive. an alternative would be to split the vector into
447  // chunks of, say, 128 elements, convert a chunk at a time and set it
448  // in the output vector using VecSetValues. since 128 elements is
449  // small enough, this could easily be allocated on the stack (as a
450  // local variable) which would make the whole thing much more
451  // efficient.
452  //
453  // a second way to make things faster is for the special case that
454  // number==PetscScalar. we could then declare a specialization of
455  // this template, and omit the conversion. the problem with this is
456  // that the best we can do is to use VecSetValues, but this isn't
457  // very efficient either: it wants to see an array of indices, which
458  // in this case a) again takes up a whole lot of memory on the heap,
459  // and b) is totally dumb since its content would simply be the
460  // sequence 0,1,2,3,...,n. the best of all worlds would probably be a
461  // function in Petsc that would take a pointer to an array of
462  // PetscScalar values and simply copy n elements verbatim into the
463  // vector...
464  for (size_type i = 0; i < v.size(); ++i)
465  (*this)(i) = v(i);
466 
467  compress(::VectorOperation::insert);
468 
469  return *this;
470  }
