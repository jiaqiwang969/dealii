85  {
86  Assert(dim >= 2, ExcNotImplemented());

199  template <int dim>
200  void AdvectionProblem<dim>::integrate_cell_term(DoFInfo & dinfo,
201  CellInfo &info)
202  {
203  const FEValuesBase<dim> & fe_values = info.fe_values();
204  FullMatrix<double> & local_matrix = dinfo.matrix(0).matrix;
205  const std::vector<double> &JxW = fe_values.get_JxW_values();
206 
207  for (unsigned int point = 0; point < fe_values.n_quadrature_points; ++point)
208  {
209  const Tensor<1, dim> beta_at_q_point =
210  beta(fe_values.quadrature_point(point));
211 
212  for (unsigned int i = 0; i < fe_values.dofs_per_cell; ++i)
213  for (unsigned int j = 0; j < fe_values.dofs_per_cell; ++j)
214  local_matrix(i, j) += -beta_at_q_point *
215  fe_values.shape_grad(i, point) *
216  fe_values.shape_value(j, point) *
217  JxW[point];
218  }
219  }

221  template <int dim>
222  void AdvectionProblem<dim>::integrate_boundary_term(DoFInfo & dinfo,
223  CellInfo &info)
224  {
225  const FEValuesBase<dim> &fe_face_values = info.fe_values();
226  FullMatrix<double> & local_matrix = dinfo.matrix(0).matrix;
227  Vector<double> & local_vector = dinfo.vector(0).block(0);
228 
229  const std::vector<double> & JxW = fe_face_values.get_JxW_values();
230  const std::vector<Tensor<1, dim>> &normals =
231  fe_face_values.get_normal_vectors();
232 
233  std::vector<double> g(fe_face_values.n_quadrature_points);
234 
235  static BoundaryValues<dim> boundary_function;
236  boundary_function.value_list(fe_face_values.get_quadrature_points(), g);
237 
238  for (unsigned int point = 0; point < fe_face_values.n_quadrature_points;
239  ++point)
240  {
241  const double beta_dot_n =
242  beta(fe_face_values.quadrature_point(point)) * normals[point];
243  if (beta_dot_n > 0)
244  for (unsigned int i = 0; i < fe_face_values.dofs_per_cell; ++i)
245  for (unsigned int j = 0; j < fe_face_values.dofs_per_cell; ++j)
246  local_matrix(i, j) += beta_dot_n *
247  fe_face_values.shape_value(j, point) *
248  fe_face_values.shape_value(i, point) *
249  JxW[point];
250  else
251  for (unsigned int i = 0; i < fe_face_values.dofs_per_cell; ++i)
252  local_vector(i) += -beta_dot_n *
253  g[point] *
254  fe_face_values.shape_value(i, point) *
255  JxW[point];
256  }
257  }

259  template <int dim>
260  void AdvectionProblem<dim>::integrate_face_term(DoFInfo & dinfo1,
261  DoFInfo & dinfo2,
262  CellInfo &info1,
263  CellInfo &info2)
264  {
265  const FEValuesBase<dim> &fe_face_values = info1.fe_values();
266  const unsigned int dofs_per_cell = fe_face_values.dofs_per_cell;
267 
268  const FEValuesBase<dim> &fe_face_values_neighbor = info2.fe_values();
269  const unsigned int neighbor_dofs_per_cell =
270  fe_face_values_neighbor.dofs_per_cell;
271 
272  FullMatrix<double> &u1_v1_matrix = dinfo1.matrix(0, false).matrix;
273  FullMatrix<double> &u2_v1_matrix = dinfo1.matrix(0, true).matrix;
274  FullMatrix<double> &u1_v2_matrix = dinfo2.matrix(0, true).matrix;
275  FullMatrix<double> &u2_v2_matrix = dinfo2.matrix(0, false).matrix;
276 
277 
278  const std::vector<double> & JxW = fe_face_values.get_JxW_values();
279  const std::vector<Tensor<1, dim>> &normals =
280  fe_face_values.get_normal_vectors();
281 
282  for (unsigned int point = 0; point < fe_face_values.n_quadrature_points;
283  ++point)
284  {
285  const double beta_dot_n =
286  beta(fe_face_values.quadrature_point(point)) * normals[point];
287  if (beta_dot_n > 0)
288  {
289  for (unsigned int i = 0; i < dofs_per_cell; ++i)
290  for (unsigned int j = 0; j < dofs_per_cell; ++j)
291  u1_v1_matrix(i, j) += beta_dot_n *
292  fe_face_values.shape_value(j, point) *
293  fe_face_values.shape_value(i, point) *
294  JxW[point];
295 
296  for (unsigned int k = 0; k < neighbor_dofs_per_cell; ++k)
297  for (unsigned int j = 0; j < dofs_per_cell; ++j)
298  u1_v2_matrix(k, j) +=
299  -beta_dot_n *
300  fe_face_values.shape_value(j, point) *
301  fe_face_values_neighbor.shape_value(k, point) *
302  JxW[point];
303  }
304  else
305  {
306  for (unsigned int i = 0; i < dofs_per_cell; ++i)
307  for (unsigned int l = 0; l < neighbor_dofs_per_cell; ++l)
308  u2_v1_matrix(i, l) +=
309  beta_dot_n *
310  fe_face_values_neighbor.shape_value(l, point) *
311  fe_face_values.shape_value(i, point) *
312  JxW[point];
313 
314  for (unsigned int k = 0; k < neighbor_dofs_per_cell; ++k)
315  for (unsigned int l = 0; l < neighbor_dofs_per_cell; ++l)
316  u2_v2_matrix(k, l) +=
317  -beta_dot_n *
318  fe_face_values_neighbor.shape_value(l, point) *
319  fe_face_values_neighbor.shape_value(k, point) *
320  JxW[point];
321  }
322  }

417  assemble_system();
