238  template <int dim>
239  template <class Iterator>
240  void LaplaceProblem<dim>::cell_worker(const Iterator & cell,
241  ScratchData<dim> &scratch_data,
242  CopyData & copy_data)
243  {
244  FEValues<dim> &fe_values = scratch_data.fe_values;
245  fe_values.reinit(cell);
246 
247  const unsigned int dofs_per_cell = fe_values.get_fe().n_dofs_per_cell();
248  const unsigned int n_q_points = fe_values.get_quadrature().size();
249 
250  copy_data.reinit(cell, dofs_per_cell);
251 
252  const std::vector<double> &JxW = fe_values.get_JxW_values();
253 
254  for (unsigned int q = 0; q < n_q_points; ++q)
255  {
256  const double coefficient =
257  (fe_values.get_quadrature_points()[q][0] < 0.0) ? 1.0 : 0.1;
258 
259  for (unsigned int i = 0; i < dofs_per_cell; ++i)
260  {
261  for (unsigned int j = 0; j < dofs_per_cell; ++j)
262  {
263  copy_data.cell_matrix(i, j) +=
264  coefficient *
265  (fe_values.shape_grad(i, q) * fe_values.shape_grad(j, q)) *
266  JxW[q];
267  }
268  copy_data.cell_rhs(i) += 1.0 * fe_values.shape_value(i, q) * JxW[q];
269  }
270  }
271  }

317  void LaplaceProblem<dim>::assemble_multigrid()
318  {
319  MappingQ1<dim> mapping;
320  const unsigned int n_levels = triangulation.n_levels();
321 
322  std::vector<AffineConstraints<double>> boundary_constraints(n_levels);
323  for (unsigned int level = 0; level < n_levels; ++level)
324  {
325  IndexSet dofset;
326  DoFTools::extract_locally_relevant_level_dofs(dof_handler,
327  level,
328  dofset);
329  boundary_constraints[level].reinit(dofset);
330  boundary_constraints[level].add_lines(
331  mg_constrained_dofs.get_refinement_edge_indices(level));
332  boundary_constraints[level].add_lines(
333  mg_constrained_dofs.get_boundary_indices(level));
334  boundary_constraints[level].close();
335  }
336 
337  auto cell_worker =
338  [&](const typename DoFHandler<dim>::level_cell_iterator &cell,
339  ScratchData<dim> & scratch_data,
340  CopyData & copy_data) {
341  this->cell_worker(cell, scratch_data, copy_data);
342  };
343 
344  auto copier = [&](const CopyData &cd) {
345  boundary_constraints[cd.level].distribute_local_to_global(
346  cd.cell_matrix, cd.local_dof_indices, mg_matrices[cd.level]);
347 
348  const unsigned int dofs_per_cell = cd.local_dof_indices.size();
349 
350  for (unsigned int i = 0; i < dofs_per_cell; ++i)
351  for (unsigned int j = 0; j < dofs_per_cell; ++j)
352  if (mg_constrained_dofs.is_interface_matrix_entry(
353  cd.level, cd.local_dof_indices[i], cd.local_dof_indices[j]))
354  {
355  mg_interface_matrices[cd.level].add(cd.local_dof_indices[i],
356  cd.local_dof_indices[j],

358  }
359  };
360 
361  const unsigned int n_gauss_points = degree + 1;
362 
363  ScratchData<dim> scratch_data(mapping,
364  fe,
365  n_gauss_points,
366  update_values | update_gradients |

368  update_quadrature_points);

382  template <int dim>
383  void LaplaceProblem<dim>::solve()
384  {
385  MGTransferPrebuilt<Vector<double>> mg_transfer(mg_constrained_dofs);
386  mg_transfer.build(dof_handler);
387 
388  FullMatrix<double> coarse_matrix;
389  coarse_matrix.copy_from(mg_matrices[0]);
390  MGCoarseGridHouseholder<double, Vector<double>> coarse_grid_solver;
391  coarse_grid_solver.initialize(coarse_matrix);
392 
393  using Smoother = PreconditionSOR<SparseMatrix<double>>;
394  mg::SmootherRelaxation<Smoother, Vector<double>> mg_smoother;
395  mg_smoother.initialize(mg_matrices);
396  mg_smoother.set_steps(2);
397  mg_smoother.set_symmetric(true);
398 
399  mg::Matrix<Vector<double>> mg_matrix(mg_matrices);
400  mg::Matrix<Vector<double>> mg_interface_up(mg_interface_matrices);
401  mg::Matrix<Vector<double>> mg_interface_down(mg_interface_matrices);
402 
403  Multigrid<Vector<double>> mg(
404  mg_matrix, coarse_grid_solver, mg_transfer, mg_smoother, mg_smoother);
405  mg.set_edge_matrices(mg_interface_down, mg_interface_up);
406 
407  PreconditionMG<dim, Vector<double>, MGTransferPrebuilt<Vector<double>>>
408  preconditioner(dof_handler, mg, mg_transfer);
409 
410  SolverControl solver_control(1000, 1e-12);
411  SolverCG<Vector<double>> solver(solver_control);
412 
413  solution = 0;
414 
415  solver.solve(system_matrix, solution, system_rhs, preconditioner);
416  std::cout << " Number of CG iterations: " << solver_control.last_step()
417  << "\n"
418  << std::endl;
419  constraints.distribute(solution);
420  }
