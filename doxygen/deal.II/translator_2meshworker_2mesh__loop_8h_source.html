395  auto cell_action = [&](const CellIteratorBaseType &cell,
396  ScratchData & scratch,
397  CopyData & copy) {
398  // First reset the CopyData class to the empty copy_data given by the
399  // user.
400  copy = sample_copy_data;
401 
402  // Store the dimension in which we are working for later use
403  const auto dim = cell->get_triangulation().dimension;
404 
405  const bool ignore_subdomain =
406  (cell->get_triangulation().locally_owned_subdomain() ==
407  numbers::invalid_subdomain_id);
408 
409  types::subdomain_id current_subdomain_id =
410  (cell->is_level_cell() ? cell->level_subdomain_id() :
411  cell->subdomain_id());
412 
413  const bool own_cell =
414  ignore_subdomain ||
415  (current_subdomain_id ==
416  cell->get_triangulation().locally_owned_subdomain());
417 
418  if ((!ignore_subdomain) &&
419  (current_subdomain_id == numbers::artificial_subdomain_id))
420  return;
421 
422  if (!(flags & (cells_after_faces)) &&
423  (((flags & (assemble_own_cells)) && own_cell) ||
424  ((flags & assemble_ghost_cells) && !own_cell)))
425  cell_worker(cell, scratch, copy);
426 
427  if (flags & (work_on_faces | work_on_boundary))
428  for (const unsigned int face_no : cell->face_indices())
429  {
430  if (cell->at_boundary(face_no) &&
431  !cell->has_periodic_neighbor(face_no))
432  {
433  // only integrate boundary faces of own cells
434  if ((flags & assemble_boundary_faces) && own_cell)
435  boundary_worker(cell, face_no, scratch, copy);
436  }
437  else
438  {
439  // interior face, potentially assemble
440  TriaIterator<typename CellIteratorBaseType::AccessorType>
441  neighbor = cell->neighbor_or_periodic_neighbor(face_no);
442 
443  types::subdomain_id neighbor_subdomain_id =
444  numbers::artificial_subdomain_id;
445  if (neighbor->is_level_cell())
446  neighbor_subdomain_id = neighbor->level_subdomain_id();
447  // subdomain id is only valid for active cells
448  else if (neighbor->is_active())
449  neighbor_subdomain_id = neighbor->subdomain_id();
450 
451  const bool own_neighbor =
452  ignore_subdomain ||
453  (neighbor_subdomain_id ==
454  cell->get_triangulation().locally_owned_subdomain());
455 
456  // skip all faces between two ghost cells
457  if (!own_cell && !own_neighbor)
458  continue;
459 
460  // skip if the user doesn't want faces between own cells
461  if (own_cell && own_neighbor &&
462  !(flags & (assemble_own_interior_faces_both |
463  assemble_own_interior_faces_once)))
464  continue;
465 
466  // skip face to ghost
467  if (own_cell != own_neighbor &&
468  !(flags &
469  (assemble_ghost_faces_both | assemble_ghost_faces_once)))
470  continue;
471 
472  // Deal with refinement edges from the refined side. Assuming
473  // one-irregular meshes, this situation should only occur if
474  // both cells are active.
475  const bool periodic_neighbor =
476  cell->has_periodic_neighbor(face_no);
477 
478  if (dim > 1 && ((!periodic_neighbor &&
479  cell->neighbor_is_coarser(face_no) &&
480  neighbor->is_active()) ||
481  (periodic_neighbor &&
482  cell->periodic_neighbor_is_coarser(face_no) &&
483  neighbor->is_active())))
484  {
485  Assert(cell->is_active(), ExcInternalError());
486 
487  // skip if only one processor needs to assemble the face
488  // to a ghost cell and the fine cell is not ours.
489  if (!own_cell && (flags & assemble_ghost_faces_once))
490  continue;
491 
492  const std::pair<unsigned int, unsigned int>
493  neighbor_face_no =
494  periodic_neighbor ?
495  cell->periodic_neighbor_of_coarser_periodic_neighbor(
496  face_no) :
497  cell->neighbor_of_coarser_neighbor(face_no);
498 
499  face_worker(cell,
500  face_no,
501  numbers::invalid_unsigned_int,
502  neighbor,
503  neighbor_face_no.first,
504  neighbor_face_no.second,
505  scratch,
506  copy);
507 
508  if (flags & assemble_own_interior_faces_both)
509  {
510  // If own faces are to be assembled from both sides,
511  // call the faceworker again with swapped arguments.
512  // This is because we won't be looking at an adaptively
513  // refined edge coming from the other side.
514  face_worker(neighbor,
515  neighbor_face_no.first,
516  neighbor_face_no.second,
517  cell,
518  face_no,
519  numbers::invalid_unsigned_int,
520  scratch,
521  copy);
522  }
523  }
524  else if (dim == 1 && cell->level() > neighbor->level())
525  {
526  // In one dimension, there is no other check to do
527  const unsigned int neighbor_face_no =
528  periodic_neighbor ?
529  cell->periodic_neighbor_face_no(face_no) :
530  cell->neighbor_face_no(face_no);
531  Assert(periodic_neighbor ||
532  neighbor->face(neighbor_face_no) ==
533  cell->face(face_no),
534  ExcInternalError());
535 
536  face_worker(cell,
537  face_no,
538  numbers::invalid_unsigned_int,
539  neighbor,
540  neighbor_face_no,
541  numbers::invalid_unsigned_int,
542  scratch,
543  copy);
544 
545  if (flags & assemble_own_interior_faces_both)
546  {
547  // If own faces are to be assembled from both sides,
548  // call the faceworker again with swapped arguments.
549  face_worker(neighbor,
550  neighbor_face_no,
551  numbers::invalid_unsigned_int,
552  cell,
553  face_no,
554  numbers::invalid_unsigned_int,
555  scratch,
556  copy);
557  }
558  }
559  else
560  {
561  // If iterator is active and neighbor is refined, skip
562  // internal face.
563  if (::internal::is_active_iterator(cell) &&
564  neighbor->has_children())
565  continue;
566 
567  // Now neighbor is on the same refinement level.
568  // Double check.
569  Assert(!cell->neighbor_is_coarser(face_no),
570  ExcInternalError());
571 
572  // If we own both cells only do faces from one side (unless
573  // AssembleFlags says otherwise). Here, we rely on cell
574  // comparison that will look at cell->index().
575  if (own_cell && own_neighbor &&
576  (flags & assemble_own_interior_faces_once) &&
577  (neighbor < cell))
578  continue;
579 
580  // We only look at faces to ghost on the same level once
581  // (only where own_cell=true and own_neighbor=false)
582  if (!own_cell)
583  continue;
584 
585  // now only one processor assembles faces_to_ghost. We let
586  // the processor with the smaller (level-)subdomain id
587  // assemble the face.
588  if (own_cell && !own_neighbor &&
589  (flags & assemble_ghost_faces_once) &&
590  (neighbor_subdomain_id < current_subdomain_id))
591  continue;
592 
593  const unsigned int neighbor_face_no =
594  periodic_neighbor ?
595  cell->periodic_neighbor_face_no(face_no) :
596  cell->neighbor_face_no(face_no);
597  Assert(periodic_neighbor ||
598  neighbor->face(neighbor_face_no) ==
599  cell->face(face_no),
600  ExcInternalError());
601 
602  face_worker(cell,
603  face_no,
604  numbers::invalid_unsigned_int,
605  neighbor,
606  neighbor_face_no,
607  numbers::invalid_unsigned_int,
608  scratch,
609  copy);
610  }
611  }
612  } // faces
613 
614  // Execute the cell_worker if faces are handled before cells
615  if ((flags & cells_after_faces) &&
616  (((flags & assemble_own_cells) && own_cell) ||
617  ((flags & assemble_ghost_cells) && !own_cell)))
618  cell_worker(cell, scratch, copy);
619  };
620

622  WorkStream::run(begin,
623  end,
624  cell_action,
625  copier,
626  sample_scratch_data,
627  sample_copy_data,
628  queue_length,
629  chunk_size);
630  }
