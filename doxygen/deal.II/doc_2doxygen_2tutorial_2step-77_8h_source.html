392  * <a name="AssemblingandfactorizingtheJacobianmatrix"></a>
393  * <h4>Assembling and factorizing the Jacobian matrix</h4>
394  *
395 
396  *
397  * The following function is then responsible for assembling and factorizing
398  * the Jacobian matrix. The first half of the function is in essence the
399  * `assemble_system()` function of @ref step_15 "step-15", except that it does not deal with
400  * also forming a right hand side vector (i.e., the residual) since we do not
401  * always have to do these operations at the same time.
402  *
403 
404  *
405  * We put the whole assembly functionality into a code block enclosed by curly
406  * braces so that we can use a TimerOutput::Scope variable to measure how much
407  * time is spent in this code block, excluding everything that happens in this
408  * function after the matching closing brace `}`.
409  *
410  * @code
411  * template <int dim>
412  * void MinimalSurfaceProblem<dim>::compute_and_factorize_jacobian(
413  * const Vector<double> &evaluation_point)
414  * {
415  * {
416  * TimerOutput::Scope t(computing_timer, "assembling the Jacobian");
417  *
418  * std::cout << " Computing Jacobian matrix" << std::endl;
419  *
420  * const QGauss<dim> quadrature_formula(fe.degree + 1);
421  *
422  * jacobian_matrix = 0;
423  *
424  * FEValues<dim> fe_values(fe,
425  * quadrature_formula,
426  * update_gradients | update_quadrature_points |
427  * update_JxW_values);
428  *
429  * const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
430  * const unsigned int n_q_points = quadrature_formula.size();
431  *
432  * FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
433  *
434  * std::vector<Tensor<1, dim>> evaluation_point_gradients(n_q_points);
435  *
436  * std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
437  *
438  * for (const auto &cell : dof_handler.active_cell_iterators())
439  * {
440  * cell_matrix = 0;
441  *
442  * fe_values.reinit(cell);
443  *
444  * fe_values.get_function_gradients(evaluation_point,
445  * evaluation_point_gradients);
446  *
447  * for (unsigned int q = 0; q < n_q_points; ++q)
448  * {
449  * const double coeff =
450  * 1.0 / std::sqrt(1 + evaluation_point_gradients[q] *
451  * evaluation_point_gradients[q]);
452  *
453  * for (unsigned int i = 0; i < dofs_per_cell; ++i)
454  * {
455  * for (unsigned int j = 0; j < dofs_per_cell; ++j)
456  * cell_matrix(i, j) +=
457  * (((fe_values.shape_grad(i, q) // ((\nabla \phi_i
458  * * coeff // * a_n
459  * * fe_values.shape_grad(j, q)) // * \nabla \phi_j)
460  * - // -
461  * (fe_values.shape_grad(i, q) // (\nabla \phi_i
462  * * coeff * coeff * coeff // * a_n^3
463  * *
464  * (fe_values.shape_grad(j, q) // * (\nabla \phi_j
465  * * evaluation_point_gradients[q]) // * \nabla u_n)
466  * * evaluation_point_gradients[q])) // * \nabla u_n)))
467  * * fe_values.JxW(q)); // * dx
468  * }
469  * }
470  *
471  * cell->get_dof_indices(local_dof_indices);
472  * hanging_node_constraints.distribute_local_to_global(cell_matrix,
473  * local_dof_indices,
474  * jacobian_matrix);
475  * }
476  *
477  * std::map<types::global_dof_index, double> boundary_values;
478  * VectorTools::interpolate_boundary_values(dof_handler,
479  * 0,
480  * Functions::ZeroFunction<dim>(),
481  * boundary_values);
482  * Vector<double> dummy_solution(dof_handler.n_dofs());
483  * Vector<double> dummy_rhs(dof_handler.n_dofs());
484  * MatrixTools::apply_boundary_values(boundary_values,
485  * jacobian_matrix,
486  * dummy_solution,
487  * dummy_rhs);
488  * }

503  *
504

530  * <a name="Computingtheresidualvector"></a>
531  * <h4>Computing the residual vector</h4>
532  *
533 
534  *
535  * The second part of what `assemble_system()` used to do in @ref step_15 "step-15" is
536  * computing the residual vector, i.e., the right hand side vector of the
537  * Newton linear systems. We have broken this out of the previous function,
538  * but the following function will be easy to understand if you understood
539  * what `assemble_system()` in @ref step_15 "step-15" did. Importantly, however, we need to
540  * compute the residual not linearized around the current solution vector, but
541  * whatever we get from KINSOL. This is necessary for operations such as line
542  * search where we want to know what the residual @f$F(U^k + \alpha_k \delta
543  * U^K)@f$ is for different values of @f$\alpha_k@f$; KINSOL in those cases simply
544  * gives us the argument to the function @f$F@f$ and we then compute the residual
545  * @f$F(\cdot)@f$ at this point.
546  *
547 
548  *
549  * The function prints the norm of the so-computed residual at the end as a
550  * way for us to follow along the progress of the program.
551  *
552  * @code
553  * template <int dim>
554  * void MinimalSurfaceProblem<dim>::compute_residual(
555  * const Vector<double> &evaluation_point,
556  * Vector<double> & residual)
557  * {
558  * TimerOutput::Scope t(computing_timer, "assembling the residual");
559  *
560  * std::cout << " Computing residual vector..." << std::flush;
561  *
562  * const QGauss<dim> quadrature_formula(fe.degree + 1);
563  * FEValues<dim> fe_values(fe,
564  * quadrature_formula,
565  * update_gradients | update_quadrature_points |
566  * update_JxW_values);
567  *
568  * const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
569  * const unsigned int n_q_points = quadrature_formula.size();
570  *
571  * Vector<double> cell_residual(dofs_per_cell);
572  * std::vector<Tensor<1, dim>> evaluation_point_gradients(n_q_points);
573  *
574  * std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
575  *
576  * for (const auto &cell : dof_handler.active_cell_iterators())
577  * {
578  * cell_residual = 0;
579  * fe_values.reinit(cell);
580  *
581  * fe_values.get_function_gradients(evaluation_point,
582  * evaluation_point_gradients);
583  *
584  *
585  * for (unsigned int q = 0; q < n_q_points; ++q)
586  * {
587  * const double coeff =
588  * 1.0 / std::sqrt(1 + evaluation_point_gradients[q] *
589  * evaluation_point_gradients[q]);
590  *
591  * for (unsigned int i = 0; i < dofs_per_cell; ++i)
592  * cell_residual(i) = (fe_values.shape_grad(i, q) // \nabla \phi_i
593  * * coeff // * a_n
594  * * evaluation_point_gradients[q] // * u_n
595  * * fe_values.JxW(q)); // * dx
596  * }
597  *
598  * cell->get_dof_indices(local_dof_indices);
599  * for (unsigned int i = 0; i < dofs_per_cell; ++i)
600  * residual(local_dof_indices[i]) += cell_residual(i);
601  * }
602  *
603  * hanging_node_constraints.condense(residual);
604  *
605  * for (const types::global_dof_index i :
606  * DoFTools::extract_boundary_dofs(dof_handler))
607  * residual(i) = 0;
608  *
609  * for (const types::global_dof_index i :
610  * DoFTools::extract_hanging_node_dofs(dof_handler))
611  * residual(i) = 0;
612  *
613  * std::cout << " norm=" << residual.l2_norm() << std::endl;

621  * <a name="SolvinglinearsystemswiththeJacobianmatrix"></a>
622  * <h4>Solving linear systems with the Jacobian matrix</h4>
623  *
624 
625  *
626  * Next up is the function that implements the solution of a linear system
627  * with the Jacobian matrix. Since we have already factored the matrix when we
628  * built the matrix, solving a linear system comes down to applying the
629  * inverse matrix to the given right hand side vector: This is what the
630  * SparseDirectUMFPACK::vmult() function does that we use here. Following
631  * this, we have to make sure that we also address the values of hanging nodes
632  * in the solution vector, and this is done using
633  * AffineConstraints::distribute().
634  *
635 
636  *
637  * The function takes an additional, but unused, argument `tolerance` that
638  * indicates how accurately we have to solve the linear system. The meaning of
639  * this argument is discussed in the introduction in the context of the
640  * "Eisenstat Walker trick", but since we are using a direct rather than an
641  * iterative solver, we are not using this opportunity to solve linear systems
642  * only inexactly.
643  *
644  * @code
645  * template <int dim>
646  * void MinimalSurfaceProblem<dim>::solve(const Vector<double> &rhs,
647  * Vector<double> & solution,
648  * const double /*tolerance*/)
649  * {
650  * TimerOutput::Scope t(computing_timer, "linear system solve");
651  *
652  * std::cout << " Solving linear system" << std::endl;
653  *
654  * jacobian_matrix_factorization->vmult(solution, rhs);
655  *
656  * hanging_node_constraints.distribute(solution);
657  * }
658  *
659  *
660  *
661  * @endcode

757  * <a name="Therunfunctionandtheoveralllogicoftheprogram"></a>
758  * <h4>The run() function and the overall logic of the program</h4>
759  *
760 
761  *
762  * The only function that *really* is interesting in this program is the one
763  * that drives the overall algorithm of starting on a coarse mesh, doing some
764  * mesh refinement cycles, and on each mesh using KINSOL to find the solution
765  * of the nonlinear algebraic equation we obtain from discretization on this
766  * mesh. The `refine_mesh()` function above makes sure that the solution on
767  * one mesh is used as the starting guess on the next mesh. We also use a
768  * TimerOutput object to measure how much time every operation on each mesh
769  * costs, and reset the timer at the beginning of each cycle.
770  *
771 
772  *
773  * As discussed in the introduction, it is not necessary to solve problems on
774  * coarse meshes particularly accurately since these will only solve as
775  * starting guesses for the next mesh. As a consequence, we will use a target
776  * tolerance of
777  * @f$\tau=10^{-3} \frac{1}{10^k}@f$ for the @f$k@f$th mesh refinement cycle.
778  *
779 
780  *
781  * All of this is encoded in the first part of this function:
782  *
783  * @code
784  * template <int dim>
785  * void MinimalSurfaceProblem<dim>::run()
786  * {
787  * GridGenerator::hyper_ball(triangulation);
788  * triangulation.refine_global(2);
789  *
790  * setup_system(/*initial_step=*/true);
791  * set_boundary_values();
792  *
793  * for (unsigned int refinement_cycle = 0; refinement_cycle < 6;
794  * ++refinement_cycle)
795  * {
796  * computing_timer.reset();
797  * std::cout << "Mesh refinement step " << refinement_cycle << std::endl;
798  *
799  * if (refinement_cycle != 0)
800  * refine_mesh();
801  *
802  * const double target_tolerance = 1e-3 * std::pow(0.1, refinement_cycle);
803  * std::cout << " Target_tolerance: " << target_tolerance << std::endl
804  * << std::endl;
805  *
806  * @endcode
807  *
808  * This is where the fun starts. At the top we create the KINSOL solver
809  * object and feed it with an object that encodes a number of additional
810  * specifics (of which we only change the nonlinear tolerance we want to
811  * reach; but you might want to look into what other members the
812  * SUNDIALS::KINSOL::AdditionalData class has and play with them).
813  *
814  * @code
815  * {
816  * typename SUNDIALS::KINSOL<Vector<double>>::AdditionalData
817  * additional_data;
818  * additional_data.function_tolerance = target_tolerance;
819  *
820  * SUNDIALS::KINSOL<Vector<double>> nonlinear_solver(additional_data);
821  *
822  * @endcode
823  *
824  * Then we have to describe the operations that were already mentioned
825  * in the introduction. In essence, we have to teach KINSOL how to (i)
826  * resize a vector to the correct size, (ii) compute the residual
827  * vector, (iii) compute the Jacobian matrix (during which we also
828  * compute its factorization), and (iv) solve a linear system with the
829  * Jacobian.
830  *
831 
832  *
833  * All four of these operations are represented by member variables of
834  * the SUNDIALS::KINSOL class that are of type `std::function`, i.e.,
835  * they are objects to which we can assign a pointer to a function or,
836  * as we do here, a "lambda function" that takes the appropriate
837  * arguments and returns the appropriate information. By convention,
838  * KINSOL wants that functions doing something nontrivial return an
839  * integer where zero indicates success. It turns out that we can do
840  * all of this in just 25 lines of code.
841  *
842 
843  *
844  * (If you're not familiar what "lambda functions" are, take
845  * a look at @ref step_12 "step-12" or at the
846  * [wikipedia page](https://en.wikipedia.org/wiki/Anonymous_function)
847  * on the subject. The idea of lambda functions is that one
848  * wants to define a function with a certain set of
849  * arguments, but (i) not make it a named functions because,
850  * typically, the function is used in only one place and it
851  * seems unnecessary to give it a global name; and (ii) that
852  * the function has access to some of the variables that
853  * exist at the place where it is defined, including member
854  * variables. The syntax of lambda functions is awkward, but
855  * ultimately quite useful.)
856  *
857 
858  *
859  * At the very end of the code block we then tell KINSOL to go to work
860  * and solve our problem. The member functions called from the
861  * 'residual', 'setup_jacobian', and 'solve_jacobian_system' functions
862  * will then print output to screen that allows us to follow along
863  * with the progress of the program.
864  *
865  * @code
866  * nonlinear_solver.reinit_vector = [&](Vector<double> &x) {
867  * x.reinit(dof_handler.n_dofs());
868  * };
869  *
870  * nonlinear_solver.residual =
871  * [&](const Vector<double> &evaluation_point,
872  * Vector<double> & residual) {
873  * compute_residual(evaluation_point, residual);
874  *
875  * return 0;
876  * };
877  *
878  * nonlinear_solver.setup_jacobian =
879  * [&](const Vector<double> &current_u,
880  * const Vector<double> & /*current_f*/) {
881  * compute_and_factorize_jacobian(current_u);
882  *
883  * return 0;
884  * };
885  *
886  * nonlinear_solver.solve_with_jacobian = [&](const Vector<double> &rhs,
887  * Vector<double> & dst,
888  * const double tolerance) {
889  * this->solve(rhs, dst, tolerance);
890  *
891  * return 0;
892  * };
893  *
894  * nonlinear_solver.solve(current_solution);

897  * @endcode

903  *
