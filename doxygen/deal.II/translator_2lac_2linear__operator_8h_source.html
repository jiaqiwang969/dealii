418 LinearOperator<Range, Domain, Payload>
419 operator+(const LinearOperator<Range, Domain, Payload> &first_op,
420  const LinearOperator<Range, Domain, Payload> &second_op)
421 {
422  if (first_op.is_null_operator)
423  {
424  return second_op;
425  }
426  else if (second_op.is_null_operator)
427  {
428  return first_op;
429  }

433  static_cast<const Payload &>(first_op) +

437  return_op.reinit_domain_vector = first_op.reinit_domain_vector;

522 template <typename Range, typename Domain, typename Payload>
523 LinearOperator<Range, Domain, Payload>
524 operator*(typename Range::value_type number,
525  const LinearOperator<Range, Domain, Payload> &op)
526 {
527  static_assert(
528  std::is_convertible<typename Range::value_type,
529  typename Domain::value_type>::value,
530  "Range and Domain must have implicitly convertible 'value_type's");
531 
532  if (op.is_null_operator)
533  {
534  return op;
535  }
536  else if (number == 0.)
537  {
538  return null_operator(op);
539  }
540  else
541  {
542  LinearOperator<Range, Domain, Payload> return_op = op;
543 
544  // ensure to have valid computation objects by catching number and op by
545  // value
546 
547  return_op.vmult = [number, op](Range &v, const Domain &u) {
548  op.vmult(v, u);
549  v *= number;
550  };
551 
552  return_op.vmult_add = [number, op](Range &v, const Domain &u) {
553  v /= number;
554  op.vmult_add(v, u);
555  v *= number;
556  };
557 
558  return_op.Tvmult = [number, op](Domain &v, const Range &u) {
559  op.Tvmult(v, u);
560  v *= number;
561  };
562 
563  return_op.Tvmult_add = [number, op](Domain &v, const Range &u) {
564  v /= number;
565  op.Tvmult_add(v, u);
566  v *= number;
567  };
568 
569  return return_op;
570  }

644  static_cast<const Payload &>(first_op) *
