138  template <int dim>
139  bool ConstitutiveLaw<dim>::get_stress_strain_tensor(
140  const SymmetricTensor<2, dim> &strain_tensor,
141  SymmetricTensor<4, dim> & stress_strain_tensor) const
142  {
143  Assert(dim == 3, ExcNotImplemented());
144 
145  SymmetricTensor<2, dim> stress_tensor;
146  stress_tensor =
147  (stress_strain_tensor_kappa + stress_strain_tensor_mu) * strain_tensor;
148 
149  const SymmetricTensor<2, dim> deviator_stress_tensor =
150  deviator(stress_tensor);
151  const double deviator_stress_tensor_norm = deviator_stress_tensor.norm();
152 
153  stress_strain_tensor = stress_strain_tensor_mu;
154  if (deviator_stress_tensor_norm > sigma_0)
155  {
156  const double beta = sigma_0 / deviator_stress_tensor_norm;
157  stress_strain_tensor *= (gamma + (1 - gamma) * beta);
158  }
159 
160  stress_strain_tensor += stress_strain_tensor_kappa;
161 
162  return (deviator_stress_tensor_norm > sigma_0);
163  }

379  template <int dim>
380  double BitmapFile<dim>::get_value(const double x, const double y) const
381  {
382  const int ix = std::min(std::max(static_cast<int>(x / hx), 0), nx - 2);
383  const int iy = std::min(std::max(static_cast<int>(y / hy), 0), ny - 2);
384 
385  const double xi = std::min(std::max((x - ix * hx) / hx, 1.), 0.);
386  const double eta = std::min(std::max((y - iy * hy) / hy, 1.), 0.);
387 
388  return ((1 - xi) * (1 - eta) * get_pixel_value(ix, iy) +
389  xi * (1 - eta) * get_pixel_value(ix + 1, iy) +
390  (1 - xi) * eta * get_pixel_value(ix, iy + 1) +
391  xi * eta * get_pixel_value(ix + 1, iy + 1));
392  }

965  quadrature_formula,

967  update_JxW_values);

970  face_quadrature_formula,
971  update_values | update_quadrature_points |
972  update_JxW_values);
973

975  const unsigned int n_q_points = quadrature_formula.size();
976  const unsigned int n_face_q_points = face_quadrature_formula.size();
977 
978  const EquationData::BoundaryForce<dim> boundary_force;
979  std::vector<Vector<double>> boundary_force_values(n_face_q_points,

994  cell_rhs = 0;

1006  stress_strain_tensor_linearized,

1010  {

1012  stress_strain_tensor_linearized *
1013  fe_values[displacement].symmetric_gradient(i, q_point);

1019  fe_values.JxW(q_point));

1022  ((stress_phi_i -
1023  stress_strain_tensor *

1025  q_point)) *
1026  strain_tensor[q_point] * fe_values.JxW(q_point));

1028  }
1029 
1030  for (const auto &face : cell->face_iterators())
1031  if (face->at_boundary() && face->boundary_id() == 1)
1032  {

1034

1036  fe_values_face.get_quadrature_points(),
1037  boundary_force_values);
1038 
1039  for (unsigned int q_point = 0; q_point < n_face_q_points;
1040  ++q_point)
1041  {

1043  rhs_values[2] = boundary_force_values[q_point][2];
1044  for (unsigned int i = 0; i < dofs_per_cell; ++i)
1045  cell_rhs(i) +=
1046  (fe_values_face[displacement].value(i, q_point) *
1047  rhs_values * fe_values_face.JxW(q_point));
1048  }
1049  }
1050 
1051  cell->get_dof_indices(local_dof_indices);
1052  all_constraints.distribute_local_to_global(cell_matrix,
1053  cell_rhs,
1054  local_dof_indices,
1055  newton_matrix,
1056  newton_rhs,
1057  true);
1058  }
1059 
1060  newton_matrix.compress(VectorOperation::add);
1061  newton_rhs.compress(VectorOperation::add);
