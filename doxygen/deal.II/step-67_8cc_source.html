104  template <int dim>
105  double ExactSolution<dim>::value(const Point<dim> & x,
106  const unsigned int component) const
107  {
108  const double t = this->get_time();
109 
110  switch (testcase)
111  {
112  case 0:
113  {
114  Assert(dim == 2, ExcNotImplemented());
115  const double beta = 5;
116 
117  Point<dim> x0;
118  x0[0] = 5.;
119  const double radius_sqr =
120  (x - x0).norm_square() - 2. * (x[0] - x0[0]) * t + t * t;
121  const double factor =
122  beta / (numbers::PI * 2) * std::exp(1. - radius_sqr);
123  const double density_log = std::log2(
124  std::abs(1. - (gamma - 1.) / gamma * 0.25 * factor * factor));
125  const double density = std::exp2(density_log * (1. / (gamma - 1.)));
126  const double u = 1. - factor * (x[1] - x0[1]);
127  const double v = factor * (x[0] - t - x0[0]);
128 
129  if (component == 0)
130  return density;
131  else if (component == 1)
132  return density * u;
133  else if (component == 2)
134  return density * v;
135  else
136  {
137  const double pressure =
138  std::exp2(density_log * (gamma / (gamma - 1.)));
139  return pressure / (gamma - 1.) +
140  0.5 * (density * u * u + density * v * v);
141  }
142  }
143 
144  case 1:
145  {
146  if (component == 0)
147  return 1.;
148  else if (component == 1)
149  return 0.4;
150  else if (component == dim + 1)
151  return 3.097857142857143;
152  else
153  return 0.;
154  }
155 
156  default:
157  Assert(false, ExcNotImplemented());
158  return 0.;
159  }

267  template <int dim, typename Number>
268  inline DEAL_II_ALWAYS_INLINE
269  Number
270  euler_pressure(const Tensor<1, dim + 2, Number> &conserved_variables)
271  {
272  const Tensor<1, dim, Number> velocity =
273  euler_velocity<dim>(conserved_variables);
274 
275  Number rho_u_dot_u = conserved_variables[1] * velocity[0];
276  for (unsigned int d = 1; d < dim; ++d)
277  rho_u_dot_u += conserved_variables[1 + d] * velocity[d];
278 
279  return (gamma - 1.) * (conserved_variables[dim + 1] - 0.5 * rho_u_dot_u);
280  }

282  template <int dim, typename Number>
283  inline DEAL_II_ALWAYS_INLINE
284  Tensor<1, dim + 2, Tensor<1, dim, Number>>
285  euler_flux(const Tensor<1, dim + 2, Number> &conserved_variables)
286  {
287  const Tensor<1, dim, Number> velocity =
288  euler_velocity<dim>(conserved_variables);
289  const Number pressure = euler_pressure<dim>(conserved_variables);
290 
291  Tensor<1, dim + 2, Tensor<1, dim, Number>> flux;
292  for (unsigned int d = 0; d < dim; ++d)
293  {
294  flux[0][d] = conserved_variables[1 + d];
295  for (unsigned int e = 0; e < dim; ++e)
296  flux[e + 1][d] = conserved_variables[e + 1] * velocity[d];
297  flux[d + 1][d] += pressure;
298  flux[dim + 1][d] =
299  velocity[d] * (conserved_variables[dim + 1] + pressure);
300  }
301 
302  return flux;
303  }

317  template <int dim, typename Number>
318  inline DEAL_II_ALWAYS_INLINE
319  Tensor<1, dim + 2, Number>
320  euler_numerical_flux(const Tensor<1, dim + 2, Number> &u_m,
321  const Tensor<1, dim + 2, Number> &u_p,
322  const Tensor<1, dim, Number> & normal)
323  {
324  const auto velocity_m = euler_velocity<dim>(u_m);
325  const auto velocity_p = euler_velocity<dim>(u_p);
326 
327  const auto pressure_m = euler_pressure<dim>(u_m);
328  const auto pressure_p = euler_pressure<dim>(u_p);
329 
330  const auto flux_m = euler_flux<dim>(u_m);
331  const auto flux_p = euler_flux<dim>(u_p);
332 
333  switch (numerical_flux_type)
334  {
335  case lax_friedrichs_modified:
336  {
337  const auto lambda =
338  0.5 * std::sqrt(std::max(velocity_p.norm_square() +
339  gamma * pressure_p * (1. / u_p[0]),
340  velocity_m.norm_square() +
341  gamma * pressure_m * (1. / u_m[0])));
342 
343  return 0.5 * (flux_m * normal + flux_p * normal) +
344  0.5 * lambda * (u_m - u_p);
345  }
346 
347  case harten_lax_vanleer:
348  {
349  const auto avg_velocity_normal =
350  0.5 * ((velocity_m + velocity_p) * normal);
351  const auto avg_c = std::sqrt(std::abs(
352  0.5 * gamma *
353  (pressure_p * (1. / u_p[0]) + pressure_m * (1. / u_m[0]))));
354  const Number s_pos =
355  std::max(Number(), avg_velocity_normal + avg_c);
356  const Number s_neg =
357  std::min(Number(), avg_velocity_normal - avg_c);
358  const Number inverse_s = Number(1.) / (s_pos - s_neg);
359 
360  return inverse_s *
361  ((s_pos * (flux_m * normal) - s_neg * (flux_p * normal)) -
362  s_pos * s_neg * (u_m - u_p));
363  }
364 
365  default:
366  {
367  Assert(false, ExcNotImplemented());
368  return {};
369  }
370  }
371  }

976  template <int dim, int degree, int n_points_1d>
977  double EulerOperator<dim, degree, n_points_1d>::compute_cell_transport_speed(
978  const LinearAlgebra::distributed::Vector<Number> &solution) const
979  {
980  TimerOutput::Scope t(timer, "compute transport speed");
981  Number max_transport = 0;
982  FEEvaluation<dim, degree, degree + 1, dim + 2, Number> phi(data, 0, 1);
983 
984  for (unsigned int cell = 0; cell < data.n_cell_batches(); ++cell)
985  {
986  phi.reinit(cell);
987  phi.gather_evaluate(solution, EvaluationFlags::values);
988  VectorizedArray<Number> local_max = 0.;
989  for (unsigned int q = 0; q < phi.n_q_points; ++q)
990  {
991  const auto solution = phi.get_value(q);
992  const auto velocity = euler_velocity<dim>(solution);
993  const auto pressure = euler_pressure<dim>(solution);
994 
995  const auto inverse_jacobian = phi.inverse_jacobian(q);
996  const auto convective_speed = inverse_jacobian * velocity;
997  VectorizedArray<Number> convective_limit = 0.;
998  for (unsigned int d = 0; d < dim; ++d)
999  convective_limit =
1000  std::max(convective_limit, std::abs(convective_speed[d]));
1001 
1002  const auto speed_of_sound =
1003  std::sqrt(gamma * pressure * (1. / solution[0]));
1004 
1005  Tensor<1, dim, VectorizedArray<Number>> eigenvector;
1006  for (unsigned int d = 0; d < dim; ++d)
1007  eigenvector[d] = 1.;
1008  for (unsigned int i = 0; i < 5; ++i)
1009  {
1010  eigenvector = transpose(inverse_jacobian) *
1011  (inverse_jacobian * eigenvector);
1012  VectorizedArray<Number> eigenvector_norm = 0.;
1013  for (unsigned int d = 0; d < dim; ++d)
1014  eigenvector_norm =
1015  std::max(eigenvector_norm, std::abs(eigenvector[d]));
1016  eigenvector /= eigenvector_norm;
1017  }
1018  const auto jac_times_ev = inverse_jacobian * eigenvector;
1019  const auto max_eigenvalue = std::sqrt(
1020  (jac_times_ev * jac_times_ev) / (eigenvector * eigenvector));
1021  local_max =
1022  std::max(local_max,
1023  max_eigenvalue * speed_of_sound + convective_limit);
1024  }
1025 
1026  for (unsigned int v = 0; v < data.n_active_entries_per_cell_batch(cell);
1027  ++v)
1028  for (unsigned int d = 0; d < 3; ++d)
1029  max_transport = std::max(max_transport, local_max[v]);
1030  }
1031 
1032  max_transport = Utilities::MPI::max(max_transport, MPI_COMM_WORLD);
1033 
1034  return max_transport;
