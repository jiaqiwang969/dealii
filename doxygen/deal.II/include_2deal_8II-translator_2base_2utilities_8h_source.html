462  template <typename T>
463  constexpr T
464  pow(const T base, const int iexp)
465  {
466 #if defined(DEBUG) && !defined(DEAL_II_CXX14_CONSTEXPR_BUG)
467  // Up to __builtin_expect this is the same code as in the 'Assert' macro.
468  // The call to __builtin_expect turns out to be problematic.
469  if (!(iexp >= 0))
470  ::deal_II_exceptions::internals::issue_error_noreturn(
471  ::deal_II_exceptions::internals::abort_or_throw_on_exception,
472  __FILE__,
473  __LINE__,
474  __PRETTY_FUNCTION__,
475  "iexp>=0",
476  "ExcMessage(\"The exponent must not be negative!\")",
477  ExcMessage("The exponent must not be negative!"));
478 #endif
479  // The "exponentiation by squaring" algorithm used below has to be
480  // compressed to one statement due to C++11's restrictions on constexpr
481  // functions. A more descriptive version would be:
482  //
483  // <code>
484  // if (iexp <= 0)
485  // return 1;
486  //
487  // // avoid overflow of one additional recursion with pow(base * base, 0)
488  // if (iexp == 1)
489  // return base;
490  //
491  // // if the current exponent is not divisible by two,
492  // // we need to account for that.
493  // const unsigned int prefactor = (iexp % 2 == 1) ? base : 1;
494  //
495  // // a^b = (a*a)^(b/2) for b even
496  // // a^b = a*(a*a)^((b-1)/2 for b odd
497  // return prefactor * ::Utilities::pow(base*base, iexp/2);
498  // </code>
499 
500  static_assert(std::is_integral<T>::value, "Only integral types supported");
501 
502  return iexp <= 0 ?
503  1 :
504  (iexp == 1 ? base :
505  (((iexp % 2 == 1) ? base : 1) *
506  ::Utilities::pow(base * base, iexp / 2)));
507  }

1101  inline Iterator
1102  lower_bound(Iterator first, Iterator last, const T &val, const Comp comp)
1103  {
1104  // verify that the two iterators are properly ordered. since
1105  // we need operator- for the iterator type anyway, do the
1106  // test as follows, rather than via 'last >= first'
1107  Assert(last - first >= 0,
1108  ExcMessage(
1109  "The given iterators do not satisfy the proper ordering."));
1110 
1111  unsigned int len = static_cast<unsigned int>(last - first);
1112 
1113  if (len == 0)
1114  return first;
1115 
1116  while (true)
1117  {
1118  // if length equals 8 or less,
1119  // then do a rolled out
1120  // search. use a switch without
1121  // breaks for that and roll-out
1122  // the loop somehow
1123  if (len < 8)
1124  {
1125  switch (len)
1126  {
1127  case 7:
1128  if (!comp(*first, val))
1129  return first;
1130  ++first;
1131  DEAL_II_FALLTHROUGH;
1132  case 6:
1133  if (!comp(*first, val))
1134  return first;
1135  ++first;
1136  DEAL_II_FALLTHROUGH;
1137  case 5:
1138  if (!comp(*first, val))
1139  return first;
1140  ++first;
1141  DEAL_II_FALLTHROUGH;
1142  case 4:
1143  if (!comp(*first, val))
1144  return first;
1145  ++first;
1146  DEAL_II_FALLTHROUGH;
1147  case 3:
1148  if (!comp(*first, val))
1149  return first;
1150  ++first;
1151  DEAL_II_FALLTHROUGH;
1152  case 2:
1153  if (!comp(*first, val))
1154  return first;
1155  ++first;
1156  DEAL_II_FALLTHROUGH;
1157  case 1:
1158  if (!comp(*first, val))
1159  return first;
1160  return first + 1;
1161  default:
1162  // indices seem
1163  // to not be
1164  // sorted
1165  // correctly!? or
1166  // did len
1167  // become==0
1168  // somehow? that
1169  // shouldn't have
1170  // happened
1171  Assert(false, ExcInternalError());
1172  }

1199  size_t
1200  pack(const T & object,
1201  std::vector<char> &dest_buffer,
1202  const bool allow_compression)
1203  {
1204  std::size_t size = 0;
1205 
1206  // see if the object is small and copyable via memcpy. if so, use
1207  // this fast path. otherwise, we have to go through the BOOST
1208  // serialization machinery
1209 #ifdef DEAL_II_HAVE_CXX17
1210  if constexpr (std::is_trivially_copyable<T>() && sizeof(T) < 256)
1211 #else
1212  if (std::is_trivially_copyable<T>() && sizeof(T) < 256)
1213 #endif
1214  {
1215  (void)allow_compression;
1216  const std::size_t previous_size = dest_buffer.size();
1217  dest_buffer.resize(previous_size + sizeof(T));
1218 
1219  std::memcpy(dest_buffer.data() + previous_size, &object, sizeof(T));
1220 
1221  size = sizeof(T);
1222  }
1223  else
1224  {
1225  // use buffer as the target of a compressing
1226  // stream into which we serialize the current object
1227  const std::size_t previous_size = dest_buffer.size();
1228  {
1229  boost::iostreams::filtering_ostreambuf fosb;
1230 #ifdef DEAL_II_WITH_ZLIB
1231  if (allow_compression)
1232  fosb.push(boost::iostreams::gzip_compressor());
1233 #else
1234  (void)allow_compression;
1235 #endif
1236  fosb.push(boost::iostreams::back_inserter(dest_buffer));
1237 
1238  boost::archive::binary_oarchive boa(fosb);
1239  boa << object;
1240  // the stream object has to be destroyed before the return statement
1241  // to ensure that all data has been written in the buffer
1242  }
1243  size = dest_buffer.size() - previous_size;
1244  }
