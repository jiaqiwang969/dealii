165  double df_dx(const double x, const double y)
166  {
167  Assert(x != 0.0, ExcDivideByZero());
168  return y * std::sin(y / x) / (x * x);

177  {

595  template <int dim, Differentiation::AD::NumberTypes ADTypeCode>
596  void
597  Magnetoelastic_Constitutive_Law_AD<dim, ADTypeCode>::update_internal_data(
598  const SymmetricTensor<2, dim> &C,
599  const Tensor<1, dim> & H,
600  const DiscreteTime &)
601  {
602  Assert(determinant(C) > 0, ExcInternalError());
603 
604  ad_helper.reset();
605 
606  ad_helper.register_independent_variable(H, H_components);
607  ad_helper.register_independent_variable(C, C_components);
608 
609  const Tensor<1, dim, ADNumberType> H_ad =
610  ad_helper.get_sensitive_variables(H_components);
611  const SymmetricTensor<2, dim, ADNumberType> C_ad =
612  ad_helper.get_sensitive_variables(C_components);
613 
614  const ADNumberType det_F_ad = std::sqrt(determinant(C_ad));
615  const SymmetricTensor<2, dim, ADNumberType> C_inv_ad = invert(C_ad);
616  AssertThrow(det_F_ad > ADNumberType(0.0),
617  ExcMessage("Volumetric Jacobian must be positive."));
618 
619  const ADNumberType f_mu_e_ad =
620  1.0 + (this->get_mu_e_inf() / this->get_mu_e() - 1.0) *
621  std::tanh((2.0 * H_ad * H_ad) /
622  (this->get_mu_e_h_sat() * this->get_mu_e_h_sat()));
623 
624  const ADNumberType psi_ad =
625  0.5 * this->get_mu_e() * f_mu_e_ad *
626  (trace(C_ad) - dim - 2.0 * std::log(det_F_ad))

816  template <int dim>
817  void Magnetoviscoelastic_Constitutive_Law_SD<dim>::initialize_optimizer()
818  {
819  const Differentiation::SD::Expression det_F_sd =
820  std::sqrt(determinant(C_sd));
821  const SymmetricTensor<2, dim, Differentiation::SD::Expression> C_inv_sd =
822  invert(C_sd);
823 
824  const Differentiation::SD::Expression f_mu_e_sd =
825  1.0 +
826  (mu_e_inf_sd / mu_e_sd - 1.0) *
827  std::tanh((2.0 * H_sd * H_sd) / (mu_e_h_sat_sd * mu_e_h_sat_sd));
828 
829  const Differentiation::SD::Expression psi_ME_sd =
830  0.5 * mu_e_sd * f_mu_e_sd *
831  (trace(C_sd) - dim - 2.0 * std::log(det_F_sd)) +
832  lambda_e_sd * std::log(det_F_sd) * std::log(det_F_sd) -
833  0.5 * this->get_mu_0() * mu_r_sd * det_F_sd * (H_sd * C_inv_sd * H_sd);
834 
835  const Differentiation::SD::Expression f_mu_v_sd =
836  1.0 +
837  (mu_v_inf_sd / mu_v_sd - 1.0) *
838  std::tanh((2.0 * H_sd * H_sd) / (mu_v_h_sat_sd * mu_v_h_sat_sd));
839 
840  const Differentiation::SD::Expression psi_MVE_sd =
841  0.5 * mu_v_sd * f_mu_v_sd *
842  (Q_t_sd * (std::pow(det_F_sd, -2.0 / dim) * C_sd) - dim -
843  std::log(determinant(Q_t_sd)));
844 
845  psi_sd = psi_ME_sd + psi_MVE_sd;
846 
847 
848  B_sd = -Differentiation::SD::differentiate(psi_sd, H_sd);
849  S_sd = 2.0 * Differentiation::SD::differentiate(psi_sd, C_sd);
850 
851  const SymmetricTensor<2, dim, Differentiation::SD::Expression>
852  Q_t_sd_explicit =
853  (1.0 / (1.0 + delta_t_sd / tau_v_sd)) *
854  (Q_t1_sd +
855  (delta_t_sd / tau_v_sd * std::pow(det_F_sd, 2.0 / dim) * C_inv_sd));
856 
857  const Differentiation::SD::types::substitution_map
858  substitution_map_explicit = Differentiation::SD::make_substitution_map(
859  std::make_pair(Q_t_sd, Q_t_sd_explicit));
860 
861  BB_sd = symmetrize(Differentiation::SD::differentiate(
862  Differentiation::SD::substitute(B_sd, substitution_map_explicit),
863  H_sd));
864  PP_sd = -Differentiation::SD::differentiate(
865  Differentiation::SD::substitute(S_sd, substitution_map_explicit), H_sd);
866  HH_sd =
867  2.0 *
868  Differentiation::SD::differentiate(
869  Differentiation::SD::substitute(S_sd, substitution_map_explicit),
870  C_sd);
871 
872  optimizer.register_symbols(
873  Differentiation::SD::Utilities::extract_symbols(
874  make_substitution_map({}, {}, 0)));
875 
876  optimizer.register_functions(psi_sd, B_sd, S_sd, BB_sd, PP_sd, HH_sd);
877 
878  optimizer.optimize();
879  }

999  template <int dim>
1000  void Magnetoelastic_Constitutive_Law<dim>::update_internal_data(
1001  const SymmetricTensor<2, dim> &C,
1002  const Tensor<1, dim> & H,
1003  const DiscreteTime &)
1004  {
1005  const double det_F = std::sqrt(determinant(C));
1006  const SymmetricTensor<2, dim> C_inv = invert(C);
1007  AssertThrow(det_F > 0.0,
1008  ExcMessage("Volumetric Jacobian must be positive."));
1009 
1010  const double two_h_dot_h_div_h_sat_squ =
1011  (2.0 * H * H) / (this->get_mu_e_h_sat() * this->get_mu_e_h_sat());
1012  const double tanh_two_h_dot_h_div_h_sat_squ =
1013  std::tanh(two_h_dot_h_div_h_sat_squ);
1014 
1015  const double f_mu_e =
1016  1.0 + (this->get_mu_e_inf() / this->get_mu_e() - 1.0) *
1017  tanh_two_h_dot_h_div_h_sat_squ;
1018 
1019  const double dtanh_two_h_dot_h_div_h_sat_squ =
1020  std::pow(1.0 / std::cosh(two_h_dot_h_div_h_sat_squ), 2.0);
1021  const Tensor<1, dim> dtwo_h_dot_h_div_h_sat_squ_dH =
1022  2.0 * 2.0 / (this->get_mu_e_h_sat() * this->get_mu_e_h_sat()) * H;
1023 
1024  const Tensor<1, dim> df_mu_e_dH =
1025  (this->get_mu_e_inf() / this->get_mu_e() - 1.0) *
1026  (dtanh_two_h_dot_h_div_h_sat_squ * dtwo_h_dot_h_div_h_sat_squ_dH);
1027 
1028  const double d2tanh_two_h_dot_h_div_h_sat_squ =
1029  -2.0 * tanh_two_h_dot_h_div_h_sat_squ * dtanh_two_h_dot_h_div_h_sat_squ;
1030  const SymmetricTensor<2, dim> d2two_h_dot_h_div_h_sat_squ_dH_dH =
1031  2.0 * 2.0 / (this->get_mu_e_h_sat() * this->get_mu_e_h_sat()) *
1032  Physics::Elasticity::StandardTensors<dim>::I;
1033 
1034  const SymmetricTensor<2, dim> d2f_mu_e_dH_dH =
1035  (this->get_mu_e_inf() / this->get_mu_e() - 1.0) *
1036  (d2tanh_two_h_dot_h_div_h_sat_squ *
1037  symmetrize(outer_product(dtwo_h_dot_h_div_h_sat_squ_dH,
1038  dtwo_h_dot_h_div_h_sat_squ_dH)) +
1039  dtanh_two_h_dot_h_div_h_sat_squ * d2two_h_dot_h_div_h_sat_squ_dH_dH);
1040 
1041  const double log_det_F = std::log(det_F);
1042  const double tr_C = trace(C);
1043  const Tensor<1, dim> C_inv_dot_H = C_inv * H;
1044  const double H_dot_C_inv_dot_H = H * C_inv_dot_H;
1045 
1046  const SymmetricTensor<2, dim> d_tr_C_dC =
1047  Physics::Elasticity::StandardTensors<dim>::I;
1048  const SymmetricTensor<2, dim> ddet_F_dC = 0.5 * det_F * C_inv;
1049  const SymmetricTensor<2, dim> dlog_det_F_dC = 0.5 * C_inv;
1050 
1051  const Tensor<1, dim> dH_dot_C_inv_dot_H_dH = 2.0 * C_inv_dot_H;
1052 
1053  SymmetricTensor<4, dim> dC_inv_dC;
1054  for (unsigned int A = 0; A < dim; ++A)
1055  for (unsigned int B = A; B < dim; ++B)
1056  for (unsigned int C = 0; C < dim; ++C)
1057  for (unsigned int D = C; D < dim; ++D)
1058  dC_inv_dC[A][B][C][D] -=
1059  0.5 * (C_inv[A][C] * C_inv[B][D]
1060  + C_inv[A][D] * C_inv[B][C]);
1061 
1062  const SymmetricTensor<2, dim> dH_dot_C_inv_dot_H_dC =
1063  -symmetrize(outer_product(C_inv_dot_H, C_inv_dot_H));
1064 
1065  const SymmetricTensor<4, dim> d2log_det_F_dC_dC = 0.5 * dC_inv_dC;
1066 
1067  const SymmetricTensor<4, dim> d2det_F_dC_dC =
1068  0.5 * (outer_product(C_inv, ddet_F_dC) + det_F * dC_inv_dC);
1069 
1070  const SymmetricTensor<2, dim> d2H_dot_C_inv_dot_H_dH_dH = 2.0 * C_inv;
1071 
1072  Tensor<3, dim> d2H_dot_C_inv_dot_H_dC_dH;
1073  for (unsigned int A = 0; A < dim; ++A)
1074  for (unsigned int B = 0; B < dim; ++B)
1075  for (unsigned int C = 0; C < dim; ++C)
1076  d2H_dot_C_inv_dot_H_dC_dH[A][B][C] -=
1077  C_inv[A][C] * C_inv_dot_H[B] +
1078  C_inv_dot_H[A] * C_inv[B][C];
1079 
1080  SymmetricTensor<4, dim> d2H_dot_C_inv_dot_H_dC_dC;
1081  for (unsigned int A = 0; A < dim; ++A)
1082  for (unsigned int B = A; B < dim; ++B)
1083  for (unsigned int C = 0; C < dim; ++C)
1084  for (unsigned int D = C; D < dim; ++D)
1085  d2H_dot_C_inv_dot_H_dC_dC[A][B][C][D] +=
1086  0.5 * (C_inv_dot_H[A] * C_inv_dot_H[C] * C_inv[B][D] +
1087  C_inv_dot_H[A] * C_inv_dot_H[D] * C_inv[B][C] +
1088  C_inv_dot_H[B] * C_inv_dot_H[C] * C_inv[A][D] +
1089  C_inv_dot_H[B] * C_inv_dot_H[D] * C_inv[A][C]);
1090 
1091  psi =
1092  (0.5 * this->get_mu_e() * f_mu_e) *
1093  (tr_C - dim - 2.0 * std::log(det_F)) +
1094  this->get_lambda_e() * (std::log(det_F) * std::log(det_F)) -
1095  (0.5 * this->get_mu_0() * this->get_mu_r()) * det_F * (H * C_inv * H);
1096 
1097  B = -(0.5 * this->get_mu_e() * (tr_C - dim - 2.0 * log_det_F)) *
1098  df_mu_e_dH
1099  + 0.5 * this->get_mu_0() * this->get_mu_r() * det_F *
1100  dH_dot_C_inv_dot_H_dH;
1101 
1102  S = 2.0 * (0.5 * this->get_mu_e() * f_mu_e) *
1103  (d_tr_C_dC - 2.0 * dlog_det_F_dC)
1104  + 2.0 * this->get_lambda_e() * (2.0 * log_det_F * dlog_det_F_dC)
1105  - 2.0 * (0.5 * this->get_mu_0() * this->get_mu_r()) *
1106  (H_dot_C_inv_dot_H * ddet_F_dC
1107  + det_F * dH_dot_C_inv_dot_H_dC);
1108 
1109  BB = -(0.5 * this->get_mu_e() * (tr_C - dim - 2.0 * log_det_F)) *
1110  d2f_mu_e_dH_dH
1111  + 0.5 * this->get_mu_0() * this->get_mu_r() * det_F *
1112  d2H_dot_C_inv_dot_H_dH_dH;
1113 
1114  PP = -2.0 * (0.5 * this->get_mu_e()) *
1115  outer_product(Tensor<2, dim>(d_tr_C_dC - 2.0 * dlog_det_F_dC),
1116  df_mu_e_dH)
1117  +
1118  2.0 * (0.5 * this->get_mu_0() * this->get_mu_r()) *
1119  (outer_product(Tensor<2, dim>(ddet_F_dC), dH_dot_C_inv_dot_H_dH)
1120  + det_F * d2H_dot_C_inv_dot_H_dC_dH);
