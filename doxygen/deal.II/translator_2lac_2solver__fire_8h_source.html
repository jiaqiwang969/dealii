265 template <typename VectorType>
266 template <typename PreconditionerType>
267 void
268 SolverFIRE<VectorType>::solve(
269  const std::function<double(VectorType &, const VectorType &)> &compute,
270  VectorType & x,
271  const PreconditionerType &inverse_mass_matrix)
272 {
273  LogStream::Prefix prefix("FIRE");
274 
275  // FIRE algorithm constants
276  const double DELAYSTEP = 5;
277  const double TIMESTEP_GROW = 1.1;
278  const double TIMESTEP_SHRINK = 0.5;
279  const double ALPHA_0 = 0.1;
280  const double ALPHA_SHRINK = 0.99;
281 
282  using real_type = typename VectorType::real_type;
283 
284  typename VectorMemory<VectorType>::Pointer v(this->memory);
285  typename VectorMemory<VectorType>::Pointer g(this->memory);
286 
287  // Set velocities to zero but not gradients
288  // as we are going to compute them soon.
289  v->reinit(x, false);
290  g->reinit(x, true);
291 
292  // Refer to v and g with some readable names.
293  VectorType &velocities = *v;
294  VectorType &gradients = *g;
295 
296  // Update gradients for the new x.
297  compute(gradients, x);
298 
299  unsigned int iter = 0;
300 
301  SolverControl::State conv = SolverControl::iterate;
302  conv = this->iteration_status(iter, gradients * gradients, x);
303  if (conv != SolverControl::iterate)
304  return;
305 
306  // Refer to additional data members with some readable names.
307  const auto &maximum_timestep = additional_data.maximum_timestep;
308  double timestep = additional_data.initial_timestep;
309 
310  // First scaling factor.
311  double alpha = ALPHA_0;
312 
313  unsigned int previous_iter_with_positive_v_dot_g = 0;
314 
315  while (conv == SolverControl::iterate)
316  {
317  ++iter;
318  // Euler integration step.
319  x.add(timestep, velocities); // x += dt * v
320  inverse_mass_matrix.vmult(gradients, gradients); // g = M^{-1} * g
321  velocities.add(-timestep, gradients); // v -= dt * h
322 
323  // Compute gradients for the new x.
324  compute(gradients, x);
325 
326  const real_type gradient_norm_squared = gradients * gradients;
327  conv = this->iteration_status(iter, gradient_norm_squared, x);
328  if (conv != SolverControl::iterate)
329  break;
330 
331  // v_dot_g = V * G
332  const real_type v_dot_g = velocities * gradients;
333 
334  if (v_dot_g < 0.)
335  {
336  const real_type velocities_norm_squared = velocities * velocities;
337 
338  // Check if we divide by zero in DEBUG mode.
339  Assert(gradient_norm_squared > 0., ExcInternalError());
340 
341  // beta = - alpha |V|/|G|
342  const real_type beta =
343  -alpha * std::sqrt(velocities_norm_squared / gradient_norm_squared);
344 
345  // V = (1-alpha) V + beta G.
346  velocities.sadd(1. - alpha, beta, gradients);
347 
348  if (iter - previous_iter_with_positive_v_dot_g > DELAYSTEP)
349  {
350  // Increase timestep and decrease alpha.
351  timestep = std::min(timestep * TIMESTEP_GROW, maximum_timestep);
352  alpha *= ALPHA_SHRINK;
353  }
354  }
355  else
356  {
357  // Decrease timestep, reset alpha and set V = 0.
358  previous_iter_with_positive_v_dot_g = iter;
359  timestep *= TIMESTEP_SHRINK;
360  alpha = ALPHA_0;
361  velocities = 0.;
362  }
363 
364  real_type vmax = velocities.linfty_norm();
365 
366  // Change timestep if any dof would move more than maximum_linfty_norm.
367  if (vmax > 0.)
368  {
369  const double minimal_timestep =
370  additional_data.maximum_linfty_norm / vmax;
371  if (minimal_timestep < timestep)
372  timestep = minimal_timestep;
373  }
374 
375  print_vectors(iter, x, velocities, gradients);
376 
377  } // While we need to iterate.
378 
379  // In the case of failure: throw exception.
380  if (conv != SolverControl::success)
381  AssertThrow(false,
382  SolverControl::NoConvergence(iter, gradients * gradients));
383 }

390 SolverFIRE<VectorType>::solve(const MatrixType & A,
391  VectorType & x,
392  const VectorType & b,
393  const PreconditionerType &preconditioner)
394 {
395  std::function<double(VectorType &, const VectorType &)> compute_func =
396  [&](VectorType &g, const VectorType &x) -> double {
397  // Residual of the quadratic form @f$ \frac{1}{2} xAx - xb @f$.
398  // G = b - Ax
399  A.residual(g, x, b);
400 
401  // Gradient G = Ax -b.
402  g *= -1.;
403 
404  // The quadratic form @f$\frac{1}{2} xAx - xb @f$.
405  return 0.5 * A.matrix_norm_square(x) - x * b;
406  };
407 
408  this->solve(compute_func, x, preconditioner);
409 }
