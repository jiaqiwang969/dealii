511  * <a name="SineGordonProblemSineGordonProblem"></a>
512  * <h4>SineGordonProblem::SineGordonProblem</h4>
513  *
514 
515  *
516  * This is the constructor of the SineGordonProblem class. The time interval
517  * and time step size are defined here. Moreover, we use the degree of the
518  * finite element that we defined at the top of the program to initialize a
519  * FE_Q finite element based on Gauss-Lobatto support points. These points
520  * are convenient because in conjunction with a QGaussLobatto quadrature
521  * rule of the same order they give a diagonal mass matrix without
522  * compromising accuracy too much (note that the integration is inexact,
523  * though), see also the discussion in the introduction. Note that FE_Q
524  * selects the Gauss-Lobatto nodal points by default due to their improved
525  * conditioning versus equidistant points. To make things more explicit, we
526  * state the selection of the nodal points nonetheless.
527  *
528  * @code
529  * template <int dim>
530  * SineGordonProblem<dim>::SineGordonProblem()
531  * : pcout(std::cout, Utilities::MPI::this_mpi_process(MPI_COMM_WORLD) == 0)
532  * ,
533  * #ifdef DEAL_II_WITH_P4EST
534  * triangulation(MPI_COMM_WORLD)
535  * ,
536  * #endif
537  * fe(QGaussLobatto<1>(fe_degree + 1))
538  * , dof_handler(triangulation)

550  * <a name="SineGordonProblemmake_grid_and_dofs"></a>
551  * <h4>SineGordonProblem::make_grid_and_dofs</h4>
552  *
553 
554  *
555  * As in @ref step_25 "step-25" this functions sets up a cube grid in <code>dim</code>
556  * dimensions of extent @f$[-15,15]@f$. We refine the mesh more in the center of
557  * the domain since the solution is concentrated there. We first refine all
558  * cells whose center is within a radius of 11, and then refine once more
559  * for a radius 6. This simple ad hoc refinement could be done better by
560  * adapting the mesh to the solution using error estimators during the time
561  * stepping as done in other example programs, and using
562  * parallel::distributed::SolutionTransfer to transfer the solution to the
563  * new mesh.
564  *
565  * @code
566  * template <int dim>
567  * void SineGordonProblem<dim>::make_grid_and_dofs()
568  * {
569  * GridGenerator::hyper_cube(triangulation, -15, 15);
570  * triangulation.refine_global(n_global_refinements);
571  * {
572  * typename Triangulation<dim>::active_cell_iterator
573  * cell = triangulation.begin_active(),
574  * end_cell = triangulation.end();
575  * for (; cell != end_cell; ++cell)
576  * if (cell->is_locally_owned())
577  * if (cell->center().norm() < 11)
578  * cell->set_refine_flag();
579  * triangulation.execute_coarsening_and_refinement();
580  *
581  * cell = triangulation.begin_active();
582  * end_cell = triangulation.end();
583  * for (; cell != end_cell; ++cell)
584  * if (cell->is_locally_owned())
585  * if (cell->center().norm() < 6)
586  * cell->set_refine_flag();
587  * triangulation.execute_coarsening_and_refinement();
588  * }
589  *
590  * pcout << " Number of global active cells: "
591  * #ifdef DEAL_II_WITH_P4EST
592  * << triangulation.n_global_active_cells()
593  * #else
594  * << triangulation.n_active_cells()
595  * #endif
596  * << std::endl;
597  *
598  * dof_handler.distribute_dofs(fe);
599  *
600  * pcout << " Number of degrees of freedom: " << dof_handler.n_dofs()
601  * << std::endl;
602  *
603  *
604  * @endcode
605  *
606  * We generate hanging node constraints for ensuring continuity of the
607  * solution. As in @ref step_40 "step-40", we need to equip the constraint matrix with
608  * the IndexSet of locally relevant degrees of freedom to avoid it to
609  * consume too much memory for big problems. Next, the <code> MatrixFree
610  * </code> object for the problem is set up. Note that we specify a
611  * particular scheme for shared-memory parallelization (hence one would
612  * use multithreading for intra-node parallelism and not MPI; we here
613  * choose the standard option &mdash; if we wanted to disable shared
614  * memory parallelization even in case where there is more than one TBB
615  * thread available in the program, we would choose
616  * MatrixFree::AdditionalData::TasksParallelScheme::none). Also note that,
617  * instead of using the default QGauss quadrature argument, we supply a
618  * QGaussLobatto quadrature formula to enable the desired
619  * behavior. Finally, three solution vectors are initialized. MatrixFree
620  * expects a particular layout of ghost indices (as it handles index
621  * access in MPI-local numbers that need to match between the vector and
622  * MatrixFree), so we just ask it to initialize the vectors to be sure the
623  * ghost exchange is properly handled.

625  * @code

652  * <a name="SineGordonProblemoutput_results"></a>
653  * <h4>SineGordonProblem::output_results</h4>
654  *
655 
656  *
657  * This function prints the norm of the solution and writes the solution
658  * vector to a file. The norm is standard (except for the fact that we need
659  * to accumulate the norms over all processors for the parallel grid which
660  * we do via the VectorTools::compute_global_error() function), and the
661  * second is similar to what we did in @ref step_40 "step-40" or @ref step_37 "step-37". Note that we can
662  * use the same vector for output as the one used during computations: The
663  * vectors in the matrix-free framework always provide full information on
664  * all locally owned cells (this is what is needed in the local evaluations,
665  * too), including ghost vector entries on these cells. This is the only
666  * data that is needed in the VectorTools::integrate_difference() function
667  * as well as in DataOut. The only action to take at this point is to make
668  * sure that the vector updates its ghost values before we read from
669  * them, and to reset ghost values once done. This is a feature present only
670  * in the LinearAlgebra::distributed::Vector class. Distributed vectors with
671  * PETSc and Trilinos, on the other hand, need to be copied to special
672  * vectors including ghost values (see the relevant section in @ref step_40 "step-40"). If
673  * we also wanted to access all degrees of freedom on ghost cells (e.g. when
674  * computing error estimators that use the jump of solution over cell
675  * boundaries), we would need more information and create a vector
676  * initialized with locally relevant dofs just as in @ref step_40 "step-40". Observe also
677  * that we need to distribute constraints for output - they are not filled
678  * during computations (rather, they are interpolated on the fly in the
679  * matrix-free method FEEvaluation::read_dof_values()).

722  * <a name="SineGordonProblemrun"></a>
723  * <h4>SineGordonProblem::run</h4>
724  *
725 
726  *
727  * This function is called by the main function and steps into the
728  * subroutines of the class.
729  *
730 
731  *
732  * After printing some information about the parallel setup, the first
733  * action is to set up the grid and the cell operator. Then, the time step
734  * is computed from the CFL number given in the constructor and the finest
735  * mesh size. The finest mesh size is computed as the diameter of the last
736  * cell in the triangulation, which is the last cell on the finest level of
737  * the mesh. This is only possible for meshes where all elements on a level
738  * have the same size, otherwise, one needs to loop over all cells. Note
739  * that we need to query all the processors for their finest cell since
740  * not all processors might hold a region where the mesh is at the finest
741  * level. Then, we readjust the time step a little to hit the final time
742  * exactly.
743  *
744  * @code
745  * template <int dim>
746  * void SineGordonProblem<dim>::run()
747  * {
748  * {
749  * pcout << "Number of MPI ranks: "
750  * << Utilities::MPI::n_mpi_processes(MPI_COMM_WORLD) << std::endl;
751  * pcout << "Number of threads on each rank: "
752  * << MultithreadInfo::n_threads() << std::endl;
753  * const unsigned int n_vect_doubles = VectorizedArray<double>::size();
754  * const unsigned int n_vect_bits = 8 * sizeof(double) * n_vect_doubles;
755  * pcout << "Vectorization over " << n_vect_doubles
756  * << " doubles = " << n_vect_bits << " bits ("
757  * << Utilities::System::get_current_vectorization_level() << ")"
758  * << std::endl
759  * << std::endl;
760  * }
761  * make_grid_and_dofs();
762  *
763  * const double local_min_cell_diameter =
764  * triangulation.last()->diameter() / std::sqrt(dim);
765  * const double global_min_cell_diameter =
766  * -Utilities::MPI::max(-local_min_cell_diameter, MPI_COMM_WORLD);
767  * time_step = cfl_number * global_min_cell_diameter;
768  * time_step = (final_time - time) / (int((final_time - time) / time_step));
769  * pcout << " Time step size: " << time_step
770  * << ", finest cell: " << global_min_cell_diameter << std::endl
771  * << std::endl;
772  *
773  * @endcode
774  *
775  * Next the initial value is set. Since we have a two-step time stepping
776  * method, we also need a value of the solution at time-time_step. For
777  * accurate results, one would need to compute this from the time
778  * derivative of the solution at initial time, but here we ignore this
779  * difficulty and just set it to the initial value function at that
780  * artificial time.
781  *
782 
783  *
784  * We then go on by writing the initial state to file and collecting
785  * the two starting solutions in a <tt>std::vector</tt> of pointers that
786  * get later consumed by the SineGordonOperation::apply() function. Next,
787  * an instance of the <code> SineGordonOperation class </code> based on
788  * the finite element degree specified at the top of this file is set up.
789  *
790  * @code
791  * VectorTools::interpolate(mapping,
792  * dof_handler,
793  * InitialCondition<dim>(1, time),
794  * solution);
795  * VectorTools::interpolate(mapping,
796  * dof_handler,
797  * InitialCondition<dim>(1, time - time_step),
798  * old_solution);
799  * output_results(0);
800  *
801  * std::vector<LinearAlgebra::distributed::Vector<double> *>
802  * previous_solutions({&old_solution, &old_old_solution});
803  *
804  * SineGordonOperation<dim, fe_degree> sine_gordon_op(matrix_free_data,
805  * time_step);
806  *
807  * @endcode
808  *
809  * Now loop over the time steps. In each iteration, we shift the solution
810  * vectors by one and call the `apply` function of the
811  * `SineGordonOperator` class. Then, we write the solution to a file. We
812  * clock the wall times for the computational time needed as wall as the
813  * time needed to create the output and report the numbers when the time
814  * stepping is finished.
815  *
816 
817  *
818  * Note how this shift is implemented: We simply call the swap method on
819  * the two vectors which swaps only some pointers without the need to copy
820  * data around, a relatively expensive operation within an explicit time
821  * stepping method. Let us see what happens in more detail: First, we
822  * exchange <code>old_solution</code> with <code>old_old_solution</code>,
823  * which means that <code>old_old_solution</code> gets
824  * <code>old_solution</code>, which is what we expect. Similarly,
825  * <code>old_solution</code> gets the content from <code>solution</code>
826  * in the next step. After this, <code>solution</code> holds
827  * <code>old_old_solution</code>, but that will be overwritten during this
828  * step.
829  *
830  * @code
831  * unsigned int timestep_number = 1;
832  *
833  * Timer timer;
834  * double wtime = 0;
835  * double output_time = 0;
836  * for (time += time_step; time <= final_time;
837  * time += time_step, ++timestep_number)
838  * {
839  * timer.restart();
840  * old_old_solution.swap(old_solution);
841  * old_solution.swap(solution);
842  * sine_gordon_op.apply(solution, previous_solutions);
843  * wtime += timer.wall_time();
844  *
845  * timer.restart();
846  * if (timestep_number % output_timestep_skip == 0)
847  * output_results(timestep_number / output_timestep_skip);
848  *
849  * output_time += timer.wall_time();
850  * }
851  * timer.restart();
852  * output_results(timestep_number / output_timestep_skip + 1);
853  * output_time += timer.wall_time();
854  *
855  * pcout << std::endl
856  * << " Performed " << timestep_number << " time steps." << std::endl;
857  *
858  * pcout << " Average wallclock time per time step: "
859  * << wtime / timestep_number << "s" << std::endl;
860  *
861  * pcout << " Spent " << output_time << "s on output and " << wtime
862  * << "s on computations." << std::endl;
863  * }
864  * } // namespace Step48
865  *
866  *
867  *
868  * @endcode
