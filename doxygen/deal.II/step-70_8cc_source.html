430  template <int dim, int spacedim>
431  void read_grid_and_cad_files(const std::string &grid_file_name,
432  const std::string &ids_and_cad_file_names,
433  Triangulation<dim, spacedim> &tria)
434  {
435  GridIn<dim, spacedim> grid_in;
436  grid_in.attach_triangulation(tria);
437  grid_in.read(grid_file_name);
438 
439 #ifdef DEAL_II_WITH_OPENCASCADE
440  using map_type = std::map<types::manifold_id, std::string>;
441  using Converter = Patterns::Tools::Convert<map_type>;
442 
443  for (const auto &pair : Converter::to_value(ids_and_cad_file_names))
444  {
445  const auto &manifold_id = pair.first;
446  const auto &cad_file_name = pair.second;
447 
448  const auto extension = boost::algorithm::to_lower_copy(
449  cad_file_name.substr(cad_file_name.find_last_of('.') + 1));
450 
451  TopoDS_Shape shape;
452  if (extension == "iges" || extension == "igs")
453  shape = OpenCASCADE::read_IGES(cad_file_name);
454  else if (extension == "step" || extension == "stp")
455  shape = OpenCASCADE::read_STEP(cad_file_name);
456  else
457  AssertThrow(false,
458  ExcNotImplemented("We found an extension that we "
459  "do not recognize as a CAD file "
460  "extension. Bailing out."));
461 
462  const auto n_elements = OpenCASCADE::count_elements(shape);
463  if ((std::get<0>(n_elements) == 0))
464  tria.set_manifold(
465  manifold_id,
466  OpenCASCADE::ArclengthProjectionLineManifold<dim, spacedim>(shape));
467  else if (spacedim == 3)
468  {
469  const auto t = reinterpret_cast<Triangulation<dim, 3> *>(&tria);
470  t->set_manifold(manifold_id,
471  OpenCASCADE::NormalToMeshProjectionManifold<dim, 3>(
472  shape));
473  }
474  else
475  tria.set_manifold(manifold_id,
476  OpenCASCADE::NURBSPatchManifold<dim, spacedim>(
477  TopoDS::Face(shape)));
478  }
479 #else
480  (void)ids_and_cad_file_names;
481  AssertThrow(false, ExcNotImplemented("Generation of the grid failed."));
482 #endif
483  }

767  template <int dim, int spacedim>
768  void StokesImmersedProblem<dim, spacedim>::assemble_stokes_system()
769  {
770  system_matrix = 0;
771  preconditioner_matrix = 0;
772  system_rhs = 0;
773 
774  TimerOutput::Scope t(computing_timer, "Assemble Stokes terms");
775 
776  FEValues<spacedim> fe_values(*fluid_fe,
777  *fluid_quadrature_formula,
778  update_values | update_gradients |
779  update_quadrature_points |
780  update_JxW_values);
781 
782  const unsigned int dofs_per_cell = fluid_fe->n_dofs_per_cell();
783  const unsigned int n_q_points = fluid_quadrature_formula->size();
784 
785  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
786  FullMatrix<double> cell_matrix2(dofs_per_cell, dofs_per_cell);
787  Vector<double> cell_rhs(dofs_per_cell);
788 
789  std::vector<Vector<double>> rhs_values(n_q_points,
790  Vector<double>(spacedim + 1));
791 
792  std::vector<Tensor<2, spacedim>> grad_phi_u(dofs_per_cell);
793  std::vector<double> div_phi_u(dofs_per_cell);
794  std::vector<double> phi_p(dofs_per_cell);
795 
796  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
797  const FEValuesExtractors::Vector velocities(0);
798  const FEValuesExtractors::Scalar pressure(spacedim);
799 
800  for (const auto &cell : fluid_dh.active_cell_iterators())
801  if (cell->is_locally_owned())
802  {
803  cell_matrix = 0;
804  cell_matrix2 = 0;
805  cell_rhs = 0;
806 
807  fe_values.reinit(cell);
808  par.rhs.vector_value_list(fe_values.get_quadrature_points(),
809  rhs_values);
810  for (unsigned int q = 0; q < n_q_points; ++q)
811  {
812  for (unsigned int k = 0; k < dofs_per_cell; ++k)
813  {
814  grad_phi_u[k] = fe_values[velocities].gradient(k, q);
815  div_phi_u[k] = fe_values[velocities].divergence(k, q);
816  phi_p[k] = fe_values[pressure].value(k, q);
817  }
818 
819  for (unsigned int i = 0; i < dofs_per_cell; ++i)
820  {
821  for (unsigned int j = 0; j < dofs_per_cell; ++j)
822  {
823  cell_matrix(i, j) +=
824  (par.viscosity *
825  scalar_product(grad_phi_u[i], grad_phi_u[j]) -
826  div_phi_u[i] * phi_p[j] - phi_p[i] * div_phi_u[j]) *
827  fe_values.JxW(q);
828 
829  cell_matrix2(i, j) += 1.0 / par.viscosity * phi_p[i] *
830  phi_p[j] * fe_values.JxW(q);
831  }
832 
833  const unsigned int component_i =
834  fluid_fe->system_to_component_index(i).first;
835  cell_rhs(i) += fe_values.shape_value(i, q) *
836  rhs_values[q](component_i) * fe_values.JxW(q);
837  }
838  }
839 
840 
841  cell->get_dof_indices(local_dof_indices);
842  constraints.distribute_local_to_global(cell_matrix,
843  cell_rhs,
844  local_dof_indices,
845  system_matrix,
846  system_rhs);
847 
848  constraints.distribute_local_to_global(cell_matrix2,
849  local_dof_indices,
850  preconditioner_matrix);
851  }
852 
853  system_matrix.compress(VectorOperation::add);
854  preconditioner_matrix.compress(VectorOperation::add);
855  system_rhs.compress(VectorOperation::add);
856  }

859  template <int dim, int spacedim>
860  void StokesImmersedProblem<dim, spacedim>::assemble_nitsche_restriction()
861  {
862  TimerOutput::Scope t(computing_timer, "Assemble Nitsche terms");
863 
864  const FEValuesExtractors::Vector velocities(0);
865  const FEValuesExtractors::Scalar pressure(spacedim);
866 
867  SolidVelocity<spacedim> solid_velocity(par.angular_velocity);
868 
869  std::vector<types::global_dof_index> fluid_dof_indices(
870  fluid_fe->n_dofs_per_cell());
871 
872  FullMatrix<double> local_matrix(fluid_fe->n_dofs_per_cell(),
873  fluid_fe->n_dofs_per_cell());
874  ::Vector<double> local_rhs(fluid_fe->n_dofs_per_cell());
875 
876  const auto penalty_parameter =
877  1.0 / GridTools::minimal_cell_diameter(fluid_tria);
878 
879  auto particle = solid_particle_handler.begin();
880  while (particle != solid_particle_handler.end())
881  {
882  local_matrix = 0;
883  local_rhs = 0;
884 
885  const auto &cell = particle->get_surrounding_cell(fluid_tria);
886  const auto &dh_cell =
887  typename DoFHandler<spacedim>::cell_iterator(*cell, &fluid_dh);
888  dh_cell->get_dof_indices(fluid_dof_indices);
889 
890  const auto pic = solid_particle_handler.particles_in_cell(cell);
891  Assert(pic.begin() == particle, ExcInternalError());
892  for (const auto &p : pic)
893  {
894  const auto &ref_q = p.get_reference_location();
895  const auto &real_q = p.get_location();
896  const auto &JxW = p.get_properties()[0];
897 
898  for (unsigned int i = 0; i < fluid_fe->n_dofs_per_cell(); ++i)
899  {
900  const auto comp_i =
901  fluid_fe->system_to_component_index(i).first;
902  if (comp_i < spacedim)
903  {
904  for (unsigned int j = 0; j < fluid_fe->n_dofs_per_cell();
905  ++j)
906  {
907  const auto comp_j =
908  fluid_fe->system_to_component_index(j).first;
909  if (comp_i == comp_j)
910  local_matrix(i, j) +=
911  penalty_parameter * par.penalty_term *
912  fluid_fe->shape_value(i, ref_q) *
913  fluid_fe->shape_value(j, ref_q) * JxW;
914  }
915  local_rhs(i) += penalty_parameter * par.penalty_term *
916  solid_velocity.value(real_q, comp_i) *
917  fluid_fe->shape_value(i, ref_q) * JxW;
918  }
919  }
920  }
921 
922  constraints.distribute_local_to_global(local_matrix,
923  local_rhs,
924  fluid_dof_indices,
925  system_matrix,
926  system_rhs);
927  particle = pic.end();
928  }
929 
930  system_matrix.compress(VectorOperation::add);
931  system_rhs.compress(VectorOperation::add);
932  }

937  void StokesImmersedProblem<dim, spacedim>::solve()
938  {
939  TimerOutput::Scope t(computing_timer, "Solve");
940 
941  LA::MPI::PreconditionAMG prec_A;
942  {
943  LA::MPI::PreconditionAMG::AdditionalData data;
944 
945 #ifdef USE_PETSC_LA
946  data.symmetric_operator = true;
947 #endif
948  prec_A.initialize(system_matrix.block(0, 0), data);
949  }
950 
951  LA::MPI::PreconditionAMG prec_S;
952  {
953  LA::MPI::PreconditionAMG::AdditionalData data;
954 
955 #ifdef USE_PETSC_LA
956  data.symmetric_operator = true;
957 #endif
958  prec_S.initialize(preconditioner_matrix.block(1, 1), data);
959  }
960 
961  const auto A = linear_operator<LA::MPI::Vector>(system_matrix.block(0, 0));
962  const auto amgA = linear_operator(A, prec_A);
963 
964  const auto S =
965  linear_operator<LA::MPI::Vector>(preconditioner_matrix.block(1, 1));
966  const auto amgS = linear_operator(S, prec_S);
967 
968  ReductionControl inner_solver_control(100,
969  1e-8 * system_rhs.l2_norm(),
970  1.e-2);
971  SolverCG<LA::MPI::Vector> cg(inner_solver_control);
972 
973  const auto invS = inverse_operator(S, cg, amgS);
974 
975  const auto P = block_diagonal_operator<2, LA::MPI::BlockVector>(
976  std::array<
977  ::LinearOperator<typename LA::MPI::BlockVector::BlockType>,
978  2>{{amgA, amgS}});
979 
980  SolverControl solver_control(system_matrix.m(),
981  1e-10 * system_rhs.l2_norm());
982 
983  SolverFGMRES<LA::MPI::BlockVector> solver(solver_control);
984 
985  constraints.set_zero(solution);
986 
987  solver.solve(system_matrix, solution, system_rhs, P);
988 
989 
990  pcout << " Solved in " << solver_control.last_step() << " iterations."
991  << std::endl;
992 
993  constraints.distribute(solution);
994 
995  locally_relevant_solution = solution;
996  const double mean_pressure =
997  VectorTools::compute_mean_value(fluid_dh,
998  QGauss<spacedim>(par.velocity_degree + 2),
999  locally_relevant_solution,
1000  spacedim);
1001  solution.block(1).add(-mean_pressure);
1002  locally_relevant_solution.block(1) = solution.block(1);
1003  }
