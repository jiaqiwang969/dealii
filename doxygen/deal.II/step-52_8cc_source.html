154  void Diffusion::assemble_system()
155  {
156  system_matrix = 0.;
157  mass_matrix = 0.;
158 
159  const QGauss<2> quadrature_formula(fe_degree + 1);
160 
161  FEValues<2> fe_values(fe,
162  quadrature_formula,
163  update_values | update_gradients | update_JxW_values);
164 
165 
166  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
167  const unsigned int n_q_points = quadrature_formula.size();
168 
169  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
170  FullMatrix<double> cell_mass_matrix(dofs_per_cell, dofs_per_cell);
171 
172  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
173 
174  for (const auto &cell : dof_handler.active_cell_iterators())
175  {
176  cell_matrix = 0.;
177  cell_mass_matrix = 0.;
178 
179  fe_values.reinit(cell);
180 
181  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
182  for (unsigned int i = 0; i < dofs_per_cell; ++i)
183  for (unsigned int j = 0; j < dofs_per_cell; ++j)
184  {
185  cell_matrix(i, j) +=
186  ((-diffusion_coefficient * // (-D
187  fe_values.shape_grad(i, q_point) * // * grad phi_i
188  fe_values.shape_grad(j, q_point) // * grad phi_j
189  - absorption_cross_section * // -Sigma
190  fe_values.shape_value(i, q_point) * // * phi_i
191  fe_values.shape_value(j, q_point)) // * phi_j)
192  * fe_values.JxW(q_point)); // * dx
193  cell_mass_matrix(i, j) += fe_values.shape_value(i, q_point) *
194  fe_values.shape_value(j, q_point) *
195  fe_values.JxW(q_point);
196  }
197 
198  cell->get_dof_indices(local_dof_indices);
199 
200  constraint_matrix.distribute_local_to_global(cell_matrix,
201  local_dof_indices,
202  system_matrix);
203  constraint_matrix.distribute_local_to_global(cell_mass_matrix,
204  local_dof_indices,
205  mass_matrix);
206  }
207 
208  inverse_mass_matrix.initialize(mass_matrix);
209  }

213  double Diffusion::get_source(const double time, const Point<2> &point) const
214  {
215  const double intensity = 10.;
216  const double frequency = numbers::PI / 10.;
217  const double b = 5.;
218  const double x = point(0);
219 
220  return intensity *
221  (frequency * std::cos(frequency * time) * (b * x - x * x) +
222  std::sin(frequency * time) *
223  (absorption_cross_section * (b * x - x * x) +
224  2. * diffusion_coefficient));
225  }

229  Vector<double> Diffusion::evaluate_diffusion(const double time,
230  const Vector<double> &y) const
231  {
232  Vector<double> tmp(dof_handler.n_dofs());
233  tmp = 0.;
234  system_matrix.vmult(tmp, y);
235 
236  const QGauss<2> quadrature_formula(fe_degree + 1);
237 
238  FEValues<2> fe_values(fe,
239  quadrature_formula,
240  update_values | update_quadrature_points |
241  update_JxW_values);
242 
243 
244  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
245  const unsigned int n_q_points = quadrature_formula.size();
246 
247  Vector<double> cell_source(dofs_per_cell);
248 
249  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
250 
251  for (const auto &cell : dof_handler.active_cell_iterators())
252  {
253  cell_source = 0.;
254 
255  fe_values.reinit(cell);
256 
257  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
258  {
259  const double source =
260  get_source(time, fe_values.quadrature_point(q_point));
261  for (unsigned int i = 0; i < dofs_per_cell; ++i)
262  cell_source(i) += fe_values.shape_value(i, q_point) * // phi_i(x)
263  source * // * S(x)
264  fe_values.JxW(q_point); // * dx
265  }
266 
267  cell->get_dof_indices(local_dof_indices);
268 
269  constraint_matrix.distribute_local_to_global(cell_source,
270  local_dof_indices,
271  tmp);
272  }
273 
274  Vector<double> value(dof_handler.n_dofs());
275  inverse_mass_matrix.vmult(value, tmp);
276 
277  return value;
278  }

281  Vector<double> Diffusion::id_minus_tau_J_inverse(const double /*time*/,
282  const double tau,
283  const Vector<double> &y)
284  {
285  SparseDirectUMFPACK inverse_mass_minus_tau_Jacobian;
286 
287  mass_minus_tau_Jacobian.copy_from(mass_matrix);
288  mass_minus_tau_Jacobian.add(-tau, system_matrix);
289 
290  inverse_mass_minus_tau_Jacobian.initialize(mass_minus_tau_Jacobian);
291 
292  Vector<double> tmp(dof_handler.n_dofs());
293  mass_matrix.vmult(tmp, y);
294 
295  Vector<double> result(y);
296  inverse_mass_minus_tau_Jacobian.vmult(result, tmp);
297 
298  return result;

328  method_name = "backward_euler";

331  case TimeStepping::IMPLICIT_MIDPOINT:

333  method_name = "implicit_midpoint";

337  {
338  method_name = "sdirk";
339  break;

342  {
343  method_name = "heun_euler";
344  break;
345  }
346  case TimeStepping::BOGACKI_SHAMPINE:
347  {
348  method_name = "bocacki_shampine";
349  break;

351  case TimeStepping::DOPRI:

353  method_name = "dopri";
354  break;
355  }
356  case TimeStepping::FEHLBERG:
357  {
358  method_name = "fehlberg";
359  break;
360  }
361  case TimeStepping::CASH_KARP:

395  pvd_filename = "solution_" + method_name + ".pvd";
396  }

398  std::ofstream pvd_output(pvd_filename);
399  DataOutBase::write_pvd_record(pvd_output, times_and_names);

403  void Diffusion::explicit_method(const TimeStepping::runge_kutta_method method,
404  const unsigned int n_time_steps,
405  const double initial_time,
406  const double final_time)
407  {
408  const double time_step =
409  (final_time - initial_time) / static_cast<double>(n_time_steps);
410 
411  solution = 0.;
412  constraint_matrix.distribute(solution);
413 
414  TimeStepping::ExplicitRungeKutta<Vector<double>> explicit_runge_kutta(
415  method);
416  output_results(initial_time, 0, method);
417  DiscreteTime time(initial_time, final_time, time_step);
418  while (time.is_at_end() == false)
419  {
420  explicit_runge_kutta.evolve_one_time_step(
421  [this](const double time, const Vector<double> &y) {
422  return this->evaluate_diffusion(time, y);
423  },
424  time.get_current_time(),
425  time.get_next_step_size(),
426  solution);
427  time.advance_time();
428 
429  constraint_matrix.distribute(solution);
430 
431  if (time.get_step_number() % 10 == 0)
432  output_results(time.get_current_time(),
433  time.get_step_number(),
434  method);
435  }
436  }

480  unsigned int Diffusion::embedded_explicit_method(
481  const TimeStepping::runge_kutta_method method,
482  const unsigned int n_time_steps,
483  const double initial_time,
484  const double final_time)
485  {
486  const double time_step =
487  (final_time - initial_time) / static_cast<double>(n_time_steps);
488  const double coarsen_param = 1.2;
489  const double refine_param = 0.8;
490  const double min_delta = 1e-8;
491  const double max_delta = 10 * time_step;
492  const double refine_tol = 1e-1;
493  const double coarsen_tol = 1e-5;
494 
495  solution = 0.;
496  constraint_matrix.distribute(solution);
497 
498  TimeStepping::EmbeddedExplicitRungeKutta<Vector<double>>
499  embedded_explicit_runge_kutta(method,
500  coarsen_param,
501  refine_param,
502  min_delta,
503  max_delta,
504  refine_tol,
505  coarsen_tol);
506  output_results(initial_time, 0, method);
507  DiscreteTime time(initial_time, final_time, time_step);
508  while (time.is_at_end() == false)
509  {
510  const double new_time =
511  embedded_explicit_runge_kutta.evolve_one_time_step(
512  [this](const double time, const Vector<double> &y) {
513  return this->evaluate_diffusion(time, y);
514  },
515  time.get_current_time(),
516  time.get_next_step_size(),
517  solution);
518  time.set_next_step_size(new_time - time.get_current_time());
519  time.advance_time();
520 
521  constraint_matrix.distribute(solution);
522 
523  if (time.get_step_number() % 10 == 0)
524  output_results(time.get_current_time(),
525  time.get_step_number(),
526  method);
527 
528  time.set_desired_next_step_size(
529  embedded_explicit_runge_kutta.get_status().delta_t_guess);
530  }
531 
532  return time.get_step_number();
533  }

541

544  if (face->at_boundary())
545  {
546  if ((face->center()[0] == 0.) || (face->center()[0] == 5.))
