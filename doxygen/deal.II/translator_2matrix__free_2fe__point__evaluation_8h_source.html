701 template <int n_components, int dim, int spacedim, typename Number>
702 FEPointEvaluation<n_components, dim, spacedim, Number>::FEPointEvaluation(
703  const Mapping<dim> & mapping,
704  const FiniteElement<dim> &fe,
705  const UpdateFlags update_flags,
706  const unsigned int first_selected_component)
707  : mapping(&mapping)
708  , mapping_q_generic(
709  dynamic_cast<const MappingQGeneric<dim, spacedim> *>(&mapping))
710  , fe(&fe)
711  , update_flags(update_flags)
712  , update_flags_mapping(update_default)
713 {
714  bool same_base_element = true;
715  unsigned int base_element_number = 0;
716  unsigned int component = 0;
717  for (; base_element_number < fe.n_base_elements(); ++base_element_number)
718  if (component + fe.element_multiplicity(base_element_number) >
719  first_selected_component)
720  {
721  if (first_selected_component + n_components >
722  component + fe.element_multiplicity(base_element_number))
723  same_base_element = false;
724  break;
725  }
726  else
727  component += fe.element_multiplicity(base_element_number);
728  if (mapping_q_generic != nullptr &&
729  internal::FEPointEvaluation::is_fast_path_supported(
730  fe, base_element_number) &&
731  same_base_element)
732  {
733  internal::MatrixFreeFunctions::ShapeInfo<double> shape_info;
734 
735  shape_info.reinit(QMidpoint<1>(), fe, base_element_number);
736  renumber = shape_info.lexicographic_numbering;
737  dofs_per_component = shape_info.dofs_per_component_on_cell;
738  poly = internal::FEPointEvaluation::get_polynomial_space(
739  fe.base_element(base_element_number));
740 
741  polynomials_are_hat_functions =
742  (poly.size() == 2 && poly[0].value(0.) == 1. &&
743  poly[0].value(1.) == 0. && poly[1].value(0.) == 0. &&
744  poly[1].value(1.) == 1.);
745  }
746  else
747  {
748  nonzero_shape_function_component.resize(fe.n_dofs_per_cell());
749  for (unsigned int d = 0; d < n_components; ++d)
750  {
751  const unsigned int component = first_selected_component + d;
752  for (unsigned int i = 0; i < fe.n_dofs_per_cell(); ++i)
753  {
754  const bool is_primitive = fe.is_primitive() || fe.is_primitive(i);
755  if (is_primitive)
756  nonzero_shape_function_component[i][d] =
757  (component == fe.system_to_component_index(i).first);
758  else
759  nonzero_shape_function_component[i][d] =
760  (fe.get_nonzero_components(i)[component] == true);
761  }
762  }
763  }
764 
765  // translate update flags
766  if (update_flags & update_jacobians)
767  update_flags_mapping |= update_jacobians;
768  if (update_flags & update_gradients ||
769  update_flags & update_inverse_jacobians)
770  update_flags_mapping |= update_inverse_jacobians;
771  if (update_flags & update_quadrature_points)
772  update_flags_mapping |= update_quadrature_points;
773 }

819 template <int n_components, int dim, int spacedim, typename Number>
820 void
821 FEPointEvaluation<n_components, dim, spacedim, Number>::evaluate(
822  const ArrayView<const Number> & solution_values,
823  const EvaluationFlags::EvaluationFlags &evaluation_flag)
824 {
825  if (unit_points.empty())
826  return;
827 
828  AssertDimension(solution_values.size(), fe->dofs_per_cell);
829  if (((evaluation_flag & EvaluationFlags::values) ||
830  (evaluation_flag & EvaluationFlags::gradients)) &&
831  !poly.empty())
832  {
833  // fast path with tensor product evaluation
834  if (solution_renumbered.size() != dofs_per_component)
835  solution_renumbered.resize(dofs_per_component);
836  for (unsigned int comp = 0; comp < n_components; ++comp)
837  for (unsigned int i = 0; i < dofs_per_component; ++i)
838  internal::FEPointEvaluation::
839  EvaluatorTypeTraits<dim, n_components, Number>::read_value(
840  solution_values[renumber[comp * dofs_per_component + i]],
841  comp,
842  solution_renumbered[i]);
843 
844  // unit gradients are currently only implemented with the fast tensor
845  // path
846  unit_gradients.resize(unit_points.size(),
847  numbers::signaling_nan<gradient_type>());
848 
849  const std::size_t n_points = unit_points.size();
850  const std::size_t n_lanes = VectorizedArray<Number>::size();
851  for (unsigned int i = 0; i < n_points; i += n_lanes)
852  {
853  // convert to vectorized format
854  Point<dim, VectorizedArray<Number>> vectorized_points;
855  for (unsigned int j = 0; j < n_lanes && i + j < n_points; ++j)
856  for (unsigned int d = 0; d < dim; ++d)
857  vectorized_points[d][j] = unit_points[i + j][d];
858 
859  // compute
860  const auto val_and_grad =
861  internal::evaluate_tensor_product_value_and_gradient(
862  poly,
863  solution_renumbered,
864  vectorized_points,
865  polynomials_are_hat_functions);
866 
867  // convert back to standard format
868  if (evaluation_flag & EvaluationFlags::values)
869  for (unsigned int j = 0; j < n_lanes && i + j < n_points; ++j)
870  internal::FEPointEvaluation::
871  EvaluatorTypeTraits<dim, n_components, Number>::set_value(
872  val_and_grad.first, j, values[i + j]);
873  if (evaluation_flag & EvaluationFlags::gradients)
874  {
875  Assert(update_flags & update_gradients ||
876  update_flags & update_inverse_jacobians,
877  ExcNotInitialized());
878  for (unsigned int j = 0; j < n_lanes && i + j < n_points; ++j)
879  {
880  Assert(update_flags_mapping & update_inverse_jacobians,
881  ExcNotInitialized());
882  internal::FEPointEvaluation::EvaluatorTypeTraits<
883  dim,
884  n_components,
885  Number>::set_gradient(val_and_grad.second,
886  j,
887  unit_gradients[i + j]);
888  gradients[i + j] = apply_transformation(
889  mapping_data.inverse_jacobians[i + j].transpose(),
890  unit_gradients[i + j]);
891  }
892  }
893  }
894  }
895  else if ((evaluation_flag & EvaluationFlags::values) ||
896  (evaluation_flag & EvaluationFlags::gradients))
897  {
898  // slow path with FEValues
899  Assert(fe_values.get() != nullptr,
900  ExcMessage(
901  "Not initialized. Please call FEPointEvaluation::reinit()!"));
902 
903  if (evaluation_flag & EvaluationFlags::values)
904  {
905  values.resize(unit_points.size());
906  std::fill(values.begin(), values.end(), value_type());
907  for (unsigned int i = 0; i < fe->n_dofs_per_cell(); ++i)
908  {
909  const Number value = solution_values[i];
910  for (unsigned int d = 0; d < n_components; ++d)
911  if (nonzero_shape_function_component[i][d] &&
912  (fe->is_primitive(i) || fe->is_primitive()))
913  for (unsigned int q = 0; q < unit_points.size(); ++q)
914  internal::FEPointEvaluation::
915  EvaluatorTypeTraits<dim, n_components, Number>::access(
916  values[q], d) += fe_values->shape_value(i, q) * value;
917  else if (nonzero_shape_function_component[i][d])
918  for (unsigned int q = 0; q < unit_points.size(); ++q)
919  internal::FEPointEvaluation::
920  EvaluatorTypeTraits<dim, n_components, Number>::access(
921  values[q], d) +=
922  fe_values->shape_value_component(i, q, d) * value;
923  }
924  }
925 
926  if (evaluation_flag & EvaluationFlags::gradients)
927  {
928  gradients.resize(unit_points.size());
929  std::fill(gradients.begin(), gradients.end(), gradient_type());
930  for (unsigned int i = 0; i < fe->n_dofs_per_cell(); ++i)
931  {
