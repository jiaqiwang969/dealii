261  double fractional_flow(const double S, const double viscosity)
262  {
263  Assert((S >= 0) && (S <= 1),
264  ExcMessage("Saturation is outside its physically valid range."));
265 
266  return S * S / (S * S + viscosity * (1 - S) * (1 - S));
267  }

269 
270  double fractional_flow_derivative(const double S, const double viscosity)
271  {
272  Assert((S >= 0) && (S <= 1),
273  ExcMessage("Saturation is outside its physically valid range."));
274 
275  const double temp = (S * S + viscosity * (1 - S) * (1 - S));
276 
277  const double numerator =
278  2.0 * S * temp - S * S * (2.0 * S - 2.0 * viscosity * (1 - S));
279  const double denominator = std::pow(temp, 2.0);
280 
281  const double F_prime = numerator / denominator;
282 
283  Assert(F_prime >= 0, ExcInternalError());
284 
285  return F_prime;
286  }
