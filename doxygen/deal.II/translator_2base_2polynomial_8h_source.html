796  number value = coefficients.back();
797  for (int k = m - 2; k >= 0; --k)

817  inline void
818  Polynomial<number>::value(const Number2 x,
819  const unsigned int n_derivatives,
820  Number2 * values) const
821  {
822  // evaluate Lagrange polynomial and derivatives
823  if (in_lagrange_product_form == true)
824  {
825  // to compute the value and all derivatives of a polynomial of the
826  // form (x-x_1)*(x-x_2)*...*(x-x_n), expand the derivatives like
827  // automatic differentiation does.
828  const unsigned int n_supp = lagrange_support_points.size();
829  const number weight = lagrange_weight;
830  switch (n_derivatives)
831  {
832  default:
833  values[0] = 1.;
834  for (unsigned int d = 1; d <= n_derivatives; ++d)
835  values[d] = 0.;
836  for (unsigned int i = 0; i < n_supp; ++i)
837  {
838  const Number2 v = x - lagrange_support_points[i];
839 
840  // multiply by (x-x_i) and compute action on all derivatives,
841  // too (inspired from automatic differentiation: implement the
842  // product rule for the old value and the new variable 'v',
843  // i.e., expand value v and derivative one). since we reuse a
844  // value from the next lower derivative from the steps before,
845  // need to start from the highest derivative
846  for (unsigned int k = n_derivatives; k > 0; --k)
847  values[k] = (values[k] * v + values[k - 1]);
848  values[0] *= v;
849  }
850  // finally, multiply by the weight in the Lagrange
851  // denominator. Could be done instead of setting values[0] = 1
852  // above, but that gives different accumulation of round-off
853  // errors (multiplication is not associative) compared to when we
854  // computed the weight, and hence a basis function might not be
855  // exactly one at the center point, which is nice to have. We also
856  // multiply derivatives by k! to transform the product p_n =
857  // p^(n)(x)/k! into the actual form of the derivative
858  {
859  number k_factorial = 1;
860  for (unsigned int k = 0; k <= n_derivatives; ++k)
861  {
862  values[k] *= k_factorial * weight;
863  k_factorial *= static_cast<number>(k + 1);
864  }
865  }
866  break;
867 
868  // manually implement case 0 (values only), case 1 (value + first
869  // derivative), and case 2 (up to second derivative) since they
870  // might be called often. then, we can unroll the inner loop and
871  // keep the temporary results as local variables to help the
872  // compiler with the pointer aliasing analysis.
873  case 0:
874  {
875  Number2 value = 1.;
876  for (unsigned int i = 0; i < n_supp; ++i)
877  {
878  const Number2 v = x - lagrange_support_points[i];
879  value *= v;
880  }
881  values[0] = weight * value;
882  break;
883  }
884 
885  case 1:
886  {
887  Number2 value = 1.;
888  Number2 derivative = 0.;
889  for (unsigned int i = 0; i < n_supp; ++i)
890  {
891  const Number2 v = x - lagrange_support_points[i];
892  derivative = derivative * v + value;
893  value *= v;
894  }
895  values[0] = weight * value;
896  values[1] = weight * derivative;
897  break;
898  }
899 
900  case 2:
901  {
902  Number2 value = 1.;
903  Number2 derivative = 0.;
904  Number2 second = 0.;
905  for (unsigned int i = 0; i < n_supp; ++i)
906  {
907  const Number2 v = x - lagrange_support_points[i];
908  second = second * v + derivative;
909  derivative = derivative * v + value;
910  value *= v;
911  }
912  values[0] = weight * value;
913  values[1] = weight * derivative;
914  values[2] = static_cast<number>(2) * weight * second;
915  break;
916  }
917  }
918  return;
919  }

924  // Horner scheme
925  const unsigned int m = coefficients.size();
926  std::vector<Number2> a(coefficients.size());
927  std::copy(coefficients.begin(), coefficients.end(), a.begin());

932  // the highest power is
933  const unsigned int min_valuessize_m = std::min(n_derivatives + 1, m);
934  for (unsigned int j = 0; j < min_valuessize_m; ++j)
935  {
936  for (int k = m - 2; k >= static_cast<int>(j); --k)

938  values[j] = static_cast<number>(j_factorial) * a[j];

941  }
942

965  template <typename Number>
966  Number
967  jacobi_polynomial_value(const unsigned int degree,
968  const int alpha,
969  const int beta,
970  const Number x)
971  {
972  Assert(alpha >= 0 && beta >= 0,
973  ExcNotImplemented("Negative alpha/beta coefficients not supported"));
974  // the Jacobi polynomial is evaluated using a recursion formula.
975  Number p0, p1;
976 
977  // The recursion formula is defined for the interval [-1, 1], so rescale
978  // to that interval here
979  const Number xeval = Number(-1) + 2. * x;
980 
981  // initial values P_0(x), P_1(x):
982  p0 = 1.0;
983  if (degree == 0)
984  return p0;
985  p1 = ((alpha + beta + 2) * xeval + (alpha - beta)) / 2;
986  if (degree == 1)
987  return p1;
988 
989  for (unsigned int i = 1; i < degree; ++i)
990  {
991  const Number v = 2 * i + (alpha + beta);
992  const Number a1 = 2 * (i + 1) * (i + (alpha + beta + 1)) * v;
993  const Number a2 = (v + 1) * (alpha * alpha - beta * beta);
994  const Number a3 = v * (v + 1) * (v + 2);
995  const Number a4 = 2 * (i + alpha) * (i + beta) * (v + 2);
996 
997  const Number pn = ((a2 + a3 * xeval) * p1 - a4 * p0) / a1;
998  p0 = p1;
999  p1 = pn;
1000  }
1001  return p1;
1002  }

1006  template <typename Number>
1007  std::vector<Number>
1008  jacobi_polynomial_roots(const unsigned int degree,
1009  const int alpha,
1010  const int beta)
1011  {
1012  std::vector<Number> x(degree, 0.5);
1013 
1014  // compute zeros with a Newton algorithm.
1015 
1016  // Set tolerance. For long double we might not always get the additional
1017  // precision in a run time environment (e.g. with valgrind), so we must
1018  // limit the tolerance to double. Since we do a Newton iteration, doing
1019  // one more iteration after the residual has indicated convergence will be
1020  // enough for all number types due to the quadratic convergence of
1021  // Newton's method
1022 
1023  const Number tolerance =
1024  4 * std::max(static_cast<Number>(std::numeric_limits<double>::epsilon()),
1025  std::numeric_limits<Number>::epsilon());
1026 
1027  // The following implementation follows closely the one given in the
1028  // appendix of the book by Karniadakis and Sherwin: Spectral/hp element
1029  // methods for computational fluid dynamics (Oxford University Press,
1030  // 2005)
1031 
1032  // If symmetric, we only need to compute the half of points
1033  const unsigned int n_points = (alpha == beta ? degree / 2 : degree);
1034  for (unsigned int k = 0; k < n_points; ++k)
1035  {
1036  // we take the zeros of the Chebyshev polynomial (alpha=beta=-0.5) as
1037  // initial values, corrected by the initial value
1038  Number r = 0.5 - 0.5 * std::cos(static_cast<Number>(2 * k + 1) /
1039  (2 * degree) * numbers::PI);
1040  if (k > 0)
1041  r = (r + x[k - 1]) / 2;
1042 
1043  unsigned int converged = numbers::invalid_unsigned_int;
1044  for (unsigned int it = 1; it < 1000; ++it)
1045  {
1046  Number s = 0.;
1047  for (unsigned int i = 0; i < k; ++i)
1048  s += 1. / (r - x[i]);
1049 
1050  // derivative of P_n^{alpha,beta}, rescaled to [0, 1]
1051  const Number J_x =
1052  (alpha + beta + degree + 1) *
1053  jacobi_polynomial_value(degree - 1, alpha + 1, beta + 1, r);
1054 
1055  // value of P_n^{alpha,beta}
1056  const Number f = jacobi_polynomial_value(degree, alpha, beta, r);
1057  const Number delta = f / (f * s - J_x);
1058  r += delta;
1059  if (converged == numbers::invalid_unsigned_int &&
1060  std::abs(delta) < tolerance)
1061  converged = it;
1062 
1063  // do one more iteration to ensure accuracy also for tighter
1064  // types than double (e.g. long double)
1065  if (it == converged + 1)
1066  break;
1067  }
1068 
1069  Assert(converged != numbers::invalid_unsigned_int,
1070  ExcMessage("Newton iteration for zero of Jacobi polynomial "
1071  "did not converge."));
1072 
1073  x[k] = r;
1074  }
1075 
1076  // in case we assumed symmetry, fill up the missing values
1077  for (unsigned int k = n_points; k < degree; ++k)
1078  x[k] = 1.0 - x[degree - k - 1];
1079 
1080  return x;
1081  }
