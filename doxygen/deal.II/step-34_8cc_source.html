335  template <int dim>
336  void BEMProblem<dim>::assemble_system()
337  {
338  FEValues<dim - 1, dim> fe_v(mapping,
339  fe,
340  *quadrature,
341  update_values | update_normal_vectors |
342  update_quadrature_points | update_JxW_values);
343 
344  const unsigned int n_q_points = fe_v.n_quadrature_points;
345 
346  std::vector<types::global_dof_index> local_dof_indices(
347  fe.n_dofs_per_cell());
348 
349  std::vector<Vector<double>> cell_wind(n_q_points, Vector<double>(dim));
350  double normal_wind;
351 
352  Vector<double> local_matrix_row_i(fe.n_dofs_per_cell());
353 
354 
355  std::vector<Point<dim>> support_points(dof_handler.n_dofs());
356  DoFTools::map_dofs_to_support_points<dim - 1, dim>(mapping,
357  dof_handler,
358  support_points);
359 
360 
361  for (const auto &cell : dof_handler.active_cell_iterators())
362  {
363  fe_v.reinit(cell);
364  cell->get_dof_indices(local_dof_indices);
365 
366  const std::vector<Point<dim>> &q_points = fe_v.get_quadrature_points();
367  const std::vector<Tensor<1, dim>> &normals = fe_v.get_normal_vectors();
368  wind.vector_value_list(q_points, cell_wind);
369 
370  for (unsigned int i = 0; i < dof_handler.n_dofs(); ++i)
371  {
372  local_matrix_row_i = 0;
373 
374  bool is_singular = false;
375  unsigned int singular_index = numbers::invalid_unsigned_int;
376 
377  for (unsigned int j = 0; j < fe.n_dofs_per_cell(); ++j)
378  if (local_dof_indices[j] == i)
379  {
380  singular_index = j;
381  is_singular = true;
382  break;
383  }
384 
385  if (is_singular == false)
386  {
387  for (unsigned int q = 0; q < n_q_points; ++q)
388  {
389  normal_wind = 0;
390  for (unsigned int d = 0; d < dim; ++d)
391  normal_wind += normals[q][d] * cell_wind[q](d);
392 
393  const Tensor<1, dim> R = q_points[q] - support_points[i];
394 
395  system_rhs(i) += (LaplaceKernel::single_layer(R) *
396  normal_wind * fe_v.JxW(q));
397 
398  for (unsigned int j = 0; j < fe.n_dofs_per_cell(); ++j)
399 
400  local_matrix_row_i(j) -=
401  ((LaplaceKernel::double_layer(R) * normals[q]) *
402  fe_v.shape_value(j, q) * fe_v.JxW(q));
403  }
404  }
405  else
406  {
407  Assert(singular_index != numbers::invalid_unsigned_int,
408  ExcInternalError());
409 
410  const Quadrature<dim - 1> &singular_quadrature =
411  get_singular_quadrature(cell, singular_index);
412 
413  FEValues<dim - 1, dim> fe_v_singular(
414  mapping,
415  fe,
416  singular_quadrature,
417  update_jacobians | update_values | update_normal_vectors |
418  update_quadrature_points);
419 
420  fe_v_singular.reinit(cell);
421 
422  std::vector<Vector<double>> singular_cell_wind(
423  singular_quadrature.size(), Vector<double>(dim));
424 
425  const std::vector<Tensor<1, dim>> &singular_normals =
426  fe_v_singular.get_normal_vectors();
427  const std::vector<Point<dim>> &singular_q_points =
428  fe_v_singular.get_quadrature_points();
429 
430  wind.vector_value_list(singular_q_points, singular_cell_wind);
431 
432  for (unsigned int q = 0; q < singular_quadrature.size(); ++q)
433  {
434  const Tensor<1, dim> R =
435  singular_q_points[q] - support_points[i];
436  double normal_wind = 0;
437  for (unsigned int d = 0; d < dim; ++d)
438  normal_wind +=
439  (singular_cell_wind[q](d) * singular_normals[q][d]);
440 
441  system_rhs(i) += (LaplaceKernel::single_layer(R) *
442  normal_wind * fe_v_singular.JxW(q));
443 
444  for (unsigned int j = 0; j < fe.n_dofs_per_cell(); ++j)
445  {
446  local_matrix_row_i(j) -=
447  ((LaplaceKernel::double_layer(R) *
448  singular_normals[q]) *
449  fe_v_singular.shape_value(j, q) *
450  fe_v_singular.JxW(q));
451  }
452  }
453  }
454 
455  for (unsigned int j = 0; j < fe.n_dofs_per_cell(); ++j)
456  system_matrix(i, local_dof_indices[j]) += local_matrix_row_i(j);
457  }
458  }
459 
460 
461  Vector<double> ones(dof_handler.n_dofs());
462  ones.add(-1.);
463 
464  system_matrix.vmult(alpha, ones);
465  alpha.add(1);
466  for (unsigned int i = 0; i < dof_handler.n_dofs(); ++i)
467  system_matrix(i, i) += alpha(i);
468  }

558  template <int dim>
559  void BEMProblem<dim>::compute_exterior_solution()
560  {
561  Triangulation<dim> external_tria;
562  GridGenerator::hyper_cube(external_tria, -2, 2);
563 
564  FE_Q<dim> external_fe(1);
565  DoFHandler<dim> external_dh(external_tria);
566  Vector<double> external_phi;
567 
568  external_tria.refine_global(external_refinement);
569  external_dh.distribute_dofs(external_fe);
570  external_phi.reinit(external_dh.n_dofs());
571 
572  FEValues<dim - 1, dim> fe_v(mapping,
573  fe,
574  *quadrature,
575  update_values | update_normal_vectors |
576  update_quadrature_points | update_JxW_values);
577 
578  const unsigned int n_q_points = fe_v.n_quadrature_points;
579 
580  std::vector<types::global_dof_index> dofs(fe.n_dofs_per_cell());
581 
582  std::vector<double> local_phi(n_q_points);
583  std::vector<double> normal_wind(n_q_points);
584  std::vector<Vector<double>> local_wind(n_q_points, Vector<double>(dim));
585 
586  std::vector<Point<dim>> external_support_points(external_dh.n_dofs());
587  DoFTools::map_dofs_to_support_points<dim>(StaticMappingQ1<dim>::mapping,
588  external_dh,
589  external_support_points);
590 
591  for (const auto &cell : dof_handler.active_cell_iterators())
592  {
593  fe_v.reinit(cell);
594 
595  const std::vector<Point<dim>> &q_points = fe_v.get_quadrature_points();
596  const std::vector<Tensor<1, dim>> &normals = fe_v.get_normal_vectors();
597 
598  cell->get_dof_indices(dofs);
599  fe_v.get_function_values(phi, local_phi);
600 
601  wind.vector_value_list(q_points, local_wind);
602 
603  for (unsigned int q = 0; q < n_q_points; ++q)
604  {
605  normal_wind[q] = 0;
606  for (unsigned int d = 0; d < dim; ++d)
607  normal_wind[q] += normals[q][d] * local_wind[q](d);
608  }
609 
610  for (unsigned int i = 0; i < external_dh.n_dofs(); ++i)
611  for (unsigned int q = 0; q < n_q_points; ++q)
612  {
613  const Tensor<1, dim> R = q_points[q] - external_support_points[i];
614 
615  external_phi(i) +=
616  ((LaplaceKernel::single_layer(R) * normal_wind[q] +
617  (LaplaceKernel::double_layer(R) * normals[q]) *
618  local_phi[q]) *
619  fe_v.JxW(q));
620  }
621  }
622 
623  DataOut<dim> data_out;
624 
625  data_out.attach_dof_handler(external_dh);
626  data_out.add_data_vector(external_phi, "external_phi");
627  data_out.build_patches();
628 
629  const std::string filename = std::to_string(dim) + "d_external.vtk";
630  std::ofstream file(filename);
631 
632  data_out.write_vtk(file);
633  }
