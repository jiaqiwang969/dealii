617 template <typename CellIteratorType, typename DataType>
618 template <typename T>
619 inline void
620 CellDataStorage<CellIteratorType, DataType>::initialize(
621  const CellIteratorType &cell,
622  const unsigned int n_q_points)
623 {
624  static_assert(std::is_base_of<DataType, T>::value,
625  "User's T class should be derived from user's DataType class");
626  // The first time this method is called, it has to initialize the reference
627  // to the triangulation object
628  if (!tria)
629  tria = &cell->get_triangulation();
630  Assert(&cell->get_triangulation() == tria, ExcTriangulationMismatch());
631 
632  const auto key = cell->id();
633  if (map.find(key) == map.end())
634  {
635  map[key] = std::vector<std::shared_ptr<DataType>>(n_q_points);
636  // we need to initialize one-by-one as the std::vector<>(q, T())
637  // will end with a single same T object stored in each element of the
638  // vector:
639  const auto it = map.find(key);
640  for (unsigned int q = 0; q < n_q_points; q++)
641  it->second[q] = std::make_shared<T>();
642  }
643 }

662 template <typename CellIteratorType, typename DataType>
663 inline bool
664 CellDataStorage<CellIteratorType, DataType>::erase(const CellIteratorType &cell)
665 {
666  const auto key = cell->id();
667  const auto it = map.find(key);
668  if (it == map.end())
669  return false;
670  Assert(&cell->get_triangulation() == tria, ExcTriangulationMismatch());
671  for (unsigned int i = 0; i < it->second.size(); i++)
672  {
673  Assert(
674  it->second[i].unique(),
675  ExcMessage(
676  "Can not erase the cell data multiple objects reference its data."));
677  }
678 
679  return (map.erase(key) == 1);
680 }

710 inline std::vector<std::shared_ptr<T>>
711 CellDataStorage<CellIteratorType, DataType>::get_data(
712  const CellIteratorType &cell)
713 {
714  static_assert(std::is_base_of<DataType, T>::value,
715  "User's T class should be derived from user's DataType class");
716  Assert(&cell->get_triangulation() == tria, ExcTriangulationMismatch());
717 
718  const auto it = map.find(cell->id());
719  Assert(it != map.end(), ExcMessage("Could not find data for the cell"));
720 
721  // It would be nice to have a specialized version of this function for
722  // T==DataType. However explicit (i.e full) specialization of a member
723  // template is only allowed when the enclosing class is also explicitly (i.e
724  // fully) specialized. Thus, stick with copying of shared pointers even when
725  // the T==DataType:
726  std::vector<std::shared_ptr<T>> res(it->second.size());
727  for (unsigned int q = 0; q < res.size(); q++)
728  {
729  res[q] = std::dynamic_pointer_cast<T>(it->second[q]);
730  Assert(res[q], ExcCellDataTypeMismatch());
731  }
732  return res;
733 }

829 /* Pack cell data of type @p DataType stored using @p data_storage in @p cell at each quadrature point to @p matrix_data. Here @p matrix_data is a matrix whose first index corresponds to different quadrature points on the cell whereas the second index represents different values stored at each quadrature point in the DataType class.
830 
831 *
832 * */
833 template <typename CellIteratorType, typename DataType>
834 inline void
835 pack_cell_data(const CellIteratorType & cell,
836  const CellDataStorage<CellIteratorType, DataType> *data_storage,
837  FullMatrix<double> & matrix_data)
838 {
839  static_assert(
840  std::is_base_of<TransferableQuadraturePointData, DataType>::value,
841  "User's DataType class should be derived from QPData");
842 
843  if (const auto qpd = data_storage->try_get_data(cell))
844  {
845  const unsigned int m = qpd->size();
846  Assert(m > 0, ExcInternalError());
847  const unsigned int n = (*qpd)[0]->number_of_values();
848  matrix_data.reinit(m, n);
849 
850  std::vector<double> single_qp_data(n);
851  for (unsigned int q = 0; q < m; ++q)
852  {
853  (*qpd)[q]->pack_values(single_qp_data);
854  AssertDimension(single_qp_data.size(), n);
855 
856  for (unsigned int i = 0; i < n; ++i)
857  matrix_data(q, i) = single_qp_data[i];
858  }
859  }
860  else
861  {
862  matrix_data.reinit({0, 0});
863  }
864 }

894  (*qpd)[q]->unpack_values(single_qp_data);
895  }

1003  template <int dim, typename DataType>
1004  inline void
1005  ContinuousQuadratureDataTransfer<dim, DataType>::unpack_function(
1006  const typename parallel::distributed::Triangulation<dim>::cell_iterator
1007  &cell,
1008  const typename parallel::distributed::Triangulation<dim>::CellStatus
1009  status,
1010  const boost::iterator_range<std::vector<char>::const_iterator>
1011  &data_range)
1012  {
1013  Assert((status !=
1014  parallel::distributed::Triangulation<dim, dim>::CELL_COARSEN),
1015  ExcNotImplemented());
1016  (void)status;
1017 
1018  matrix_dofs =
1019  Utilities::unpack<FullMatrix<double>>(data_range.begin(),
1020  data_range.end(),
1021  /*allow_compression=*/ false);
1022  const unsigned int number_of_values = matrix_dofs.n();
1023  if (number_of_values == 0)
1024  return;
1025 
1026  matrix_quadrature.reinit(n_q_points, number_of_values);
1027 
1028  if (cell->has_children())
1029  {
1030  // we need to first use prolongation matrix to get dofvalues on child
1031  // cells based on dofvalues stored in the parent's data_store
1032  matrix_dofs_child.reinit(projection_fe->n_dofs_per_cell(),
1033  number_of_values);
1034  for (unsigned int child = 0; child < cell->n_children(); ++child)
1035  if (cell->child(child)->is_locally_owned())
1036  {
1037  projection_fe
1038  ->get_prolongation_matrix(child, cell->refinement_case())
1039  .mmult(matrix_dofs_child, matrix_dofs);
1040 
1041  // now we do the usual business of evaluating FE on quadrature
1042  // points:
1043  project_to_qp_matrix.mmult(matrix_quadrature,
1044  matrix_dofs_child);
1045 
1046  // finally, put back into the map:
1047  unpack_to_cell_data(cell->child(child),
1048  matrix_quadrature,
1049  data_storage);
1050  }
1051  }
1052  else
1053  {
1054  // if there are no children, evaluate FE field at
1055  // rhs_quadrature points.
1056  project_to_qp_matrix.mmult(matrix_quadrature, matrix_dofs);
1057 
1058  // finally, put back into the map:
1059  unpack_to_cell_data(cell, matrix_quadrature, data_storage);
1060  }
1061  }
