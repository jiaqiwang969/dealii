715  inline Domination operator&(const Domination d1, const Domination d2)
716  {
717  // go through the entire list of possibilities. note that if we were into
718  // speed, obfuscation and cared enough, we could implement this operator
719  // by doing a bitwise & (and) if we gave these values to the enum values:
720  // neither_element_dominates=0, this_element_dominates=1,
721  // other_element_dominates=2, either_element_can_dominate=3
722  // =this_element_dominates|other_element_dominates
723  switch (d1)
724  {
725  case this_element_dominates:
726  if ((d2 == this_element_dominates) ||
727  (d2 == either_element_can_dominate) || (d2 == no_requirements))
728  return this_element_dominates;
729  else
730  return neither_element_dominates;
731 
732  case other_element_dominates:
733  if ((d2 == other_element_dominates) ||
734  (d2 == either_element_can_dominate) || (d2 == no_requirements))
735  return other_element_dominates;
736  else
737  return neither_element_dominates;
738 
739  case neither_element_dominates:
740  return neither_element_dominates;
741 
742  case either_element_can_dominate:
743  if (d2 == no_requirements)
744  return either_element_can_dominate;
745  else
746  return d2;
747 
748  case no_requirements:
749  return d2;
750 
751  default:
752  // shouldn't get here
753  Assert(false, ExcInternalError());
754  }
755 
756  return neither_element_dominates;
757  }
