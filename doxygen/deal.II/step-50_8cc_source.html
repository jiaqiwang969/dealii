214 template <int dim>
215 template <typename number>
216 std::shared_ptr<Table<2, VectorizedArray<number>>>
217 Coefficient<dim>::make_coefficient_table(
218  const MatrixFree<dim, number, VectorizedArray<number>> &mf_storage) const
219 {
220  auto coefficient_table =
221  std::make_shared<Table<2, VectorizedArray<number>>>();
222 
223  FEEvaluation<dim, -1, 0, 1, number> fe_eval(mf_storage);
224 
225  const unsigned int n_cells = mf_storage.n_cell_batches();
226  const unsigned int n_q_points = fe_eval.n_q_points;
227 
228  coefficient_table->reinit(n_cells, 1);
229 
230  for (unsigned int cell = 0; cell < n_cells; ++cell)
231  {
232  fe_eval.reinit(cell);
233 
234  VectorizedArray<number> average_value = 0.;
235  for (unsigned int q = 0; q < n_q_points; ++q)
236  average_value += value(fe_eval.quadrature_point(q));
237  average_value /= n_q_points;
238 
239  (*coefficient_table)(cell, 0) = average_value;
240  }
241 
242  return coefficient_table;

435 void LaplaceProblem<dim, degree>::setup_system()
436 {
437  TimerOutput::Scope timing(computing_timer, "Setup");
438 
439  dof_handler.distribute_dofs(fe);
440 
441  DoFTools::extract_locally_relevant_dofs(dof_handler, locally_relevant_dofs);
442  locally_owned_dofs = dof_handler.locally_owned_dofs();
443 
444  solution.reinit(locally_owned_dofs, mpi_communicator);
445  right_hand_side.reinit(locally_owned_dofs, mpi_communicator);
446  constraints.reinit(locally_relevant_dofs);
447  DoFTools::make_hanging_node_constraints(dof_handler, constraints);
448 
449  VectorTools::interpolate_boundary_values(
450  mapping, dof_handler, 0, Functions::ZeroFunction<dim>(), constraints);
451  constraints.close();
452 
453  switch (settings.solver)
454  {
455  case Settings::gmg_mf:
456  {
457  typename MatrixFree<dim, double>::AdditionalData additional_data;
458  additional_data.tasks_parallel_scheme =
459  MatrixFree<dim, double>::AdditionalData::none;
460  additional_data.mapping_update_flags =
461  (update_gradients | update_JxW_values | update_quadrature_points);
462  std::shared_ptr<MatrixFree<dim, double>> mf_storage =
463  std::make_shared<MatrixFree<dim, double>>();
464  mf_storage->reinit(mapping,
465  dof_handler,
466  constraints,
467  QGauss<1>(degree + 1),
468  additional_data);
469 
470  mf_system_matrix.initialize(mf_storage);
471 
472  const Coefficient<dim> coefficient;
473  mf_system_matrix.set_coefficient(
474  coefficient.make_coefficient_table(*mf_storage));

477  }
478

481  {

485 
486  SparsityTools::distribute_sparsity_pattern(dsp,
487  locally_owned_dofs,
488  mpi_communicator,
489  locally_relevant_dofs);
490

494  mpi_communicator);

664 template <int dim, int degree>
665 void LaplaceProblem<dim, degree>::assemble_system()
666 {
667  TimerOutput::Scope timing(computing_timer, "Assemble");
668 
669  const QGauss<dim> quadrature_formula(degree + 1);
670 
671  FEValues<dim> fe_values(fe,
672  quadrature_formula,
673  update_values | update_gradients |
674  update_quadrature_points | update_JxW_values);
675 
676  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
677  const unsigned int n_q_points = quadrature_formula.size();
678 
679  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
680  Vector<double> cell_rhs(dofs_per_cell);
681 
682  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
683 
684  const Coefficient<dim> coefficient;
685  RightHandSide<dim> rhs;
686  std::vector<double> rhs_values(n_q_points);
687 
688  for (const auto &cell : dof_handler.active_cell_iterators())
689  if (cell->is_locally_owned())
690  {
691  cell_matrix = 0;
692  cell_rhs = 0;
693 
694  fe_values.reinit(cell);
695 
696  const double coefficient_value =
697  coefficient.average_value(fe_values.get_quadrature_points());
698  rhs.value_list(fe_values.get_quadrature_points(), rhs_values);
699 
700  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
701  for (unsigned int i = 0; i < dofs_per_cell; ++i)
702  {
703  for (unsigned int j = 0; j < dofs_per_cell; ++j)
704  cell_matrix(i, j) +=
705  coefficient_value * // epsilon(x)
706  fe_values.shape_grad(i, q_point) * // * grad phi_i(x)
707  fe_values.shape_grad(j, q_point) * // * grad phi_j(x)
708  fe_values.JxW(q_point); // * dx
709 
710  cell_rhs(i) +=
711  fe_values.shape_value(i, q_point) * // grad phi_i(x)
712  rhs_values[q_point] * // * f(x)
713  fe_values.JxW(q_point); // * dx
714  }
715 
716  cell->get_dof_indices(local_dof_indices);
717  constraints.distribute_local_to_global(cell_matrix,
718  cell_rhs,
719  local_dof_indices,
720  system_matrix,
721  right_hand_side);
722  }
723 
724  system_matrix.compress(VectorOperation::add);
725  right_hand_side.compress(VectorOperation::add);

750

793  mg_interface_in[cell->level()].add(local_dof_indices[i],
794  local_dof_indices[j],
795  cell_matrix(i, j));
796  }

1125 template <int dim, int degree>
1126 void LaplaceProblem<dim, degree>::estimate()
1127 {
1128  TimerOutput::Scope timing(computing_timer, "Estimate");
1129 
1130  VectorType temp_solution;
1131  temp_solution.reinit(locally_owned_dofs,
1132  locally_relevant_dofs,
1133  mpi_communicator);
1134  temp_solution = solution;
1135 
1136  const Coefficient<dim> coefficient;
1137 
1138  estimated_error_square_per_cell.reinit(triangulation.n_active_cells());
1139 
1140  using Iterator = typename DoFHandler<dim>::active_cell_iterator;
1141 
1142  auto cell_worker = [&](const Iterator & cell,
1143  ScratchData<dim> &scratch_data,
1144  CopyData & copy_data) {
1145  FEValues<dim> &fe_values = scratch_data.fe_values;
1146  fe_values.reinit(cell);
1147 
1148  RightHandSide<dim> rhs;
1149  const double rhs_value = rhs.value(cell->center());
1150 
1151  const double nu = coefficient.value(cell->center());
1152 
1153  std::vector<Tensor<2, dim>> hessians(fe_values.n_quadrature_points);
1154  fe_values.get_function_hessians(temp_solution, hessians);
1155 
1156  copy_data.cell_index = cell->active_cell_index();
1157 
1158  double residual_norm_square = 0.;
1159  for (unsigned k = 0; k < fe_values.n_quadrature_points; ++k)
1160  {
1161  const double residual = (rhs_value + nu * trace(hessians[k]));
1162  residual_norm_square += residual * residual * fe_values.JxW(k);
1163  }
1164 
1165  copy_data.value =
1166  cell->diameter() * cell->diameter() * residual_norm_square;
1167  };
1168 
1169  auto face_worker = [&](const Iterator & cell,
1170  const unsigned int &f,
1171  const unsigned int &sf,
1172  const Iterator & ncell,
1173  const unsigned int &nf,
1174  const unsigned int &nsf,
1175  ScratchData<dim> & scratch_data,
1176  CopyData & copy_data) {
1177  FEInterfaceValues<dim> &fe_interface_values =
1178  scratch_data.fe_interface_values;
1179  fe_interface_values.reinit(cell, f, sf, ncell, nf, nsf);
1180 
1181  copy_data.face_data.emplace_back();
1182  CopyData::FaceData &copy_data_face = copy_data.face_data.back();
1183 
1184  copy_data_face.cell_indices[0] = cell->active_cell_index();
1185  copy_data_face.cell_indices[1] = ncell->active_cell_index();
1186 
1187  const double coeff1 = coefficient.value(cell->center());
1188  const double coeff2 = coefficient.value(ncell->center());
1189 
1190  std::vector<Tensor<1, dim>> grad_u[2];
1191 
1192  for (unsigned int i = 0; i < 2; ++i)
1193  {
1194  grad_u[i].resize(fe_interface_values.n_quadrature_points);
1195  fe_interface_values.get_fe_face_values(i).get_function_gradients(
1196  temp_solution, grad_u[i]);
1197  }
1198 
1199  double jump_norm_square = 0.;
1200 
1201  for (unsigned int qpoint = 0;
1202  qpoint < fe_interface_values.n_quadrature_points;
1203  ++qpoint)
1204  {
1205  const double jump =
1206  coeff1 * grad_u[0][qpoint] * fe_interface_values.normal(qpoint) -
1207  coeff2 * grad_u[1][qpoint] * fe_interface_values.normal(qpoint);
1208 
1209  jump_norm_square += jump * jump * fe_interface_values.JxW(qpoint);
1210  }
1211 
1212  const double h = cell->face(f)->measure();
1213  copy_data_face.values[0] = 0.5 * h * jump_norm_square;
1214  copy_data_face.values[1] = copy_data_face.values[0];
1215  };
1216 
1217  auto copier = [&](const CopyData &copy_data) {
1218  if (copy_data.cell_index != numbers::invalid_unsigned_int)
1219  estimated_error_square_per_cell[copy_data.cell_index] += copy_data.value;
1220 
1221  for (auto &cdf : copy_data.face_data)
1222  for (unsigned int j = 0; j < 2; ++j)
1223  estimated_error_square_per_cell[cdf.cell_indices[j]] += cdf.values[j];
1224  };
1225 
1226  const unsigned int n_gauss_points = degree + 1;
1227  ScratchData<dim> scratch_data(mapping,
1228  fe,
1229  n_gauss_points,
1230  update_hessians | update_quadrature_points |
1231  update_JxW_values,
1232  update_values | update_gradients |
1233  update_JxW_values | update_normal_vectors);
1234  CopyData copy_data;
1235 
1236  MeshWorker::mesh_loop(dof_handler.begin_active(),
1237  dof_handler.end(),
1238  cell_worker,
1239  copier,
1240  scratch_data,
1241  copy_data,
1242  MeshWorker::assemble_own_cells |
1243  MeshWorker::assemble_ghost_faces_both |
1244  MeshWorker::assemble_own_interior_faces_once,
1245  /*boundary_worker=*/nullptr,
1246  face_worker);
1247 
1248  const double global_error_estimate =
1249  std::sqrt(Utilities::MPI::sum(estimated_error_square_per_cell.l1_norm(),
1250  mpi_communicator));
1251  pcout << " Global error estimate: " << global_error_estimate
1252  << std::endl;
1253 }
