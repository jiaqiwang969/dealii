340  template <int dim>
341  void AdvectionProblem<dim>::local_assemble_system(
342  const typename DoFHandler<dim>::active_cell_iterator &cell,
343  AssemblyScratchData & scratch_data,
344  AssemblyCopyData & copy_data)
345  {
346  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
347  const unsigned int n_q_points =
348  scratch_data.fe_values.get_quadrature().size();
349  const unsigned int n_face_q_points =
350  scratch_data.fe_face_values.get_quadrature().size();
351 
352  copy_data.cell_matrix.reinit(dofs_per_cell, dofs_per_cell);
353  copy_data.cell_rhs.reinit(dofs_per_cell);
354 
355  copy_data.local_dof_indices.resize(dofs_per_cell);
356 
357  scratch_data.fe_values.reinit(cell);
358 
359  scratch_data.advection_field.value_list(
360  scratch_data.fe_values.get_quadrature_points(),
361  scratch_data.advection_directions);
362  scratch_data.right_hand_side.value_list(
363  scratch_data.fe_values.get_quadrature_points(), scratch_data.rhs_values);
364 
365  const double delta = 0.1 * cell->diameter();
366 
367  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
368  for (unsigned int i = 0; i < dofs_per_cell; ++i)
369  {
370  const auto &sd = scratch_data;
371  for (unsigned int j = 0; j < dofs_per_cell; ++j)
372  copy_data.cell_matrix(i, j) +=
373  ((sd.fe_values.shape_value(i, q_point) + // (phi_i +
374  delta * (sd.advection_directions[q_point] * // delta beta
375  sd.fe_values.shape_grad(i, q_point))) * // grad phi_i)
376  sd.advection_directions[q_point] * // beta
377  sd.fe_values.shape_grad(j, q_point)) * // grad phi_j
378  sd.fe_values.JxW(q_point); // dx
379 
380  copy_data.cell_rhs(i) +=
381  (sd.fe_values.shape_value(i, q_point) + // (phi_i +
382  delta * (sd.advection_directions[q_point] * // delta beta
383  sd.fe_values.shape_grad(i, q_point))) * // grad phi_i)
384  sd.rhs_values[q_point] * // f
385  sd.fe_values.JxW(q_point); // dx
386  }
387 
388  for (const auto &face : cell->face_iterators())
389  if (face->at_boundary())
390  {
391  scratch_data.fe_face_values.reinit(cell, face);
392 
393  scratch_data.boundary_values.value_list(
394  scratch_data.fe_face_values.get_quadrature_points(),
395  scratch_data.face_boundary_values);
396  scratch_data.advection_field.value_list(
397  scratch_data.fe_face_values.get_quadrature_points(),
398  scratch_data.face_advection_directions);
399 
400  for (unsigned int q_point = 0; q_point < n_face_q_points; ++q_point)
401  if (scratch_data.fe_face_values.normal_vector(q_point) *
402  scratch_data.face_advection_directions[q_point] <
403  0.)
404  for (unsigned int i = 0; i < dofs_per_cell; ++i)
405  {
406  for (unsigned int j = 0; j < dofs_per_cell; ++j)
407  copy_data.cell_matrix(i, j) -=
408  (scratch_data.face_advection_directions[q_point] *
409  scratch_data.fe_face_values.normal_vector(q_point) *
410  scratch_data.fe_face_values.shape_value(i, q_point) *
411  scratch_data.fe_face_values.shape_value(j, q_point) *
412  scratch_data.fe_face_values.JxW(q_point));
413 
414  copy_data.cell_rhs(i) -=
415  (scratch_data.face_advection_directions[q_point] *
416  scratch_data.fe_face_values.normal_vector(q_point) *
417  scratch_data.face_boundary_values[q_point] *
418  scratch_data.fe_face_values.shape_value(i, q_point) *
419  scratch_data.fe_face_values.JxW(q_point));
420  }
421  }
422 
423  cell->get_dof_indices(copy_data.local_dof_indices);
424  }

444  system_rhs.size() / 10),

593  template <int dim>
594  void GradientEstimation::estimate_cell(
595  const typename DoFHandler<dim>::active_cell_iterator &cell,
596  EstimateScratchData<dim> & scratch_data,
597  const EstimateCopyData &)
598  {
599  Tensor<2, dim> Y;
600 
601  scratch_data.fe_midpoint_value.reinit(cell);
602 
603  scratch_data.active_neighbors.clear();
604  for (const auto face_n : cell->face_indices())
605  if (!cell->at_boundary(face_n))
606  {
607  const auto face = cell->face(face_n);
608  const auto neighbor = cell->neighbor(face_n);
609 
610  if (neighbor->is_active())
611  scratch_data.active_neighbors.push_back(neighbor);
612  else
613  {
614  if (dim == 1)
615  {
616  auto neighbor_child = neighbor;
617  while (neighbor_child->has_children())
618  neighbor_child = neighbor_child->child(face_n == 0 ? 1 : 0);
619 
620  Assert(neighbor_child->neighbor(face_n == 0 ? 1 : 0) == cell,
621  ExcInternalError());
622 
623  scratch_data.active_neighbors.push_back(neighbor_child);
624  }
625  else
626  for (unsigned int subface_n = 0; subface_n < face->n_children();
627  ++subface_n)
628  scratch_data.active_neighbors.push_back(
629  cell->neighbor_child_on_subface(face_n, subface_n));
630  }
631  }
632 
633  const Point<dim> this_center =
634  scratch_data.fe_midpoint_value.quadrature_point(0);
635 
636  scratch_data.fe_midpoint_value.get_function_values(
637  scratch_data.solution, scratch_data.cell_midpoint_value);
638 
639  Tensor<1, dim> projected_gradient;
640  for (const auto &neighbor : scratch_data.active_neighbors)
641  {
642  scratch_data.fe_midpoint_value.reinit(neighbor);
643  const Point<dim> neighbor_center =
644  scratch_data.fe_midpoint_value.quadrature_point(0);
645 
646  scratch_data.fe_midpoint_value.get_function_values(
647  scratch_data.solution, scratch_data.neighbor_midpoint_value);
648 
649  Tensor<1, dim> y = neighbor_center - this_center;
650  const double distance = y.norm();
651  y /= distance;
652 
653  for (unsigned int i = 0; i < dim; ++i)
654  for (unsigned int j = 0; j < dim; ++j)
