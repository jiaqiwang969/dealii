417  &boundary_points)

422  ExcMessage("Bounding Box can't be created: the points' "

430 template <int spacedim, typename Number>
431 template <class Container>
432 inline BoundingBox<spacedim, Number>::BoundingBox(const Container &points)
433 {
434  // Use the default constructor in case points is empty instead of setting
435  // things to +oo and -oo
436  if (points.size() > 0)
437  {
438  auto &min = boundary_points.first;
439  auto &max = boundary_points.second;
440  std::fill(min.begin_raw(),
441  min.end_raw(),
442  std::numeric_limits<Number>::infinity());
443  std::fill(max.begin_raw(),
444  max.end_raw(),
445  -std::numeric_limits<Number>::infinity());
446 
447  for (const Point<spacedim, Number> &point : points)
448  for (unsigned int d = 0; d < spacedim; ++d)
449  {
450  min[d] = std::min(min[d], point[d]);
451  max[d] = std::max(max[d], point[d]);
452  }
453  }
454 }
