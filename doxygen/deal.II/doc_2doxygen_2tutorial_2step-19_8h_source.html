638  * <h4>The <code>CathodeRaySimulator::make_grid</code> function</h4>
639  *
640 
641  *
642  * The next function is then responsible for generating the mesh on which
643  * we want to solve. Recall how the domain looks like:
644  * <p align="center">
645  * <img
646  * src="https://www.dealii.org/images/steps/developer/step-19.geometry.png"
647  * alt="The geometry used in this program"
648  * width="600">
649  * </p>
650  * We subdivide this geometry into a mesh of @f$4\times 2@f$ cells that looks
651  * like this:
652  * <div class=CodeFragmentInTutorialComment>
653  * @code
654  * *---*---*---*---*
655  * \ | | | |
656  * *--*---*---*---*
657  * / | | | |
658  * *---*---*---*---*
659  * @endcode
660  * </div>
661  * The way this is done is by first defining where the @f$15=5\times 3@f$
662  * vertices are located -- here, we say that they are on integer points
663  * with the middle one on the left side moved to the right by a value of
664  * `delta=0.5`.
665  *
666 
667  *
668  * In the following, we then have to say which vertices together form
669  * the 8 cells. The following code is then entirely equivalent to what
670  * we also do in @ref step_14 "step-14":
671  *
672  * @code
673  * template <int dim>
674  * void CathodeRaySimulator<dim>::make_grid()
675  * {
676  * static_assert(dim == 2,
677  * "This function is currently only implemented for 2d.");
678  *
679  * const double delta = 0.5;
680  * const unsigned int nx = 5;
681  * const unsigned int ny = 3;
682  *
683  * const std::vector<Point<dim>> vertices
684  * = {{0, 0},
685  * {1, 0},
686  * {2, 0},
687  * {3, 0},
688  * {4, 0},
689  * {delta, 1},
690  * {1, 1},
691  * {2, 1},
692  * {3, 1},
693  * {4, 1},
694  * {0, 2},
695  * {1, 2},
696  * {2, 2},
697  * {3, 2},
698  * {4, 2}};
699  * AssertDimension(vertices.size(), nx * ny);
700  *
701  * const std::vector<unsigned int> cell_vertices[(nx - 1) * (ny - 1)] = {
702  * {0, 1, nx + 0, nx + 1},
703  * {1, 2, nx + 1, nx + 2},
704  * {2, 3, nx + 2, nx + 3},
705  * {3, 4, nx + 3, nx + 4},
706  *
707  * {5, nx + 1, 2 * nx + 0, 2 * nx + 1},
708  * {nx + 1, nx + 2, 2 * nx + 1, 2 * nx + 2},
709  * {nx + 2, nx + 3, 2 * nx + 2, 2 * nx + 3},
710  * {nx + 3, nx + 4, 2 * nx + 3, 2 * nx + 4}};
711  *
712  * @endcode
713  *
714  * With these arrays out of the way, we can move to slightly higher
715  * higher-level data structures. We create a vector of CellData
716  * objects that store for each cell to be created the vertices in
717  * question as well as the @ref GlossMaterialId "material id" (which
718  * we will here simply set to zero since we don't use it in the program).
719  *
720 
721  *
722  * This information is then handed to the
723  * Triangulation::create_triangulation() function, and the mesh is twice
724  * globally refined.
725  *
726  * @code
727  * std::vector<CellData<dim>> cells((nx - 1) * (ny - 1), CellData<dim>());
728  * for (unsigned int i = 0; i < cells.size(); ++i)
729  * {
730  * cells[i].vertices = cell_vertices[i];
731  * cells[i].material_id = 0;
732  * }
733  *
734  * triangulation.create_triangulation(
735  * vertices,
736  * cells,
737  * SubCellData()); // No boundary information
738  *
739  * triangulation.refine_global(2);
740  *
741  * @endcode
742  *
743  * The remaining part of the function loops over all cells and their faces,
744  * and if a face is at the boundary determines which boundary indicator
745  * should be applied to it. The various conditions should make sense if
746  * you compare the code with the picture of the geometry above.
747  *
748 
749  *
750  * Once done with this step, we refine the mesh once more globally.
751  *
752  * @code
753  * for (auto &cell : triangulation.active_cell_iterators())
754  * for (auto &face : cell->face_iterators())
755  * if (face->at_boundary())
756  * {
757  * if ((face->center()[0] > 0) && (face->center()[0] < 0.5) &&
758  * (face->center()[1] > 0) && (face->center()[1] < 2))
759  * face->set_boundary_id(BoundaryIds::cathode);
760  * else if ((face->center()[0] > 0) && (face->center()[0] < 2))
761  * face->set_boundary_id(BoundaryIds::focus_element);
762  * else if ((face->center()[0] > 4 - 1e-12) &&
763  * ((face->center()[1] > 1.5) || (face->center()[1] < 0.5)))
764  * face->set_boundary_id(BoundaryIds::anode);
765  * else
766  * face->set_boundary_id(BoundaryIds::open);
767  * }
768  *
769  * triangulation.refine_global(1);
770  * }

829  * <a name="ThecodeCathodeRaySimulatorassemble_systemcodefunction"></a>
830  * <h4>The <code>CathodeRaySimulator::assemble_system</code> function</h4>
831  *
832 
833  *
834  * The function that computes
835  * the matrix entries is again in essence a copy of the
836  * corresponding function in @ref step_6 "step-6":
837  *
838  * @code
839  * template <int dim>
840  * void CathodeRaySimulator<dim>::assemble_system()
841  * {
842  * system_matrix = 0;
843  * system_rhs = 0;
844  *
845  * const QGauss<dim> quadrature_formula(fe.degree + 1);
846  *
847  * FEValues<dim> fe_values(fe,
848  * quadrature_formula,
849  * update_values | update_gradients |
850  * update_quadrature_points | update_JxW_values);
851  *
852  * const unsigned int dofs_per_cell = fe.dofs_per_cell;
853  *
854  * FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
855  * Vector<double> cell_rhs(dofs_per_cell);
856  *
857  * std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
858  *
859  * for (const auto &cell : dof_handler.active_cell_iterators())
860  * {
861  * cell_matrix = 0;
862  * cell_rhs = 0;
863  *
864  * fe_values.reinit(cell);
865  *
866  * for (const unsigned int q_index : fe_values.quadrature_point_indices())
867  * for (const unsigned int i : fe_values.dof_indices())
868  * {
869  * for (const unsigned int j : fe_values.dof_indices())
870  * cell_matrix(i, j) +=
871  * (fe_values.shape_grad(i, q_index) * // grad phi_i(x_q)
872  * fe_values.shape_grad(j, q_index) * // grad phi_j(x_q)
873  * fe_values.JxW(q_index)); // dx
874  * }
875  *
876  * @endcode
877  *
878  * The only interesting part of this function is how it forms the right
879  * hand side of the linear system. Recall that the right hand side
880  * of the PDE is
881  * @f[
882  * \sum_p (N e)\delta(\mathbf x-\mathbf x_p),
883  * @f]
884  * where we have used @f$p@f$ to index the particles here to avoid
885  * confusion with the shape function @f$\varphi_i@f$; @f$\mathbf x_p@f$
886  * is the position of the @f$p@f$th particle.
887  *
888 
889  *
890  * When multiplied by a test function @f$\varphi_i@f$ and integrated over
891  * the domain results in a right hand side vector
892  * @f{align*}{
893  * F_i &= \int_\Omega \varphi_i (\mathbf x)\left[
894  * \sum_p (N e)\delta(\mathbf x-\mathbf x_p) \right] dx
895  * \\ &= \sum_p (N e) \varphi_i(\mathbf x_p).
896  * @f}
897  * Note that the final line no longer contains an integral, and
898  * consequently also no occurrence of @f$dx@f$ which would require the
899  * appearance of the `JxW` symbol in our code.
900  *
901 
902  *
903  * For a given cell @f$K@f$, this cell's contribution to the right hand
904  * side is then
905  * @f{align*}{
906  * F_i^K &= \sum_{p, \mathbf x_p\in K} (N e) \varphi_i(\mathbf x_p),
907  * @f}
908  * i.e., we only have to worry about those particles that are actually
909  * located on the current cell @f$K@f$.
910  *
911 
912  *
913  * In practice, what we do here is the following: If there are any
914  * particles on the current cell, then we first obtain an iterator range
915  * pointing to the first particle of that cell as well as the particle
916  * past the last one on this cell (or the end iterator) -- i.e., a
917  * half-open range as is common for C++ functions. Knowing now the list
918  * of particles, we query their reference locations (with respect to
919  * the reference cell), evaluate the shape functions in these reference
920  * locations, and compute the force according to the formula above
921  * (without any FEValues::JxW).
922  *
923 
924  *
925  * @note It is worth pointing out that calling the
926  * Particles::ParticleHandler::particles_in_cell() and
927  * Particles::ParticleHandler::n_particles_in_cell() functions is not
928  * very efficient on problems with a large number of particles. But it
929  * illustrates the easiest way to write this algorithm, and so we are
930  * willing to incur this cost for the moment for expository purposes.
931  * We discuss the issue in more detail in the
932  * <a href="#extensions">"possibilities for extensions" section</a>
933  * below, and use a better approach in @ref step_70 "step-70", for example.
934  *
935  * @code
936  * if (particle_handler.n_particles_in_cell(cell) > 0)
937  * for (const auto &particle : particle_handler.particles_in_cell(cell))
938  * {
939  * const Point<dim> &reference_location =
940  * particle.get_reference_location();
941  * for (const unsigned int i : fe_values.dof_indices())
942  * cell_rhs(i) +=
943  * (fe.shape_value(i, reference_location) * // phi_i(x_p)
944  * (-Constants::electrons_per_particle * // N
945  * Constants::electron_charge)); // e
946  * }
947  *
948  * @endcode
949  *
950  * Finally, we can copy the contributions of this cell into
951  * the global matrix and right hand side vector:
952  *
953  * @code
954  * cell->get_dof_indices(local_dof_indices);
955  * constraints.distribute_local_to_global(
956  * cell_matrix, cell_rhs, local_dof_indices, system_matrix, system_rhs);
957  * }
958  * }
959  *
960  *
961  * @endcode
962  *

1027  * <a name="CathodeRaySimulatorcreate_particles"></a>
1028  * <h4>CathodeRaySimulator::create_particles</h4>
1029  *
1030 
1031  *
1032  * Let us now turn to the functions that deal with particles. The first one
1033  * is about the creation of particles. As mentioned in the introduction,
1034  * we want to create a particle at points of the cathode if the the electric
1035  * field @f$\mathbf E=\nabla V@f$ exceeds a certain threshold, i.e., if
1036  * @f$|\mathbf E| \ge E_\text{threshold}@f$, and if furthermore the electric field
1037  * points into the domain (i.e., if @f$\mathbf E \cdot \mathbf n < 0@f$). As is
1038  * common in the finite element method, we evaluate fields (and their
1039  * derivatives) at specific evaluation points; typically, these are
1040  * "quadrature points", and so we create a "quadrature formula" that we will
1041  * use to designate the points at which we want to evaluate the solution.
1042  * Here, we will simply take QMidpoint implying that we will only check the
1043  * threshold condition at the midpoints of faces. We then use this to
1044  * initialize an object of type FEFaceValues to evaluate the solution at these
1045  * points.
1046  *
1047 
1048  *
1049  * All of this will then be used in a loop over all cells, their faces, and
1050  * specifically those faces that are at the boundary and, moreover, the
1051  * cathode part of the boundary.
1052  *
1053  * @code
1054  * template <int dim>
1055  * void CathodeRaySimulator<dim>::create_particles()
1056  * {
1057  * FEFaceValues<dim> fe_face_values(fe,
1058  * QMidpoint<dim - 1>(),
1059  * update_quadrature_points |
1060  * update_gradients |
1061  * update_normal_vectors);
1062  *
1063  * std::vector<Tensor<1, dim>> solution_gradients(
1064  * fe_face_values.n_quadrature_points);
1065  *
1066  * for (const auto &cell : dof_handler.active_cell_iterators())
1067  * for (const auto &face : cell->face_iterators())
1068  * if (face->at_boundary() &&
1069  * (face->boundary_id() == BoundaryIds::cathode))
1070  * {
1071  * fe_face_values.reinit(cell, face);
1072  *
1073  * @endcode
1074  *
1075  * So we have found a face on the cathode. Next, we let the
1076  * FEFaceValues object compute the gradient of the solution at each
1077  * "quadrature" point, and extract the electric field vector from
1078  * the gradient in the form of a Tensor variable through the methods
1079  * discussed in the
1080  * @ref vector_valued "vector-valued problems" documentation module.
1081  *
1082  * @code
1083  * const FEValuesExtractors::Scalar electric_potential(0);
1084  * fe_face_values[electric_potential].get_function_gradients(
1085  * solution, solution_gradients);
1086  * for (const unsigned int q_point :
1087  * fe_face_values.quadrature_point_indices())
1088  * {
1089  * const Tensor<1, dim> E = solution_gradients[q_point];
1090  *
1091  * @endcode
1092  *
1093  * Electrons can only escape the cathode if the electric field
1094  * strength exceeds a threshold and,
1095  * crucially, if the electric field points *into* the domain.
1096  * Once we have that checked, we create a new
1097  * Particles::Particle object at this location and insert it
1098  * into the Particles::ParticleHandler object with a unique ID.
1099  *
1100 
1101  *
1102  * The only thing that may be not obvious here is that we also
1103  * associate with this particle the location in the reference
1104  * coordinates of the cell we are currently on. This is done
1105  * because we will in downstream functions compute quantities
1106  * such as the electric field at the location of the particle
1107  * (e.g., to compute the forces that act on it when updating its
1108  * position in each time step). Evaluating a finite element
1109  * field at arbitrary coordinates is quite an expensive
1110  * operation because shape functions are really only defined on
1111  * the reference cell, and so when asking for the electric field
1112  * at an arbitrary point requires us first to determine what the
1113  * reference coordinates of that point are. To avoid having to
1114  * do this over and over, we determine these coordinates once
1115  * and for all and then store these reference coordinates
1116  * directly with the particle.
1117  *
1118  * @code
1119  * if ((E * fe_face_values.normal_vector(q_point) < 0) &&
1120  * (E.norm() > Constants::E_threshold))
1121  * {
1122  * const Point<dim> &location =
1123  * fe_face_values.quadrature_point(q_point);
1124  *
1125  * Particles::Particle<dim> new_particle;
1126  * new_particle.set_location(location);
1127  * new_particle.set_reference_location(
1128  * mapping.transform_real_to_unit_cell(cell, location));
1129  * new_particle.set_id(next_unused_particle_id);
1130  * particle_handler.insert_particle(new_particle, cell);
1131  *
1132  * ++next_unused_particle_id;
1133  * }
1134  * }
1135  * }
1136  *
1137  * @endcode
1138  *
1139  * At the end of all of these insertions, we let the `particle_handler`
1140  * update some internal statistics about the particles it stores.
1141  *
1142  * @code
1143  * particle_handler.update_cached_numbers();
1144  * }
1145  *
1146  *
1147  * @endcode
1148  *
1149  *
1150  * <a name="CathodeRaySimulatormove_particles"></a>
1151  * <h4>CathodeRaySimulator::move_particles</h4>
1152  *
1153 
1154  *
1155  * The second particle-related function is the one that moves the particles
1156  * in each time step. To do this, we have to loop over all cells, the
1157  * particles in each cell, and evaluate the electric field at each of the
1158  * particles' positions.
1159  *
1160 
1161  *
1162  * The approach used here is conceptually the same used in the
1163  * `assemble_system()` function: We loop over all cells, find the particles
1164  * located there (with the same caveat about the inefficiency of the algorithm
1165  * used here to find these particles), and use FEPointEvaluation object to
1166  * evaluate the gradient at these positions:
1167  *
1168  * @code
1169  * template <int dim>
1170  * void CathodeRaySimulator<dim>::move_particles()
1171  * {
1172  * const double dt = time.get_next_step_size();
1173  *
1174  * Vector<double> solution_values(fe.n_dofs_per_cell());
1175  * FEPointEvaluation<1, dim> evaluator(mapping, fe, update_gradients);
1176  *
1177  * for (const auto &cell : dof_handler.active_cell_iterators())
1178  * if (particle_handler.n_particles_in_cell(cell) > 0)
1179  * {
1180  * const typename Particles::ParticleHandler<
1181  * dim>::particle_iterator_range particles_in_cell =
1182  * particle_handler.particles_in_cell(cell);
1183  *
1184  * std::vector<Point<dim>> particle_positions;
1185  * for (const auto &particle : particles_in_cell)
1186  * particle_positions.push_back(particle.get_reference_location());
1187  *
1188  * cell->get_dof_values(solution, solution_values);
1189  *
1190  * @endcode
1191  *
1192  * Then we can ask the FEPointEvaluation object for the gradients of
1193  * the solution (i.e., the electric field @f$\mathbf E@f$) at these
1194  * locations and loop over the individual particles:
1195  *
1196  * @code
1197  * evaluator.reinit(cell, particle_positions);
1198  * evaluator.evaluate(make_array_view(solution_values),
1199  * EvaluationFlags::gradients);
1200  *
1201  * {
1202  * typename Particles::ParticleHandler<dim>::particle_iterator
1203  * particle = particles_in_cell.begin();
1204  * for (unsigned int particle_index = 0;

1273  *
1274 
1275  *
1276  * The final particle-related function is the one that is called whenever a
1277  * particle is lost from the simulation. This typically happens if it leaves
1278  * the domain. If that happens, this function is called both the cell (which
1279  * we can ask for its new location) and the cell it was previously on. The
1280  * function then keeps track of updating the number of particles lost in this
1281  * time step, the total number of lost particles, and then estimates whether
1282  * the particle left through the hole in the middle of the anode. We do so by
1283  * first checking whether the cell it was in last had an @f$x@f$ coordinate to the
1284  * left of the right boundary (located at @f$x=4@f$) and the particle now has a
1285  * position to the right of the right boundary. If that is so, we compute a
1286  * direction vector of its motion that is normalized so that the @f$x@f$ component
1287  * of the direction vector is equal to @f$1@f$. With this direction vector, we can
1288  * compute where it would have intersected the line @f$x=4@f$. If this intersect
1289  * is between @f$0.5@f$ and @f$1.5@f$, then we claim that the particle left through
1290  * the hole and increment a counter.

1307  * (current_location - approximate_previous_location) /
1308  * (current_location[0] - approximate_previous_location[0]);

1311  * approximate_previous_location[1] +
1312  * (4 - approximate_previous_location[0]) * direction[1];

1326  *
1327 
1328  *
1329  * As discussed at length in the introduction, we need to respect a time step
1330  * condition whereby particles can not move further than one cell in one time
1331  * step. To ensure that this is the case, we again first compute the maximal
1332  * speed of all particles on each cell, and divide the cell size by that
1333  * speed. We then compute the next time step size as the minimum of this
1334  * quantity over all cells, using the safety factor discussed in the
1335  * introduction, and set this as the desired time step size using the
1336  * DiscreteTime::set_desired_time_step_size() function.

1756 `assemble_system()`、`move_particles()`和`update_timestep_size()`函数都调用 Particles::ParticleHandler::particles_in_cell() 和 Particles::ParticleHandler::n_particles_in_cell() ，查询位于当前单元上的粒子信息。虽然这很方便，但也很低效。为了理解为什么会这样，我们需要知道粒子是如何存储在 Particles::ParticleHandler: 中的，即在一个数据结构中，粒子是以某种线性方式按它们所在的单元进行排序的。因此，为了找到与给定单元相关的粒子，这些函数需要搜索给定单元上的第一个（也可能是最后一个）粒子--这种努力需要花费 @f${\cal O}(\log N)@f$ 次操作，其中 @f$N@f$ 是粒子的数量。但这是在每个单元上重复的；假设对于大型计算来说，单元和粒子的数量大致成正比，那么这些函数调用的累积成本是 @f${\cal O}(N \log N)@f$ ，因此大于我们应该对程序的所有部分进行的 @f${\cal O}(N)@f$ 成本。
