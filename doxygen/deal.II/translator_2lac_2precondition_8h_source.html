1949  void
1950  apply_to_subrange(const std::size_t begin, const std::size_t end) const
1951  {
1952  // To circumvent a bug in gcc
1953  // (https://gcc.gnu.org/bugzilla/show_bug.cgi?id=63945), we create
1954  // copies of the variables factor1 and factor2 and do not check based on
1955  // factor1.
1956  const Number factor1 = this->factor1;
1957  const Number factor1_plus_1 = 1. + this->factor1;
1958  const Number factor2 = this->factor2;
1959  if (iteration_index == 0)
1960  {
1961  DEAL_II_OPENMP_SIMD_PRAGMA
1962  for (std::size_t i = begin; i < end; ++i)
1963  solution[i] = factor2 * matrix_diagonal_inverse[i] * rhs[i];
1964  }
1965  else if (iteration_index == 1)
1966  {
1967  // x^{n+1} = x^{n} + f_1 * x^{n} + f_2 * P^{-1} * (b-A*x^{n})
1968  DEAL_II_OPENMP_SIMD_PRAGMA
1969  for (std::size_t i = begin; i < end; ++i)
1970  // for efficiency reason, write back to temp_vector that is
1971  // already read (avoid read-for-ownership)
1972  tmp_vector[i] =
1973  factor1_plus_1 * solution[i] +
1974  factor2 * matrix_diagonal_inverse[i] * (rhs[i] - tmp_vector[i]);
1975  }
1976  else
1977  {
1978  // x^{n+1} = x^{n} + f_1 * (x^{n}-x^{n-1})
1979  // + f_2 * P^{-1} * (b-A*x^{n})
1980  DEAL_II_OPENMP_SIMD_PRAGMA
1981  for (std::size_t i = begin; i < end; ++i)
1982  solution_old[i] =
1983  factor1_plus_1 * solution[i] - factor1 * solution_old[i] +
1984  factor2 * matrix_diagonal_inverse[i] * (rhs[i] - tmp_vector[i]);
1985  }
1986  }

1989  const Number * matrix_diagonal_inverse;
1990  const unsigned int iteration_index;
1991  const Number factor1;
1992  const Number factor2;
1993  mutable Number * solution_old;
1994  mutable Number * tmp_vector;
1995  mutable Number * solution;
1996  };
