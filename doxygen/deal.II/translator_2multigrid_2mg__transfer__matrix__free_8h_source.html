525 template <int dim, typename Number>
526 template <typename Number2, int spacedim>
527 void
528 MGTransferMatrixFree<dim, Number>::interpolate_to_mg(
529  const DoFHandler<dim, spacedim> & dof_handler,
530  MGLevelObject<LinearAlgebra::distributed::Vector<Number>> &dst,
531  const LinearAlgebra::distributed::Vector<Number2> & src) const
532 {
533  const unsigned int min_level = dst.min_level();
534  const unsigned int max_level = dst.max_level();
535 
536  Assert(max_level == dof_handler.get_triangulation().n_global_levels() - 1,
537  ExcDimensionMismatch(
538  max_level, dof_handler.get_triangulation().n_global_levels() - 1));
539 
540  const FiniteElement<dim, spacedim> &fe = dof_handler.get_fe();
541 
542  for (unsigned int level = min_level; level <= max_level; ++level)
543  if (dst[level].size() != dof_handler.n_dofs(level) ||
544  dst[level].locally_owned_size() !=
545  dof_handler.locally_owned_mg_dofs(level).n_elements())
546  dst[level].reinit(this->vector_partitioners[level]);
547 
548  // copy fine level vector to active cells in MG hierarchy
549  this->copy_to_mg(dof_handler, dst, src, true);
550 
551  // FIXME: maybe need to store hanging nodes constraints per level?
552  // MGConstrainedDoFs does NOT keep this info right now, only periodicity
553  // constraints...
554 
555  // do the transfer from level to level-1:
556  dst[max_level].update_ghost_values();
557  for (unsigned int level = max_level; level > min_level; --level)
558  {
559  // auxiliary vector which always has ghost elements
560  const LinearAlgebra::distributed::Vector<Number> *input = nullptr;
561  LinearAlgebra::distributed::Vector<Number> ghosted_fine;
562  if (dst[level].get_partitioner().get() ==
563  this->vector_partitioners[level].get())
564  input = &dst[level];
565  else
566  {
567  ghosted_fine.reinit(this->vector_partitioners[level]);
568  ghosted_fine.copy_locally_owned_data_from(dst[level]);
569  ghosted_fine.update_ghost_values();
570  input = &ghosted_fine;
571  }
572 
573  std::vector<Number> dof_values_coarse(fe.n_dofs_per_cell());
574  Vector<Number> dof_values_fine(fe.n_dofs_per_cell());
575  Vector<Number> tmp(fe.n_dofs_per_cell());
576  std::vector<types::global_dof_index> dof_indices(fe.n_dofs_per_cell());
577  for (const auto &cell : dof_handler.cell_iterators_on_level(level - 1))
578  if (cell->is_locally_owned_on_level())
579  {
580  // if we get to a cell without children (== active), we can
581  // skip it as there values should be already set by the
582  // equivalent of copy_to_mg()
583  if (cell->is_active())
584  continue;
585 
586  std::fill(dof_values_coarse.begin(), dof_values_coarse.end(), 0.);
587  for (unsigned int child = 0; child < cell->n_children(); ++child)
588  {
589  cell->child(child)->get_mg_dof_indices(dof_indices);
590  for (unsigned int i = 0; i < fe.n_dofs_per_cell(); ++i)
591  dof_values_fine(i) = (*input)(dof_indices[i]);
592  fe.get_restriction_matrix(child, cell->refinement_case())
593  .vmult(tmp, dof_values_fine);
594  for (unsigned int i = 0; i < fe.n_dofs_per_cell(); ++i)
595  if (fe.restriction_is_additive(i))
596  dof_values_coarse[i] += tmp[i];
597  else if (tmp(i) != 0.)
598  dof_values_coarse[i] = tmp[i];
599  }
600  cell->get_mg_dof_indices(dof_indices);
601  for (unsigned int i = 0; i < fe.n_dofs_per_cell(); ++i)
602  if (dof_handler.locally_owned_mg_dofs(level - 1).is_element(
603  dof_indices[i]))
604  dst[level - 1](dof_indices[i]) = dof_values_coarse[i];
605  }
606 
607  dst[level - 1].update_ghost_values();
608  }
609 }

635 template <int dim, typename Number>
636 template <typename Number2, int spacedim>
637 void
638 MGTransferBlockMatrixFree<dim, Number>::copy_to_mg(
639  const std::vector<const DoFHandler<dim, spacedim> *> & dof_handler,
640  MGLevelObject<LinearAlgebra::distributed::BlockVector<Number>> &dst,
641  const LinearAlgebra::distributed::BlockVector<Number2> & src) const
642 {
643  const unsigned int n_blocks = src.n_blocks();
644  AssertDimension(dof_handler.size(), n_blocks);
645 
646  if (n_blocks == 0)
647  return;
648 
649  const unsigned int min_level = dst.min_level();
650  const unsigned int max_level = dst.max_level();
651 
652  // this function is normally called within the Multigrid class with
653  // dst == defect level block vector. At first run this vector is not
654  // initialized. Do this below:
655  {
656  const parallel::TriangulationBase<dim, spacedim> *tria =
657  (dynamic_cast<const parallel::TriangulationBase<dim, spacedim> *>(
658  &(dof_handler[0]->get_triangulation())));
659  for (unsigned int i = 1; i < n_blocks; ++i)
660  AssertThrow(
661  (dynamic_cast<const parallel::TriangulationBase<dim, spacedim> *>(
662  &(dof_handler[0]->get_triangulation())) == tria),
663  ExcMessage("The DoFHandler use different Triangulations!"));
664 
665  MGLevelObject<bool> do_reinit;
666  do_reinit.resize(min_level, max_level);
667  for (unsigned int level = min_level; level <= max_level; ++level)
668  {
669  do_reinit[level] = false;
670  if (dst[level].n_blocks() != n_blocks)
671  {
672  do_reinit[level] = true;
673  continue; // level
674  }
675  for (unsigned int b = 0; b < n_blocks; ++b)
676  {
677  LinearAlgebra::distributed::Vector<Number> &v = dst[level].block(b);
678  if (v.size() !=
679  dof_handler[b]->locally_owned_mg_dofs(level).size() ||
680  v.locally_owned_size() !=
681  dof_handler[b]->locally_owned_mg_dofs(level).n_elements())
682  {
683  do_reinit[level] = true;
684  break; // b
685  }
686  }
687  }
688 
689  for (unsigned int level = min_level; level <= max_level; ++level)
690  {
691  if (do_reinit[level])
692  {
693  dst[level].reinit(n_blocks);
694  for (unsigned int b = 0; b < n_blocks; ++b)
695  {
696  LinearAlgebra::distributed::Vector<Number> &v =
697  dst[level].block(b);
698  v.reinit(dof_handler[b]->locally_owned_mg_dofs(level),
699  dof_handler[b]->get_communicator());
700  }
701  dst[level].collect_sizes();
702  }
703  else
704  dst[level] = 0;
705  }
706  }
707 
708  // FIXME: this a quite ugly as we need a temporary object:
709  MGLevelObject<LinearAlgebra::distributed::Vector<Number>> dst_non_block(
710  min_level, max_level);
711 
712  for (unsigned int b = 0; b < n_blocks; ++b)
713  {
714  for (unsigned int l = min_level; l <= max_level; ++l)
715  dst_non_block[l].reinit(dst[l].block(b));
716  const unsigned int data_block = same_for_all ? 0 : b;
717  matrix_free_transfer_vector[data_block].copy_to_mg(*dof_handler[b],
718  dst_non_block,
719  src.block(b));
720 
721  for (unsigned int l = min_level; l <= max_level; ++l)
722  dst[l].block(b) = dst_non_block[l];
723  }
724 }
