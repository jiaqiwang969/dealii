406  for (const auto &cell : triangulation.active_cell_iterators())

465  template <int dim>
466  void TopLevel<dim>::assemble_system()
467  {
468  system_rhs = 0;
469  system_matrix = 0;
470 
471  FEValues<dim> fe_values(fe,
472  quadrature_formula,
473  update_values | update_gradients |
474  update_quadrature_points | update_JxW_values);
475 
476  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
477  const unsigned int n_q_points = quadrature_formula.size();
478 
479  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
480  Vector<double> cell_rhs(dofs_per_cell);
481 
482  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
483 
484  BodyForce<dim> body_force;
485  std::vector<Vector<double>> body_force_values(n_q_points,
486  Vector<double>(dim));
487 
488  for (const auto &cell : dof_handler.active_cell_iterators())
489  if (cell->is_locally_owned())
490  {
491  cell_matrix = 0;
492  cell_rhs = 0;
493 
494  fe_values.reinit(cell);
495 
496  for (unsigned int i = 0; i < dofs_per_cell; ++i)
497  for (unsigned int j = 0; j < dofs_per_cell; ++j)
498  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
499  {
500  const SymmetricTensor<2, dim>
501  eps_phi_i = get_strain(fe_values, i, q_point),
502  eps_phi_j = get_strain(fe_values, j, q_point);
503 
504  cell_matrix(i, j) += (eps_phi_i *
505  stress_strain_tensor *
506  eps_phi_j
507  ) *
508  fe_values.JxW(q_point);
509  }
510 
511 
512  const PointHistory<dim> *local_quadrature_points_data =
513  reinterpret_cast<PointHistory<dim> *>(cell->user_pointer());
514  body_force.vector_value_list(fe_values.get_quadrature_points(),
515  body_force_values);
516  for (unsigned int i = 0; i < dofs_per_cell; ++i)
517  {
518  const unsigned int component_i =
519  fe.system_to_component_index(i).first;
520 
521  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
522  {
523  const SymmetricTensor<2, dim> &old_stress =
524  local_quadrature_points_data[q_point].old_stress;
525 
526  cell_rhs(i) +=
527  (body_force_values[q_point](component_i) *
528  fe_values.shape_value(i, q_point) -
529  old_stress * get_strain(fe_values, i, q_point)) *
530  fe_values.JxW(q_point);
531  }
532  }
533 
534  cell->get_dof_indices(local_dof_indices);
535 
536  hanging_node_constraints.distribute_local_to_global(cell_matrix,
537  cell_rhs,
538  local_dof_indices,
539  system_matrix,
540  system_rhs);
541  }
542 
543  system_matrix.compress(VectorOperation::add);
544  system_rhs.compress(VectorOperation::add);
545 
546 
547 
548 
549  FEValuesExtractors::Scalar z_component(dim - 1);
550  std::map<types::global_dof_index, double> boundary_values;
551  VectorTools::interpolate_boundary_values(dof_handler,
552  0,
553  Functions::ZeroFunction<dim>(dim),
554  boundary_values);
555  VectorTools::interpolate_boundary_values(
556  dof_handler,
557  1,
558  IncrementalBoundaryValues<dim>(present_time, present_timestep),
559  boundary_values,
560  fe.component_mask(z_component));
561 
562  PETScWrappers::MPI::Vector tmp(locally_owned_dofs, mpi_communicator);
563  MatrixTools::apply_boundary_values(
564  boundary_values, system_matrix, tmp, system_rhs, false);
565  incremental_displacement = tmp;
566  }
