282  template <int dim>
283  void BiharmonicProblem<dim>::assemble_system()
284  {
285  using Iterator = typename DoFHandler<dim>::active_cell_iterator;
286 
287  auto cell_worker = [&](const Iterator & cell,
288  ScratchData<dim> &scratch_data,
289  CopyData & copy_data) {
290  copy_data.cell_matrix = 0;
291  copy_data.cell_rhs = 0;
292 
293  FEValues<dim> &fe_values = scratch_data.fe_values;
294  fe_values.reinit(cell);
295 
296  cell->get_dof_indices(copy_data.local_dof_indices);
297 
298  const ExactSolution::RightHandSide<dim> right_hand_side;
299 
300  const unsigned int dofs_per_cell =
301  scratch_data.fe_values.get_fe().n_dofs_per_cell();
302 
303  for (unsigned int qpoint = 0; qpoint < fe_values.n_quadrature_points;
304  ++qpoint)
305  {
306  for (unsigned int i = 0; i < dofs_per_cell; ++i)
307  {
308  const Tensor<2, dim> &hessian_i =
309  fe_values.shape_hessian(i, qpoint);
310 
311  for (unsigned int j = 0; j < dofs_per_cell; ++j)
312  {
313  const Tensor<2, dim> &hessian_j =
314  fe_values.shape_hessian(j, qpoint);
315 
316  copy_data.cell_matrix(i, j) +=
317  scalar_product(hessian_i, // nabla^2 phi_i(x)
318  hessian_j) * // nabla^2 phi_j(x)
319  fe_values.JxW(qpoint); // dx
320  }
321 
322  copy_data.cell_rhs(i) +=
323  fe_values.shape_value(i, qpoint) * // phi_i(x)
324  right_hand_side.value(
325  fe_values.quadrature_point(qpoint)) * // f(x)
326  fe_values.JxW(qpoint); // dx
327  }
328  }
329  };

332  auto face_worker = [&](const Iterator & cell,
333  const unsigned int &f,
334  const unsigned int &sf,
335  const Iterator & ncell,
336  const unsigned int &nf,
337  const unsigned int &nsf,
338  ScratchData<dim> & scratch_data,
339  CopyData & copy_data) {
340  FEInterfaceValues<dim> &fe_interface_values =
341  scratch_data.fe_interface_values;
342  fe_interface_values.reinit(cell, f, sf, ncell, nf, nsf);
343 
344  copy_data.face_data.emplace_back();
345  CopyData::FaceData &copy_data_face = copy_data.face_data.back();
346 
347  copy_data_face.joint_dof_indices =
348  fe_interface_values.get_interface_dof_indices();
349 
350  const unsigned int n_interface_dofs =
351  fe_interface_values.n_current_interface_dofs();
352  copy_data_face.cell_matrix.reinit(n_interface_dofs, n_interface_dofs);
353 
354  const unsigned int p = fe.degree;
355  const double gamma_over_h =
356  std::max((1.0 * p * (p + 1) /
357  cell->extent_in_direction(
358  GeometryInfo<dim>::unit_normal_direction[f])),
359  (1.0 * p * (p + 1) /
360  ncell->extent_in_direction(
361  GeometryInfo<dim>::unit_normal_direction[nf])));
362 
363  for (unsigned int qpoint = 0;
364  qpoint < fe_interface_values.n_quadrature_points;
365  ++qpoint)
366  {
367  const auto &n = fe_interface_values.normal(qpoint);
368 
369  for (unsigned int i = 0; i < n_interface_dofs; ++i)
370  {
371  const double av_hessian_i_dot_n_dot_n =
372  (fe_interface_values.average_hessian(i, qpoint) * n * n);
373  const double jump_grad_i_dot_n =
374  (fe_interface_values.jump_gradient(i, qpoint) * n);
375 
376  for (unsigned int j = 0; j < n_interface_dofs; ++j)
377  {
378  const double av_hessian_j_dot_n_dot_n =
379  (fe_interface_values.average_hessian(j, qpoint) * n * n);
380  const double jump_grad_j_dot_n =
381  (fe_interface_values.jump_gradient(j, qpoint) * n);
382 
383  copy_data_face.cell_matrix(i, j) +=
384  (-av_hessian_i_dot_n_dot_n // - {grad^2 v n n }
385  * jump_grad_j_dot_n // [grad u n]
386  - av_hessian_j_dot_n_dot_n // - {grad^2 u n n }
387  * jump_grad_i_dot_n // [grad v n]
388  + // +
389  gamma_over_h * // gamma/h
390  jump_grad_i_dot_n * // [grad v n]
391  jump_grad_j_dot_n) * // [grad u n]
392  fe_interface_values.JxW(qpoint); // dx
393  }
394  }
395  }
396  };

400  const unsigned int &face_no,

410 
411  const unsigned int n_dofs =
412  fe_interface_values.n_current_interface_dofs();
413  copy_data_face.joint_dof_indices =
414  fe_interface_values.get_interface_dof_indices();
415 
416  copy_data_face.cell_matrix.reinit(n_dofs, n_dofs);
417 
418  const std::vector<double> &JxW = fe_interface_values.get_JxW_values();
419  const std::vector<Tensor<1, dim>> &normals =
420  fe_interface_values.get_normal_vectors();
421 
422 
423  const ExactSolution::Solution<dim> exact_solution;
424  std::vector<Tensor<1, dim>> exact_gradients(q_points.size());
425  exact_solution.gradient_list(q_points, exact_gradients);
426 
427 
428  const unsigned int p = fe.degree;
429  const double gamma_over_h =
430  (1.0 * p * (p + 1) /
431  cell->extent_in_direction(
432  GeometryInfo<dim>::unit_normal_direction[face_no]));
433 
434  for (unsigned int qpoint = 0; qpoint < q_points.size(); ++qpoint)
435  {
436  const auto &n = normals[qpoint];
437 
438  for (unsigned int i = 0; i < n_dofs; ++i)
439  {
440  const double av_hessian_i_dot_n_dot_n =
441  (fe_interface_values.average_hessian(i, qpoint) * n * n);
442  const double jump_grad_i_dot_n =
443  (fe_interface_values.jump_gradient(i, qpoint) * n);
444 
445  for (unsigned int j = 0; j < n_dofs; ++j)
446  {
447  const double av_hessian_j_dot_n_dot_n =
448  (fe_interface_values.average_hessian(j, qpoint) * n * n);
449  const double jump_grad_j_dot_n =
450  (fe_interface_values.jump_gradient(j, qpoint) * n);
451 
452  copy_data_face.cell_matrix(i, j) +=
453  (-av_hessian_i_dot_n_dot_n // - {grad^2 v n n}
454  * jump_grad_j_dot_n // [grad u n]
455  - av_hessian_j_dot_n_dot_n // - {grad^2 u n n}
456  * jump_grad_i_dot_n // [grad v n]
457  + gamma_over_h // gamma/h
458  * jump_grad_i_dot_n // [grad v n]
459  * jump_grad_j_dot_n // [grad u n]
460  ) *
461  JxW[qpoint]; // dx
462  }
463 
464  copy_data.cell_rhs(i) +=
465  (-av_hessian_i_dot_n_dot_n * // - {grad^2 v n n }
466  (exact_gradients[qpoint] * n) // (grad u_exact . n)
467  + // +
468  gamma_over_h // gamma/h
469  * jump_grad_i_dot_n // [grad v n]
470  * (exact_gradients[qpoint] * n) // (grad u_exact . n)
471  ) *
472  JxW[qpoint]; // dx
473  }
474  }
475  };
