191  template <int dim>
192  SymmetricTensor<4, dim> get_stiffness_tensor(const double lambda,
193  const double mu)
194  {
195  SymmetricTensor<4, dim> stiffness_tensor;
196  for (unsigned int i = 0; i < dim; ++i)
197  for (unsigned int j = 0; j < dim; ++j)
198  for (unsigned int k = 0; k < dim; ++k)
199  for (unsigned int l = 0; l < dim; ++l)
200  stiffness_tensor[i][j][k][l] =
201  (((i == k) && (j == l) ? mu : 0.0) +
202  ((i == l) && (j == k) ? mu : 0.0) +
203  ((i == j) && (k == l) ? lambda : 0.0));
204  return stiffness_tensor;
205  }

332  template <int dim>
333  std::complex<double> PML<dim>::value(const Point<dim> & p,
334  const unsigned int component) const
335  {
336  double calculated_pml_x_coeff = 0;
337  double calculated_pml_y_coeff = 0;
338 
339  if ((component == 0) && pml_x)
340  {
341  const double pml_x_start_position = dimension_x / 2 - pml_width_x;
342  if (std::abs(p[0]) > pml_x_start_position)
343  {
344  const double x_prime = std::abs(p[0]) - pml_x_start_position;
345  calculated_pml_x_coeff =
346  a_coeff_x * std::pow(x_prime, pml_coeff_degree);
347  }
348  }
349 
350  if ((component == 1) && pml_y)
351  {
352  const double pml_y_start_position = dimension_y / 2 - pml_width_y;
353  if (std::abs(p[1]) > pml_y_start_position)
354  {
355  const double y_prime = std::abs(p[1]) - pml_y_start_position;
356  calculated_pml_y_coeff =
357  a_coeff_y * std::pow(y_prime, pml_coeff_degree);
358  }
359  }
360 
361  return 1. + std::max(calculated_pml_x_coeff, calculated_pml_y_coeff) *
362  std::complex<double>(0., 1.);
363  }

388  template <int dim>
389  double Rho<dim>::value(const Point<dim> &p,
390  const unsigned int /*component*/) const
391  {
392  double elastic_constant;
393  if (dim == 2)
394  {
395  elastic_constant = 4 * mu * (lambda + mu) / (lambda + 2 * mu);
396  }
397  else if (dim == 3)
398  {
399  elastic_constant = mu * (3 * lambda + 2 * mu) / (lambda + mu);
400  }
401  else
402  {
403  Assert(false, ExcInternalError());
404  }
405  const double material_a_speed_of_sound =
406  std::sqrt(elastic_constant / material_a_rho);
407  const double material_a_wavelength =
408  material_a_speed_of_sound / cavity_resonance_frequency;
409  const double material_b_speed_of_sound =
410  std::sqrt(elastic_constant / material_b_rho);
411  const double material_b_wavelength =
412  material_b_speed_of_sound / cavity_resonance_frequency;
413 
414  for (unsigned int idx = 0; idx < nb_mirror_pairs; idx++)
415  {
416  const double layer_transition_center =
417  material_a_wavelength / 2 +
418  idx * (material_b_wavelength / 4 + material_a_wavelength / 4);
419  if (std::abs(p[0]) >=
420  (layer_transition_center - average_rho_width / 2) &&
421  std::abs(p[0]) <= (layer_transition_center + average_rho_width / 2))
422  {
423  const double coefficient =
424  (std::abs(p[0]) -
425  (layer_transition_center - average_rho_width / 2)) /
426  average_rho_width;
427  return (1 - coefficient) * material_a_rho +
428  coefficient * material_b_rho;
429  }
430  }
431 
432  for (unsigned int idx = 0; idx < nb_mirror_pairs; idx++)
433  {
434  const double layer_transition_center =
435  material_a_wavelength / 2 +
436  idx * (material_b_wavelength / 4 + material_a_wavelength / 4) +
437  material_b_wavelength / 4;
438  if (std::abs(p[0]) >=
439  (layer_transition_center - average_rho_width / 2) &&
440  std::abs(p[0]) <= (layer_transition_center + average_rho_width / 2))
441  {
442  const double coefficient =
443  (std::abs(p[0]) -
444  (layer_transition_center - average_rho_width / 2)) /
445  average_rho_width;
446  return (1 - coefficient) * material_b_rho +
447  coefficient * material_a_rho;
448  }
449  }
450 
451  if (std::abs(p[0]) <= material_a_wavelength / 2)
452  {
453  return material_a_rho;
454  }
455 
456  for (unsigned int idx = 0; idx < nb_mirror_pairs; idx++)
457  {
458  const double layer_center =
459  material_a_wavelength / 2 +
460  idx * (material_b_wavelength / 4 + material_a_wavelength / 4) +
461  material_b_wavelength / 4 + material_a_wavelength / 8;
462  const double layer_width = material_a_wavelength / 4;
463  if (std::abs(p[0]) >= (layer_center - layer_width / 2) &&
464  std::abs(p[0]) <= (layer_center + layer_width / 2))
465  {
466  return material_a_rho;
467  }
468  }
469 
470  for (unsigned int idx = 0; idx < nb_mirror_pairs; idx++)
471  {
472  const double layer_center =
473  material_a_wavelength / 2 +
474  idx * (material_b_wavelength / 4 + material_a_wavelength / 4) +
475  material_b_wavelength / 8;
476  const double layer_width = material_b_wavelength / 4;
477  if (std::abs(p[0]) >= (layer_center - layer_width / 2) &&
478  std::abs(p[0]) <= (layer_center + layer_width / 2))
479  {
480  return material_b_rho;
481  }
482  }
483 
484  return material_a_rho;
485  }

604  template <int dim>
605  void ElasticWave<dim>::assemble_system(const double omega,
606  const bool calculate_quadrature_data)
607  {
608  TimerOutput::Scope t(computing_timer, "assembly");
609 
610  FEValues<dim> fe_values(fe,
611  quadrature_formula,
612  update_values | update_gradients |
613  update_quadrature_points | update_JxW_values);
614  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
615  const unsigned int n_q_points = quadrature_formula.size();
616 
617  FullMatrix<std::complex<double>> cell_matrix(dofs_per_cell, dofs_per_cell);
618  Vector<std::complex<double>> cell_rhs(dofs_per_cell);
619 
620  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
621 
622  std::vector<Vector<double>> rhs_values(n_q_points, Vector<double>(dim));
623  std::vector<double> rho_values(n_q_points);
624  std::vector<Vector<std::complex<double>>> pml_values(
625  n_q_points, Vector<std::complex<double>>(dim));
626 
627  const SymmetricTensor<4, dim> stiffness_tensor =
628  get_stiffness_tensor<dim>(parameters.lambda, parameters.mu);
629 
630  const FEValuesExtractors::Vector displacement(0);
631 
632  for (const auto &cell : dof_handler.active_cell_iterators())
633  if (cell->is_locally_owned())
634  {
635  cell_matrix = 0;
636  cell_rhs = 0;
637 
638  if (calculate_quadrature_data)
639  {
640  fe_values.reinit(cell);
641 
642  parameters.right_hand_side.vector_value_list(
643  fe_values.get_quadrature_points(), rhs_values);
644  parameters.rho.value_list(fe_values.get_quadrature_points(),
645  rho_values);
646  parameters.pml.vector_value_list(
647  fe_values.get_quadrature_points(), pml_values);
648  }
649 
650  QuadratureCache<dim> *local_quadrature_points_data =
651  reinterpret_cast<QuadratureCache<dim> *>(cell->user_pointer());
652  Assert(local_quadrature_points_data >= &quadrature_cache.front(),
653  ExcInternalError());
654  Assert(local_quadrature_points_data <= &quadrature_cache.back(),
655  ExcInternalError());
656  for (unsigned int q = 0; q < n_q_points; ++q)
657  {
658  QuadratureCache<dim> &quadrature_data =
659  local_quadrature_points_data[q];
