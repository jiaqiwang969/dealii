96  void TriangulationOnCAD::read_domain()
97  {
98  TopoDS_Shape bow_surface = OpenCASCADE::read_IGES(cad_file_name, 1e-3);
99 
100 
101  const double tolerance = OpenCASCADE::get_shape_tolerance(bow_surface) * 5;
102 
103  std::vector<TopoDS_Compound> compounds;
104  std::vector<TopoDS_CompSolid> compsolids;
105  std::vector<TopoDS_Solid> solids;
106  std::vector<TopoDS_Shell> shells;
107  std::vector<TopoDS_Wire> wires;
108 
109  OpenCASCADE::extract_compound_shapes(
110  bow_surface, compounds, compsolids, solids, shells, wires);
111 
112  std::ifstream in;
113 
114  in.open(initial_mesh_filename);
115 
116  GridIn<2, 3> gi;
117  gi.attach_triangulation(tria);
118  gi.read_vtk(in);
119 
120  output_results(0);
121 
122  Triangulation<2, 3>::active_cell_iterator cell = tria.begin_active();
123  cell->set_manifold_id(1);
124 
125  for (const auto &face : cell->face_iterators())
126  face->set_manifold_id(2);
127 
128 
129  Assert(
130  wires.size() > 0,
131  ExcMessage(
132  "I could not find any wire in the CAD file you gave me. Bailing out."));
133 
134  OpenCASCADE::ArclengthProjectionLineManifold<2, 3> line_projector(
135  wires[0], tolerance);
136 
137  tria.set_manifold(2, line_projector);
138 
139  switch (surface_projection_kind)
140  {
141  case NormalProjection:
142  {
143  OpenCASCADE::NormalProjectionManifold<2, 3> normal_projector(
144  bow_surface, tolerance);
145  tria.set_manifold(1, normal_projector);
146 
147  break;
148  }
149 
150  case DirectionalProjection:
151  {
152  OpenCASCADE::DirectionalProjectionManifold<2, 3>
153  directional_projector(bow_surface,
154  Point<3>(0.0, 1.0, 0.0),
155  tolerance);
156  tria.set_manifold(1, directional_projector);
157 
158  break;
159  }
160 
161  case NormalToMeshProjection:
162  {
163  OpenCASCADE::NormalToMeshProjectionManifold<2, 3>
164  normal_to_mesh_projector(bow_surface, tolerance);
165  tria.set_manifold(1, normal_to_mesh_projector);
166 
167  break;
168  }
169 
170  default:
171  AssertThrow(false, ExcInternalError());
172  }
173  }
