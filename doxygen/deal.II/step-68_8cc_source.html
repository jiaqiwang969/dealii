142  template <int dim>
143  void Vortex<dim>::vector_value(const Point<dim> &point,
144  Vector<double> & values) const
145  {
146  const double T = 4;
147  const double t = this->get_time();
148 
149  const double px = numbers::PI * point(0);
150  const double py = numbers::PI * point(1);
151  const double pt = numbers::PI / T * t;
152 
153  values[0] = -2 * cos(pt) * pow(sin(px), 2) * sin(py) * cos(py);
154  values[1] = 2 * cos(pt) * pow(sin(py), 2) * sin(px) * cos(px);
155  if (dim == 3)
156  {
157  values[2] = 0;
158  }
159  }

231  template <int dim>
232  unsigned int ParticleTracking<dim>::cell_weight(
233  const typename parallel::distributed::Triangulation<dim>::cell_iterator
234  & cell,
235  const typename parallel::distributed::Triangulation<dim>::CellStatus status)
236  const
237  {
238  if (!cell->is_locally_owned())
239  return 0;
240 
241  const unsigned int particle_weight = 10000;
242 
243  if (status == parallel::distributed::Triangulation<dim>::CELL_PERSIST ||
244  status == parallel::distributed::Triangulation<dim>::CELL_REFINE)
245  {
246  const unsigned int n_particles_in_cell =
247  particle_handler.n_particles_in_cell(cell);
248  return n_particles_in_cell * particle_weight;
249  }
250  else if (status == parallel::distributed::Triangulation<dim>::CELL_COARSEN)
251  {
252  unsigned int n_particles_in_cell = 0;
253 
254  for (unsigned int child_index = 0; child_index < cell->n_children();
255  ++child_index)
256  n_particles_in_cell +=
257  particle_handler.n_particles_in_cell(cell->child(child_index));
258 
259  return n_particles_in_cell * particle_weight;
260  }
261 
262  Assert(false, ExcInternalError());
263  return 0;
264  }

356  void ParticleTracking<dim>::euler_step_analytical(const double dt)
357  {
358  const unsigned int this_mpi_rank =
359  Utilities::MPI::this_mpi_process(mpi_communicator);
360  Vector<double> particle_velocity(dim);
361 
362  for (auto &particle : particle_handler)
363  {
364  Point<dim> particle_location = particle.get_location();
365  velocity.vector_value(particle_location, particle_velocity);
366 
367  for (int d = 0; d < dim; ++d)
368  particle_location[d] += particle_velocity[d] * dt;
369 
370  particle.set_location(particle_location);
371 
372  ArrayView<double> properties = particle.get_properties();
373  for (int d = 0; d < dim; ++d)
374  properties[d] = particle_velocity[d];

381  template <int dim>
382  void ParticleTracking<dim>::euler_step_interpolated(const double dt)
383  {
384  Vector<double> local_dof_values(fluid_fe.dofs_per_cell);
385 
386  auto particle = particle_handler.begin();
387  while (particle != particle_handler.end())
388  {
389  const auto cell =
390  particle->get_surrounding_cell(background_triangulation);
391  const auto dh_cell =
392  typename DoFHandler<dim>::cell_iterator(*cell, &fluid_dh);
393 
394  dh_cell->get_dof_values(velocity_field, local_dof_values);
395 
396  const auto pic = particle_handler.particles_in_cell(cell);
397  Assert(pic.begin() == particle, ExcInternalError());
398  for (auto &p : pic)
399  {
400  const Point<dim> reference_location = p.get_reference_location();
401  Tensor<1, dim> particle_velocity;
402  for (unsigned int j = 0; j < fluid_fe.dofs_per_cell; ++j)
403  {
404  const auto comp_j = fluid_fe.system_to_component_index(j);
405 
406  particle_velocity[comp_j.first] +=
407  fluid_fe.shape_value(j, reference_location) *
408  local_dof_values[j];
409  }
410 
411  Point<dim> particle_location = particle->get_location();
412  for (int d = 0; d < dim; ++d)
413  particle_location[d] += particle_velocity[d] * dt;
414  p.set_location(particle_location);
415 
416  ArrayView<double> properties = p.get_properties();
417  for (int d = 0; d < dim; ++d)
418  properties[d] = particle_velocity[d];
419 
420  properties[dim] =
421  Utilities::MPI::this_mpi_process(mpi_communicator);
422 
423  ++particle;
424  }
425  }

497  template <int dim>
498  void ParticleTracking<dim>::run()
499  {
500  DiscreteTime discrete_time(0, par.final_time, par.time_step);
501 
502  generate_particles();
503 
504  pcout << "Repartitioning triangulation after particle generation"
505  << std::endl;
506  background_triangulation.repartition();
507 
508  if (interpolated_velocity)
509  {
510  setup_background_dofs();
511  interpolate_function_to_field();
512  euler_step_interpolated(0.);
513  }
514  else
515  euler_step_analytical(0.);
516 
517  output_particles(discrete_time.get_step_number());
518  if (interpolated_velocity)
519  output_background(discrete_time.get_step_number());
520 
521  while (!discrete_time.is_at_end())
522  {
523  discrete_time.advance_time();
524  velocity.set_time(discrete_time.get_previous_time());
525 
526  if ((discrete_time.get_step_number() % par.repartition_frequency) == 0)
527  {
528  background_triangulation.repartition();
529  if (interpolated_velocity)
530  setup_background_dofs();
531  }
532 
533  if (interpolated_velocity)
534  {
535  interpolate_function_to_field();
536  euler_step_interpolated(discrete_time.get_previous_step_size());
537  }
538  else
539  euler_step_analytical(discrete_time.get_previous_step_size());
540 
541  particle_handler.sort_particles_into_subdomains_and_cells();
542 
543  if ((discrete_time.get_step_number() % par.output_frequency) == 0)
544  {
545  output_particles(discrete_time.get_step_number());
546  if (interpolated_velocity)
547  output_background(discrete_time.get_step_number());
548  }
549  }
550  }
