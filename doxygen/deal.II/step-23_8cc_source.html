289  template <int dim>
290  void WaveEquation<dim>::run()
291  {
292  setup_system();
293 
294  VectorTools::project(dof_handler,
295  constraints,
296  QGauss<dim>(fe.degree + 1),
297  InitialValuesU<dim>(),
298  old_solution_u);
299  VectorTools::project(dof_handler,
300  constraints,
301  QGauss<dim>(fe.degree + 1),
302  InitialValuesV<dim>(),
303  old_solution_v);
304 
305  Vector<double> tmp(solution_u.size());
306  Vector<double> forcing_terms(solution_u.size());
307 
308  for (; time <= 5; time += time_step, ++timestep_number)
309  {
310  std::cout << "Time step " << timestep_number << " at t=" << time
311  << std::endl;
312 
313  mass_matrix.vmult(system_rhs, old_solution_u);
314 
315  mass_matrix.vmult(tmp, old_solution_v);
316  system_rhs.add(time_step, tmp);
317 
318  laplace_matrix.vmult(tmp, old_solution_u);
319  system_rhs.add(-theta * (1 - theta) * time_step * time_step, tmp);
320 
321  RightHandSide<dim> rhs_function;
322  rhs_function.set_time(time);
323  VectorTools::create_right_hand_side(dof_handler,
324  QGauss<dim>(fe.degree + 1),
325  rhs_function,
326  tmp);
327  forcing_terms = tmp;
328  forcing_terms *= theta * time_step;
329 
330  rhs_function.set_time(time - time_step);
331  VectorTools::create_right_hand_side(dof_handler,
332  QGauss<dim>(fe.degree + 1),
333  rhs_function,
334  tmp);
335 
336  forcing_terms.add((1 - theta) * time_step, tmp);
337 
338  system_rhs.add(theta * time_step, forcing_terms);
339 
340  {
341  BoundaryValuesU<dim> boundary_values_u_function;
342  boundary_values_u_function.set_time(time);
343 
344  std::map<types::global_dof_index, double> boundary_values;
345  VectorTools::interpolate_boundary_values(dof_handler,
346  0,
347  boundary_values_u_function,
348  boundary_values);
349 
350  matrix_u.copy_from(mass_matrix);
351  matrix_u.add(theta * theta * time_step * time_step, laplace_matrix);
352  MatrixTools::apply_boundary_values(boundary_values,
353  matrix_u,
354  solution_u,
355  system_rhs);
356  }
357  solve_u();
358 
359 
360  laplace_matrix.vmult(system_rhs, solution_u);
361  system_rhs *= -theta * time_step;
362 
363  mass_matrix.vmult(tmp, old_solution_v);
364  system_rhs += tmp;
365 
366  laplace_matrix.vmult(tmp, old_solution_u);
367  system_rhs.add(-time_step * (1 - theta), tmp);
368 
369  system_rhs += forcing_terms;
370 
371  {
372  BoundaryValuesV<dim> boundary_values_v_function;
373  boundary_values_v_function.set_time(time);
374 
375  std::map<types::global_dof_index, double> boundary_values;
376  VectorTools::interpolate_boundary_values(dof_handler,
377  0,
378  boundary_values_v_function,
379  boundary_values);
380  matrix_v.copy_from(mass_matrix);
381  MatrixTools::apply_boundary_values(boundary_values,
382  matrix_v,
383  solution_v,
384  system_rhs);
385  }
386  solve_v();
387 
388  output_results();
389 
390  std::cout << " Total energy: "
391  << (mass_matrix.matrix_norm_square(solution_v) +
392  laplace_matrix.matrix_norm_square(solution_u)) /
393  2
394  << std::endl;
395 
396  old_solution_u = solution_u;
397  old_solution_v = solution_v;
398  }
399  }
