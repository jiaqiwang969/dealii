245 template <typename VectorType>
246 void
247 SolverBFGS<VectorType>::solve(
248  const std::function<typename VectorType::value_type(const VectorType &x,
249  VectorType &f)> &compute,
250  VectorType & x)
251 {
252  // Also see scipy Fortran implementation
253  // https://github.com/scipy/scipy/blob/master/scipy/optimize/lbfgsb_src/lbfgsb.f
254  // and Octave-optim implementation:
255  // https://sourceforge.net/p/octave/optim/ci/default/tree/src/__bfgsmin.cc
256  LogStream::Prefix prefix("BFGS");
257 
258  // default line search:
259  bool first_step = true;
260  Number f_prev = 0.;
261  // provide default line search if no signal was attached
262  VectorType x0;
263  if (line_search_signal.empty())
264  {
265  x0.reinit(x);
266  const auto default_line_min =
267  [&](Number &f, VectorType &x, VectorType &g, const VectorType &p) {
268  const Number f0 = f;
269  const Number g0 = g * p;
270  Assert(g0 < 0,
271  ExcMessage(
272  "Function does not decrease along the current direction"));
273 
274  // save current solution value (to be used in line_search):
275  x0 = x;
276 
277  // see scipy implementation
278  // https://docs.scipy.org/doc/scipy/reference/generated/scipy.optimize.line_search.html#scipy.optimize.line_search
279  // and Eq. 2.6.8 in Fletcher 2013, Practical methods of optimization
280  Number df = f_prev - f;
281  Assert(first_step || df >= 0.,
282  ExcMessage("Function value is not decreasing"));
283  df = std::max(df, 100. * std::numeric_limits<Number>::epsilon());
284  // guess a reasonable first step:
285  const Number a1 =
286  (first_step ? 1. : std::min(1., -1.01 * 2. * df / g0));
287  Assert(a1 > 0., ExcInternalError());
288  f_prev = f;
289 
290  // 1D line-search function
291  const auto line_func =
292  [&](const Number &x_line) -> std::pair<Number, Number> {
293  x = x0;
294  x.add(x_line, p);
295  f = compute(x, g);
296  const Number g_line = g * p;
297  return std::make_pair(f, g_line);
298  };
299 
300  // loose line search:
301  const auto res = LineMinimization::line_search<Number>(
302  line_func,
303  f0,
304  g0,
305  LineMinimization::poly_fit<Number>,
306  a1,
307  0.9,
308  0.001);
309 
310  if (first_step)
311  first_step = false;
312 
313  return res.first;
314  };
315  this->connect_line_search_slot(default_line_min);
316  }
317 
318  // FIXME: Octave has convergence in terms of:
319  // function change tolerance, default 1e-12
320  // parameter change tolerance, default 1e-6
321  // gradient tolerance, default 1e-5
322  // SolverBase and/or SolverControl need extension
323 
324  VectorType g(x), p(x), y_k(x), s_k(x);
325 
326  std::vector<Number> c1;
327  c1.reserve(additional_data.max_history_size);
328 
329  // limited history
330  FiniteSizeHistory<VectorType> y(additional_data.max_history_size);
331  FiniteSizeHistory<VectorType> s(additional_data.max_history_size);
332  FiniteSizeHistory<Number> rho(additional_data.max_history_size);
333 
334  unsigned int m = 0;
335  Number f;
336 
337  SolverControl::State conv = SolverControl::iterate;
338  unsigned int k = 0;
339 
340  f = compute(x, g);
341 
342  conv = this->iteration_status(k, g.l2_norm(), x);
343  if (conv != SolverControl::iterate)
344  return;
345 
346  while (conv == SolverControl::iterate)
347  {
348  if (additional_data.debug_output)
349  deallog << "Iteration " << k << " history " << m << std::endl
350  << "f=" << f << std::endl;
351 
352  // 1. Two loop recursion to calculate p = - H*g
353  c1.resize(m);
354  p = g;
355  // first loop:
356  for (unsigned int i = 0; i < m; ++i)
357  {
358  c1[i] = rho[i] * (s[i] * p);
359  p.add(-c1[i], y[i]);
360  }
361  // H0
362  if (!preconditioner_signal.empty())
363  preconditioner_signal(p, s, y);
364 
365  // second loop:
366  for (int i = m - 1; i >= 0; --i)
367  {
368  Assert(i >= 0, ExcInternalError());
369  const Number c2 = rho[i] * (y[i] * p);
370  p.add(c1[i] - c2, s[i]);
371  }
372  p *= -1.;
373 
374  // 2. Line search
375  s_k = x;
376  y_k = g;
377  const Number alpha = line_search_signal(f, x, g, p)
378  .get(); // <-- signals return boost::optional
379  s_k.sadd(-1, 1, x);
380  y_k.sadd(-1, 1, g);
381 
382  if (additional_data.debug_output)
383  deallog << "Line search a=" << alpha << " f=" << f << std::endl;
384 
385  // 3. Check convergence
386  k++;
387  const Number g_l2 = g.l2_norm();
388  conv = this->iteration_status(k, g_l2, x);
389  if (conv != SolverControl::iterate)
390  break;
391 
392  // 4. Store s, y, rho
393  const Number curvature = s_k * y_k;
394  if (additional_data.debug_output)
395  deallog << "Curvature " << curvature << std::endl;
396 
397  if (curvature > 0. && additional_data.max_history_size > 0)
398  {
399  s.add(s_k);
400  y.add(y_k);
401  rho.add(1. / curvature);
402  m = s.size();
403 
404  Assert(y.size() == m, ExcInternalError());
405  Assert(rho.size() == m, ExcInternalError());
406  }
407 
408  Assert(m <= additional_data.max_history_size, ExcInternalError());
409  }
410 
411  // In the case of failure: throw exception.
412  AssertThrow(conv == SolverControl::success,
413  SolverControl::NoConvergence(k, g.l2_norm()));
414 }
