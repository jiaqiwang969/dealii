154  void DGTransportEquation<dim>::assemble_cell_term(
155  const FEValues<dim> &fe_v,
156  FullMatrix<double> & ui_vi_matrix,
157  Vector<double> & cell_vector) const
158  {
159  const std::vector<double> &JxW = fe_v.get_JxW_values();
160 
161  std::vector<Point<dim>> beta(fe_v.n_quadrature_points);
162  std::vector<double> rhs(fe_v.n_quadrature_points);
163 
164  beta_function.value_list(fe_v.get_quadrature_points(), beta);
165  rhs_function.value_list(fe_v.get_quadrature_points(), rhs);
166 
167  for (unsigned int point = 0; point < fe_v.n_quadrature_points; ++point)
168  for (unsigned int i = 0; i < fe_v.dofs_per_cell; ++i)
169  {
170  for (unsigned int j = 0; j < fe_v.dofs_per_cell; ++j)
171  ui_vi_matrix(i, j) -= beta[point] * fe_v.shape_grad(i, point) *
172  fe_v.shape_value(j, point) * JxW[point];
173 
174  cell_vector(i) +=
175  rhs[point] * fe_v.shape_value(i, point) * JxW[point];

180  template <int dim>
181  void DGTransportEquation<dim>::assemble_boundary_term(
182  const FEFaceValues<dim> &fe_v,
183  FullMatrix<double> & ui_vi_matrix,
184  Vector<double> & cell_vector) const
185  {
186  const std::vector<double> & JxW = fe_v.get_JxW_values();
187  const std::vector<Tensor<1, dim>> &normals = fe_v.get_normal_vectors();
188 
189  std::vector<Point<dim>> beta(fe_v.n_quadrature_points);
190  std::vector<double> g(fe_v.n_quadrature_points);
191 
192  beta_function.value_list(fe_v.get_quadrature_points(), beta);
193  boundary_function.value_list(fe_v.get_quadrature_points(), g);
194 
195  for (unsigned int point = 0; point < fe_v.n_quadrature_points; ++point)
196  {
197  const double beta_n = beta[point] * normals[point];
198  if (beta_n > 0)
199  for (unsigned int i = 0; i < fe_v.dofs_per_cell; ++i)
200  for (unsigned int j = 0; j < fe_v.dofs_per_cell; ++j)
201  ui_vi_matrix(i, j) += beta_n * fe_v.shape_value(j, point) *
202  fe_v.shape_value(i, point) * JxW[point];
203  else
204  for (unsigned int i = 0; i < fe_v.dofs_per_cell; ++i)
205  cell_vector(i) -=
206  beta_n * g[point] * fe_v.shape_value(i, point) * JxW[point];
207  }
208  }

212  void DGTransportEquation<dim>::assemble_face_term(
213  const FEFaceValuesBase<dim> &fe_v,
214  const FEFaceValuesBase<dim> &fe_v_neighbor,

242  }
243  else

245  for (unsigned int i = 0; i < fe_v.dofs_per_cell; ++i)
246  for (unsigned int l = 0; l < fe_v_neighbor.dofs_per_cell; ++l)
247  ue_vi_matrix(i, l) += beta_n *
248  fe_v_neighbor.shape_value(l, point) *
249  fe_v.shape_value(i, point) * JxW[point];
250 
251  for (unsigned int k = 0; k < fe_v_neighbor.dofs_per_cell; ++k)
252  for (unsigned int l = 0; l < fe_v_neighbor.dofs_per_cell; ++l)
253  ue_ve_matrix(k, l) -=
254  beta_n * fe_v_neighbor.shape_value(l, point) *
255  fe_v_neighbor.shape_value(k, point) * JxW[point];

341  std::vector<types::global_dof_index> dofs(dofs_per_cell);
342  std::vector<types::global_dof_index> dofs_neighbor(dofs_per_cell);

345  update_quadrature_points |
346  update_JxW_values;
347 
348  const UpdateFlags face_update_flags =
349  update_values | update_quadrature_points | update_JxW_values |
350  update_normal_vectors;
351 
352  const UpdateFlags neighbor_face_update_flags = update_values;
353 
354  FEValues<dim> fe_v(mapping, fe, quadrature, update_flags);
355  FEFaceValues<dim> fe_v_face(mapping,
356  fe,
357  face_quadrature,
358  face_update_flags);
359  FESubfaceValues<dim> fe_v_subface(mapping,
360  fe,
361  face_quadrature,

364  fe,
365  face_quadrature,
366  neighbor_face_update_flags);

368

370  FullMatrix<double> ue_vi_matrix(dofs_per_cell, dofs_per_cell);
371 
372  FullMatrix<double> ui_ve_matrix(dofs_per_cell, dofs_per_cell);
373  FullMatrix<double> ue_ve_matrix(dofs_per_cell, dofs_per_cell);
374 
375  Vector<double> cell_vector(dofs_per_cell);
376 
377  for (const auto &cell : dof_handler.active_cell_iterators())
378  {
379  ui_vi_matrix = 0;
380  cell_vector = 0;
381 
382  fe_v.reinit(cell);
383 
384  dg.assemble_cell_term(fe_v, ui_vi_matrix, cell_vector);
385 
386  cell->get_dof_indices(dofs);
387 
388  for (const auto face_no : cell->face_indices())
389  {
390  const auto face = cell->face(face_no);
391 
392  if (face->at_boundary())
393  {
394  fe_v_face.reinit(cell, face_no);
395 
396  dg.assemble_boundary_term(fe_v_face, ui_vi_matrix, cell_vector);
397  }
398  else
399  {
400  Assert(cell->neighbor(face_no).state() == IteratorState::valid,
401  ExcInternalError());
402  const auto neighbor = cell->neighbor(face_no);
403 
404  if (face->has_children())
405  {
406  const unsigned int neighbor2 =
407  cell->neighbor_face_no(face_no);
408 
409  for (unsigned int subface_no = 0;
410  subface_no < face->n_active_descendants();
411  ++subface_no)
412  {
413  const auto neighbor_child =
414  cell->neighbor_child_on_subface(face_no, subface_no);
415  Assert(!neighbor_child->has_children(),
416  ExcInternalError());
417 
418  ue_vi_matrix = 0;
419  ui_ve_matrix = 0;
420  ue_ve_matrix = 0;
421 
422  fe_v_subface.reinit(cell, face_no, subface_no);
423  fe_v_face_neighbor.reinit(neighbor_child, neighbor2);
424 
425  dg.assemble_face_term(fe_v_subface,
426  fe_v_face_neighbor,
427  ui_vi_matrix,
428  ue_vi_matrix,
429  ui_ve_matrix,
430  ue_ve_matrix);
431 
432  neighbor_child->get_dof_indices(dofs_neighbor);
433 
434  for (unsigned int i = 0; i < dofs_per_cell; ++i)
435  for (unsigned int j = 0; j < dofs_per_cell; ++j)
436  {
437  system_matrix.add(dofs[i],
438  dofs_neighbor[j],
439  ue_vi_matrix(i, j));
440  system_matrix.add(dofs_neighbor[i],
441  dofs[j],
442  ui_ve_matrix(i, j));
443  system_matrix.add(dofs_neighbor[i],
444  dofs_neighbor[j],
445  ue_ve_matrix(i, j));
446  }
447  }
448  }
449  else
450  {
451  if (dim > 1 && cell->neighbor_is_coarser(face_no))
452  continue;
453 
454  if (((dim > 1) && (cell->index() < neighbor->index())) ||
455  ((dim == 1) && ((cell->level() < neighbor->level()) ||
456  ((cell->level() == neighbor->level()) &&
457  (cell->index() < neighbor->index())))))
458  {
459  const unsigned int neighbor2 =
460  cell->neighbor_of_neighbor(face_no);
461 
462  ue_vi_matrix = 0;
463  ui_ve_matrix = 0;
464  ue_ve_matrix = 0;
465 
466  fe_v_face.reinit(cell, face_no);
467  fe_v_face_neighbor.reinit(neighbor, neighbor2);
468 
469  dg.assemble_face_term(fe_v_face,
470  fe_v_face_neighbor,
471  ui_vi_matrix,
472  ue_vi_matrix,
473  ui_ve_matrix,
474  ue_ve_matrix);
475 
476  neighbor->get_dof_indices(dofs_neighbor);
477 
478  for (unsigned int i = 0; i < dofs_per_cell; ++i)
479  for (unsigned int j = 0; j < dofs_per_cell; ++j)
480  {
481  system_matrix.add(dofs[i],
482  dofs_neighbor[j],
483  ue_vi_matrix(i, j));
484  system_matrix.add(dofs_neighbor[i],
485  dofs[j],
486  ui_ve_matrix(i, j));

489  ue_ve_matrix(i, j));

493  }
