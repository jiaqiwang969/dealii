242 template <int dim, int spacedim>
243 inline void
244 MGConstrainedDoFs::initialize(
245  const DoFHandler<dim, spacedim> &dof,
246  const MGLevelObject<IndexSet> & level_relevant_dofs)
247 {
248  boundary_indices.clear();
249  refinement_edge_indices.clear();
250  level_constraints.clear();
251  user_constraints.clear();
252 
253  const unsigned int nlevels = dof.get_triangulation().n_global_levels();
254  const unsigned int min_level = level_relevant_dofs.min_level();
255  const unsigned int max_level = (level_relevant_dofs.max_level() == 0) ?
256  nlevels - 1 :
257  level_relevant_dofs.max_level();
258  const bool user_level_dofs =
259  (level_relevant_dofs.max_level() == 0) ? false : true;
260 
261  // At this point level_constraint and refinement_edge_indices are empty.
262  refinement_edge_indices.resize(nlevels);
263  level_constraints.resize(nlevels);
264  user_constraints.resize(nlevels);
265  for (unsigned int l = min_level; l <= max_level; ++l)
266  {
267  if (user_level_dofs)
268  {
269  level_constraints[l].reinit(level_relevant_dofs[l]);
270  }
271  else
272  {
273  IndexSet relevant_dofs;
274  DoFTools::extract_locally_relevant_level_dofs(dof, l, relevant_dofs);
275  level_constraints[l].reinit(relevant_dofs);
276  }
277 
278  // Loop through relevant cells and faces finding those which are periodic
279  // neighbors.
280  typename DoFHandler<dim, spacedim>::cell_iterator cell = dof.begin(l),
281  endc = dof.end(l);
282  for (; cell != endc; ++cell)
283  if (cell->level_subdomain_id() != numbers::artificial_subdomain_id)
284  {
285  for (auto f : cell->face_indices())
286  if (cell->has_periodic_neighbor(f) &&
287  cell->periodic_neighbor(f)->level() == cell->level())
288  {
289  if (cell->is_locally_owned_on_level())
290  {
291  Assert(
292  cell->periodic_neighbor(f)->level_subdomain_id() !=
293  numbers::artificial_subdomain_id,
294  ExcMessage(
295  "Periodic neighbor of a locally owned cell must either be owned or ghost."));
296  }
297  // Cell is a level-ghost and its neighbor is a
298  // level-artificial cell nothing to do here
299  else if (cell->periodic_neighbor(f)->level_subdomain_id() ==
300  numbers::artificial_subdomain_id)
301  {
302  Assert(cell->is_locally_owned_on_level() == false,
303  ExcInternalError());
304  continue;
305  }
306 
307  const unsigned int dofs_per_face =
308  dof.get_fe(0).n_dofs_per_face(f);
309  std::vector<types::global_dof_index> dofs_1(dofs_per_face);
310  std::vector<types::global_dof_index> dofs_2(dofs_per_face);
311 
312  cell->periodic_neighbor(f)
313  ->face(cell->periodic_neighbor_face_no(f))
314  ->get_mg_dof_indices(l, dofs_1, 0);
315  cell->face(f)->get_mg_dof_indices(l, dofs_2, 0);
316  // Store periodicity information in the level
317  // AffineConstraints object. Skip DoFs for which we've
318  // previously entered periodicity constraints already; this
319  // can happen, for example, for a vertex dof at a periodic
320  // boundary that we visit from more than one cell
321  for (unsigned int i = 0; i < dofs_per_face; ++i)
322  if (level_constraints[l].can_store_line(dofs_2[i]) &&
323  level_constraints[l].can_store_line(dofs_1[i]) &&
324  !level_constraints[l].is_constrained(dofs_2[i]) &&
325  !level_constraints[l].is_constrained(dofs_1[i]))
326  {
327  level_constraints[l].add_line(dofs_2[i]);
328  level_constraints[l].add_entry(dofs_2[i],
329  dofs_1[i],
330  1.);
331  }
332  }
333  }
334  level_constraints[l].close();
335 
336  // Initialize with empty IndexSet of correct size
337  refinement_edge_indices[l] = IndexSet(dof.n_dofs(l));
338  }
339 
340  MGTools::extract_inner_interface_dofs(dof, refinement_edge_indices);
341 }

366 template <int dim, int spacedim>
367 inline void
368 MGConstrainedDoFs::make_no_normal_flux_constraints(
369  const DoFHandler<dim, spacedim> &dof,
370  const types::boundary_id bid,
371  const unsigned int first_vector_component)
372 {
373  // For a given boundary id, find which vector component is on the boundary
374  // and set a zero boundary constraint for those degrees of freedom.
375  const unsigned int n_components = dof.get_fe_collection().n_components();
376  AssertIndexRange(first_vector_component + dim - 1, n_components);
377 
378  ComponentMask comp_mask(n_components, false);
379 
380 
381  typename Triangulation<dim>::face_iterator
382  face = dof.get_triangulation().begin_face(),
383  endf = dof.get_triangulation().end_face();
384  for (; face != endf; ++face)
385  if (face->at_boundary() && face->boundary_id() == bid)
386  for (unsigned int d = 0; d < dim; ++d)
387  {
388  Tensor<1, dim, double> unit_vec;
389  unit_vec[d] = 1.0;
390 
391  const Tensor<1, dim> normal_vec =
392  face->get_manifold().normal_vector(face, face->center());
393 
394  if (std::abs(std::abs(unit_vec * normal_vec) - 1.0) < 1e-10)
395  comp_mask.set(d + first_vector_component, true);
396  else
397  Assert(
398  std::abs(unit_vec * normal_vec) < 1e-10,
399  ExcMessage(
400  "We can currently only support no normal flux conditions "
401  "for a specific boundary id if all faces are normal to the "
402  "x, y, or z axis."));
403  }
404 
405  Assert(comp_mask.n_selected_components() == 1,
406  ExcMessage(
407  "We can currently only support no normal flux conditions "
408  "for a specific boundary id if all faces are facing in the "
409  "same direction, i.e., a boundary normal to the x-axis must "
410  "have a different boundary id than a boundary normal to the "
411  "y- or z-axis and so on. If the mesh here was produced using "
412  "GridGenerator::..., setting colorize=true during mesh generation "
413  "and calling make_no_normal_flux_constraints() for each no normal "
414  "flux boundary will fulfill the condition."));
415 
416  this->make_zero_boundary_constraints(dof, {bid}, comp_mask);
417 }
