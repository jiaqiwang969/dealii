146  Point<3> AfricaGeometry::push_forward_wgs84(const Point<3> &phi_theta_d) const
147  {
148  const double phi = phi_theta_d[0];
149  const double theta = phi_theta_d[1];
150  const double d = phi_theta_d[2];
151 
152  const double R_bar = R / std::sqrt(1 - (ellipticity * ellipticity *
153  std::sin(theta) * std::sin(theta)));
154 
155  return {(R_bar + d) * std::cos(phi) * std::cos(theta),
156  (R_bar + d) * std::sin(phi) * std::cos(theta),
157  ((1 - ellipticity * ellipticity) * R_bar + d) * std::sin(theta)};
158  }
159 
160  Point<3> AfricaGeometry::pull_back_wgs84(const Point<3> &x) const
161  {
162  const double b = std::sqrt(R * R * (1 - ellipticity * ellipticity));
163  const double ep = std::sqrt((R * R - b * b) / (b * b));
164  const double p = std::sqrt(x(0) * x(0) + x(1) * x(1));
165  const double th = std::atan2(R * x(2), b * p);
166  const double phi = std::atan2(x(1), x(0));
167  const double theta =
168  std::atan2(x(2) + ep * ep * b * std::pow(std::sin(th), 3),
169  (p -
170  (ellipticity * ellipticity * R * std::pow(std::cos(th), 3))));
171  const double R_bar =
172  R / (std::sqrt(1 - ellipticity * ellipticity * std::sin(theta) *
173  std::sin(theta)));
174  const double R_plus_d = p / std::cos(theta);
175 
176  Point<3> phi_theta_d;
177  if (phi < 0)
178  phi_theta_d[0] = phi + 2 * numbers::PI;
179  else if (phi > 2 * numbers::PI)
180  phi_theta_d[0] = phi - 2 * numbers::PI;
181  else
182  phi_theta_d[0] = phi;
183  phi_theta_d[1] = theta;
184  phi_theta_d[2] = R_plus_d - R_bar;
185  return phi_theta_d;
186  }

207  void run()
208  {
209  AfricaGeometry geometry;
210  Triangulation<3> triangulation;
211 
212  {
213  const Point<3> corner_points[2] = {
214  Point<3>(26 * numbers::PI / 180, -10 * numbers::PI / 180, -500000),
215  Point<3>(35 * numbers::PI / 180, 5 * numbers::PI / 180, 0)};
216  std::vector<unsigned int> subdivisions(3);

218  subdivisions[1] = 2;
219  subdivisions[2] = 1;
220  GridGenerator::subdivided_hyper_rectangle(
221  triangulation, subdivisions, corner_points[0], corner_points[1], true);
222 
223  GridTools::transform(
224  [&geometry](const Point<3> &chart_point) {
225  return geometry.push_forward(chart_point);
226  },
227  triangulation);
228  }
229 
230  triangulation.set_manifold(0, geometry);
231  for (const auto &cell : triangulation.active_cell_iterators())
232  cell->set_all_manifold_ids(0);
233 
234  for (unsigned int i = 0; i < 6; ++i)
235  {
236  for (const auto &cell : triangulation.active_cell_iterators())
237  for (const auto &face : cell->face_iterators())
238  if (face->boundary_id() == 5)
239  {
240  cell->set_refine_flag();
241  break;
242  }
243  triangulation.execute_coarsening_and_refinement();
244 
245  std::cout << "Refinement step " << i + 1 << ": "
