1209  template <typename T>
1210  std::map<unsigned int, T>
1211  some_to_some(const MPI_Comm & comm,
1212  const std::map<unsigned int, T> &objects_to_send)
1213  {
1214 # ifndef DEAL_II_WITH_MPI
1215  (void)comm;
1216  Assert(objects_to_send.size() < 2,
1217  ExcMessage("Cannot send to more than one processor."));
1218  Assert(objects_to_send.find(0) != objects_to_send.end() ||
1219  objects_to_send.size() == 0,
1220  ExcMessage("Can only send to myself or to nobody."));
1221  return objects_to_send;
1222 # else
1223  const auto my_proc = this_mpi_process(comm);
1224 
1225  std::map<unsigned int, T> received_objects;
1226 
1227  std::vector<unsigned int> send_to;
1228  send_to.reserve(objects_to_send.size());
1229  for (const auto &m : objects_to_send)
1230  if (m.first == my_proc)
1231  received_objects[my_proc] = m.second;
1232  else
1233  send_to.emplace_back(m.first);
1234 
1235  const unsigned int n_point_point_communications =
1236  Utilities::MPI::compute_n_point_to_point_communications(comm, send_to);
1237 
1238  // Protect the following communication:
1239  static CollectiveMutex mutex;
1240  CollectiveMutex::ScopedLock lock(mutex, comm);
1241 
1242  // If we have something to send, or we expect something from other
1243  // processors, we need to visit one of the two scopes below. Otherwise,
1244  // no other action is required by this mpi process, and we can safely
1245  // return.
1246  if (send_to.size() == 0 && n_point_point_communications == 0)
1247  return received_objects;
1248 
1249  const int mpi_tag =
1250  internal::Tags::compute_point_to_point_communication_pattern;
1251 
1252  // Sending buffers
1253  std::vector<std::vector<char>> buffers_to_send(send_to.size());
1254  std::vector<MPI_Request> buffer_send_requests(send_to.size());
1255  {
1256  unsigned int i = 0;
1257  for (const auto &rank_obj : objects_to_send)
1258  if (rank_obj.first != my_proc)
1259  {
1260  const auto &rank = rank_obj.first;
1261  buffers_to_send[i] = Utilities::pack(rank_obj.second,
1262  /*allow_compression=*/ false);
1263  const int ierr = MPI_Isend(buffers_to_send[i].data(),
1264  buffers_to_send[i].size(),
1265  MPI_CHAR,
1266  rank,
1267  mpi_tag,
1268  comm,
1269  &buffer_send_requests[i]);
1270  AssertThrowMPI(ierr);
1271  ++i;
1272  }
1273  }
1274 
1275  // Fill the output map
1276  {
1277  std::vector<char> buffer;
1278  // We do this on a first come/first served basis
1279  for (unsigned int i = 0; i < n_point_point_communications; ++i)
1280  {
1281  // Probe what's going on. Take data from the first available sender
1282  MPI_Status status;
1283  int ierr = MPI_Probe(MPI_ANY_SOURCE, mpi_tag, comm, &status);
1284  AssertThrowMPI(ierr);
1285 
1286  // Length of the message
1287  int len;
1288  ierr = MPI_Get_count(&status, MPI_CHAR, &len);
1289  AssertThrowMPI(ierr);
1290  buffer.resize(len);
1291 
1292  // Source rank
1293  const unsigned int rank = status.MPI_SOURCE;
1294 
1295  // Actually receive the message
1296  ierr = MPI_Recv(buffer.data(),
1297  len,
1298  MPI_CHAR,
1299  status.MPI_SOURCE,
1300  status.MPI_TAG,
1301  comm,
1302  MPI_STATUS_IGNORE);
1303  AssertThrowMPI(ierr);
1304  Assert(received_objects.find(rank) == received_objects.end(),
1305  ExcInternalError(
1306  "I should not receive again from this rank"));
1307  received_objects[rank] =
1308  Utilities::unpack<T>(buffer,
1309  /*allow_compression=*/ false);
1310  }
1311  }
1312 
1313  // Wait to have sent all objects.
1314  const int ierr = MPI_Waitall(send_to.size(),
1315  buffer_send_requests.data(),
1316  MPI_STATUSES_IGNORE);
1317  AssertThrowMPI(ierr);
1318 
1319  return received_objects;
1320 # endif // deal.II with MPI
1321  }

1323  template <typename T>
1324  std::vector<T>
1325  all_gather(const MPI_Comm &comm, const T &object)
1326  {
1327  if (job_supports_mpi() == false)
1328  return {object};
1329 
1330 # ifndef DEAL_II_WITH_MPI
1331  (void)comm;
1332  std::vector<T> v(1, object);
1333  return v;
1334 # else
1335  const auto n_procs = ::Utilities::MPI::n_mpi_processes(comm);
1336 
1337  std::vector<char> buffer = Utilities::pack(object);
1338 
1339  int n_local_data = buffer.size();
1340 
1341  // Vector to store the size of loc_data_array for every process
1342  std::vector<int> size_all_data(n_procs, 0);
1343 
1344  // Exchanging the size of each buffer
1345  int ierr = MPI_Allgather(
1346  &n_local_data, 1, MPI_INT, size_all_data.data(), 1, MPI_INT, comm);
1347  AssertThrowMPI(ierr);
1348 
1349  // Now computing the displacement, relative to recvbuf,
1350  // at which to store the incoming buffer
1351  std::vector<int> rdispls(n_procs);
1352  rdispls[0] = 0;
1353  for (unsigned int i = 1; i < n_procs; ++i)
1354  rdispls[i] = rdispls[i - 1] + size_all_data[i - 1];
1355 
1356  // Step 3: exchange the buffer:
1357  std::vector<char> received_unrolled_buffer(rdispls.back() +
1358  size_all_data.back());
1359 
1360  ierr = MPI_Allgatherv(buffer.data(),
1361  n_local_data,
1362  MPI_CHAR,
1363  received_unrolled_buffer.data(),
1364  size_all_data.data(),
1365  rdispls.data(),
1366  MPI_CHAR,
1367  comm);
1368  AssertThrowMPI(ierr);
1369 
1370  std::vector<T> received_objects(n_procs);
1371  for (unsigned int i = 0; i < n_procs; ++i)
1372  {
1373  std::vector<char> local_buffer(received_unrolled_buffer.begin() +
1374  rdispls[i],
1375  received_unrolled_buffer.begin() +
1376  rdispls[i] + size_all_data[i]);
1377  received_objects[i] = Utilities::unpack<T>(local_buffer);
1378  }
1379 
1380  return received_objects;
1381 # endif
1382  }

1385  std::vector<T>
1386  gather(const MPI_Comm & comm,
1387  const T & object_to_send,
1388  const unsigned int root_process)
1389  {
1390 # ifndef DEAL_II_WITH_MPI
1391  (void)comm;
1392  (void)root_process;
1393  std::vector<T> v(1, object_to_send);
1394  return v;
1395 # else
1396  const auto n_procs = ::Utilities::MPI::n_mpi_processes(comm);
1397  const auto my_rank = ::Utilities::MPI::this_mpi_process(comm);
1398 
1399  AssertIndexRange(root_process, n_procs);
1400 
1401  std::vector<char> buffer = Utilities::pack(object_to_send);
1402  int n_local_data = buffer.size();
1403 
1404  // Vector to store the size of loc_data_array for every process
1405  // only the root process needs to allocate memory for that purpose
1406  std::vector<int> size_all_data;
1407  if (my_rank == root_process)
1408  size_all_data.resize(n_procs, 0);
1409 
1410  // Exchanging the size of each buffer
1411  int ierr = MPI_Gather(&n_local_data,
1412  1,
1413  MPI_INT,
1414  size_all_data.data(),
1415  1,
1416  MPI_INT,
1417  root_process,
1418  comm);
1419  AssertThrowMPI(ierr);
1420 
1421  // Now computing the displacement, relative to recvbuf,
1422  // at which to store the incoming buffer; only for root
1423  std::vector<int> rdispls;
1424  if (my_rank == root_process)
1425  {
1426  rdispls.resize(n_procs, 0);
1427  for (unsigned int i = 1; i < n_procs; ++i)
1428  rdispls[i] = rdispls[i - 1] + size_all_data[i - 1];
1429  }
1430  // exchange the buffer:
1431  std::vector<char> received_unrolled_buffer;
1432  if (my_rank == root_process)
1433  received_unrolled_buffer.resize(rdispls.back() + size_all_data.back());
1434 
1435  ierr = MPI_Gatherv(buffer.data(),
1436  n_local_data,
1437  MPI_CHAR,
1438  received_unrolled_buffer.data(),
1439  size_all_data.data(),
1440  rdispls.data(),
1441  MPI_CHAR,
1442  root_process,
1443  comm);
1444  AssertThrowMPI(ierr);
1445 
1446  std::vector<T> received_objects;
1447 
1448  if (my_rank == root_process)
1449  {
1450  received_objects.resize(n_procs);
1451 
1452  for (unsigned int i = 0; i < n_procs; ++i)
1453  {
1454  const std::vector<char> local_buffer(
1455  received_unrolled_buffer.begin() + rdispls[i],
1456  received_unrolled_buffer.begin() + rdispls[i] +
1457  size_all_data[i]);
1458  received_objects[i] = Utilities::unpack<T>(local_buffer);
1459  }
1460  }
1461  return received_objects;
1462 # endif
1463  }
