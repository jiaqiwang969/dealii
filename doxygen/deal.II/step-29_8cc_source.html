240  {
241  std::cout << "Generating grid... ";
242  Timer timer;
243 
244  prm.enter_subsection("Mesh & geometry parameters");
245 
246  const double focal_distance = prm.get_double("Focal distance");
247  const unsigned int n_refinements = prm.get_integer("Number of refinements");
248 
249  prm.leave_subsection();
250 
251  const Point<dim> transducer =
252  (dim == 2) ? Point<dim>(0.5, 0.0) : Point<dim>(0.5, 0.5, 0.0);
253  const Point<dim> focal_point = (dim == 2) ?
254  Point<dim>(0.5, focal_distance) :
255  Point<dim>(0.5, 0.5, focal_distance);
256 
257 
258  GridGenerator::subdivided_hyper_cube(triangulation, 5, 0, 1);
259 
260  for (auto &cell : triangulation.cell_iterators())
261  for (const auto &face : cell->face_iterators())
262  if (face->at_boundary() &&
263  ((face->center() - transducer).norm_square() < 0.01))
264  {
265  face->set_boundary_id(1);
266  face->set_manifold_id(1);
267  }
268  triangulation.set_manifold(1, SphericalManifold<dim>(focal_point));
269 
270  triangulation.refine_global(n_refinements);
271 
272  timer.stop();
273  std::cout << "done (" << timer.cpu_time() << "s)" << std::endl;
274 
275  std::cout << " Number of active cells: " << triangulation.n_active_cells()

305  template <int dim>
306  void UltrasoundProblem<dim>::assemble_system()
307  {
308  std::cout << "Assembling system matrix... ";
309  Timer timer;
310 
311 
312  prm.enter_subsection("Physical constants");
313 
314  const double omega = prm.get_double("omega"), c = prm.get_double("c");
315 
316  prm.leave_subsection();
317 
318  QGauss<dim> quadrature_formula(fe.degree + 1);
319  QGauss<dim - 1> face_quadrature_formula(fe.degree + 1);
320 
321  const unsigned int n_q_points = quadrature_formula.size(),
322  n_face_q_points = face_quadrature_formula.size(),
323  dofs_per_cell = fe.n_dofs_per_cell();
324 
325  FEValues<dim> fe_values(fe,
326  quadrature_formula,
327  update_values | update_gradients |
328  update_JxW_values);
329 
330  FEFaceValues<dim> fe_face_values(fe,
331  face_quadrature_formula,
332  update_values | update_JxW_values);
333 
334  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
335  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
336 
337  for (const auto &cell : dof_handler.active_cell_iterators())
338  {
339  cell_matrix = 0;
340  fe_values.reinit(cell);
341 
342  for (unsigned int i = 0; i < dofs_per_cell; ++i)
343  {
344  for (unsigned int j = 0; j < dofs_per_cell; ++j)
345  {
346  if (fe.system_to_component_index(i).first ==
347  fe.system_to_component_index(j).first)
348  {
349 
350  for (unsigned int q_point = 0; q_point < n_q_points;
351  ++q_point)
352  cell_matrix(i, j) +=
353  (((fe_values.shape_value(i, q_point) *
354  fe_values.shape_value(j, q_point)) *
355  (-omega * omega) +
356  (fe_values.shape_grad(i, q_point) *
357  fe_values.shape_grad(j, q_point)) *
358  c * c) *
359  fe_values.JxW(q_point));
360 
361  }
362  }
363  }
364 
365 
366  for (const auto face_no : cell->face_indices())
367  if (cell->face(face_no)->at_boundary() &&
368  (cell->face(face_no)->boundary_id() == 0))
369  {
370  fe_face_values.reinit(cell, face_no);
371 
372 
373  for (unsigned int i = 0; i < dofs_per_cell; ++i)
374  for (unsigned int j = 0; j < dofs_per_cell; ++j)
375  if ((fe.system_to_component_index(i).first !=
376  fe.system_to_component_index(j).first) &&
377  fe.has_support_on_face(i, face_no) &&
378  fe.has_support_on_face(j, face_no))
379 
380  for (unsigned int q_point = 0; q_point < n_face_q_points;
381  ++q_point)
382  cell_matrix(i, j) +=
383  ((fe.system_to_component_index(i).first == 0) ? -1 :
384  1) *
385  fe_face_values.shape_value(i, q_point) *
386  fe_face_values.shape_value(j, q_point) * c * omega *
387  fe_face_values.JxW(q_point);
388  }
389 
390  cell->get_dof_indices(local_dof_indices);
391 
392 
393  for (unsigned int i = 0; i < dofs_per_cell; ++i)
394  for (unsigned int j = 0; j < dofs_per_cell; ++j)
395  system_matrix.add(local_dof_indices[i],
396  local_dof_indices[j],
397  cell_matrix(i, j));
398  }
399 
400 
401  std::map<types::global_dof_index, double> boundary_values;
402  VectorTools::interpolate_boundary_values(dof_handler,
403  1,
404  DirichletBoundaryValues<dim>(),
405  boundary_values);
406 
407  MatrixTools::apply_boundary_values(boundary_values,
408  system_matrix,
409  solution,
410  system_rhs);
411 
412  timer.stop();
413  std::cout << "done (" << timer.cpu_time() << "s)" << std::endl;
414  }
