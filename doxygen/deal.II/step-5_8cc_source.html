120 template <int dim>
121 void Step5<dim>::assemble_system()
122 {
123  QGauss<dim> quadrature_formula(fe.degree + 1);
124 
125  FEValues<dim> fe_values(fe,
126  quadrature_formula,
127  update_values | update_gradients |
128  update_quadrature_points | update_JxW_values);
129 
130  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
131 
132  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
133  Vector<double> cell_rhs(dofs_per_cell);
134 
135  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
136 
137  for (const auto &cell : dof_handler.active_cell_iterators())
138  {
139  cell_matrix = 0.;
140  cell_rhs = 0.;
141 
142  fe_values.reinit(cell);
143 
144  for (const unsigned int q_index : fe_values.quadrature_point_indices())
145  {
146  const double current_coefficient =
147  coefficient(fe_values.quadrature_point(q_index));
148  for (const unsigned int i : fe_values.dof_indices())
149  {
150  for (const unsigned int j : fe_values.dof_indices())
151  cell_matrix(i, j) +=
152  (current_coefficient * // a(x_q)
153  fe_values.shape_grad(i, q_index) * // grad phi_i(x_q)
154  fe_values.shape_grad(j, q_index) * // grad phi_j(x_q)
155  fe_values.JxW(q_index)); // dx
156 
157  cell_rhs(i) += (fe_values.shape_value(i, q_index) * // phi_i(x_q)
158  1.0 * // f(x_q)
159  fe_values.JxW(q_index)); // dx
160  }
161  }
162 
163 
164  cell->get_dof_indices(local_dof_indices);
165  for (const unsigned int i : fe_values.dof_indices())
166  {
167  for (const unsigned int j : fe_values.dof_indices())
168  system_matrix.add(local_dof_indices[i],
169  local_dof_indices[j],
170  cell_matrix(i, j));
171 
172  system_rhs(local_dof_indices[i]) += cell_rhs(i);
173  }
174  }
175 
176  std::map<types::global_dof_index, double> boundary_values;
177  VectorTools::interpolate_boundary_values(dof_handler,
178  0,
179  Functions::ZeroFunction<dim>(),
180  boundary_values);
181  MatrixTools::apply_boundary_values(boundary_values,
182  system_matrix,
183  solution,
184  system_rhs);
185 }

198  solver.solve(system_matrix, solution, system_rhs, preconditioner);
