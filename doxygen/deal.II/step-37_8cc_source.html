154  template <int dim, int fe_degree, typename number>
155  void LaplaceOperator<dim, fe_degree, number>::evaluate_coefficient(
156  const Coefficient<dim> &coefficient_function)
157  {
158  const unsigned int n_cells = this->data->n_cell_batches();
159  FEEvaluation<dim, fe_degree, fe_degree + 1, 1, number> phi(*this->data);
160 
161  coefficient.reinit(n_cells, phi.n_q_points);
162  for (unsigned int cell = 0; cell < n_cells; ++cell)
163  {
164  phi.reinit(cell);
165  for (unsigned int q = 0; q < phi.n_q_points; ++q)
166  coefficient(cell, q) =
167  coefficient_function.value(phi.quadrature_point(q));
168  }

174  template <int dim, int fe_degree, typename number>
175  void LaplaceOperator<dim, fe_degree, number>::local_apply(
176  const MatrixFree<dim, number> & data,
177  LinearAlgebra::distributed::Vector<number> & dst,
178  const LinearAlgebra::distributed::Vector<number> &src,
179  const std::pair<unsigned int, unsigned int> & cell_range) const
180  {
181  FEEvaluation<dim, fe_degree, fe_degree + 1, 1, number> phi(data);
182 
183  for (unsigned int cell = cell_range.first; cell < cell_range.second; ++cell)
184  {
185  AssertDimension(coefficient.size(0), data.n_cell_batches());
186  AssertDimension(coefficient.size(1), phi.n_q_points);
187 
188  phi.reinit(cell);
189  phi.read_dof_values(src);
190  phi.evaluate(EvaluationFlags::gradients);
191  for (unsigned int q = 0; q < phi.n_q_points; ++q)
192  phi.submit_gradient(coefficient(cell, q) * phi.get_gradient(q), q);
193  phi.integrate(EvaluationFlags::gradients);
194  phi.distribute_local_to_global(dst);
195  }
196  }

210  template <int dim, int fe_degree, typename number>
211  void LaplaceOperator<dim, fe_degree, number>::compute_diagonal()
212  {
213  this->inverse_diagonal_entries.reset(
214  new DiagonalMatrix<LinearAlgebra::distributed::Vector<number>>());
215  LinearAlgebra::distributed::Vector<number> &inverse_diagonal =
216  this->inverse_diagonal_entries->get_vector();
217  this->data->initialize_dof_vector(inverse_diagonal);
218  unsigned int dummy = 0;
219  this->data->cell_loop(&LaplaceOperator::local_compute_diagonal,
220  this,
221  inverse_diagonal,
222  dummy);
223 
224  this->set_constrained_entries_to_one(inverse_diagonal);
225 
226  for (unsigned int i = 0; i < inverse_diagonal.locally_owned_size(); ++i)
227  {
228  Assert(inverse_diagonal.local_element(i) > 0.,
229  ExcMessage("No diagonal entry in a positive definite operator "
230  "should be zero"));
231  inverse_diagonal.local_element(i) =
232  1. / inverse_diagonal.local_element(i);
233  }
234  }

238  template <int dim, int fe_degree, typename number>
239  void LaplaceOperator<dim, fe_degree, number>::local_compute_diagonal(
240  const MatrixFree<dim, number> & data,
241  LinearAlgebra::distributed::Vector<number> &dst,
242  const unsigned int &,
243  const std::pair<unsigned int, unsigned int> &cell_range) const
244  {
245  FEEvaluation<dim, fe_degree, fe_degree + 1, 1, number> phi(data);
246 
247  AlignedVector<VectorizedArray<number>> diagonal(phi.dofs_per_cell);
248 
249  for (unsigned int cell = cell_range.first; cell < cell_range.second; ++cell)
250  {
251  AssertDimension(coefficient.size(0), data.n_cell_batches());
252  AssertDimension(coefficient.size(1), phi.n_q_points);
253 
254  phi.reinit(cell);
255  for (unsigned int i = 0; i < phi.dofs_per_cell; ++i)
256  {
257  for (unsigned int j = 0; j < phi.dofs_per_cell; ++j)
258  phi.submit_dof_value(VectorizedArray<number>(), j);
259  phi.submit_dof_value(make_vectorized_array<number>(1.), i);
260 
261  phi.evaluate(EvaluationFlags::gradients);
262  for (unsigned int q = 0; q < phi.n_q_points; ++q)
263  phi.submit_gradient(coefficient(cell, q) * phi.get_gradient(q),
264  q);
265  phi.integrate(EvaluationFlags::gradients);
266  diagonal[i] = phi.get_dof_value(i);
267  }
268  for (unsigned int i = 0; i < phi.dofs_per_cell; ++i)
269  phi.submit_dof_value(diagonal[i], i);
270  phi.distribute_local_to_global(dst);
271  }
272  }

346  template <int dim>
347  void LaplaceProblem<dim>::setup_system()
348  {
349  Timer time;
350  setup_time = 0;
351 
352  system_matrix.clear();
353  mg_matrices.clear_elements();
354 
355  dof_handler.distribute_dofs(fe);
356  dof_handler.distribute_mg_dofs();
357 
358  pcout << "Number of degrees of freedom: " << dof_handler.n_dofs()
359  << std::endl;
360 
361  IndexSet locally_relevant_dofs;
362  DoFTools::extract_locally_relevant_dofs(dof_handler, locally_relevant_dofs);
363 
364  constraints.clear();
365  constraints.reinit(locally_relevant_dofs);
366  DoFTools::make_hanging_node_constraints(dof_handler, constraints);
367  VectorTools::interpolate_boundary_values(
368  mapping, dof_handler, 0, Functions::ZeroFunction<dim>(), constraints);
369  constraints.close();
370  setup_time += time.wall_time();
371  time_details << "Distribute DoFs & B.C. (CPU/wall) " << time.cpu_time()
372  << "s/" << time.wall_time() << "s" << std::endl;
373  time.restart();
374 
375  {
376  typename MatrixFree<dim, double>::AdditionalData additional_data;
377  additional_data.tasks_parallel_scheme =
378  MatrixFree<dim, double>::AdditionalData::none;
379  additional_data.mapping_update_flags =
380  (update_gradients | update_JxW_values | update_quadrature_points);
381  std::shared_ptr<MatrixFree<dim, double>> system_mf_storage(
382  new MatrixFree<dim, double>());
383  system_mf_storage->reinit(mapping,
384  dof_handler,
385  constraints,
386  QGauss<1>(fe.degree + 1),
387  additional_data);
388  system_matrix.initialize(system_mf_storage);
389  }
390 
391  system_matrix.evaluate_coefficient(Coefficient<dim>());
392 
393  system_matrix.initialize_dof_vector(solution);
394  system_matrix.initialize_dof_vector(system_rhs);
395 
396  setup_time += time.wall_time();
397  time_details << "Setup matrix-free system (CPU/wall) " << time.cpu_time()
398  << "s/" << time.wall_time() << "s" << std::endl;
399  time.restart();
400 
401  const unsigned int nlevels = triangulation.n_global_levels();
402  mg_matrices.resize(0, nlevels - 1);
403 
404  std::set<types::boundary_id> dirichlet_boundary;
405  dirichlet_boundary.insert(0);
406  mg_constrained_dofs.initialize(dof_handler);
407  mg_constrained_dofs.make_zero_boundary_constraints(dof_handler,
408  dirichlet_boundary);
409 
410  for (unsigned int level = 0; level < nlevels; ++level)
411  {
412  IndexSet relevant_dofs;
413  DoFTools::extract_locally_relevant_level_dofs(dof_handler,
414  level,
415  relevant_dofs);
416  AffineConstraints<double> level_constraints;
417  level_constraints.reinit(relevant_dofs);
418  level_constraints.add_lines(
419  mg_constrained_dofs.get_boundary_indices(level));
420  level_constraints.close();
421 
422  typename MatrixFree<dim, float>::AdditionalData additional_data;
423  additional_data.tasks_parallel_scheme =
424  MatrixFree<dim, float>::AdditionalData::none;
425  additional_data.mapping_update_flags =
426  (update_gradients | update_JxW_values | update_quadrature_points);
427  additional_data.mg_level = level;
428  std::shared_ptr<MatrixFree<dim, float>> mg_mf_storage_level(
429  new MatrixFree<dim, float>());
430  mg_mf_storage_level->reinit(mapping,
431  dof_handler,
432  level_constraints,
433  QGauss<1>(fe.degree + 1),
434  additional_data);
435 
436  mg_matrices[level].initialize(mg_mf_storage_level,
437  mg_constrained_dofs,
438  level);
439  mg_matrices[level].evaluate_coefficient(Coefficient<dim>());
440  }
441  setup_time += time.wall_time();
442  time_details << "Setup matrix-free levels (CPU/wall) " << time.cpu_time()
443  << "s/" << time.wall_time() << "s" << std::endl;
444  }

477  template <int dim>
478  void LaplaceProblem<dim>::solve()
479  {
480  Timer time;
481  MGTransferMatrixFree<dim, float> mg_transfer(mg_constrained_dofs);
482  mg_transfer.build(dof_handler);
483  setup_time += time.wall_time();
484  time_details << "MG build transfer time (CPU/wall) " << time.cpu_time()
485  << "s/" << time.wall_time() << "s\n";
486  time.restart();
487 
488  using SmootherType =
489  PreconditionChebyshev<LevelMatrixType,
490  LinearAlgebra::distributed::Vector<float>>;
491  mg::SmootherRelaxation<SmootherType,
492  LinearAlgebra::distributed::Vector<float>>
493  mg_smoother;
494  MGLevelObject<typename SmootherType::AdditionalData> smoother_data;
495  smoother_data.resize(0, triangulation.n_global_levels() - 1);
496  for (unsigned int level = 0; level < triangulation.n_global_levels();
497  ++level)
498  {
499  if (level > 0)
500  {
501  smoother_data[level].smoothing_range = 15.;
502  smoother_data[level].degree = 5;
503  smoother_data[level].eig_cg_n_iterations = 10;
504  }
505  else
506  {
507  smoother_data[0].smoothing_range = 1e-3;
508  smoother_data[0].degree = numbers::invalid_unsigned_int;
509  smoother_data[0].eig_cg_n_iterations = mg_matrices[0].m();
510  }
511  mg_matrices[level].compute_diagonal();
512  smoother_data[level].preconditioner =
513  mg_matrices[level].get_matrix_diagonal_inverse();
514  }
515  mg_smoother.initialize(mg_matrices, smoother_data);
516 
517  MGCoarseGridApplySmoother<LinearAlgebra::distributed::Vector<float>>
518  mg_coarse;
519  mg_coarse.initialize(mg_smoother);
520 
521  mg::Matrix<LinearAlgebra::distributed::Vector<float>> mg_matrix(
522  mg_matrices);
523 
524  MGLevelObject<MatrixFreeOperators::MGInterfaceOperator<LevelMatrixType>>
525  mg_interface_matrices;
526  mg_interface_matrices.resize(0, triangulation.n_global_levels() - 1);
527  for (unsigned int level = 0; level < triangulation.n_global_levels();
528  ++level)
529  mg_interface_matrices[level].initialize(mg_matrices[level]);
530  mg::Matrix<LinearAlgebra::distributed::Vector<float>> mg_interface(
531  mg_interface_matrices);
532 
533  Multigrid<LinearAlgebra::distributed::Vector<float>> mg(
534  mg_matrix, mg_coarse, mg_transfer, mg_smoother, mg_smoother);
535  mg.set_edge_matrices(mg_interface, mg_interface);
536 
537  PreconditionMG<dim,
538  LinearAlgebra::distributed::Vector<float>,
539  MGTransferMatrixFree<dim, float>>
540  preconditioner(dof_handler, mg, mg_transfer);
541 
542 
543  SolverControl solver_control(100, 1e-12 * system_rhs.l2_norm());
544  SolverCG<LinearAlgebra::distributed::Vector<double>> cg(solver_control);
545  setup_time += time.wall_time();
546  time_details << "MG build smoother time (CPU/wall) " << time.cpu_time()
547  << "s/" << time.wall_time() << "s\n";
548  pcout << "Total setup time (wall) " << setup_time << "s\n";
549 
550  time.reset();
551  time.start();
552  constraints.set_zero(solution);
553  cg.solve(system_matrix, solution, system_rhs, preconditioner);
554 
555  constraints.distribute(solution);
556 
557  pcout << "Time solve (" << solver_control.last_step() << " iterations)"
558  << (solver_control.last_step() < 10 ? " " : " ") << "(CPU/wall) "
559  << time.cpu_time() << "s/" << time.wall_time() << "s\n";
560  }
