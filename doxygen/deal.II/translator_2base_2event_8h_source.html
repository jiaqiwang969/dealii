204  inline bool
205  Event::test(const Event &event) const
206  {
207  // First, test all_true in this
208  if (all_true)
209  return true;
210 
211  const unsigned int n = flags.size();
212  const unsigned int m = event.flags.size();
213  const unsigned int n_min = (n < m) ? n : m;
214 
215  // Now, if all_true set in the
216  // other, then all must be true
217  // in this
218  if (event.all_true)
219  {
220  // Non existing flags are
221  // always assumed false
222  if (m > n)
223  return false;
224 
225  // Test all flags separately
226  // and return false if one is
227  // not set
228  return std::find(flags.begin(), flags.end(), false) == flags.end();
229  }
230 
231  // Finally, compare each flag
232  // separately
233  for (unsigned int i = 0; i < n_min; ++i)
234  if (event.flags[i] && !flags[i])
235  return false;
236  for (unsigned int i = n_min; i < m; ++i)
237  if (event.flags[i])
238  return false;
239  return true;
240  }
