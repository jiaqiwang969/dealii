1999 template <typename number>
2000 inline void
2001 AffineConstraints<number>::add_line(const size_type line_n)
2002 {
2003  Assert(sorted == false, ExcMatrixIsClosed());
2004 
2005  // the following can happen when we compute with distributed meshes and dof
2006  // handlers and we constrain a degree of freedom whose number we don't have
2007  // locally. if we don't abort here the program will try to allocate several
2008  // terabytes of memory to resize the various arrays below :-)
2009  Assert(line_n != numbers::invalid_size_type, ExcInternalError());
2010  const size_type line_index = calculate_line_index(line_n);
2011 
2012  // check whether line already exists; it may, in which case we can just quit
2013  if (is_constrained(line_n))
2014  return;
