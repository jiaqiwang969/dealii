1016  inline void
1017  Accessor::advance()
1018  {
1019  const auto chunk_size = sparsity_pattern->get_chunk_size();
1020  Assert(chunk_row < chunk_size && chunk_col < chunk_size,
1021  ExcIteratorPastEnd());
1022  Assert(reduced_accessor.row() * chunk_size + chunk_row <
1023  sparsity_pattern->n_rows() &&
1024  reduced_accessor.column() * chunk_size + chunk_col <
1025  sparsity_pattern->n_cols(),
1026  ExcIteratorPastEnd());
1027  if (chunk_size == 1)
1028  {
1029  reduced_accessor.advance();
1030  return;
1031  }
1032 
1033  ++chunk_col;
1034 
1035  // end of chunk
1036  if (chunk_col == chunk_size ||
1037  reduced_accessor.column() * chunk_size + chunk_col ==
1038  sparsity_pattern->n_cols())
1039  {
1040  const auto reduced_row = reduced_accessor.row();
1041  // end of row
1042  if (reduced_accessor.linear_index + 1 ==
1043  reduced_accessor.container->rowstart[reduced_row + 1])
1044  {
1045  ++chunk_row;
1046 
1047  chunk_col = 0;
1048 
1049  // end of chunk rows or end of matrix
1050  if (chunk_row == chunk_size ||
1051  (reduced_row * chunk_size + chunk_row ==
1052  sparsity_pattern->n_rows()))
1053  {
1054  chunk_row = 0;
1055  reduced_accessor.advance();
1056  }
1057  // go back to the beginning of the same reduced row but with
1058  // chunk_row increased by one
1059  else
1060  reduced_accessor.linear_index =
1061  reduced_accessor.container->rowstart[reduced_row];
1062  }
1063  // advance within chunk
1064  else
1065  {
1066  reduced_accessor.advance();
1067  chunk_col = 0;
1068  }
1069  }
1070  }

1202 template <typename ForwardIterator>
1203 void
1204 ChunkSparsityPattern::copy_from(const size_type n_rows,
1205  const size_type n_cols,
1206  const ForwardIterator begin,
1207  const ForwardIterator end,
1208  const size_type chunk_size)
1209 {
1210  Assert(static_cast<size_type>(std::distance(begin, end)) == n_rows,
1211  ExcIteratorRange(std::distance(begin, end), n_rows));
1212 
1213  // first determine row lengths for each row. if the matrix is quadratic,
1214  // then we might have to add an additional entry for the diagonal, if that
1215  // is not yet present. as we have to call compress anyway later on, don't
1216  // bother to check whether that diagonal entry is in a certain row or not
1217  const bool is_square = (n_rows == n_cols);
1218  std::vector<size_type> row_lengths;
1219  row_lengths.reserve(n_rows);
1220  for (ForwardIterator i = begin; i != end; ++i)
1221  row_lengths.push_back(std::distance(i->begin(), i->end()) +
1222  (is_square ? 1 : 0));
1223  reinit(n_rows, n_cols, row_lengths, chunk_size);
1224 
1225  // now enter all the elements into the matrix
1226  size_type row = 0;
1227  using inner_iterator =
1228  typename std::iterator_traits<ForwardIterator>::value_type::const_iterator;
1229  for (ForwardIterator i = begin; i != end; ++i, ++row)
1230  {
1231  const inner_iterator end_of_row = i->end();
1232  for (inner_iterator j = i->begin(); j != end_of_row; ++j)
1233  {
1234  const size_type col =
1235  internal::SparsityPatternTools::get_column_index_from_iterator(*j);
1236  Assert(col < n_cols, ExcInvalidIndex(col, n_cols));
1237 
1238  add(row, col);
1239  }
1240  }
1241 
1242  // finally compress everything. this also sorts the entries within each row
1243  compress();
1244 }
