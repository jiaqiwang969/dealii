94  template <int dim>
95  class Solution : public Function<dim>, protected SolutionBase<dim>
96  {
97  public:
98  virtual double value(const Point<dim> &p,
99  const unsigned int /*component*/ = 0) const override
100  {
101  double sum = 0;
102  for (unsigned int i = 0; i < this->n_source_centers; ++i)
103  {
104  const Tensor<1, dim> x_minus_xi = p - this->source_centers[i];
105  sum +=
106  std::exp(-x_minus_xi.norm_square() / (this->width * this->width));
107  }
108 
109  return sum /
110  std::pow(2. * numbers::PI * this->width * this->width, dim / 2.);
111  }
112 
113  virtual Tensor<1, dim>
114  gradient(const Point<dim> &p,
115  const unsigned int /*component*/ = 0) const override
116  {
117  Tensor<1, dim> sum;
118  for (unsigned int i = 0; i < this->n_source_centers; ++i)
119  {
120  const Tensor<1, dim> x_minus_xi = p - this->source_centers[i];
121 
122  sum +=
123  (-2 / (this->width * this->width) *
124  std::exp(-x_minus_xi.norm_square() / (this->width * this->width)) *
125  x_minus_xi);
126  }
127 
128  return sum /
129  std::pow(2. * numbers::PI * this->width * this->width, dim / 2.);
130  }

192  class RightHandSide : public Function<dim>, protected SolutionBase<dim>
193  {
194  public:
195  virtual double value(const Point<dim> &p,
196  const unsigned int /*component*/ = 0) const override
197  {
198  ConvectionVelocity<dim> convection_velocity;
199  Tensor<1, dim> convection = convection_velocity.value(p);
200  double sum = 0;
201  for (unsigned int i = 0; i < this->n_source_centers; ++i)
202  {
203  const Tensor<1, dim> x_minus_xi = p - this->source_centers[i];
204 
205  sum +=

208  (this->width * this->width) *
209  std::exp(-x_minus_xi.norm_square() / (this->width * this->width)));
210  }
211

532  template <int dim>
533  void HDG<dim>::assemble_system_one_cell(
534  const typename DoFHandler<dim>::active_cell_iterator &cell,
535  ScratchData & scratch,
536  PerTaskData & task_data)
537  {
538  typename DoFHandler<dim>::active_cell_iterator loc_cell(&triangulation,
539  cell->level(),
540  cell->index(),
541  &dof_handler_local);
542 
543  const unsigned int n_q_points =
544  scratch.fe_values_local.get_quadrature().size();
545  const unsigned int n_face_q_points =
546  scratch.fe_face_values_local.get_quadrature().size();
547 
548  const unsigned int loc_dofs_per_cell =
549  scratch.fe_values_local.get_fe().n_dofs_per_cell();
550 
551  const FEValuesExtractors::Vector fluxes(0);
552  const FEValuesExtractors::Scalar scalar(dim);
553 
554  scratch.ll_matrix = 0;
555  scratch.l_rhs = 0;
556  if (!task_data.trace_reconstruct)
557  {
558  scratch.lf_matrix = 0;
559  scratch.fl_matrix = 0;
560  task_data.cell_matrix = 0;
561  task_data.cell_vector = 0;
562  }
563  scratch.fe_values_local.reinit(loc_cell);
564 
565  for (unsigned int q = 0; q < n_q_points; ++q)
566  {
567  const double rhs_value = scratch.right_hand_side.value(
568  scratch.fe_values_local.quadrature_point(q));
569  const Tensor<1, dim> convection = scratch.convection_velocity.value(
570  scratch.fe_values_local.quadrature_point(q));
571  const double JxW = scratch.fe_values_local.JxW(q);
572  for (unsigned int k = 0; k < loc_dofs_per_cell; ++k)
573  {
574  scratch.q_phi[k] = scratch.fe_values_local[fluxes].value(k, q);
575  scratch.q_phi_div[k] =
576  scratch.fe_values_local[fluxes].divergence(k, q);
577  scratch.u_phi[k] = scratch.fe_values_local[scalar].value(k, q);
578  scratch.u_phi_grad[k] =
579  scratch.fe_values_local[scalar].gradient(k, q);
580  }
581  for (unsigned int i = 0; i < loc_dofs_per_cell; ++i)
582  {
583  for (unsigned int j = 0; j < loc_dofs_per_cell; ++j)
584  scratch.ll_matrix(i, j) +=
585  (scratch.q_phi[i] * scratch.q_phi[j] -
586  scratch.q_phi_div[i] * scratch.u_phi[j] +
587  scratch.u_phi[i] * scratch.q_phi_div[j] -
588  (scratch.u_phi_grad[i] * convection) * scratch.u_phi[j]) *
589  JxW;
590  scratch.l_rhs(i) += scratch.u_phi[i] * rhs_value * JxW;
591  }
592  }
593 
594  for (const auto face_no : cell->face_indices())
595  {
596  scratch.fe_face_values_local.reinit(loc_cell, face_no);
597  scratch.fe_face_values.reinit(cell, face_no);
598 
599  if (task_data.trace_reconstruct)
600  scratch.fe_face_values.get_function_values(solution,
601  scratch.trace_values);
602 
603  for (unsigned int q = 0; q < n_face_q_points; ++q)
604  {
605  const double JxW = scratch.fe_face_values.JxW(q);
606  const Point<dim> quadrature_point =
607  scratch.fe_face_values.quadrature_point(q);
608  const Tensor<1, dim> normal =
609  scratch.fe_face_values.normal_vector(q);
610  const Tensor<1, dim> convection =
611  scratch.convection_velocity.value(quadrature_point);
612 
613  const double tau_stab = (5. + std::abs(convection * normal));
614 
615  for (unsigned int k = 0;
616  k < scratch.fe_local_support_on_face[face_no].size();
617  ++k)
618  {
619  const unsigned int kk =
620  scratch.fe_local_support_on_face[face_no][k];
