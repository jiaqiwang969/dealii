706  template <typename T>
707  struct NumberType
708  {
709  static constexpr DEAL_II_ALWAYS_INLINE DEAL_II_CUDA_HOST_DEV const T &
710  value(const T &t)
711  {
712  return t;
713  }
714 
715  // Below are generic functions that allows an overload for any
716  // type U that is transformable to type T. This is particularly
717  // useful when needing to cast exotic number types
718  // (e.g. auto-differentiable or symbolic numbers) to a floating
719  // point one, such as might happen when converting between tensor
720  // types.
721 
722  // Type T is constructible from F.
723  template <typename F>
724  static constexpr DEAL_II_ALWAYS_INLINE DEAL_II_CUDA_HOST_DEV T
725  value(const F &f,
726  typename std::enable_if<
727  !std::is_same<typename std::decay<T>::type,
728  typename std::decay<F>::type>::value &&
729  std::is_constructible<T, F>::value>::type * = nullptr)
730  {
731  return T(f);
732  }
733 
734  // Type T is explicitly convertible (but not constructible) from F.
735  template <typename F>
736  static constexpr DEAL_II_ALWAYS_INLINE T
737  value(const F &f,
738  typename std::enable_if<
739  !std::is_same<typename std::decay<T>::type,
740  typename std::decay<F>::type>::value &&
741  !std::is_constructible<T, F>::value &&
742  is_explicitly_convertible<const F, T>::value>::type * = nullptr)
743  {
744  return static_cast<T>(f);
745  }
746 
747  // Sacado doesn't provide any conversion operators, so we have
748  // to extract the value and perform further conversions from there.
749  // To be safe, we extend this to other possible AD numbers that
750  // might fall into the same category.
751  template <typename F>
752  static T
753  value(const F &f,
754  typename std::enable_if<
755  !std::is_same<typename std::decay<T>::type,
756  typename std::decay<F>::type>::value &&
757  !std::is_constructible<T, F>::value &&
758  !is_explicitly_convertible<const F, T>::value &&
759  Differentiation::AD::is_ad_number<F>::value>::type * = nullptr)
760  {
761  return Differentiation::AD::internal::NumberType<T>::value(f);
762  }
763  };
