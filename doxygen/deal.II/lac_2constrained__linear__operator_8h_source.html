246 template <typename Range, typename Domain, typename Payload>
247 LinearOperator<Range, Domain, Payload>
248 constrained_linear_operator(

250  const LinearOperator<Range, Domain, Payload> & linop)

296 constrained_right_hand_side(

307  const auto Ct = transpose_operator(C);

309  GrowingVectorMemory<Domain> vector_memory;

311  linop.reinit_domain_vector(*k, /*bool fast=*/ false);

315  };
