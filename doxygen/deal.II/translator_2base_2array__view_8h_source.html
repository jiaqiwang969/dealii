371  inline bool
372  is_in_correct_memory_space(const void *const ptr)
373  {
374 #ifndef DEAL_II_COMPILER_CUDA_AWARE
375  (void)ptr;
376  static_assert(std::is_same<MemorySpaceType, MemorySpace::Host>::value,
377  "If the compiler doesn't understand CUDA code, "
378  "the only possible memory space is 'MemorySpace::Host'!");
379  return true;
380 #else
381  cudaPointerAttributes attributes;
382  const cudaError_t cuda_error = cudaPointerGetAttributes(&attributes, ptr);
383  if (cuda_error != cudaErrorInvalidValue)
384  {
385  AssertCuda(cuda_error);
386  if (std::is_same<MemorySpaceType, MemorySpace::Host>::value)
387  return (attributes.type == cudaMemoryTypeHost) ||
388  (attributes.type == cudaMemoryTypeUnregistered);
389  else
390  return attributes.type == cudaMemoryTypeDevice;
391  }
392  else
393  {
394  // ignore and reset the error since host pointers produce an error
395  cudaGetLastError();
396  return std::is_same<MemorySpaceType, MemorySpace::Host>::value;
397  }

711 template <typename Iterator, typename MemorySpaceType = MemorySpace::Host>
712 ArrayView<typename std::remove_reference<
713  typename std::iterator_traits<Iterator>::reference>::type,
714  MemorySpaceType>
715 make_array_view(const Iterator begin, const Iterator end)
716 {
717  static_assert(
718  std::is_same<typename std::iterator_traits<Iterator>::iterator_category,
719  typename std::random_access_iterator_tag>::value,
720  "The provided iterator should be a random access iterator.");
721  Assert(begin <= end,
722  ExcMessage(
723  "The beginning of the array view should be before the end."));
724  Assert(internal::ArrayViewHelper::is_contiguous(begin, end),
725  ExcMessage("The provided range isn't contiguous in memory!"));
726  // the reference type, not the value type, knows the constness of the iterator
727  return ArrayView<typename std::remove_reference<
728  typename std::iterator_traits<Iterator>::reference>::type,
729  MemorySpaceType>(std::addressof(*begin), end - begin);
