1467 template <typename InitFunctionObject, typename LoopFunctionObject>
1468 void
1469 TimeDependent::do_loop(InitFunctionObject init_function,
1470  LoopFunctionObject loop_function,
1471  const TimeSteppingData &timestepping_data,
1472  const Direction direction)
1473 {
1474  // the following functions looks quite
1475  // disrupted due to the recurring switches
1476  // for forward and backward running loops.
1477  //
1478  // I chose to switch at every place where
1479  // it is needed, since it is so easy
1480  // to overlook something when you change
1481  // some code at one place when it needs
1482  // to be changed at a second place, here
1483  // for the other direction, also.
1484 
1485  const unsigned int n_timesteps = timesteps.size();
1486 
1487  // initialize the time steps for
1488  // a round of this loop
1489  for (unsigned int step = 0; step < n_timesteps; ++step)
1490  switch (direction)
1491  {
1492  case forward:
1493  init_function((&*timesteps[step]));
1494  break;
1495  case backward:
1496  init_function((&*timesteps[n_timesteps - step - 1]));
1497  break;
1498  };
1499 
1500 
1501  // wake up the first few time levels
1502  for (int step = -static_cast<int>(timestepping_data.look_ahead); step < 0;
1503  ++step)
1504  for (int look_ahead = 0;
1505  look_ahead <= static_cast<int>(timestepping_data.look_ahead);
1506  ++look_ahead)
1507  switch (direction)
1508  {
1509  case forward:
1510  if (step + look_ahead >= 0)
1511  timesteps[step + look_ahead]->wake_up(look_ahead);
1512  break;
1513  case backward:
1514  if (n_timesteps - (step + look_ahead) < n_timesteps)
1515  timesteps[n_timesteps - (step + look_ahead)]->wake_up(look_ahead);
1516  break;
1517  };
1518 
1519 
1520  for (unsigned int step = 0; step < n_timesteps; ++step)
1521  {
1522  // first thing: wake up the
1523  // timesteps ahead as necessary
1524  for (unsigned int look_ahead = 0;
1525  look_ahead <= timestepping_data.look_ahead;
1526  ++look_ahead)
1527  switch (direction)
1528  {
1529  case forward:
1530  if (step + look_ahead < n_timesteps)
1531  timesteps[step + look_ahead]->wake_up(look_ahead);
1532  break;
1533  case backward:
1534  if (n_timesteps > (step + look_ahead))
1535  timesteps[n_timesteps - (step + look_ahead) - 1]->wake_up(
1536  look_ahead);
1537  break;
1538  };
1539 
1540 
1541  // actually do the work
1542  switch (direction)
1543  {
1544  case forward:
1545  loop_function((&*timesteps[step]));
1546  break;
1547  case backward:
1548  loop_function((&*timesteps[n_timesteps - step - 1]));
1549  break;
1550  };
1551 
1552  // let the timesteps behind sleep
1553  for (unsigned int look_back = 0; look_back <= timestepping_data.look_back;
1554  ++look_back)
1555  switch (direction)
1556  {
1557  case forward:
1558  if (step >= look_back)
1559  timesteps[step - look_back]->sleep(look_back);
1560  break;
1561  case backward:
1562  if (n_timesteps - (step - look_back) <= n_timesteps)
1563  timesteps[n_timesteps - (step - look_back) - 1]->sleep(
1564  look_back);
1565  break;
1566  }
1567  }
1568 
1569  // make the last few timesteps sleep
1570  for (int step = n_timesteps;
1571  step < static_cast<int>(n_timesteps + timestepping_data.look_back);
1572  ++step)
1573  for (int look_back = 0;
1574  look_back <= static_cast<int>(timestepping_data.look_back);
1575  ++look_back)
1576  switch (direction)
1577  {
1578  case forward:
1579  if ((step - look_back >= 0) &&
1580  (step - look_back < static_cast<int>(n_timesteps)))
1581  timesteps[step - look_back]->sleep(look_back);
1582  break;
1583  case backward:
1584  if ((step - look_back >= 0) &&
1585  (step - look_back < static_cast<int>(n_timesteps)))
1586  timesteps[n_timesteps - (step - look_back) - 1]->sleep(look_back);
1587  break;
1588  };
1589 }
