1159 template <int dim, int patch_dim, int spacedim, int patch_spacedim>
1160 template <int dim2, int spacedim2>
1161 void
1162 DataOut_DoFData<dim, patch_dim, spacedim, patch_spacedim>::merge_patches(
1163  const DataOut_DoFData<dim2, patch_dim, spacedim2, patch_spacedim> &source,
1164  const Point<patch_spacedim> & shift)
1165 {
1166  const std::vector<Patch> &source_patches = source.get_patches();
1167  Assert((patches.size() != 0) && (source_patches.size() != 0),
1168  ExcMessage("When calling this function, both the current "
1169  "object and the one being merged need to have a "
1170  "nonzero number of patches associated with it. "
1171  "Either you called this function on objects that "
1172  "are empty, or you may have forgotten to call "
1173  "the 'build_patches()' function."));
1174  // Check equality of component names
1175  Assert(get_dataset_names() == source.get_dataset_names(),
1176  Exceptions::DataOutImplementation::ExcIncompatibleDatasetNames());
1177 
1178  // Make sure patches are compatible. Ideally, we would check that all input
1179  // patches from both collections are all compatible, but we'll be content
1180  // with checking that just the first ones from both sources are.
1181  //
1182  // We check compatibility by testing that both sets of patches result
1183  // from the same number of subdivisions, and that they have the same
1184  // number of source vectors (they really should, since we already checked
1185  // that there are the same number of source components above, but you
1186  // never know). This implies that the data should have the same number of
1187  // columns. They should really have the same number of rows as well,
1188  // but depending on whether a patch has points included or not, the
1189  // number of rows may or may not include coordinates for the points,
1190  // and the comparison has to account for that because in each source
1191  // stream, the patches may include some that have points included.
1192  Assert(patches[0].n_subdivisions == source_patches[0].n_subdivisions,
1193  Exceptions::DataOutImplementation::ExcIncompatiblePatchLists());
1194  Assert(patches[0].data.n_cols() == source_patches[0].data.n_cols(),
1195  Exceptions::DataOutImplementation::ExcIncompatiblePatchLists());
1196  Assert((patches[0].data.n_rows() +
1197  (patches[0].points_are_available ? 0 : patch_spacedim)) ==
1198  (source_patches[0].data.n_rows() +
1199  (source_patches[0].points_are_available ? 0 : patch_spacedim)),
1200  Exceptions::DataOutImplementation::ExcIncompatiblePatchLists());
1201 
1202  // check equality of the vector data
1203  // specifications
1204  Assert(get_nonscalar_data_ranges().size() ==
1205  source.get_nonscalar_data_ranges().size(),
1206  ExcMessage("Both sources need to declare the same components "
1207  "as vectors."));
1208  for (unsigned int i = 0; i < get_nonscalar_data_ranges().size(); ++i)
1209  {
1210  Assert(std::get<0>(get_nonscalar_data_ranges()[i]) ==
1211  std::get<0>(source.get_nonscalar_data_ranges()[i]),
1212  ExcMessage("Both sources need to declare the same components "
1213  "as vectors."));
1214  Assert(std::get<1>(get_nonscalar_data_ranges()[i]) ==
1215  std::get<1>(source.get_nonscalar_data_ranges()[i]),
1216  ExcMessage("Both sources need to declare the same components "
1217  "as vectors."));
1218  Assert(std::get<2>(get_nonscalar_data_ranges()[i]) ==
1219  std::get<2>(source.get_nonscalar_data_ranges()[i]),
1220  ExcMessage("Both sources need to declare the same components "
1221  "as vectors."));
1222  }
1223 
1224  // merge patches. store old number
1225  // of elements, since we need to
1226  // adjust patch numbers, etc
1227  // afterwards
