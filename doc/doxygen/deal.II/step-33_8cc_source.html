107  template <typename InputVector>
108  static typename InputVector::value_type
109  compute_kinetic_energy(const InputVector &W)
110  {
111  typename InputVector::value_type kinetic_energy = 0;
112  for (unsigned int d = 0; d < dim; ++d)
113  kinetic_energy +=
114  W[first_momentum_component + d] * W[first_momentum_component + d];
115  kinetic_energy *= 1. / (2 * W[density_component]);
116 
117  return kinetic_energy;

131  template <typename InputVector>
132  static void compute_flux_matrix(const InputVector &W,
133  ndarray<typename InputVector::value_type,
134  EulerEquations<dim>::n_components,
135  dim> & flux)
136  {
137  const typename InputVector::value_type pressure = compute_pressure(W);
138 
139  for (unsigned int d = 0; d < dim; ++d)
140  {
141  for (unsigned int e = 0; e < dim; ++e)
142  flux[first_momentum_component + d][e] =
143  W[first_momentum_component + d] *
144  W[first_momentum_component + e] / W[density_component];
145 
146  flux[first_momentum_component + d][d] += pressure;
147  }
148 
149  for (unsigned int d = 0; d < dim; ++d)
150  flux[density_component][d] = W[first_momentum_component + d];
151 
152  for (unsigned int d = 0; d < dim; ++d)
153  flux[energy_component][d] = W[first_momentum_component + d] /
154  W[density_component] *
155  (W[energy_component] + pressure);
156  }

219  template <typename DataVector>
220  static void
221  compute_Wminus(const std::array<BoundaryKind, n_components> &boundary_kind,
222  const Tensor<1, dim> & normal_vector,
223  const DataVector & Wplus,
224  const Vector<double> &boundary_values,
225  const DataVector & Wminus)
226  {
227  for (unsigned int c = 0; c < n_components; c++)
228  switch (boundary_kind[c])
229  {
230  case inflow_boundary:
231  {
232  Wminus[c] = boundary_values(c);
233  break;
234  }
235 
236  case outflow_boundary:
237  {
238  Wminus[c] = Wplus[c];
239  break;
240  }
241 
242  case pressure_boundary:
243  {
244  const typename DataVector::value_type density =
245  (boundary_kind[density_component] == inflow_boundary ?
246  boundary_values(density_component) :
247  Wplus[density_component]);
248 
249  typename DataVector::value_type kinetic_energy = 0;
250  for (unsigned int d = 0; d < dim; ++d)
251  if (boundary_kind[d] == inflow_boundary)
252  kinetic_energy += boundary_values(d) * boundary_values(d);
253  else
254  kinetic_energy += Wplus[d] * Wplus[d];
255  kinetic_energy *= 1. / 2. / density;
256 
257  Wminus[c] =
258  boundary_values(c) / (gas_gamma - 1.0) + kinetic_energy;
259 
260  break;
261  }
262 
263  case no_penetration_boundary:
264  {
265  typename DataVector::value_type vdotn = 0;
266  for (unsigned int d = 0; d < dim; d++)
267  {
268  vdotn += Wplus[d] * normal_vector[d];
269  }
270 
271  Wminus[c] = Wplus[c] - 2.0 * vdotn * normal_vector[c];
272  break;
273  }
274 
275  default:
276  Assert(false, ExcNotImplemented());
277  }

350  template <int dim>
351  void EulerEquations<dim>::Postprocessor::evaluate_vector_field(
352  const DataPostprocessorInputs::Vector<dim> &inputs,
353  std::vector<Vector<double>> & computed_quantities) const
354  {
355  const unsigned int n_quadrature_points = inputs.solution_values.size();
356 
357  if (do_schlieren_plot == true)
358  Assert(inputs.solution_gradients.size() == n_quadrature_points,
359  ExcInternalError());
360 
361  Assert(computed_quantities.size() == n_quadrature_points,
362  ExcInternalError());
363 
364  Assert(inputs.solution_values[0].size() == n_components,
365  ExcInternalError());
366 
367  if (do_schlieren_plot == true)
368  {
369  Assert(computed_quantities[0].size() == dim + 2, ExcInternalError());
370  }
371  else
372  {
373  Assert(computed_quantities[0].size() == dim + 1, ExcInternalError());
374  }
375 
376  for (unsigned int q = 0; q < n_quadrature_points; ++q)
377  {
378  const double density = inputs.solution_values[q](density_component);
379 
380  for (unsigned int d = 0; d < dim; ++d)
381  computed_quantities[q](d) =
382  inputs.solution_values[q](first_momentum_component + d) / density;
383 
384  computed_quantities[q](dim) =
385  compute_pressure(inputs.solution_values[q]);
386 
387  if (do_schlieren_plot == true)
388  computed_quantities[q](dim + 1) =
389  inputs.solution_gradients[q][density_component] *
390  inputs.solution_gradients[q][density_component];
391  }

1145  void ConservationLaw<dim>::assemble_cell_term(
1146  const FEValues<dim> & fe_v,
1147  const std::vector<types::global_dof_index> &dof_indices)
1148  {
1149  const unsigned int dofs_per_cell = fe_v.dofs_per_cell;
1150  const unsigned int n_q_points = fe_v.n_quadrature_points;
1151 
1152  Table<2, Sacado::Fad::DFad<double>> W(n_q_points,
1153  EulerEquations<dim>::n_components);
1154 
1155  Table<2, double> W_old(n_q_points, EulerEquations<dim>::n_components);
1156 
1157  Table<3, Sacado::Fad::DFad<double>> grad_W(
1158  n_q_points, EulerEquations<dim>::n_components, dim);
1159 
1160  Table<3, double> grad_W_old(n_q_points,
1161  EulerEquations<dim>::n_components,
1162  dim);
1163 
1164  std::vector<double> residual_derivatives(dofs_per_cell);
1165 
1166  std::vector<Sacado::Fad::DFad<double>> independent_local_dof_values(
1167  dofs_per_cell);
1168  for (unsigned int i = 0; i < dofs_per_cell; ++i)
1169  independent_local_dof_values[i] = current_solution(dof_indices[i]);
1170 
1171  for (unsigned int i = 0; i < dofs_per_cell; ++i)
1172  independent_local_dof_values[i].diff(i, dofs_per_cell);
1173 
1174  for (unsigned int q = 0; q < n_q_points; ++q)
1175  for (unsigned int c = 0; c < EulerEquations<dim>::n_components; ++c)
1176  {
1177  W[q][c] = 0;
1178  W_old[q][c] = 0;
1179  for (unsigned int d = 0; d < dim; ++d)
1180  {
1181  grad_W[q][c][d] = 0;
1182  grad_W_old[q][c][d] = 0;
1183  }
1184  }
1185 
1186  for (unsigned int q = 0; q < n_q_points; ++q)
1187  for (unsigned int i = 0; i < dofs_per_cell; ++i)
1188  {
1189  const unsigned int c =
1190  fe_v.get_fe().system_to_component_index(i).first;
1191 
1192  W[q][c] += independent_local_dof_values[i] *
1193  fe_v.shape_value_component(i, q, c);
1194  W_old[q][c] +=
1195  old_solution(dof_indices[i]) * fe_v.shape_value_component(i, q, c);
1196 
1197  for (unsigned int d = 0; d < dim; d++)
1198  {
1199  grad_W[q][c][d] += independent_local_dof_values[i] *
1200  fe_v.shape_grad_component(i, q, c)[d];
1201  grad_W_old[q][c][d] += old_solution(dof_indices[i]) *
1202  fe_v.shape_grad_component(i, q, c)[d];
1203  }
1204  }
1205 
1206 
1207 
1208  std::vector<ndarray<Sacado::Fad::DFad<double>,
1209  EulerEquations<dim>::n_components,
1210  dim>>
1211  flux(n_q_points);
1212 
1213  std::vector<ndarray<double, EulerEquations<dim>::n_components, dim>>
1214  flux_old(n_q_points);
1215 
1216  std::vector<
1217  std::array<Sacado::Fad::DFad<double>, EulerEquations<dim>::n_components>>
1218  forcing(n_q_points);
1219 
1220  std::vector<std::array<double, EulerEquations<dim>::n_components>>
1221  forcing_old(n_q_points);
1222 
1223  for (unsigned int q = 0; q < n_q_points; ++q)
1224  {
1225  EulerEquations<dim>::compute_flux_matrix(W_old[q], flux_old[q]);
1226  EulerEquations<dim>::compute_forcing_vector(W_old[q], forcing_old[q]);
1227  EulerEquations<dim>::compute_flux_matrix(W[q], flux[q]);
1228  EulerEquations<dim>::compute_forcing_vector(W[q], forcing[q]);
1229  }
1230 
1231 
1232  for (unsigned int i = 0; i < fe_v.dofs_per_cell; ++i)
1233  {
1234  Sacado::Fad::DFad<double> R_i = 0;
1235 
1236  const unsigned int component_i =
1237  fe_v.get_fe().system_to_component_index(i).first;
1238 
1239 
1240  for (unsigned int point = 0; point < fe_v.n_quadrature_points; ++point)
1241  {
1242  if (parameters.is_stationary == false)
1243  R_i += 1.0 / parameters.time_step *
1244  (W[point][component_i] - W_old[point][component_i]) *
1245  fe_v.shape_value_component(i, point, component_i) *
1246  fe_v.JxW(point);
1247 
1248  for (unsigned int d = 0; d < dim; d++)
1249  R_i -=
1250  (parameters.theta * flux[point][component_i][d] +
1251  (1.0 - parameters.theta) * flux_old[point][component_i][d]) *
1252  fe_v.shape_grad_component(i, point, component_i)[d] *
1253  fe_v.JxW(point);
1254 
1255  for (unsigned int d = 0; d < dim; d++)
1256  R_i +=
1257  1.0 *
1258  std::pow(fe_v.get_cell()->diameter(),
1259  parameters.diffusion_power) *
1260  (parameters.theta * grad_W[point][component_i][d] +
1261  (1.0 - parameters.theta) * grad_W_old[point][component_i][d]) *
1262  fe_v.shape_grad_component(i, point, component_i)[d] *
1263  fe_v.JxW(point);
1264 
1265  R_i -=
1266  (parameters.theta * forcing[point][component_i] +
1267  (1.0 - parameters.theta) * forcing_old[point][component_i]) *
1268  fe_v.shape_value_component(i, point, component_i) *
1269  fe_v.JxW(point);
1270  }
1271 
1272  for (unsigned int k = 0; k < dofs_per_cell; ++k)
1273  residual_derivatives[k] = R_i.fastAccessDx(k);
1274  system_matrix.add(dof_indices[i], dof_indices, residual_derivatives);
1275  right_hand_side(dof_indices[i]) -= R_i.val();
1276  }
