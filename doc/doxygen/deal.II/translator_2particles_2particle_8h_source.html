507  template <int dim, int spacedim>
508  inline void
509  Particle<dim, spacedim>::set_property_pool(
510  PropertyPool<dim, spacedim> &new_property_pool)
511  {
512  // First, we do want to save any properties that may
513  // have previously been set, and copy them over to the memory allocated
514  // on the new pool.
515  //
516  // It is possible that a particle currently has no properties -- for
517  // example if it has been created without an associated property
518  // pool (i.e., uses the default global pool which does not store any
519  // properties) but that the new pool has properties. In that case,
520  // there is simply nothing to transfer -- but the register_particle()
521  // call here will make sure that the newly allocated properties are
522  // zero-initialized.
523  const typename PropertyPool<dim, spacedim>::Handle new_handle =
524  new_property_pool.register_particle();
525 
526  const Point<spacedim> location = get_location();
527  const Point<dim> reference_location = get_reference_location();
528  const types::particle_index id = get_id();
529 
530  if ( /* old pool */ has_properties())
531  {
532  ArrayView<const double> old_properties = this->get_properties();
533  ArrayView<double> new_properties =
534  new_property_pool.get_properties(new_handle);
535  std::copy(old_properties.cbegin(),
536  old_properties.cend(),
537  new_properties.begin());
538  }
539 
540  // Now release the old memory handle
541  property_pool->deregister_particle(property_pool_handle);
542 
543 
544  // Then set the pointer to the property pool we want to use. Also set the
545  // handle to any properties.
546  property_pool = &new_property_pool;
547  property_pool_handle = new_handle;
548 
549  // Now also store the saved locations
550  set_location(location);
551  set_reference_location(reference_location);
552  set_id(id);
553  }

570  inline bool
571  Particle<dim, spacedim>::has_properties() const
572  {
573  // Particles always have a property pool associated with them,
574  // but we can access properties only if there is a valid handle.
