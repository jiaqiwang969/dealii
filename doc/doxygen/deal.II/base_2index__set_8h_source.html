1484 inline std::ptrdiff_t
1485 IndexSet::ElementIterator::
1486 operator-(const IndexSet::ElementIterator &other) const
1487 {
1488  Assert(index_set == other.index_set,
1489  ExcMessage(
1490  "Can not compare iterators belonging to different IndexSets"));
1491  if (*this == other)
1492  return 0;
1493  if (!(*this < other))
1494  return -(other - *this);
1495 
1496  // only other can be equal to end() because of the checks above.
1497  Assert(is_valid(), ExcInternalError());
1498 
1499  // Note: we now compute how far advance *this in "*this < other" to get other,
1500  // so we need to return -c at the end.
1501 
1502  // first finish the current range:
1503  std::ptrdiff_t c = index_set->ranges[range_idx].end - idx;
1504 
1505  // now walk in steps of ranges (need to start one behind our current one):
1506  for (size_type range = range_idx + 1;
1507  range < index_set->ranges.size() && range <= other.range_idx;
1508  ++range)
1509  c += index_set->ranges[range].end - index_set->ranges[range].begin;
1510 
1511  Assert(
1512  other.range_idx < index_set->ranges.size() ||

1611 IndexSet::at(const size_type global_index) const
1612 {

1615 
1616  if (ranges.empty())
1617  return end();
1618

1620  ranges.begin() + largest_range;
1621 
1622  Range r(global_index, global_index + 1);
1623  // This optimization makes the bounds for lower_bound smaller by checking
1624  // the largest range first.
1625  std::vector<Range>::const_iterator range_begin, range_end;
1626  if (global_index < main_range->begin)
1627  {
1628  range_begin = ranges.begin();
1629  range_end = main_range;
1630  }
1631  else
1632  {
1633  range_begin = main_range;
1634  range_end = ranges.end();
1635  }
1636 
1637  // This will give us the first range p=[a,b[ with b>=global_index using
1638  // a binary search
1639  const std::vector<Range>::const_iterator p =
1640  Utilities::lower_bound(range_begin, range_end, r, Range::end_compare);
1641 
1642  // We couldn't find a range, which means we have no range that contains
1643  // global_index and also no range behind it, meaning we need to return end().
1644  if (p == ranges.end())
1645  return end();

1647  // Finally, we can have two cases: Either global_index is not in [a,b[,
1648  // which means we need to return an iterator to a because global_index, ...,
1649  // a-1 is not in the IndexSet (if branch). Alternatively, global_index is in
1650  // [a,b[ and we will return an iterator pointing directly at global_index
1651  // (else branch).
1652  if (global_index < p->begin)
1653  return {this, static_cast<size_type>(p - ranges.begin()), p->begin};
1654  else
1655  return {this, static_cast<size_type>(p - ranges.begin()), global_index};

1782 template <typename ForwardIterator>
1783 inline void
1784 IndexSet::add_indices(const ForwardIterator &begin, const ForwardIterator &end)
1785 {
1786  if (begin == end)
1787  return;
1788 
1789  // identify ranges in the given iterator range by checking whether some
1790  // indices happen to be consecutive. to avoid quadratic complexity when
1791  // calling add_range many times (as add_range() going into the middle of an
1792  // already existing range must shift entries around), we first collect a
1793  // vector of ranges.
1794  std::vector<std::pair<size_type, size_type>> tmp_ranges;
1795  bool ranges_are_sorted = true;
1796  for (ForwardIterator p = begin; p != end;)
1797  {
1798  const size_type begin_index = *p;
1799  size_type end_index = begin_index + 1;
1800  ForwardIterator q = p;
1801  ++q;
1802  while ((q != end) && (*q == end_index))
1803  {
1804  ++end_index;
1805  ++q;
1806  }
1807 
1808  tmp_ranges.emplace_back(begin_index, end_index);
1809  p = q;
1810 
1811  // if the starting index of the next go-around of the for loop is less
1812  // than the end index of the one just identified, then we will have at
1813  // least one pair of ranges that are not sorted, and consequently the
1814  // whole collection of ranges is not sorted.
1815  if (p != end && *p < end_index)
1816  ranges_are_sorted = false;
1817  }
1818 
1819  if (!ranges_are_sorted)
1820  std::sort(tmp_ranges.begin(), tmp_ranges.end());
1821 
1822  // if we have many ranges, we first construct a temporary index set (where
1823  // we add ranges in a consecutive way, so fast), otherwise, we work with
1824  // add_range(). the number 9 is chosen heuristically given the fact that
1825  // there are typically up to 8 independent ranges when adding the degrees of
1826  // freedom on a 3D cell or 9 when adding degrees of freedom of faces. if
1827  // doing cell-by-cell additions, we want to avoid repeated calls to
1828  // IndexSet::compress() which gets called upon merging two index sets, so we
1829  // want to be in the other branch then.
1830  if (tmp_ranges.size() > 9)
1831  {
1832  IndexSet tmp_set(size());
1833  tmp_set.ranges.reserve(tmp_ranges.size());
1834  for (const auto &i : tmp_ranges)
1835  tmp_set.add_range(i.first, i.second);
1836  this->add_indices(tmp_set);
1837  }
1838  else
1839  for (const auto &i : tmp_ranges)
1840  add_range(i.first, i.second);
1841 }

1845 inline bool
1846 IndexSet::is_element(const size_type index) const
1847 {
1848  if (ranges.empty() == false)
1849  {
1850  compress();
1851 
1852  // fast check whether the index is in the largest range
1853  Assert(largest_range < ranges.size(), ExcInternalError());
1854  if (index >= ranges[largest_range].begin &&
1855  index < ranges[largest_range].end)
1856  return true;
1857 
1858  // get the element after which we would have to insert a range that
1859  // consists of all elements from this element to the end of the index
1860  // range plus one. after this call we know that if p!=end() then
1861  // p->begin<=index unless there is no such range at all
1862  //
1863  // if the searched for element is an element of this range, then we're
1864  // done. otherwise, the element can't be in one of the following ranges
1865  // because otherwise p would be a different iterator
1866  //
1867  // since we already know the position relative to the largest range (we
1868  // called compress!), we can perform the binary search on ranges with
1869  // lower/higher number compared to the largest range
1870  std::vector<Range>::const_iterator p = std::upper_bound(
1871  ranges.begin() +
1872  (index < ranges[largest_range].begin ? 0 : largest_range + 1),
1873  index < ranges[largest_range].begin ? ranges.begin() + largest_range :
1874  ranges.end(),
1875  Range(index, size() + 1));
1876 
1877  if (p == ranges.begin())
1878  return ((index >= p->begin) && (index < p->end));
1879 
1880  Assert((p == ranges.end()) || (p->begin > index), ExcInternalError());
1881 
1882  // now move to that previous range
1883  --p;
1884  Assert(p->begin <= index, ExcInternalError());
1885 
1886  return (p->end > index);
1887  }
1888 
1889  // didn't find this index, so it's not in the set
1890  return false;
1891 }
