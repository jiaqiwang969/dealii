487  const FEValuesBase<dim> &fetestn = nbeta > 0. ? fetest2 : fetest1;

489  for (unsigned j = 0; j < n1; ++j)
490  {
491  if (fe1.get_fe().is_primitive())
492  {
493  M1(i, j) += dx_nbeta * fe.shape_value(j, k) *
494  fetest.shape_value(i, k);
495  M2(i, j) -= dx_nbeta * fe.shape_value(j, k) *
496  fetestn.shape_value(i, k);
497  }
498  else

500  for (unsigned int d = 0; d < fe1.get_fe().n_components();

504  fe.shape_value_component(j, k, d) *
