1702  std::pair<std::array<Point<MeshIteratorType::AccessorType::space_dimension>,
1703  n_default_points_per_cell<MeshIteratorType>()>,
1704  std::array<double, n_default_points_per_cell<MeshIteratorType>()>>
1705  get_default_points_and_weights(const MeshIteratorType &iterator,
1706  const bool with_interpolation)
1707  {
1708  const int dim = MeshIteratorType::AccessorType::structure_dimension;
1709  const int spacedim = MeshIteratorType::AccessorType::space_dimension;
1710  constexpr std::size_t points_per_cell =
1711  n_default_points_per_cell<MeshIteratorType>();

1713  std::pair<std::array<Point<spacedim>, points_per_cell>,
1714  std::array<double, points_per_cell>>
1715  points_weights;
1716 
1717 
1718  // note that the exact weights are chosen such as to minimize the
1719  // distortion of the four new quads from the optimal shape; their
1720  // derivation and values is copied over from the
1721  // interpolation function in the mapping
1722  switch (dim)
1723  {
1724  case 1:
1725  Assert(points_weights.first.size() == 2, ExcInternalError());
1726  Assert(points_weights.second.size() == 2, ExcInternalError());
1727  points_weights.first[0] = iterator->vertex(0);
1728  points_weights.second[0] = .5;
1729  points_weights.first[1] = iterator->vertex(1);
1730  points_weights.second[1] = .5;
1731  break;
1732  case 2:
1733  Assert(points_weights.first.size() == 8, ExcInternalError());
1734  Assert(points_weights.second.size() == 8, ExcInternalError());
1735 
1736  for (unsigned int i = 0; i < 4; ++i)
1737  {
1738  points_weights.first[i] = iterator->vertex(i);
1739  points_weights.first[4 + i] =
1740  (iterator->line(i)->has_children() ?
1741  iterator->line(i)->child(0)->vertex(1) :
1742  iterator->line(i)->get_manifold().get_new_point_on_line(
1743  iterator->line(i)));
1744  }
1745 
1746  if (with_interpolation)
1747  {
1748  std::fill(points_weights.second.begin(),
1749  points_weights.second.begin() + 4,
1750  -0.25);
1751  std::fill(points_weights.second.begin() + 4,
1752  points_weights.second.end(),
1753  0.5);
1754  }
1755  else
1756  std::fill(points_weights.second.begin(),
1757  points_weights.second.end(),
1758  1.0 / 8.0);
1759  break;
1760  case 3:
1761  {
1762  TriaIterator<TriaAccessor<3, 3, 3>> hex =
1763  static_cast<TriaIterator<TriaAccessor<3, 3, 3>>>(iterator);
1764  const unsigned int np = GeometryInfo<dim>::vertices_per_cell +
1765  GeometryInfo<dim>::lines_per_cell +
1766  GeometryInfo<dim>::faces_per_cell;
1767  Assert(points_weights.first.size() == np, ExcInternalError());
1768  Assert(points_weights.second.size() == np, ExcInternalError());
1769  auto *sp3 = reinterpret_cast<
1770  std::array<Point<3>, n_default_points_per_cell<decltype(hex)>()>

1788  }

1790  ++i, ++j)

1794  hex->line(i)->child(0)->vertex(1) :

1798  }

1800  ++i, ++j)

1804  hex->quad(i)->isotropic_child(0)->vertex(3) :
1805  hex->quad(i)->get_manifold().get_new_point_on_quad(
1806  hex->quad(i)));

1808  }

1821  }
