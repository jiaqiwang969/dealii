1219  inline void
1220  VectorBase::extract_subvector_to(const ForwardIterator indices_begin,
1221  const ForwardIterator indices_end,
1222  OutputIterator values_begin) const
1223  {
1224  const PetscInt n_idx = static_cast<PetscInt>(indices_end - indices_begin);
1225  if (n_idx == 0)
1226  return;
1227 
1228  // if we are dealing
1229  // with a parallel vector
1230  if (ghosted)
1231  {
1232  // there is the possibility
1233  // that the vector has
1234  // ghost elements. in that
1235  // case, we first need to
1236  // figure out which
1237  // elements we own locally,
1238  // then get a pointer to
1239  // the elements that are
1240  // stored here (both the
1241  // ones we own as well as
1242  // the ghost elements). in
1243  // this array, the locally
1244  // owned elements come
1245  // first followed by the
1246  // ghost elements whose
1247  // position we can get from
1248  // an index set
1249  PetscInt begin, end;
1250  PetscErrorCode ierr = VecGetOwnershipRange(vector, &begin, &end);
1251  AssertThrow(ierr == 0, ExcPETScError(ierr));
1252 
1253  Vec locally_stored_elements = nullptr;
1254  ierr = VecGhostGetLocalForm(vector, &locally_stored_elements);
1255  AssertThrow(ierr == 0, ExcPETScError(ierr));
1256 
1257  PetscInt lsize;
1258  ierr = VecGetSize(locally_stored_elements, &lsize);
1259  AssertThrow(ierr == 0, ExcPETScError(ierr));
1260 
1261  PetscScalar *ptr;
1262  ierr = VecGetArray(locally_stored_elements, &ptr);
1263  AssertThrow(ierr == 0, ExcPETScError(ierr));
1264 
1265  for (PetscInt i = 0; i < n_idx; ++i)
1266  {
1267  const unsigned int index = *(indices_begin + i);
1268  if (index >= static_cast<unsigned int>(begin) &&
1269  index < static_cast<unsigned int>(end))
1270  {
1271  // local entry
1272  *(values_begin + i) = *(ptr + index - begin);
1273  }
1274  else
1275  {
1276  // ghost entry
1277  const unsigned int ghostidx =
1278  ghost_indices.index_within_set(index);
1279 
1280  AssertIndexRange(ghostidx + end - begin, lsize);
1281  *(values_begin + i) = *(ptr + ghostidx + end - begin);
1282  }
1283  }
1284 
1285  ierr = VecRestoreArray(locally_stored_elements, &ptr);
1286  AssertThrow(ierr == 0, ExcPETScError(ierr));
1287 
1288  ierr = VecGhostRestoreLocalForm(vector, &locally_stored_elements);
1289  AssertThrow(ierr == 0, ExcPETScError(ierr));
1290  }
1291  // if the vector is local or the
1292  // caller, then simply access the
1293  // element we are interested in
1294  else
1295  {
1296  PetscInt begin, end;
1297  PetscErrorCode ierr = VecGetOwnershipRange(vector, &begin, &end);
1298  AssertThrow(ierr == 0, ExcPETScError(ierr));
1299 
1300  PetscScalar *ptr;
1301  ierr = VecGetArray(vector, &ptr);
1302  AssertThrow(ierr == 0, ExcPETScError(ierr));
1303

1305  {
1306  const unsigned int index = *(indices_begin + i);
1307 
1308  Assert(index >= static_cast<unsigned int>(begin) &&
1309  index < static_cast<unsigned int>(end),
1310  ExcInternalError());

1313  }
