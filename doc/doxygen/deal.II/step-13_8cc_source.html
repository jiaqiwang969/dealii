392  template <int dim>
393  void Solver<dim>::local_assemble_matrix(
394  const typename DoFHandler<dim>::active_cell_iterator &cell,
395  AssemblyScratchData & scratch_data,
396  AssemblyCopyData & copy_data) const
397  {
398  const unsigned int dofs_per_cell = fe->n_dofs_per_cell();
399  const unsigned int n_q_points = quadrature->size();
400 
401  copy_data.cell_matrix.reinit(dofs_per_cell, dofs_per_cell);
402 
403  copy_data.local_dof_indices.resize(dofs_per_cell);
404 
405  scratch_data.fe_values.reinit(cell);
406 
407  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
408  for (unsigned int i = 0; i < dofs_per_cell; ++i)
409  for (unsigned int j = 0; j < dofs_per_cell; ++j)
410  copy_data.cell_matrix(i, j) +=
411  (scratch_data.fe_values.shape_grad(i, q_point) *
412  scratch_data.fe_values.shape_grad(j, q_point) *
413  scratch_data.fe_values.JxW(q_point));
414 
415  cell->get_dof_indices(copy_data.local_dof_indices);

507  template <int dim>
508  void PrimalSolver<dim>::assemble_rhs(Vector<double> &rhs) const
509  {
510  FEValues<dim> fe_values(*this->fe,
511  *this->quadrature,
512  update_values | update_quadrature_points |
513  update_JxW_values);
514 
515  const unsigned int dofs_per_cell = this->fe->n_dofs_per_cell();
516  const unsigned int n_q_points = this->quadrature->size();
517 
518  Vector<double> cell_rhs(dofs_per_cell);
519  std::vector<double> rhs_values(n_q_points);
520  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
521 
522  for (const auto &cell : this->dof_handler.active_cell_iterators())
523  {
524  cell_rhs = 0;
525  fe_values.reinit(cell);
526  rhs_function->value_list(fe_values.get_quadrature_points(),
527  rhs_values);
528 
529  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
530  for (unsigned int i = 0; i < dofs_per_cell; ++i)
531  cell_rhs(i) += fe_values.shape_value(i, q_point) *
532  rhs_values[q_point] *
533  fe_values.JxW(q_point);
534 
535  cell->get_dof_indices(local_dof_indices);
536  for (unsigned int i = 0; i < dofs_per_cell; ++i)
537  rhs(local_dof_indices[i]) += cell_rhs(i);
538  };
539  }
