200  {

237  template <int dim>
238  void
239  Estimator<dim>::face(MeshWorker::DoFInfo<dim> & dinfo1,
240  MeshWorker::DoFInfo<dim> & dinfo2,
241  typename MeshWorker::IntegrationInfo<dim> &info1,
242  typename MeshWorker::IntegrationInfo<dim> &info2) const
243  {
244  const FEValuesBase<dim> & fe = info1.fe_values();
245  const std::vector<double> & uh1 = info1.values[0][0];
246  const std::vector<double> & uh2 = info2.values[0][0];
247  const std::vector<Tensor<1, dim>> &Duh1 = info1.gradients[0][0];
248  const std::vector<Tensor<1, dim>> &Duh2 = info2.gradients[0][0];
249 
250  const unsigned int degree = fe.get_fe().tensor_degree();
251  const double penalty1 =
252  degree * (degree + 1) * dinfo1.face->measure() / dinfo1.cell->measure();
253  const double penalty2 =
254  degree * (degree + 1) * dinfo2.face->measure() / dinfo2.cell->measure();
255  const double penalty = penalty1 + penalty2;
256  const double h = dinfo1.face->measure();
257 
258  for (unsigned k = 0; k < fe.n_quadrature_points; ++k)
259  {
260  const double diff1 = uh1[k] - uh2[k];
261  const double diff2 =
262  fe.normal_vector(k) * Duh1[k] - fe.normal_vector(k) * Duh2[k];
263  dinfo1.value(0) +=
264  (penalty * diff1 * diff1 + h * diff2 * diff2) * fe.JxW(k);
265  }
266  dinfo1.value(0) = std::sqrt(dinfo1.value(0));
267  dinfo2.value(0) = dinfo1.value(0);
268  }

288  template <int dim>
289  void ErrorIntegrator<dim>::cell(
290  MeshWorker::DoFInfo<dim> & dinfo,
291  typename MeshWorker::IntegrationInfo<dim> &info) const
292  {
293  const FEValuesBase<dim> & fe = info.fe_values();
294  std::vector<Tensor<1, dim>> exact_gradients(fe.n_quadrature_points);
295  std::vector<double> exact_values(fe.n_quadrature_points);
296 
297  exact_solution.gradient_list(fe.get_quadrature_points(), exact_gradients);
298  exact_solution.value_list(fe.get_quadrature_points(), exact_values);
299 
300  const std::vector<Tensor<1, dim>> &Duh = info.gradients[0][0];
301  const std::vector<double> & uh = info.values[0][0];
302 
303  for (unsigned k = 0; k < fe.n_quadrature_points; ++k)
304  {
305  double sum = 0;
306  for (unsigned int d = 0; d < dim; ++d)
307  {
308  const double diff = exact_gradients[k][d] - Duh[k][d];
309  sum += diff * diff;
310  }
311  const double diff = exact_values[k] - uh[k];
312  dinfo.value(0) += sum * fe.JxW(k);
313  dinfo.value(1) += diff * diff * fe.JxW(k);
314  }
315  dinfo.value(0) = std::sqrt(dinfo.value(0));
316  dinfo.value(1) = std::sqrt(dinfo.value(1));
317  }
