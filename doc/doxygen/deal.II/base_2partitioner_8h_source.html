759  inline unsigned int
760  Partitioner::global_to_local(
761  const types::global_dof_index global_index) const
762  {
763  Assert(in_local_range(global_index) || is_ghost_entry(global_index),
764  ExcIndexNotPresent(global_index, my_pid));
765  if (in_local_range(global_index))
766  return static_cast<unsigned int>(global_index - local_range_data.first);
767  else if (is_ghost_entry(global_index))
768  return (locally_owned_size() +
769  static_cast<unsigned int>(
770  ghost_indices_data.index_within_set(global_index)));
771  else
772  // should only end up here in optimized mode, when we use this large
773  // number to trigger a segfault when using this method for array
774  // access
775  return numbers::invalid_unsigned_int;
776  }
