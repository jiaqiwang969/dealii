175 MGSmootherBlock<MatrixType, RelaxationType, number>::initialize(
176  const MGMatrixType & m,
177  const MGRelaxationType &s)
178 {
179  const unsigned int min = m.min_level();
180  const unsigned int max = m.max_level();
181 
182  matrices.resize(min, max);
183  smoothers.resize(min, max);
184 
185  for (unsigned int i = min; i <= max; ++i)
186  {
187  // Workaround: Unfortunately, not every "m[i]" object has a
188  // rich enough interface to populate reinit_(domain|range)_vector.
189  // Thus, apply an empty LinearOperator exemplar.
190  matrices[i] = linear_operator<BlockVector<number>>(
191  LinearOperator<BlockVector<number>>(), m[i]);
192  smoothers[i] = linear_operator<BlockVector<number>>(matrices[i], s[i]);
193  }

216 template <typename MatrixType, class RelaxationType, typename number>
217 inline void
218 MGSmootherBlock<MatrixType, RelaxationType, number>::smooth(
219  const unsigned int level,
220  BlockVector<number> & u,
221  const BlockVector<number> &rhs) const
222 {
223  LogStream::Prefix prefix("Smooth");
224 
225  unsigned int maxlevel = matrices.max_level();
226  unsigned int steps2 = this->steps;
227 
228  if (this->variable)
229  steps2 *= (1 << (maxlevel - level));
230 
231  typename VectorMemory<BlockVector<number>>::Pointer r(*this->mem);
232  typename VectorMemory<BlockVector<number>>::Pointer d(*this->mem);
233  r->reinit(u);
234  d->reinit(u);
235 
236  bool T = this->transpose;
237  if (this->symmetric && (steps2 % 2 == 0))
238  T = false;
239 
240  for (unsigned int i = 0; i < steps2; ++i)
241  {
242  if (T)
243  {
244  matrices[level].vmult(*r, u);
245  r->sadd(-1., 1., rhs);
246  smoothers[level].Tvmult(*d, *r);
247  }
248  else
249  {
250  matrices[level].vmult(*r, u);
251  r->sadd(-1., 1., rhs);
252  smoothers[level].vmult(*d, *r);
253  }
254  u += *d;
255  if (this->symmetric)
256  T = !T;
257  }
258 }
