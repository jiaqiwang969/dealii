322  template <int dim, int fe_degree, typename number>
323  void LaplaceOperator<dim, fe_degree, number>::apply_face(
324  const MatrixFree<dim, number> & data,
325  LinearAlgebra::distributed::Vector<number> & dst,
326  const LinearAlgebra::distributed::Vector<number> &src,
327  const std::pair<unsigned int, unsigned int> & face_range) const
328  {
329  FEFaceEvaluation<dim, fe_degree, fe_degree + 1, 1, number> phi_inner(data,
330  true);
331  FEFaceEvaluation<dim, fe_degree, fe_degree + 1, 1, number> phi_outer(data,
332  false);
333  for (unsigned int face = face_range.first; face < face_range.second; ++face)
334  {
335  phi_inner.reinit(face);
336  phi_inner.gather_evaluate(src,
337  EvaluationFlags::values |
338  EvaluationFlags::gradients);
339  phi_outer.reinit(face);
340  phi_outer.gather_evaluate(src,
341  EvaluationFlags::values |
342  EvaluationFlags::gradients);
343 
344  const VectorizedArray<number> inverse_length_normal_to_face =
345  0.5 * (std::abs((phi_inner.get_normal_vector(0) *
346  phi_inner.inverse_jacobian(0))[dim - 1]) +
347  std::abs((phi_outer.get_normal_vector(0) *
348  phi_outer.inverse_jacobian(0))[dim - 1]));
349  const VectorizedArray<number> sigma =
350  inverse_length_normal_to_face * get_penalty_factor();
351 
352  for (unsigned int q = 0; q < phi_inner.n_q_points; ++q)
353  {
354  const VectorizedArray<number> solution_jump =
355  (phi_inner.get_value(q) - phi_outer.get_value(q));
356  const VectorizedArray<number> average_normal_derivative =
357  (phi_inner.get_normal_derivative(q) +
358  phi_outer.get_normal_derivative(q)) *
359  number(0.5);
360  const VectorizedArray<number> test_by_value =
361  solution_jump * sigma - average_normal_derivative;
362 
363  phi_inner.submit_value(test_by_value, q);
364  phi_outer.submit_value(-test_by_value, q);
365 
366  phi_inner.submit_normal_derivative(-solution_jump * number(0.5), q);
367  phi_outer.submit_normal_derivative(-solution_jump * number(0.5), q);
368  }
369 
370  phi_inner.integrate_scatter(EvaluationFlags::values |
371  EvaluationFlags::gradients,
372  dst);
373  phi_outer.integrate_scatter(EvaluationFlags::values |
374  EvaluationFlags::gradients,
375  dst);
376  }

382  template <int dim, int fe_degree, typename number>
383  void LaplaceOperator<dim, fe_degree, number>::apply_boundary(
384  const MatrixFree<dim, number> & data,
385  LinearAlgebra::distributed::Vector<number> & dst,
386  const LinearAlgebra::distributed::Vector<number> &src,
387  const std::pair<unsigned int, unsigned int> & face_range) const
388  {
389  FEFaceEvaluation<dim, fe_degree, fe_degree + 1, 1, number> phi_inner(data,
390  true);
391  for (unsigned int face = face_range.first; face < face_range.second; ++face)
392  {
393  phi_inner.reinit(face);
394  phi_inner.gather_evaluate(src,
395  EvaluationFlags::values |
396  EvaluationFlags::gradients);
397 
398  const VectorizedArray<number> inverse_length_normal_to_face =
399  std::abs((phi_inner.get_normal_vector(0) *
400  phi_inner.inverse_jacobian(0))[dim - 1]);
401  const VectorizedArray<number> sigma =
402  inverse_length_normal_to_face * get_penalty_factor();
403 
404  const bool is_dirichlet = (data.get_boundary_id(face) == 0);
405 
406  for (unsigned int q = 0; q < phi_inner.n_q_points; ++q)
407  {
408  const VectorizedArray<number> u_inner = phi_inner.get_value(q);
409  const VectorizedArray<number> u_outer =
410  is_dirichlet ? -u_inner : u_inner;
411  const VectorizedArray<number> normal_derivative_inner =
412  phi_inner.get_normal_derivative(q);
413  const VectorizedArray<number> normal_derivative_outer =
414  is_dirichlet ? normal_derivative_inner : -normal_derivative_inner;
415  const VectorizedArray<number> solution_jump = (u_inner - u_outer);
416  const VectorizedArray<number> average_normal_derivative =
417  (normal_derivative_inner + normal_derivative_outer) * number(0.5);
418  const VectorizedArray<number> test_by_value =
419  solution_jump * sigma - average_normal_derivative;
420  phi_inner.submit_normal_derivative(-solution_jump * number(0.5), q);
421  phi_inner.submit_value(test_by_value, q);
422  }
423  phi_inner.integrate_scatter(EvaluationFlags::values |
424  EvaluationFlags::gradients,
425  dst);
426  }

432  template <int dim, int fe_degree, typename number>
433  void PreconditionBlockJacobi<dim, fe_degree, number>::initialize(
434  const LaplaceOperator<dim, fe_degree, number> &op)
435  {
436  data = op.get_matrix_free();
437 
438  std::string name = data->get_dof_handler().get_fe().get_name();
439  name.replace(name.find('<') + 1, 1, "1");
440  std::unique_ptr<FiniteElement<1>> fe_1d = FETools::get_fe_by_name<1>(name);
441 
442  const unsigned int N = fe_degree + 1;
443  FullMatrix<double> laplace_unscaled(N, N);
444  std::array<Table<2, VectorizedArray<number>>, dim> mass_matrices;
445  std::array<Table<2, VectorizedArray<number>>, dim> laplace_matrices;
446  for (unsigned int d = 0; d < dim; ++d)
447  {
448  mass_matrices[d].reinit(N, N);
449  laplace_matrices[d].reinit(N, N);
450  }
451 
452  QGauss<1> quadrature(N);
453  for (unsigned int i = 0; i < N; ++i)
454  for (unsigned int j = 0; j < N; ++j)
455  {
456  double sum_mass = 0, sum_laplace = 0;
457  for (unsigned int q = 0; q < quadrature.size(); ++q)
458  {
459  sum_mass += (fe_1d->shape_value(i, quadrature.point(q)) *
460  fe_1d->shape_value(j, quadrature.point(q))) *
461  quadrature.weight(q);
462  sum_laplace += (fe_1d->shape_grad(i, quadrature.point(q))[0] *
463  fe_1d->shape_grad(j, quadrature.point(q))[0]) *
464  quadrature.weight(q);
465  }
466  for (unsigned int d = 0; d < dim; ++d)
467  mass_matrices[d](i, j) = sum_mass;
468 
469  sum_laplace +=
470  (1. * fe_1d->shape_value(i, Point<1>()) *
471  fe_1d->shape_value(j, Point<1>()) * op.get_penalty_factor() +
472  0.5 * fe_1d->shape_grad(i, Point<1>())[0] *
473  fe_1d->shape_value(j, Point<1>()) +
474  0.5 * fe_1d->shape_grad(j, Point<1>())[0] *
475  fe_1d->shape_value(i, Point<1>()));
476 
477  sum_laplace +=
478  (1. * fe_1d->shape_value(i, Point<1>(1.0)) *
479  fe_1d->shape_value(j, Point<1>(1.0)) * op.get_penalty_factor() -
480  0.5 * fe_1d->shape_grad(i, Point<1>(1.0))[0] *
481  fe_1d->shape_value(j, Point<1>(1.0)) -
482  0.5 * fe_1d->shape_grad(j, Point<1>(1.0))[0] *
483  fe_1d->shape_value(i, Point<1>(1.0)));
484 
485  laplace_unscaled(i, j) = sum_laplace;
486  }
487 
488  cell_matrices.clear();
489  FEEvaluation<dim, fe_degree, fe_degree + 1, 1, number> phi(*data);
490  unsigned int old_mapping_data_index = numbers::invalid_unsigned_int;
491  for (unsigned int cell = 0; cell < data->n_cell_batches(); ++cell)
492  {
493  phi.reinit(cell);
494 
495  if (phi.get_mapping_data_index_offset() == old_mapping_data_index)
496  continue;
497 
498  Tensor<2, dim, VectorizedArray<number>> inverse_jacobian =
499  phi.inverse_jacobian(0);
500 
501  for (unsigned int d = 0; d < dim; ++d)
502  for (unsigned int e = 0; e < dim; ++e)
503  if (d != e)
504  for (unsigned int v = 0; v < VectorizedArray<number>::size(); ++v)
505  AssertThrow(inverse_jacobian[d][e][v] == 0.,
506  ExcNotImplemented());
507 
508  VectorizedArray<number> jacobian_determinant = inverse_jacobian[0][0];
509  for (unsigned int e = 1; e < dim; ++e)
510  jacobian_determinant *= inverse_jacobian[e][e];
511  jacobian_determinant = 1. / jacobian_determinant;
512 
513  for (unsigned int d = 0; d < dim; ++d)
514  {
515  const VectorizedArray<number> scaling_factor =
516  inverse_jacobian[d][d] * inverse_jacobian[d][d] *
517  jacobian_determinant;
518 
519 
520  for (unsigned int i = 0; i < N; ++i)
521  for (unsigned int j = 0; j < N; ++j)
522  laplace_matrices[d](i, j) =
523  scaling_factor * laplace_unscaled(i, j);
524  }
525  if (cell_matrices.size() <= phi.get_mapping_data_index_offset())
526  cell_matrices.resize(phi.get_mapping_data_index_offset() + 1);
527  cell_matrices[phi.get_mapping_data_index_offset()].reinit(
528  mass_matrices, laplace_matrices);
529  }
530  }
