870  TableEntry::get() const

893  template <class Archive>
894  void
895  TableEntry::save(Archive &ar, const unsigned int) const
896  {
897  // write first an identifier for the kind
898  // of data stored and then the actual
899  // data, in its correct data type
900  if (const int *p = boost::get<int>(&value))
901  {
902  char c = 'i';
903  ar &c &*p;
904  }
905  else if (const unsigned int *p = boost::get<unsigned int>(&value))
906  {
907  char c = 'u';
908  ar &c &*p;
909  }
910  else if (const double *p = boost::get<double>(&value))
911  {
912  char c = 'd';
913  ar &c &*p;
914  }
915  else if (const std::string *p = boost::get<std::string>(&value))
916  {
917  char c = 's';
918  ar &c &*p;
919  }
920  else if (const std::uint64_t *p = boost::get<std::uint64_t>(&value))
921  {
922  char c = 'l';
923  ar &c &*p;
924  }
925  else
926  Assert(false, ExcInternalError());
927  }

933  TableEntry::load(Archive &ar, const unsigned int)
934  {

936  // function, first read in the data type

992 template <typename T>
993 void
994 TableHandler::add_value(const std::string &key, const T value)
995 {
996  // see if the column already exists
997  if (columns.find(key) == columns.end())
998  declare_column(key);
999 
1000  if (auto_fill_mode == true)
1001  {
1002  // follow the algorithm given in the introduction to this class
1003  // of padding columns as necessary
1004  unsigned int max_col_length = 0;
1005  for (const auto &column : columns)
1006  max_col_length =
1007  std::max(max_col_length,
1008  static_cast<unsigned int>(column.second.entries.size()));
1009 
1010  while (columns[key].entries.size() + 1 < max_col_length)
1011  {
1012  columns[key].entries.push_back(internal::TableEntry(T()));
1013  internal::TableEntry &entry = columns[key].entries.back();
1014  entry.cache_string(columns[key].scientific, columns[key].precision);
1015  columns[key].max_length =
1016  std::max(columns[key].max_length,
1017  static_cast<unsigned int>(
1018  entry.get_cached_string().length()));
1019  }
1020  }
1021 
1022  // now push the value given to this function
1023  columns[key].entries.push_back(internal::TableEntry(value));
1024  internal::TableEntry &entry = columns[key].entries.back();
1025  entry.cache_string(columns[key].scientific, columns[key].precision);
1026  columns[key].max_length =
1027  std::max(columns[key].max_length,
1028  static_cast<unsigned int>(entry.get_cached_string().length()));
1029 }
