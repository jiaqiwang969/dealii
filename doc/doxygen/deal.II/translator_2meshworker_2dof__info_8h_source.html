505  template <int dim, class DOFINFO>
506  template <class ASSEMBLER>
507  inline void
508  DoFInfoBox<dim, DOFINFO>::assemble(ASSEMBLER &assembler) const
509  {
510  if (!cell_valid)
511  return;
512 
513  assembler.assemble(cell);
514  for (unsigned int i : GeometryInfo<dim>::face_indices())
515  {
516  // Only do something if data available
517  if (interior_face_available[i])
518  {
519  // If both data
520  // available, it is an
521  // interior face
522  if (exterior_face_available[i])
523  assembler.assemble(interior[i], exterior[i]);
524  else
525  assembler.assemble(interior[i]);
526  }
527  }
