101  class ExactSolution : public Function<dim>
102  {
103  public:
104  ExactSolution(const unsigned int n_components = 1, const double time = 0.)
105  : Function<dim>(n_components, time)
106  {}

109 
110  virtual double value(const Point<dim> &p,
111  const unsigned int /*component*/ = 0) const override
112  {
113  const double t = this->get_time();
114 
115  switch (dim)
116  {
117  case 1:
118  {
119  const double m = 0.5;
120  const double c1 = 0.;
121  const double c2 = 0.;
122  return -4. * std::atan(m / std::sqrt(1. - m * m) *
123  std::sin(std::sqrt(1. - m * m) * t + c2) /
124  std::cosh(m * p[0] + c1));
125  }
126 
127  case 2:
128  {
129  const double theta = numbers::PI / 4.;
130  const double lambda = 1.;
131  const double a0 = 1.;
132  const double s = 1.;
133  const double arg = p[0] * std::cos(theta) +
134  std::sin(theta) * (p[1] * std::cosh(lambda) +
135  t * std::sinh(lambda));
136  return 4. * std::atan(a0 * std::exp(s * arg));
137  }
138 
139  case 3:
140  {
141  const double theta = numbers::PI / 4;
142  const double phi = numbers::PI / 4;
143  const double tau = 1.;
144  const double c0 = 1.;
145  const double s = 1.;
146  const double arg = p[0] * std::cos(theta) +
147  p[1] * std::sin(theta) * std::cos(phi) +
148  std::sin(theta) * std::sin(phi) *
149  (p[2] * std::cosh(tau) + t * std::sinh(tau));
150  return 4. * std::atan(c0 * std::exp(s * arg));
151  }
152 
153  default:
154  Assert(false, ExcNotImplemented());
155  return -1e8;
156  }
157  }

231  template <int dim>
232  void SineGordonProblem<dim>::assemble_system()
233  {
234  system_matrix.copy_from(mass_matrix);
235  system_matrix.add(std::pow(time_step * theta, 2), laplace_matrix);
236 
237  SparseMatrix<double> tmp_matrix(sparsity_pattern);
238  compute_nl_matrix(old_solution, solution, tmp_matrix);
239  system_matrix.add(std::pow(time_step * theta, 2), tmp_matrix);
240 
241  system_rhs = 0.;
242 
243  Vector<double> tmp_vector(solution.size());
244 
245  mass_matrix.vmult(system_rhs, solution);
246  laplace_matrix.vmult(tmp_vector, solution);
247  system_rhs.add(std::pow(time_step * theta, 2), tmp_vector);
248 
249  mass_matrix.vmult(tmp_vector, old_solution);
250  system_rhs.add(-1.0, tmp_vector);
251  laplace_matrix.vmult(tmp_vector, old_solution);
252  system_rhs.add(std::pow(time_step, 2) * theta * (1 - theta), tmp_vector);
253 
254  system_rhs.add(-time_step, M_x_velocity);
255 
256  compute_nl_term(old_solution, solution, tmp_vector);
257  system_rhs.add(std::pow(time_step, 2) * theta, tmp_vector);
258 
259  system_rhs *= -1.;
260  }

263  template <int dim>
264  void SineGordonProblem<dim>::compute_nl_term(const Vector<double> &old_data,
265  const Vector<double> &new_data,
266  Vector<double> &nl_term) const
267  {
268  nl_term = 0;
269  const QGauss<dim> quadrature_formula(fe.degree + 1);
270  FEValues<dim> fe_values(fe,
271  quadrature_formula,
272  update_values | update_JxW_values |
273  update_quadrature_points);
274 
275  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
276  const unsigned int n_q_points = quadrature_formula.size();
277 
278  Vector<double> local_nl_term(dofs_per_cell);
279  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
280  std::vector<double> old_data_values(n_q_points);
281  std::vector<double> new_data_values(n_q_points);
282 
283  for (const auto &cell : dof_handler.active_cell_iterators())
284  {
285  local_nl_term = 0;
286  fe_values.reinit(cell);
287  fe_values.get_function_values(old_data, old_data_values);
288  fe_values.get_function_values(new_data, new_data_values);
289 
290  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
291  for (unsigned int i = 0; i < dofs_per_cell; ++i)
292  local_nl_term(i) +=
293  (std::sin(theta * new_data_values[q_point] +
294  (1 - theta) * old_data_values[q_point]) *
295  fe_values.shape_value(i, q_point) * fe_values.JxW(q_point));
296 
297  cell->get_dof_indices(local_dof_indices);
298 
299  for (unsigned int i = 0; i < dofs_per_cell; ++i)
300  nl_term(local_dof_indices[i]) += local_nl_term(i);
301  }
302  }

305  template <int dim>
306  void SineGordonProblem<dim>::compute_nl_matrix(
307  const Vector<double> &old_data,
308  const Vector<double> &new_data,
309  SparseMatrix<double> &nl_matrix) const
310  {
311  QGauss<dim> quadrature_formula(fe.degree + 1);
312  FEValues<dim> fe_values(fe,
313  quadrature_formula,
314  update_values | update_JxW_values |
315  update_quadrature_points);
316 
317  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
318  const unsigned int n_q_points = quadrature_formula.size();
319 
320  FullMatrix<double> local_nl_matrix(dofs_per_cell, dofs_per_cell);
321  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
322  std::vector<double> old_data_values(n_q_points);
323  std::vector<double> new_data_values(n_q_points);
324 
325  for (const auto &cell : dof_handler.active_cell_iterators())
326  {
327  local_nl_matrix = 0;
328  fe_values.reinit(cell);
329  fe_values.get_function_values(old_data, old_data_values);
330  fe_values.get_function_values(new_data, new_data_values);
331 
332  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
333  for (unsigned int i = 0; i < dofs_per_cell; ++i)
334  for (unsigned int j = 0; j < dofs_per_cell; ++j)
335  local_nl_matrix(i, j) +=
336  (std::cos(theta * new_data_values[q_point] +
337  (1 - theta) * old_data_values[q_point]) *
338  fe_values.shape_value(i, q_point) *
339  fe_values.shape_value(j, q_point) * fe_values.JxW(q_point));
340 
341  cell->get_dof_indices(local_dof_indices);
342 
343  for (unsigned int i = 0; i < dofs_per_cell; ++i)
344  for (unsigned int j = 0; j < dofs_per_cell; ++j)
345  nl_matrix.add(local_dof_indices[i],
346  local_dof_indices[j],
347  local_nl_matrix(i, j));
348  }
349  }

390  template <int dim>
391  void SineGordonProblem<dim>::run()
392  {
393  make_grid_and_dofs();
394 
395  {
396  AffineConstraints<double> constraints;
397  constraints.close();
398  VectorTools::project(dof_handler,
399  constraints,
400  QGauss<dim>(fe.degree + 1),
401  InitialValues<dim>(1, time),
402  solution);
403  }
404 
405  output_results(0);
406 
407  unsigned int timestep_number = 1;
408  for (time += time_step; time <= final_time;
409  time += time_step, ++timestep_number)
410  {
411  old_solution = solution;
412 
413  std::cout << std::endl
414  << "Time step #" << timestep_number << "; "
415  << "advancing to t = " << time << "." << std::endl;
416 
417  double initial_rhs_norm = 0.;
418  bool first_iteration = true;
419  do
420  {
421  assemble_system();
422 
423  if (first_iteration == true)
424  initial_rhs_norm = system_rhs.l2_norm();
425 
426  const unsigned int n_iterations = solve();
427 
428  solution += solution_update;
429 
430  if (first_iteration == true)
431  std::cout << " " << n_iterations;
432  else
433  std::cout << '+' << n_iterations;
434  first_iteration = false;
435  }
436  while (system_rhs.l2_norm() > 1e-6 * initial_rhs_norm);
437 
438  std::cout << " CG iterations per nonlinear step." << std::endl;
439 
440  Vector<double> tmp_vector(solution.size());
441  laplace_matrix.vmult(tmp_vector, solution);
442  M_x_velocity.add(-time_step * theta, tmp_vector);
443 
444  laplace_matrix.vmult(tmp_vector, old_solution);
445  M_x_velocity.add(-time_step * (1 - theta), tmp_vector);
446 
447  compute_nl_term(old_solution, solution, tmp_vector);
448  M_x_velocity.add(-time_step, tmp_vector);
449 
450  if (timestep_number % output_timestep_skip == 0)
451  output_results(timestep_number);
452  }
453  }
