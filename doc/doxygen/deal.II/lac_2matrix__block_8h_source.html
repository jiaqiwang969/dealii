711 template <typename MatrixType>
712 inline void
713 MatrixBlock<MatrixType>::add(const size_type gi,
714  const size_type gj,
715  const typename MatrixType::value_type value)
716 {
717  Assert(row_indices.size() != 0, ExcNotInitialized());
718  Assert(column_indices.size() != 0, ExcNotInitialized());
719 
720  const std::pair<unsigned int, size_type> bi = row_indices.global_to_local(gi);
721  const std::pair<unsigned int, size_type> bj =
722  column_indices.global_to_local(gj);
723 
724  Assert(bi.first == row, ExcBlockIndexMismatch(bi.first, row));
725  Assert(bj.first == column, ExcBlockIndexMismatch(bj.first, column));
726 
727  matrix.add(bi.second, bj.second, value);
728 }

754 template <typename MatrixType>
755 template <typename number>
756 inline void
757 MatrixBlock<MatrixType>::add(const size_type b_row,
758  const size_type n_cols,
759  const size_type *col_indices,
760  const number * values,
761  const bool,
762  const bool)
763 {
764  Assert(row_indices.size() != 0, ExcNotInitialized());
765  Assert(column_indices.size() != 0, ExcNotInitialized());
766 
767  const std::pair<unsigned int, size_type> bi =
768  row_indices.global_to_local(b_row);
769 
770  // In debug mode, we check whether
771  // all indices are in the correct
772  // block.
773 
774  // Actually, for the time being, we
775  // leave it at this. While it may
776  // not be the most efficient way,
777  // it is at least thread safe.
778  //#ifdef DEBUG
779  Assert(bi.first == row, ExcBlockIndexMismatch(bi.first, row));
780 
781  for (size_type j = 0; j < n_cols; ++j)
782  {
783  const std::pair<unsigned int, size_type> bj =
784  column_indices.global_to_local(col_indices[j]);
785  Assert(bj.first == column, ExcBlockIndexMismatch(bj.first, column));
786 
787  matrix.add(bi.second, bj.second, values[j]);
788  }
789  //#endif
790 }
