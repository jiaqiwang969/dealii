99  public:
100  virtual void vector_value(const Point<dim> &p,
101  Vector<double> & values) const override
102  {
103  Assert(values.size() == dim, ExcDimensionMismatch(values.size(), dim));
104  Assert(dim >= 2, ExcInternalError());
105 
106  Point<dim> point_1, point_2;
107  point_1(0) = 0.5;
108  point_2(0) = -0.5;
109 
110  if (((p - point_1).norm_square() < 0.2 * 0.2) ||
111  ((p - point_2).norm_square() < 0.2 * 0.2))
112  values(0) = 1;
113  else
114  values(0) = 0;
115 
116  if (p.square() < 0.2 * 0.2)
117  values(1) = 1;
118  else
119  values(1) = 0;

189  template <int dim>
190  void ElasticProblem<dim>::assemble_system()
191  {
192  QGauss<dim> quadrature_formula(fe.degree + 1);
193  FEValues<dim> fe_values(fe,
194  quadrature_formula,
195  update_values | update_gradients |
196  update_quadrature_points | update_JxW_values);
197 
198  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
199  const unsigned int n_q_points = quadrature_formula.size();
200 
201  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
202  Vector<double> cell_rhs(dofs_per_cell);
203 
204  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
205 
206  std::vector<double> lambda_values(n_q_points);
207  std::vector<double> mu_values(n_q_points);
208 
209  Functions::ConstantFunction<dim> lambda(1.), mu(1.);
210 
211  RightHandSide<dim> right_hand_side;
212  std::vector<Vector<double>> rhs_values(n_q_points, Vector<double>(dim));
213 
214 
215  for (const auto &cell : dof_handler.active_cell_iterators())
216  if (cell->subdomain_id() == this_mpi_process)
217  {
218  cell_matrix = 0;
219  cell_rhs = 0;
220 
221  fe_values.reinit(cell);
222 
223  lambda.value_list(fe_values.get_quadrature_points(), lambda_values);
224  mu.value_list(fe_values.get_quadrature_points(), mu_values);
225 
226  for (unsigned int i = 0; i < dofs_per_cell; ++i)
227  {
228  const unsigned int component_i =
229  fe.system_to_component_index(i).first;
230 
231  for (unsigned int j = 0; j < dofs_per_cell; ++j)
232  {
233  const unsigned int component_j =
234  fe.system_to_component_index(j).first;
235 
236  for (unsigned int q_point = 0; q_point < n_q_points;
237  ++q_point)
238  {
239  cell_matrix(i, j) +=
240  ((fe_values.shape_grad(i, q_point)[component_i] *
241  fe_values.shape_grad(j, q_point)[component_j] *
242  lambda_values[q_point]) +
243  (fe_values.shape_grad(i, q_point)[component_j] *
244  fe_values.shape_grad(j, q_point)[component_i] *
245  mu_values[q_point]) +
246  ((component_i == component_j) ?
247  (fe_values.shape_grad(i, q_point) *
248  fe_values.shape_grad(j, q_point) *
249  mu_values[q_point]) :
250  0)) *
251  fe_values.JxW(q_point);
252  }
253  }
254  }
255 
256  right_hand_side.vector_value_list(fe_values.get_quadrature_points(),
257  rhs_values);
258  for (unsigned int i = 0; i < dofs_per_cell; ++i)
259  {
260  const unsigned int component_i =
261  fe.system_to_component_index(i).first;
262 
263  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
264  cell_rhs(i) += fe_values.shape_value(i, q_point) *
265  rhs_values[q_point](component_i) *
266  fe_values.JxW(q_point);
267  }
268 
269  cell->get_dof_indices(local_dof_indices);
270  hanging_node_constraints.distribute_local_to_global(cell_matrix,
271  cell_rhs,
272  local_dof_indices,
273  system_matrix,
274  system_rhs);
275  }
276 
277  system_matrix.compress(VectorOperation::add);
278  system_rhs.compress(VectorOperation::add);
279 
280  std::map<types::global_dof_index, double> boundary_values;
281  VectorTools::interpolate_boundary_values(dof_handler,
282  0,
283  Functions::ZeroFunction<dim>(dim),
284  boundary_values);
285  MatrixTools::apply_boundary_values(
286  boundary_values, system_matrix, solution, system_rhs, false);
287  }

292  template <int dim>
293  unsigned int ElasticProblem<dim>::solve()
294  {
295  SolverControl solver_control(solution.size(), 1e-8 * system_rhs.l2_norm());
296  PETScWrappers::SolverCG cg(solver_control, mpi_communicator);
297 
298  PETScWrappers::PreconditionBlockJacobi preconditioner(system_matrix);
299 
300  cg.solve(system_matrix, solution, system_rhs, preconditioner);
301 
302  Vector<double> localized_solution(solution);
303 
304  hanging_node_constraints.distribute(localized_solution);
305 
306  solution = localized_solution;
307 
308  return solver_control.last_step();
309  }
