167 template <typename T>
168 void
169 FiniteSizeHistory<T>::add(const T &element)
170 {
171  std::unique_ptr<T> new_el;
172  if (data.size() < max_n_elements)
173  // have not reached the maximum number of elements yet
174  {
175  if (cache.size() == 0)
176  // nothing is cached, just copy a given element
177  {
178  new_el = std::make_unique<T>(element);
179  }
180  else
181  // something is cached, take one element and copy
182  // the user provided one there.
183  {
184  new_el = std::move(cache.back());
185  (*new_el) = element;
186 
187  cache.pop_back(); // removes a pointer that is now a nullptr anyway
188  }
189  }
190  else
191  // we reached the maximum number of elements and
192  // thus have to re-order/cycle elements currently stored
193  {
194  new_el = std::move(data.back());
195  (*new_el) = element;
196 
197  data.pop_back(); // removes a pointer that is now a nullptr anyway
198  }
199 
200  // finally insert the new one where appropriate
201  data.push_front(std::move(new_el));
202 
203  // whatever we do, we shall not store more than the maximum number of
204  // elements
205  Assert(data.size() + cache.size() <= max_n_elements, ExcInternalError());
206 }
