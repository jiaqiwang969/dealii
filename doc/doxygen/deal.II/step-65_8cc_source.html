56  template <int dim>
57  class ExactSolution : public Function<dim>
58  {
59  public:
60  virtual double value(const Point<dim> &p,
61  const unsigned int /*component*/ = 0) const override
62  {
63  if (p.norm_square() < 0.25)
64  return p.norm_square();
65  else
66  return 0.1 * p.norm_square() + (0.25 - 0.025);
67  }
68 
69  virtual Tensor<1, dim>
70  gradient(const Point<dim> &p,
71  const unsigned int /*component*/ = 0) const override
72  {
73  if (p.norm_square() < 0.25)
74  return 2. * p;
75  else
76  return 0.2 * p;
77  }
78  };

130  template <int dim>
131  void PoissonProblem<dim>::create_grid()
132  {
133  Triangulation<dim> tria_inner;
134  GridGenerator::hyper_ball(tria_inner, Point<dim>(), 0.5);
135 
136  Triangulation<dim> tria_outer;
137  GridGenerator::hyper_shell(
138  tria_outer, Point<dim>(), 0.5, std::sqrt(dim), 2 * dim);
139 
140  GridGenerator::merge_triangulations(tria_inner, tria_outer, triangulation);
141 
142  triangulation.reset_all_manifolds();
143  triangulation.set_all_manifold_ids(0);
144 
145  for (const auto &cell : triangulation.cell_iterators())
146  {
147  for (const auto &face : cell->face_iterators())
148  {
149  bool face_at_sphere_boundary = true;
150  for (const auto v : face->vertex_indices())
151  {
152  if (std::abs(face->vertex(v).norm_square() - 0.25) > 1e-12)
153  face_at_sphere_boundary = false;
154  }
155  if (face_at_sphere_boundary)
156  face->set_all_manifold_ids(1);
157  }
158  if (cell->center().norm_square() < 0.25)
159  cell->set_material_id(1);
160  else
161  cell->set_material_id(0);
162  }
163 
164  triangulation.set_manifold(1, SphericalManifold<dim>());
165 
166  TransfiniteInterpolationManifold<dim> transfinite_manifold;
167  transfinite_manifold.initialize(triangulation);
168  triangulation.set_manifold(0, transfinite_manifold);
169 
170  triangulation.refine_global(9 - 2 * dim);
171  }

207  template <int dim>
208  void PoissonProblem<dim>::assemble_system(const Mapping<dim> &mapping)
209  {
210  TimerOutput::Scope scope(timer, "Assemble linear system");
211 
212  const QGauss<dim> quadrature_formula(fe.degree + 2);
213  FEValues<dim> fe_values(mapping,
214  fe,
215  quadrature_formula,
216  update_values | update_gradients |
217  update_quadrature_points | update_JxW_values);
218 
219  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
220  const unsigned int n_q_points = quadrature_formula.size();
221 
222  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
223  Vector<double> cell_rhs(dofs_per_cell);
224  FullMatrix<double> partial_matrix(dofs_per_cell, dim * n_q_points);
225  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
226 
227  for (const auto &cell : dof_handler.active_cell_iterators())
228  {
229  cell_rhs = 0.;
230  fe_values.reinit(cell);
231 
232  for (unsigned int q_index = 0; q_index < n_q_points; ++q_index)
233  {
234  const double current_coefficient =
235  coefficient(fe_values.quadrature_point(q_index));
236  for (unsigned int i = 0; i < dofs_per_cell; ++i)
237  {
238  for (unsigned int d = 0; d < dim; ++d)
239  partial_matrix(i, q_index * dim + d) =
240  std::sqrt(fe_values.JxW(q_index) * current_coefficient) *
241  fe_values.shape_grad(i, q_index)[d];
242  cell_rhs(i) +=
243  (fe_values.shape_value(i, q_index) * // phi_i(x_q)
244  (-dim) * // f(x_q)
245  fe_values.JxW(q_index)); // dx
246  }
247  }
248 
249  partial_matrix.mTmult(cell_matrix, partial_matrix);
250 
251  cell->get_dof_indices(local_dof_indices);
252  constraints.distribute_local_to_global(
253  cell_matrix, cell_rhs, local_dof_indices, system_matrix, system_rhs);
254  }
255  }

260  void PoissonProblem<dim>::solve()
261  {
262  TimerOutput::Scope scope(timer, "Solve linear system");
263 
264  SolverControl solver_control(1000, 1e-12);
265  SolverCG<Vector<double>> solver(solver_control);
266 
267  PreconditionJacobi<SparseMatrix<double>> preconditioner;
268  preconditioner.initialize(system_matrix);
269 
270  solver.solve(system_matrix, solution, system_rhs, preconditioner);
271  constraints.distribute(solution);
272 
273  std::cout << " Number of solver iterations: "
