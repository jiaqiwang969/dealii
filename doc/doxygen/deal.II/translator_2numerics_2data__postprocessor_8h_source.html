1152  {
1153  // see if we had previously already stored a cell that has the same
1154  // data type; if so, reuse the memory location and avoid calling 'new'
1155  // inside boost::any
1156  if (typename DoFHandler<dim, spacedim>::cell_iterator *storage_location =
1157  boost::any_cast<typename DoFHandler<dim, spacedim>::cell_iterator>(
1158  &cell))
1159  *storage_location = new_cell;

1162  // data type, just let boost::any replace things
