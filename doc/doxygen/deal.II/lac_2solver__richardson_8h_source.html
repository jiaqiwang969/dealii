202 void
203 SolverRichardson<VectorType>::solve(const MatrixType & A,
204  VectorType & x,
205  const VectorType & b,
206  const PreconditionerType &preconditioner)
207 {
208  SolverControl::State conv = SolverControl::iterate;
209 
210  double last_criterion = -std::numeric_limits<double>::max();
211 
212  unsigned int iter = 0;
213 
214  // Memory allocation.
215  // 'Vr' holds the residual, 'Vd' the preconditioned residual
216  typename VectorMemory<VectorType>::Pointer Vr(this->memory);
217  typename VectorMemory<VectorType>::Pointer Vd(this->memory);
218 
219  VectorType &r = *Vr;
220  r.reinit(x);
221 
222  VectorType &d = *Vd;
223  d.reinit(x);
224 
225  LogStream::Prefix prefix("Richardson");
226 
227  // Main loop
228  while (conv == SolverControl::iterate)
229  {
230  // Do not use residual,
231  // but do it in 2 steps
232  A.vmult(r, x);
233  r.sadd(-1., 1., b);
234  preconditioner.vmult(d, r);
235 
236  // get the required norm of the (possibly preconditioned)
237  // residual
238  last_criterion = criterion(r, d);
239  conv = this->iteration_status(iter, last_criterion, x);
240  if (conv != SolverControl::iterate)
241  break;
242 
243  x.add(additional_data.omega, d);
244  print_vectors(iter, x, r, d);
245 
246  ++iter;
247  }
248 
249  // in case of failure: throw exception
250  if (conv != SolverControl::success)
251  AssertThrow(false, SolverControl::NoConvergence(iter, last_criterion));
252  // otherwise exit as normal
253 }
