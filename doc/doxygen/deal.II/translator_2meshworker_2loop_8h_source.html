277  template <class INFOBOX, class DOFINFO, int dim, int spacedim, class ITERATOR>
278  void
279  cell_action(
280  ITERATOR cell,
281  DoFInfoBox<dim, DOFINFO> &dof_info,
282  INFOBOX & info,
283  const std::function<void(DOFINFO &, typename INFOBOX::CellInfo &)>
284  &cell_worker,
285  const std::function<void(DOFINFO &, typename INFOBOX::CellInfo &)>
286  & boundary_worker,
287  const std::function<void(DOFINFO &,
288  DOFINFO &,
289  typename INFOBOX::CellInfo &,
290  typename INFOBOX::CellInfo &)> &face_worker,
291  const LoopControl & loop_control)
292  {
293  const bool ignore_subdomain =
294  (cell->get_triangulation().locally_owned_subdomain() ==
295  numbers::invalid_subdomain_id);
296 
297  types::subdomain_id csid = (cell->is_level_cell()) ?
298  cell->level_subdomain_id() :
299  cell->subdomain_id();
300 
301  const bool own_cell =
302  ignore_subdomain ||
303  (csid == cell->get_triangulation().locally_owned_subdomain());
304 
305  dof_info.reset();
306 
307  if ((!ignore_subdomain) && (csid == numbers::artificial_subdomain_id))
308  return;
309 
310  dof_info.cell.reinit(cell);
311  dof_info.cell_valid = true;
312 
313  const bool integrate_cell = (cell_worker != nullptr);
314  const bool integrate_boundary = (boundary_worker != nullptr);
315  const bool integrate_interior_face = (face_worker != nullptr);
316 
317  if (integrate_cell)
318  info.cell.reinit(dof_info.cell);
319  // Execute this, if cells
320  // have to be dealt with
321  // before faces
322  if (integrate_cell && loop_control.cells_first &&
323  ((loop_control.own_cells && own_cell) ||
324  (loop_control.ghost_cells && !own_cell)))
325  cell_worker(dof_info.cell, info.cell);
326 
327  // Call the callback function in
328  // the info box to do
329  // computations between cell and
330  // face action.
331  info.post_cell(dof_info);
332 
333  if (integrate_interior_face || integrate_boundary)
334  for (const unsigned int face_no : cell->face_indices())
335  {
336  typename ITERATOR::AccessorType::Container::face_iterator face =
337  cell->face(face_no);
338  if (cell->at_boundary(face_no) &&
339  !cell->has_periodic_neighbor(face_no))
340  {
341  // only integrate boundary faces of own cells
342  if (integrate_boundary && own_cell)
343  {
344  dof_info.interior_face_available[face_no] = true;
345  dof_info.interior[face_no].reinit(cell, face, face_no);
346  info.boundary.reinit(dof_info.interior[face_no]);
347  boundary_worker(dof_info.interior[face_no], info.boundary);
348  }
349  }
350  else if (integrate_interior_face)
351  {
352  // Interior face
353  TriaIterator<typename ITERATOR::AccessorType> neighbor =
354  cell->neighbor_or_periodic_neighbor(face_no);
355 
356  types::subdomain_id neighbid = numbers::artificial_subdomain_id;
357  if (neighbor->is_level_cell())
358  neighbid = neighbor->level_subdomain_id();
359  // subdomain id is only valid for active cells
360  else if (neighbor->is_active())
361  neighbid = neighbor->subdomain_id();
362 
363  const bool own_neighbor =
364  ignore_subdomain ||
365  (neighbid ==
366  cell->get_triangulation().locally_owned_subdomain());
367 
368  // skip all faces between two ghost cells
369  if (!own_cell && !own_neighbor)
370  continue;
371 
372  // skip if the user doesn't want faces between own cells
373  if (own_cell && own_neighbor &&
374  loop_control.own_faces == LoopControl::never)
375  continue;
376 
377  // skip face to ghost
378  if (own_cell != own_neighbor &&
379  loop_control.faces_to_ghost == LoopControl::never)
380  continue;
381 
382  // Deal with refinement edges from the refined side. Assuming
383  // one-irregular meshes, this situation should only occur if both
384  // cells are active.
385  const bool periodic_neighbor =
386  cell->has_periodic_neighbor(face_no);
387 
388  if ((!periodic_neighbor && cell->neighbor_is_coarser(face_no)) ||
389  (periodic_neighbor &&
390  cell->periodic_neighbor_is_coarser(face_no)))
391  {
392  Assert(cell->is_active(), ExcInternalError());
393  Assert(neighbor->is_active(), ExcInternalError());
394 
395  // skip if only one processor needs to assemble the face
396  // to a ghost cell and the fine cell is not ours.
397  if (!own_cell &&
398  loop_control.faces_to_ghost == LoopControl::one)
399  continue;
400 
401  const std::pair<unsigned int, unsigned int> neighbor_face_no =
402  periodic_neighbor ?
403  cell->periodic_neighbor_of_coarser_periodic_neighbor(
404  face_no) :
405  cell->neighbor_of_coarser_neighbor(face_no);
406  const typename ITERATOR::AccessorType::Container::
407  face_iterator nface =
408  neighbor->face(neighbor_face_no.first);
409 
410  dof_info.interior_face_available[face_no] = true;
411  dof_info.exterior_face_available[face_no] = true;
412  dof_info.interior[face_no].reinit(cell, face, face_no);
413  info.face.reinit(dof_info.interior[face_no]);
414  dof_info.exterior[face_no].reinit(neighbor,
415  nface,
416  neighbor_face_no.first,
417  neighbor_face_no.second);
418  info.subface.reinit(dof_info.exterior[face_no]);
419 
420  face_worker(dof_info.interior[face_no],
421  dof_info.exterior[face_no],
422  info.face,
423  info.subface);
424  }
425  else
426  {
427  // If iterator is active and neighbor is refined, skip
428  // internal face.
429  if (::internal::is_active_iterator(cell) &&
430  neighbor->has_children())
431  {
432  Assert(
433  loop_control.own_faces != LoopControl::both,
434  ExcMessage(
435  "Assembling from both sides for own_faces is not "
436  "supported with hanging nodes!"));
437  continue;
438  }
439 
440  // Now neighbor is on same level, double-check this:
441  Assert(cell->level() == neighbor->level(),
442  ExcInternalError());
443 
444  // If we own both cells only do faces from one side (unless
445  // LoopControl says otherwise). Here, we rely on cell
446  // comparison that will look at cell->index().
447  if (own_cell && own_neighbor &&
448  loop_control.own_faces == LoopControl::one &&
449  (neighbor < cell))
450  continue;
451 
452  // independent of loop_control.faces_to_ghost,
453  // we only look at faces to ghost on the same level once
454  // (only where own_cell=true and own_neighbor=false)
455  if (!own_cell)
456  continue;
457 
458  // now only one processor assembles faces_to_ghost. We let the
459  // processor with the smaller (level-)subdomain id assemble
460  // the face.
461  if (own_cell && !own_neighbor &&
462  loop_control.faces_to_ghost == LoopControl::one &&
463  (neighbid < csid))
464  continue;
465 
466  const unsigned int neighbor_face_no =
467  periodic_neighbor ?
468  cell->periodic_neighbor_face_no(face_no) :
469  cell->neighbor_face_no(face_no);
470  Assert(periodic_neighbor ||
471  neighbor->face(neighbor_face_no) == face,
472  ExcInternalError());
473  // Regular interior face
474  dof_info.interior_face_available[face_no] = true;
475  dof_info.exterior_face_available[face_no] = true;
476  dof_info.interior[face_no].reinit(cell, face, face_no);
477  info.face.reinit(dof_info.interior[face_no]);
478  dof_info.exterior[face_no].reinit(neighbor,
479  neighbor->face(
480  neighbor_face_no),
481  neighbor_face_no);
482  info.neighbor.reinit(dof_info.exterior[face_no]);
483 
484  face_worker(dof_info.interior[face_no],
485  dof_info.exterior[face_no],
486  info.face,
487  info.neighbor);
488  }
489  }
490  } // faces
491  // Call the callback function in
492  // the info box to do
493  // computations between face and
494  // cell action.
495  info.post_faces(dof_info);
496 
497  // Execute this, if faces
498  // have to be handled first
499  if (integrate_cell && !loop_control.cells_first &&
500  ((loop_control.own_cells && own_cell) ||
501  (loop_control.ghost_cells && !own_cell)))
502  cell_worker(dof_info.cell, info.cell);
503  }
