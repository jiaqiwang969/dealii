712  template <int dim, int spacedim>
713  template <class... FETypes>
714  FECollection<dim, spacedim>::FECollection(const FETypes &... fes)
715  {
716  static_assert(
717  is_base_of_all<FiniteElement<dim, spacedim>, FETypes...>::value,
718  "Not all of the input arguments of this function "
719  "are derived from FiniteElement<dim,spacedim>!");
720 
721  // loop over all of the given arguments and add the finite elements to
722  // this collection. Inlining the definition of fe_pointers causes internal
723  // compiler errors on GCC 7.1.1 so we define it separately:
724  const auto fe_pointers = {
725  (static_cast<const FiniteElement<dim, spacedim> *>(&fes))...};
726  for (const auto p : fe_pointers)
727  push_back(*p);
728  }
