101  template <int dim>
102  std::complex<double>
103  InitialValues<dim>::value(const Point<dim> & p,
104  const unsigned int component) const
105  {
106  static_assert(dim == 2, "This initial condition only works in 2d.");
107 
108  (void)component;
109  Assert(component == 0, ExcIndexRange(component, 0, 1));
110 
111  const std::vector<Point<dim>> vortex_centers = {{0, -0.3},
112  {0, +0.3},
113  {+0.3, 0},
114  {-0.3, 0}};
115 
116  const double R = 0.1;
117  const double alpha =
118  1. / (std::pow(R, dim) * std::pow(numbers::PI, dim / 2.));
119 
120  double sum = 0;
121  for (const auto &vortex_center : vortex_centers)
122  {
123  const Tensor<1, dim> distance = p - vortex_center;
124  const double r = distance.norm();
125 
126  sum += alpha * std::exp(-(r * r) / (R * R));
127  }
128 
129  return {std::sqrt(sum), 0.};
130  }

200  template <int dim>
201  void NonlinearSchroedingerEquation<dim>::assemble_matrices()
202  {
203  const QGauss<dim> quadrature_formula(fe.degree + 1);
204 
205  FEValues<dim> fe_values(fe,
206  quadrature_formula,
207  update_values | update_gradients |
208  update_quadrature_points | update_JxW_values);
209 
210  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
211  const unsigned int n_q_points = quadrature_formula.size();
212 
213  FullMatrix<std::complex<double>> cell_matrix_lhs(dofs_per_cell,
214  dofs_per_cell);
215  FullMatrix<std::complex<double>> cell_matrix_rhs(dofs_per_cell,
216  dofs_per_cell);
217 
218  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
219  std::vector<double> potential_values(n_q_points);
220  const Potential<dim> potential;
221 
222  for (const auto &cell : dof_handler.active_cell_iterators())
223  {
224  cell_matrix_lhs = std::complex<double>(0.);
225  cell_matrix_rhs = std::complex<double>(0.);
226 
227  fe_values.reinit(cell);
228 
229  potential.value_list(fe_values.get_quadrature_points(),
230  potential_values);
231 
232  for (unsigned int q_index = 0; q_index < n_q_points; ++q_index)
233  {
234  for (unsigned int k = 0; k < dofs_per_cell; ++k)
235  {
236  for (unsigned int l = 0; l < dofs_per_cell; ++l)
237  {
238  const std::complex<double> i = {0, 1};
239 
240  cell_matrix_lhs(k, l) +=
241  (-i * fe_values.shape_value(k, q_index) *
242  fe_values.shape_value(l, q_index) +
243  time_step / 4 * fe_values.shape_grad(k, q_index) *
244  fe_values.shape_grad(l, q_index) +
245  time_step / 2 * potential_values[q_index] *
246  fe_values.shape_value(k, q_index) *
247  fe_values.shape_value(l, q_index)) *
248  fe_values.JxW(q_index);
249 
250  cell_matrix_rhs(k, l) +=
251  (-i * fe_values.shape_value(k, q_index) *
252  fe_values.shape_value(l, q_index) -
253  time_step / 4 * fe_values.shape_grad(k, q_index) *
254  fe_values.shape_grad(l, q_index) -
255  time_step / 2 * potential_values[q_index] *
256  fe_values.shape_value(k, q_index) *
257  fe_values.shape_value(l, q_index)) *
258  fe_values.JxW(q_index);
259  }
260  }
261  }
262 
263  cell->get_dof_indices(local_dof_indices);
264  constraints.distribute_local_to_global(cell_matrix_lhs,
265  local_dof_indices,
266  system_matrix);
267  constraints.distribute_local_to_global(cell_matrix_rhs,
268  local_dof_indices,
269  rhs_matrix);
270  }
271  }

276  void NonlinearSchroedingerEquation<dim>::do_half_phase_step()
277  {

283  value = std::exp(-i * kappa * magnitude * magnitude * (time_step / 2)) *

424  void NonlinearSchroedingerEquation<dim>::run()
425  {
426  setup_system();
427  assemble_matrices();
428 
429  time = 0;
430  VectorTools::interpolate(dof_handler, InitialValues<dim>(), solution);
431  output_results();
432 
433  const double end_time = 1;
434  for (; time <= end_time; time += time_step)

436  ++timestep_number;

439  << std::endl;
440 
441  do_half_phase_step();
442  do_full_spatial_step();
443  do_half_phase_step();
444 
445  if (timestep_number % 1 == 0)
446  output_results();
447  }
