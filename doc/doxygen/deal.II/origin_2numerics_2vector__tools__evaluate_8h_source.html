177  template <typename T>
178  T
179  reduce(const EvaluationFlags::EvaluationFlags &flags,
180  const ArrayView<const T> & values)
181  {
182  switch (flags)
183  {
184  case EvaluationFlags::avg:
185  {
186  return std::accumulate(values.begin(), values.end(), T{}) /
187  (T(1.0) * values.size());
188  }
189  case EvaluationFlags::max:
190  return *std::max_element(values.begin(), values.end());
191  case EvaluationFlags::min:
192  return *std::min_element(values.begin(), values.end());
193  case EvaluationFlags::insert:
194  return values[0];
195  default:
196  Assert(false, ExcNotImplemented());
197  return values[0];
198  }
199  }

270 
271  const auto get_evaluator = [&](const unsigned int active_fe_index)
272  -> FEPointEvaluation<n_components, dim> & {
273  if (evaluators[active_fe_index] == nullptr)
274  evaluators[active_fe_index] =
275  std::make_unique<FEPointEvaluation<n_components, dim>>(
276  cache.get_mapping(),
277  dof_handler.get_fe(active_fe_index),
278  update_flags);
279 
280  return *evaluators[active_fe_index];
281  };
282 
283  for (unsigned int i = 0; i < cell_data.cells.size(); ++i)
284  {
285  typename DoFHandler<dim>::active_cell_iterator cell = {
286  &cache.get_triangulation(),
287  cell_data.cells[i].first,
288  cell_data.cells[i].second,
289  &dof_handler};
290 
291  const ArrayView<const Point<dim>> unit_points(
292  cell_data.reference_point_values.data() +
293  cell_data.reference_point_ptrs[i],
294  cell_data.reference_point_ptrs[i + 1] -
295  cell_data.reference_point_ptrs[i]);
296 
297  solution_values.resize(
298  dof_handler.get_fe(cell->active_fe_index()).n_dofs_per_cell());
299  cell->get_dof_values(vector,
300  solution_values.begin(),
301  solution_values.end());
302 
303  auto &evaluator = get_evaluator(cell->active_fe_index());
304 
305  evaluator.reinit(cell, unit_points);
306  evaluator.evaluate(solution_values, evaluation_flags);
307 
308  for (unsigned int q = 0; q < unit_points.size(); ++q)
309  values[q + cell_data.reference_point_ptrs[i]] =
310  process_quadrature_point(evaluator, q);
311  }
312  };

315  std::vector<value_type> buffer;
