319 template <typename Number>
320 DEAL_II_CONSTEXPR inline SymmetricTensor<4, dim, Number>
321 Physics::Elasticity::StandardTensors<dim>::Dev_P(
322  const Tensor<2, dim, Number> &F)
323 {
324  const Number det_F = determinant(F);
325  Assert(numbers::value_is_greater_than(det_F, 0.0),
326  ExcMessage("Deformation gradient has a negative determinant."));
327  const Tensor<2, dim, Number> C_ns = transpose(F) * F;
328  const SymmetricTensor<2, dim, Number> C = symmetrize(C_ns);
329  const SymmetricTensor<2, dim, Number> C_inv = symmetrize(invert(C_ns));
330 
331  // See Wriggers p46 equ 3.125 (but transpose indices)
332  SymmetricTensor<4, dim, Number> Dev_P =
333  outer_product(C, C_inv); // Dev_P = C_x_C_inv
334  Dev_P /= -dim; // Dev_P = -[1/dim]C_x_C_inv
335  Dev_P += SymmetricTensor<4, dim, Number>(S); // Dev_P = S - [1/dim]C_x_C_inv
336  Dev_P *=
337  std::pow(det_F, -2.0 / dim); // Dev_P = J^{-2/dim} [S - [1/dim]C_x_C_inv]
338 
339  return Dev_P;
340 }
