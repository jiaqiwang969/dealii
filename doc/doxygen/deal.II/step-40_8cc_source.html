171  template <int dim>
172  void LaplaceProblem<dim>::assemble_system()
173  {
174  TimerOutput::Scope t(computing_timer, "assembly");
175 
176  const QGauss<dim> quadrature_formula(fe.degree + 1);
177 
178  FEValues<dim> fe_values(fe,
179  quadrature_formula,
180  update_values | update_gradients |
181  update_quadrature_points | update_JxW_values);
182 
183  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
184  const unsigned int n_q_points = quadrature_formula.size();
185 
186  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
187  Vector<double> cell_rhs(dofs_per_cell);
188 
189  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
190 
191  for (const auto &cell : dof_handler.active_cell_iterators())
192  if (cell->is_locally_owned())
193  {
194  cell_matrix = 0.;
195  cell_rhs = 0.;
196 
197  fe_values.reinit(cell);
198 
199  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
200  {
201  const double rhs_value =
202  (fe_values.quadrature_point(q_point)[1] >
203  0.5 +
204  0.25 * std::sin(4.0 * numbers::PI *
205  fe_values.quadrature_point(q_point)[0]) ?
206  1. :
207  -1.);
208 
209  for (unsigned int i = 0; i < dofs_per_cell; ++i)
210  {
211  for (unsigned int j = 0; j < dofs_per_cell; ++j)
212  cell_matrix(i, j) += fe_values.shape_grad(i, q_point) *
213  fe_values.shape_grad(j, q_point) *
214  fe_values.JxW(q_point);
215 
216  cell_rhs(i) += rhs_value *
217  fe_values.shape_value(i, q_point) *
218  fe_values.JxW(q_point);
219  }
220  }
221 
222  cell->get_dof_indices(local_dof_indices);
223  constraints.distribute_local_to_global(cell_matrix,
224  cell_rhs,
225  local_dof_indices,
226  system_matrix,
227  system_rhs);
228  }
229 
230  system_matrix.compress(VectorOperation::add);
231  system_rhs.compress(VectorOperation::add);
232  }

238  void LaplaceProblem<dim>::solve()
239  {
240  TimerOutput::Scope t(computing_timer, "solve");
241  LA::MPI::Vector completely_distributed_solution(locally_owned_dofs,
242  mpi_communicator);
243 
244  SolverControl solver_control(dof_handler.n_dofs(), 1e-12);
245 
246 #ifdef USE_PETSC_LA
247  LA::SolverCG solver(solver_control, mpi_communicator);
248 #else
249  LA::SolverCG solver(solver_control);
250 #endif
251 
252  LA::MPI::PreconditionAMG preconditioner;
253 
254  LA::MPI::PreconditionAMG::AdditionalData data;
255 
256 #ifdef USE_PETSC_LA
257  data.symmetric_operator = true;
258 #else
259  /* Trilinos defaults are good */
260 #endif
261  preconditioner.initialize(system_matrix, data);
262 
263  solver.solve(system_matrix,
264  completely_distributed_solution,
265  system_rhs,
266  preconditioner);
267 
268  pcout << " Solved in " << solver_control.last_step() << " iterations."
269  << std::endl;
270 
271  constraints.distribute(completely_distributed_solution);
272 
273  locally_relevant_solution = completely_distributed_solution;

320  template <int dim>
321  void LaplaceProblem<dim>::run()
322  {
323  pcout << "Running with "
324 #ifdef USE_PETSC_LA
325  << "PETSc"
326 #else
327  << "Trilinos"
328 #endif
329  << " on " << Utilities::MPI::n_mpi_processes(mpi_communicator)
330  << " MPI rank(s)..." << std::endl;
331 
332  const unsigned int n_cycles = 8;
333  for (unsigned int cycle = 0; cycle < n_cycles; ++cycle)
334  {
335  pcout << "Cycle " << cycle << ':' << std::endl;
336 
337  if (cycle == 0)
338  {
339  GridGenerator::hyper_cube(triangulation);
340  triangulation.refine_global(5);
341  }
342  else
343  refine_grid();
344 
345  setup_system();
346 
347  pcout << " Number of active cells: "
348  << triangulation.n_global_active_cells() << std::endl
349  << " Number of degrees of freedom: " << dof_handler.n_dofs()
350  << std::endl;
351 
352  assemble_system();
353  solve();
354 
355  if (Utilities::MPI::n_mpi_processes(mpi_communicator) <= 32)
356  {
357  TimerOutput::Scope t(computing_timer, "output");
358  output_results(cycle);
359  }
360 
361  computing_timer.print_summary();
362  computing_timer.reset();
363 
364  pcout << std::endl;
365  }
