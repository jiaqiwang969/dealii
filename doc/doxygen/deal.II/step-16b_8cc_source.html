165  void LaplaceProblem<dim>::setup_system()
166  {
167  dof_handler.distribute_dofs(fe);
168  dof_handler.distribute_mg_dofs();
169 
170  deallog << " Number of degrees of freedom: " << dof_handler.n_dofs()
171  << " (by level: ";
172  for (unsigned int level = 0; level < triangulation.n_levels(); ++level)
173  deallog << dof_handler.n_dofs(level)
174  << (level == triangulation.n_levels() - 1 ? ")" : ", ");
175  deallog << std::endl;
176 
177  DynamicSparsityPattern dsp(dof_handler.n_dofs(), dof_handler.n_dofs());
178  DoFTools::make_sparsity_pattern(dof_handler, dsp);
179 
180  solution.reinit(dof_handler.n_dofs());
181  system_rhs.reinit(dof_handler.n_dofs());
182 
183  constraints.clear();
184  DoFTools::make_hanging_node_constraints(dof_handler, constraints);
185 
186  std::set<types::boundary_id> dirichlet_boundary_ids = {0};
187  Functions::ZeroFunction<dim> homogeneous_dirichlet_bc;
188  const std::map<types::boundary_id, const Function<dim> *>
189  dirichlet_boundary_functions = {
190  {types::boundary_id(0), &homogeneous_dirichlet_bc}};
191  VectorTools::interpolate_boundary_values(dof_handler,
192  dirichlet_boundary_functions,
193  constraints);
194  constraints.close();
195  constraints.condense(dsp);
196  sparsity_pattern.copy_from(dsp);
197  system_matrix.reinit(sparsity_pattern);
198 
199  mg_constrained_dofs.clear();
200  mg_constrained_dofs.initialize(dof_handler);
201  mg_constrained_dofs.make_zero_boundary_constraints(dof_handler,
202  dirichlet_boundary_ids);
203 
204 
205  const unsigned int n_levels = triangulation.n_levels();
206 
207  mg_interface_in.resize(0, n_levels - 1);
208  mg_interface_in.clear_elements();
209  mg_interface_out.resize(0, n_levels - 1);
210  mg_interface_out.clear_elements();
211  mg_matrices.resize(0, n_levels - 1);
212  mg_matrices.clear_elements();
213  mg_sparsity_patterns.resize(0, n_levels - 1);
214 
215  for (unsigned int level = 0; level < n_levels; ++level)
216  {
217  DynamicSparsityPattern dsp(dof_handler.n_dofs(level),
218  dof_handler.n_dofs(level));
219  MGTools::make_sparsity_pattern(dof_handler, dsp, level);
220 
221  mg_sparsity_patterns[level].copy_from(dsp);
222 
223  mg_matrices[level].reinit(mg_sparsity_patterns[level]);

225  mg_interface_out[level].reinit(mg_sparsity_patterns[level]);

232  void LaplaceProblem<dim>::assemble_system()
233  {
234  MappingQ1<dim> mapping;
235  MeshWorker::IntegrationInfoBox<dim> info_box;
236  UpdateFlags update_flags =
237  update_values | update_gradients | update_hessians;
238  info_box.add_update_flags_all(update_flags);
239  info_box.initialize(fe, mapping);
240 
241  MeshWorker::DoFInfo<dim> dof_info(dof_handler);
242 
243  MeshWorker::Assembler::SystemSimple<SparseMatrix<double>, Vector<double>>
244  assembler;
245  assembler.initialize(constraints);
246  assembler.initialize(system_matrix, system_rhs);
247 
248  LaplaceIntegrator<dim> matrix_integrator;
249  MeshWorker::integration_loop<dim, dim>(dof_handler.begin_active(),
250  dof_handler.end(),
251  dof_info,
252  info_box,
253  matrix_integrator,
254  assembler);
255 
256  for (unsigned int i = 0; i < dof_handler.n_dofs(); ++i)
257  if (constraints.is_constrained(i))
258  system_matrix.set(i, i, 1.);

301  template <int dim>
302  void LaplaceProblem<dim>::solve()
303  {
304  MGTransferPrebuilt<Vector<double>> mg_transfer(mg_constrained_dofs);
305  mg_transfer.build(dof_handler);
306 
307  FullMatrix<double> coarse_matrix;
308  coarse_matrix.copy_from(mg_matrices[0]);
309  MGCoarseGridHouseholder<double, Vector<double>> coarse_grid_solver;
310  coarse_grid_solver.initialize(coarse_matrix);
311 
312  using Smoother = PreconditionSOR<SparseMatrix<double>>;
313  mg::SmootherRelaxation<Smoother, Vector<double>> mg_smoother;
314  mg_smoother.initialize(mg_matrices);
315  mg_smoother.set_steps(2);
316  mg_smoother.set_symmetric(true);
317 
318  mg::Matrix<Vector<double>> mg_matrix(mg_matrices);
319  mg::Matrix<Vector<double>> mg_interface_up(mg_interface_in);
320  mg::Matrix<Vector<double>> mg_interface_down(mg_interface_out);
321 
322  Multigrid<Vector<double>> mg(
323  mg_matrix, coarse_grid_solver, mg_transfer, mg_smoother, mg_smoother);
324  mg.set_edge_matrices(mg_interface_down, mg_interface_up);
325 
326  PreconditionMG<dim, Vector<double>, MGTransferPrebuilt<Vector<double>>>
327  preconditioner(dof_handler, mg, mg_transfer);
328 
329  SolverControl solver_control(1000, 1e-12);
330  SolverCG<Vector<double>> solver(solver_control);
331 
332  solution = 0;
333 
334  solver.solve(system_matrix, solution, system_rhs, preconditioner);
335  constraints.distribute(solution);
336  }

376  template <int dim>
377  void LaplaceProblem<dim>::run()
378  {
379  for (unsigned int cycle = 0; cycle < 8; ++cycle)
380  {
381  deallog << "Cycle " << cycle << std::endl;
382 
383  if (cycle == 0)
384  {
385  GridGenerator::hyper_ball(triangulation);
386  triangulation.refine_global(1);
387  }
388  else
389  refine_grid();
390 
391  deallog << " Number of active cells: "
392  << triangulation.n_active_cells() << std::endl;
393 
394  setup_system();
395 
396  assemble_system();
397  assemble_multigrid();
398 
399  solve();
400  output_results(cycle);
401  }
