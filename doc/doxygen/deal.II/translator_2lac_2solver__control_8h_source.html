92  virtual void
93  print_info(std::ostream &out) const override
94  {
95  out
96  << "Iterative method reported convergence failure in step " << last_step
97  << ". The residual in the last step was " << last_residual << ".\n\n"
98  << "This error message can indicate that you have simply not allowed "
99  << "a sufficiently large number of iterations for your iterative solver "
100  << "to converge. This often happens when you increase the size of your "
101  << "problem. In such cases, the last residual will likely still be very "
102  << "small, and you can make the error go away by increasing the allowed "
103  << "number of iterations when setting up the SolverControl object that "
104  << "determines the maximal number of iterations you allow."
105  << "\n\n"
106  << "The other situation where this error may occur is when your matrix "
107  << "is not invertible (e.g., your matrix has a null-space), or if you "
108  << "try to apply the wrong solver to a matrix (e.g., using CG for a "
109  << "matrix that is not symmetric or not positive definite). In these "
110  << "cases, the residual in the last iteration is likely going to be large."
111  << std::endl;
112  }
