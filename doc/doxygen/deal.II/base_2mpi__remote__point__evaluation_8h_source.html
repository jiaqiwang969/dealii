320  buffer.size() / 2);

323  evaluation_function(buffer_1, cell_data);
324

326  for (unsigned int i = 0; i < send_permutation.size(); ++i)
327  buffer_2[send_permutation[i]] = buffer_1[i];

329  // process remote quadrature points and send them away
330  std::map<unsigned int, std::vector<char>> temp_map;
331 
332  std::vector<MPI_Request> requests;
333  requests.reserve(send_ranks.size());
334 
335  const unsigned int my_rank =
336  Utilities::MPI::this_mpi_process(tria->get_communicator());
337 
338  std::map<unsigned int, std::vector<T>> temp_recv_map;
339 
340  for (unsigned int i = 0; i < send_ranks.size(); ++i)
341  {
342  if (send_ranks[i] == my_rank)
343  {
344  // process locally-owned values
345  temp_recv_map[my_rank] =
346  std::vector<T>(buffer_2.begin() + send_ptrs[i],
347  buffer_2.begin() + send_ptrs[i + 1]);
348  continue;
349  }
350 
351  temp_map[send_ranks[i]] =
352  Utilities::pack(std::vector<T>(buffer_2.begin() + send_ptrs[i],
353  buffer_2.begin() + send_ptrs[i + 1]),
354  false);
355 
356  auto &buffer = temp_map[send_ranks[i]];
357 
358  requests.push_back(MPI_Request());
359 
360  const int ierr = MPI_Isend(buffer.data(),
361  buffer.size(),
362  MPI_CHAR,
363  send_ranks[i],
364  internal::Tags::remote_point_evaluation,
365  tria->get_communicator(),
366  &requests.back());
367  AssertThrowMPI(ierr);
368  }

371  {
372  if (recv_rank == my_rank)
373  continue;

376  int ierr = MPI_Probe(MPI_ANY_SOURCE,
377  internal::Tags::remote_point_evaluation,
378  tria->get_communicator(),
379  &status);
380  AssertThrowMPI(ierr);
381 
382  int message_length;

384  AssertThrowMPI(ierr);
385

389  buffer.size(),

394  MPI_STATUS_IGNORE);

398  Utilities::unpack<std::vector<T>>(buffer, false);
399  }
400

403  MPI_Waitall(requests.size(), requests.data(), MPI_STATUSES_IGNORE);
404  AssertThrowMPI(ierr);
405

407  auto it = recv_permutation.begin();
408  for (const auto &j : temp_recv_map)
409  for (const auto &i : j.second)
