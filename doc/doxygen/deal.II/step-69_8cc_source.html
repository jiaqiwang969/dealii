360  template <int dim>
361  void Discretization<dim>::setup()
362  {
363  TimerOutput::Scope scope(computing_timer, "discretization - setup");
364 
365  triangulation.clear();
366 
367  Triangulation<dim> tria1, tria2, tria3, tria4, tria5, tria6;
368 
369  GridGenerator::hyper_cube_with_cylindrical_hole(
370  tria1, disk_diameter / 2., disk_diameter, 0.5, 1, false);
371 
372  GridGenerator::subdivided_hyper_rectangle(
373  tria2,
374  {2, 1},
375  Point<2>(-disk_diameter, disk_diameter),
376  Point<2>(disk_diameter, height / 2.));
377 
378  GridGenerator::subdivided_hyper_rectangle(
379  tria3,
380  {2, 1},
381  Point<2>(-disk_diameter, -disk_diameter),
382  Point<2>(disk_diameter, -height / 2.));
383 
384  GridGenerator::subdivided_hyper_rectangle(
385  tria4,
386  {6, 2},
387  Point<2>(disk_diameter, -disk_diameter),
388  Point<2>(length - disk_position, disk_diameter));
389 
390  GridGenerator::subdivided_hyper_rectangle(
391  tria5,
392  {6, 1},
393  Point<2>(disk_diameter, disk_diameter),
394  Point<2>(length - disk_position, height / 2.));
395 
396  GridGenerator::subdivided_hyper_rectangle(
397  tria6,
398  {6, 1},
399  Point<2>(disk_diameter, -height / 2.),
400  Point<2>(length - disk_position, -disk_diameter));
401 
402  GridGenerator::merge_triangulations(
403  {&tria1, &tria2, &tria3, &tria4, &tria5, &tria6},
404  triangulation,
405  1.e-12,
406  true);
407 
408  triangulation.set_manifold(0, PolarManifold<2>(Point<2>()));
409 
410 
411  for (const auto &cell : triangulation.active_cell_iterators())
412  {
413  for (const auto v : cell->vertex_indices())
414  {
415  if (cell->vertex(v)[0] <= -disk_diameter + 1.e-6)
416  cell->vertex(v)[0] = -disk_position;
417  }
418  }
419 
420  for (const auto &cell : triangulation.active_cell_iterators())
421  {
422  for (const auto f : cell->face_indices())
423  {
424  const auto face = cell->face(f);
425 
426  if (face->at_boundary())
427  {
428  const auto center = face->center();
429 
430  if (center[0] > length - disk_position - 1.e-6)
431  face->set_boundary_id(Boundaries::do_nothing);
432  else if (center[0] < -disk_position + 1.e-6)
433  face->set_boundary_id(Boundaries::dirichlet);
434  else
435  face->set_boundary_id(Boundaries::free_slip);
436  }
437  }
438  }
439 
440  triangulation.refine_global(refinement);
441  }

490  const auto dofs_per_cell =
491  discretization->finite_element.n_dofs_per_cell();
492  std::vector<types::global_dof_index> dof_indices(dofs_per_cell);
493 
494  for (const auto &cell : dof_handler.active_cell_iterators())
495  {
496  if (cell->is_artificial())
497  continue;
498 
499  /* We transform the set of global dof indices on the cell to the
500  * corresponding "local" index range on the MPI process: */
501  cell->get_dof_indices(dof_indices);
502  std::transform(dof_indices.begin(),
503  dof_indices.end(),
504  dof_indices.begin(),
505  [&](types::global_dof_index index) {
506  return partitioner->global_to_local(index);
507  });
508 
509  /* And simply add, for each dof, a coupling to all other "local"
510  * dofs on the cell: */
511  for (const auto dof : dof_indices)
512  dsp.add_entries(dof, dof_indices.begin(), dof_indices.end());
513  }
514 
515  sparsity_pattern.copy_from(dsp);
516 
517  lumped_mass_matrix.reinit(sparsity_pattern);
518  norm_matrix.reinit(sparsity_pattern);
519  for (auto &matrix : cij_matrix)
520  matrix.reinit(sparsity_pattern);
521  for (auto &matrix : nij_matrix)
522  matrix.reinit(sparsity_pattern);

615  template <int dim>
616  void OfflineData<dim>::assemble()
617  {
618  lumped_mass_matrix = 0.;
619  norm_matrix = 0.;
620  for (auto &matrix : cij_matrix)
621  matrix = 0.;
622  for (auto &matrix : nij_matrix)
623  matrix = 0.;
624 
625  unsigned int dofs_per_cell =
626  discretization->finite_element.n_dofs_per_cell();
627  unsigned int n_q_points = discretization->quadrature.size();
628 
629 
630  MeshWorker::ScratchData<dim> scratch_data(
631  discretization->mapping,
632  discretization->finite_element,
633  discretization->quadrature,
634  update_values | update_gradients | update_quadrature_points |
635  update_JxW_values,
636  discretization->face_quadrature,
637  update_normal_vectors | update_values | update_JxW_values);
638 
639  {
640  TimerOutput::Scope scope(
641  computing_timer,
642  "offline_data - assemble lumped mass matrix, and c_ij");
643 
644  const auto local_assemble_system =
645  [&](const typename DoFHandler<dim>::cell_iterator &cell,
646  MeshWorker::ScratchData<dim> & scratch,
647  CopyData<dim> & copy) {
648  copy.is_artificial = cell->is_artificial();
649  if (copy.is_artificial)
650  return;
651 
652  copy.local_boundary_normal_map.clear();
653  copy.cell_lumped_mass_matrix.reinit(dofs_per_cell, dofs_per_cell);
654  for (auto &matrix : copy.cell_cij_matrix)
655  matrix.reinit(dofs_per_cell, dofs_per_cell);
656 
657  const auto &fe_values = scratch.reinit(cell);
658 
659  copy.local_dof_indices.resize(dofs_per_cell);
660  cell->get_dof_indices(copy.local_dof_indices);
661 
662  std::transform(copy.local_dof_indices.begin(),
663  copy.local_dof_indices.end(),
664  copy.local_dof_indices.begin(),
665  [&](types::global_dof_index index) {
666  return partitioner->global_to_local(index);
667  });
668 
669  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
670  {
671  const auto JxW = fe_values.JxW(q_point);
672 
673  for (unsigned int j = 0; j < dofs_per_cell; ++j)
674  {
675  const auto value_JxW =
676  fe_values.shape_value(j, q_point) * JxW;
677  const auto grad_JxW = fe_values.shape_grad(j, q_point) * JxW;
678 
679  copy.cell_lumped_mass_matrix(j, j) += value_JxW;
680 
681  for (unsigned int i = 0; i < dofs_per_cell; ++i)
682  {
683  const auto value = fe_values.shape_value(i, q_point);
684  for (unsigned int d = 0; d < dim; ++d)
685  copy.cell_cij_matrix[d](i, j) += value * grad_JxW[d];
686 
687  } /* i */
688  } /* j */
689  } /* q */
690 
691  for (const auto f : cell->face_indices())
692  {
693  const auto face = cell->face(f);
694  const auto id = face->boundary_id();
695 
696  if (!face->at_boundary())
697  continue;
698 
699  const auto &fe_face_values = scratch.reinit(cell, f);
700 
701  const unsigned int n_face_q_points =
702  fe_face_values.get_quadrature().size();
703 
704  for (unsigned int j = 0; j < dofs_per_cell; ++j)
705  {
706  if (!discretization->finite_element.has_support_on_face(j, f))
707  continue;
708 
709  Tensor<1, dim> normal;
710  if (id == Boundaries::free_slip)
711  {
712  for (unsigned int q = 0; q < n_face_q_points; ++q)

938  DEAL_II_ALWAYS_INLINE inline double
939  lambda_max_two_rarefaction(const std::array<double, 4> &riemann_data_i,
940  const std::array<double, 4> &riemann_data_j)
941  {
942  constexpr double gamma = ProblemDescription<1>::gamma;
943  const auto u_i = riemann_data_i[1];
944  const auto p_i = riemann_data_i[2];
945  const auto a_i = riemann_data_i[3];
946  const auto u_j = riemann_data_j[1];
947  const auto p_j = riemann_data_j[2];
948  const auto a_j = riemann_data_j[3];
949 
950  const double numerator = a_i + a_j - (gamma - 1.) / 2. * (u_j - u_i);
951 
952  const double denominator =
953  a_i * std::pow(p_i / p_j, -1. * (gamma - 1.) / 2. / gamma) + a_j * 1.;
954 
955  /* Formula (4.3) in Guermond-Popov-2016 */
956 
957  const double p_star =
958  p_j * std::pow(numerator / denominator, 2. * gamma / (gamma - 1));
959 
960  const double lambda1 = lambda1_minus(riemann_data_i, p_star);
961  const double lambda3 = lambda3_plus(riemann_data_j, p_star);
962 
963  /* Formula (2.11) in Guermond-Popov-2016 */
964 
965  return std::max(positive_part(lambda3), negative_part(lambda1));
966  }
