451  template <int dim>
452  double BoussinesqFlowProblem<dim>::compute_viscosity(
453  const std::vector<double> & old_temperature,
454  const std::vector<double> & old_old_temperature,
455  const std::vector<Tensor<1, dim>> &old_temperature_grads,
456  const std::vector<Tensor<1, dim>> &old_old_temperature_grads,
457  const std::vector<double> & old_temperature_laplacians,
458  const std::vector<double> & old_old_temperature_laplacians,
459  const std::vector<Tensor<1, dim>> &old_velocity_values,
460  const std::vector<Tensor<1, dim>> &old_old_velocity_values,
461  const std::vector<double> & gamma_values,
462  const double global_u_infty,
463  const double global_T_variation,
464  const double cell_diameter) const
465  {
466  constexpr double beta = 0.017 * dim;
467  constexpr double alpha = 1.0;
468 
469  if (global_u_infty == 0)
470  return 5e-3 * cell_diameter;
471 
472  const unsigned int n_q_points = old_temperature.size();
473 
474  double max_residual = 0;
475  double max_velocity = 0;
476 
477  for (unsigned int q = 0; q < n_q_points; ++q)
478  {
479  const Tensor<1, dim> u =
480  (old_velocity_values[q] + old_old_velocity_values[q]) / 2;
481 
482  const double dT_dt =
483  (old_temperature[q] - old_old_temperature[q]) / old_time_step;
484  const double u_grad_T =
485  u * (old_temperature_grads[q] + old_old_temperature_grads[q]) / 2;
486 
487  const double kappa_Delta_T =
488  EquationData::kappa *
489  (old_temperature_laplacians[q] + old_old_temperature_laplacians[q]) /
490  2;
491 
492  const double residual =
493  std::abs((dT_dt + u_grad_T - kappa_Delta_T - gamma_values[q]) *
494  std::pow((old_temperature[q] + old_old_temperature[q]) / 2,
495  alpha - 1.));
496 
497  max_residual = std::max(residual, max_residual);
498  max_velocity = std::max(std::sqrt(u * u), max_velocity);
499  }
500 
501  const double c_R = std::pow(2., (4. - 2 * alpha) / dim);
502  const double global_scaling = c_R * global_u_infty * global_T_variation *
503  std::pow(global_Omega_diameter, alpha - 2.);
504 
505  return (
506  beta * max_velocity *
507  std::min(cell_diameter,
508  std::pow(cell_diameter, alpha) * max_residual / global_scaling));
509  }
