313  void
314  BlockSchurPreconditioner<PreconditionerAType, PreconditionerSType>::vmult(
315  BlockVector<double> & dst,
316  const BlockVector<double> &src) const
317  {
318  Vector<double> utmp(src.block(0));
319 
320  {
321  SolverControl solver_control(1000, 1e-6 * src.block(1).l2_norm());
322  SolverCG<Vector<double>> cg(solver_control);
323 
324  dst.block(1) = 0.0;
325  cg.solve(schur_complement_matrix,
326  dst.block(1),
327  src.block(1),
328  preconditioner_S);
329 
330  n_iterations_S += solver_control.last_step();
331  dst.block(1) *= -1.0;
332  }
333 
334  {
335  system_matrix.block(0, 1).vmult(utmp, dst.block(1));
336  utmp *= -1.0;
337  utmp += src.block(0);
338  }
339 
340  if (do_solve_A == true)
341  {
342  SolverControl solver_control(10000, utmp.l2_norm() * 1e-4);
343  SolverCG<Vector<double>> cg(solver_control);
344 
345  dst.block(0) = 0.0;
346  cg.solve(system_matrix.block(0, 0),
347  dst.block(0),
348  utmp,
349  preconditioner_A);
350 
351  n_iterations_A += solver_control.last_step();
352  }
353  else

355  preconditioner_A.vmult(dst.block(0), utmp);

520  template <int dim>
521  void StokesProblem<dim>::assemble_system()
522  {
523  TimerOutput::Scope assemble(computing_timer, "Assemble");
524  system_matrix = 0;
525  system_rhs = 0;
526 
527  const bool assemble_pressure_mass_matrix =
528  (solver_type == SolverType::UMFPACK) ? false : true;
529 
530  QGauss<dim> quadrature_formula(pressure_degree + 2);
531 
532  FEValues<dim> fe_values(fe,
533  quadrature_formula,
534  update_values | update_quadrature_points |
535  update_JxW_values | update_gradients);
536 
537  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
538 
539  const unsigned int n_q_points = quadrature_formula.size();
540 
541  FullMatrix<double> local_matrix(dofs_per_cell, dofs_per_cell);
542  Vector<double> local_rhs(dofs_per_cell);
543 
544  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
545 
546  const RightHandSide<dim> right_hand_side;
547  std::vector<Vector<double>> rhs_values(n_q_points, Vector<double>(dim + 1));
548 
549  const FEValuesExtractors::Vector velocities(0);
550  const FEValuesExtractors::Scalar pressure(dim);
551 
552  std::vector<SymmetricTensor<2, dim>> symgrad_phi_u(dofs_per_cell);
553  std::vector<double> div_phi_u(dofs_per_cell);
554  std::vector<double> phi_p(dofs_per_cell);
555 
556  for (const auto &cell : dof_handler.active_cell_iterators())
557  {
558  fe_values.reinit(cell);
559  local_matrix = 0;
560  local_rhs = 0;
561 
562  right_hand_side.vector_value_list(fe_values.get_quadrature_points(),
563  rhs_values);
564 
565  for (unsigned int q = 0; q < n_q_points; ++q)
566  {
567  for (unsigned int k = 0; k < dofs_per_cell; ++k)
568  {
569  symgrad_phi_u[k] =
570  fe_values[velocities].symmetric_gradient(k, q);
571  div_phi_u[k] = fe_values[velocities].divergence(k, q);
572  phi_p[k] = fe_values[pressure].value(k, q);
573  }
574 
575  for (unsigned int i = 0; i < dofs_per_cell; ++i)
576  {
577  for (unsigned int j = 0; j <= i; ++j)
578  {
579  local_matrix(i, j) +=
580  (2 * (symgrad_phi_u[i] * symgrad_phi_u[j]) -
581  div_phi_u[i] * phi_p[j] - phi_p[i] * div_phi_u[j] +
582  (assemble_pressure_mass_matrix ? phi_p[i] * phi_p[j] :
583  0)) *
584  fe_values.JxW(q);
585  }
586 
587  const unsigned int component_i =
588  fe.system_to_component_index(i).first;
589  local_rhs(i) += fe_values.shape_value(i, q) *
590  rhs_values[q](component_i) * fe_values.JxW(q);
591  }
592  }
593 
594  for (unsigned int i = 0; i < dofs_per_cell; ++i)
595  for (unsigned int j = i + 1; j < dofs_per_cell; ++j)
596  local_matrix(i, j) = local_matrix(j, i);
597 
598  cell->get_dof_indices(local_dof_indices);
599  constraints.distribute_local_to_global(local_matrix,
600  local_rhs,
601  local_dof_indices,
602  system_matrix,
603  system_rhs);
604  }
605 
606  if (solver_type != SolverType::UMFPACK)
607  {
608  pressure_mass_matrix.reinit(sparsity_pattern.block(1, 1));
609  pressure_mass_matrix.copy_from(system_matrix.block(1, 1));
610  system_matrix.block(1, 1) = 0;
611  }
612  }
