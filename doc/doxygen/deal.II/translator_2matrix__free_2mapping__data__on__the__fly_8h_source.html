223  template <int dim, typename Number, typename VectorizedArrayType>
224  inline void
225  MappingDataOnTheFly<dim, Number, VectorizedArrayType>::reinit(
226  typename ::Triangulation<dim>::cell_iterator cell)
227  {
228  if (present_cell == cell)
229  return;
230  present_cell = cell;
231  fe_values.reinit(present_cell);
232  for (unsigned int q = 0; q < fe_values.get_quadrature().size(); ++q)
233  {
234  if (fe_values.get_update_flags() & update_JxW_values)
235  mapping_info_storage.JxW_values[q] = fe_values.JxW(q);
236  if (fe_values.get_update_flags() & update_jacobians)
237  {
238  Tensor<2, dim> jac = fe_values.jacobian(q);
239  jac = invert(transpose(jac));
240  for (unsigned int d = 0; d < dim; ++d)
241  for (unsigned int e = 0; e < dim; ++e)
242  mapping_info_storage.jacobians[0][q][d][e] = jac[d][e];
243  }
244  if (fe_values.get_update_flags() & update_quadrature_points)
245  for (unsigned int d = 0; d < dim; ++d)
246  mapping_info_storage.quadrature_points[q][d] =
247  fe_values.quadrature_point(q)[d];
248  if (fe_values.get_update_flags() & update_normal_vectors)
249  {
250  for (unsigned int d = 0; d < dim; ++d)
251  mapping_info_storage.normal_vectors[q][d] =
252  fe_values.normal_vector(q)[d];
253  mapping_info_storage.normals_times_jacobians[0][q] =
254  mapping_info_storage.normal_vectors[q] *
255  mapping_info_storage.jacobians[0][q];
256  }
257  }
258  }
