1895 template <int dim, int spacedim>
1896 template <typename number>
1897 types::global_dof_index
1898 DoFHandler<dim, spacedim>::n_boundary_dofs(
1899  const std::map<types::boundary_id, const Function<spacedim, number> *>
1900  &boundary_ids) const
1901 {
1902  Assert(!(dim == 2 && spacedim == 3) || this->hp_capability_enabled == false,
1903  ExcNotImplementedWithHP());
1904 
1905  // extract the set of boundary ids and forget about the function object
1906  // pointers
1907  std::set<types::boundary_id> boundary_ids_only;
1908  for (typename std::map<types::boundary_id,
1909  const Function<spacedim, number> *>::const_iterator p =
1910  boundary_ids.begin();
1911  p != boundary_ids.end();
1912  ++p)
1913  boundary_ids_only.insert(p->first);
1914 
1915  // then just hand everything over to the other function that does the work
1916  return n_boundary_dofs(boundary_ids_only);
1917 }
