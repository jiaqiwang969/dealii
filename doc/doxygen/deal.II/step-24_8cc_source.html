159  template <int dim>
160  void TATForwardProblem<dim>::setup_system()
161  {
162  const Point<dim> center;
163  GridGenerator::hyper_ball(triangulation, center, 1.);
164  triangulation.refine_global(7);
165 
166  time_step = GridTools::minimal_cell_diameter(triangulation) / wave_speed /
167  std::sqrt(1. * dim);
168 
169  std::cout << "Number of active cells: " << triangulation.n_active_cells()
170  << std::endl;
171 
172  dof_handler.distribute_dofs(fe);
173 
174  std::cout << "Number of degrees of freedom: " << dof_handler.n_dofs()
175  << std::endl
176  << std::endl;
177 
178  DynamicSparsityPattern dsp(dof_handler.n_dofs(), dof_handler.n_dofs());
179  DoFTools::make_sparsity_pattern(dof_handler, dsp);
180  sparsity_pattern.copy_from(dsp);
181 
182  system_matrix.reinit(sparsity_pattern);
183  mass_matrix.reinit(sparsity_pattern);
184  laplace_matrix.reinit(sparsity_pattern);
185 
186  MatrixCreator::create_mass_matrix(dof_handler,
187  QGauss<dim>(fe.degree + 1),
188  mass_matrix);
189  MatrixCreator::create_laplace_matrix(dof_handler,
190  QGauss<dim>(fe.degree + 1),
191  laplace_matrix);
192 
193  boundary_matrix.reinit(sparsity_pattern);
194 
195  {
196  const QGauss<dim - 1> quadrature_formula(fe.degree + 1);
197  FEFaceValues<dim> fe_values(fe,
198  quadrature_formula,
199  update_values | update_JxW_values);
200 
201  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
202  const unsigned int n_q_points = quadrature_formula.size();
203 
204  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
205 
206  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
207 
208  for (const auto &cell : dof_handler.active_cell_iterators())
209  for (const auto &face : cell->face_iterators())
210  if (face->at_boundary())
211  {
212  cell_matrix = 0;
213 
214  fe_values.reinit(cell, face);
215 
216  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
217  for (unsigned int i = 0; i < dofs_per_cell; ++i)
218  for (unsigned int j = 0; j < dofs_per_cell; ++j)
219  cell_matrix(i, j) += (fe_values.shape_value(i, q_point) *
220  fe_values.shape_value(j, q_point) *
221  fe_values.JxW(q_point));
222 
223  cell->get_dof_indices(local_dof_indices);
224  for (unsigned int i = 0; i < dofs_per_cell; ++i)
225  for (unsigned int j = 0; j < dofs_per_cell; ++j)
226  boundary_matrix.add(local_dof_indices[i],
227  local_dof_indices[j],
228  cell_matrix(i, j));
229  }
230  }
231 
232  system_matrix.copy_from(mass_matrix);
233  system_matrix.add(time_step * time_step * theta * theta * wave_speed *
234  wave_speed,
235  laplace_matrix);
236  system_matrix.add(wave_speed * theta * time_step, boundary_matrix);
237 
238 
239  solution_p.reinit(dof_handler.n_dofs());
240  old_solution_p.reinit(dof_handler.n_dofs());
241  system_rhs_p.reinit(dof_handler.n_dofs());
242 
243  solution_v.reinit(dof_handler.n_dofs());
244  old_solution_v.reinit(dof_handler.n_dofs());
245  system_rhs_v.reinit(dof_handler.n_dofs());
246 
247  constraints.close();
248  }

304  template <int dim>
305  void TATForwardProblem<dim>::run()
306  {
307  setup_system();
308 
309  VectorTools::project(dof_handler,
310  constraints,
311  QGauss<dim>(fe.degree + 1),
312  InitialValuesP<dim>(),
313  old_solution_p);
314  old_solution_v = 0;
315 
316 
317  std::ofstream detector_data("detectors.dat");
318 
319  Vector<double> tmp(solution_p.size());
320  Vector<double> G1(solution_p.size());
321  Vector<double> G2(solution_v.size());
322 
323  const double end_time = 0.7;
324  for (time = time_step; time <= end_time;
325  time += time_step, ++timestep_number)
326  {
327  std::cout << std::endl;
328  std::cout << "time_step " << timestep_number << " @ t=" << time
329  << std::endl;
330 
331  mass_matrix.vmult(G1, old_solution_p);
332  mass_matrix.vmult(tmp, old_solution_v);
333  G1.add(time_step * (1 - theta), tmp);
334 
335  mass_matrix.vmult(G2, old_solution_v);
336  laplace_matrix.vmult(tmp, old_solution_p);
337  G2.add(-wave_speed * wave_speed * time_step * (1 - theta), tmp);
338 
339  boundary_matrix.vmult(tmp, old_solution_p);
340  G2.add(wave_speed, tmp);
341 
342  system_rhs_p = G1;
343  system_rhs_p.add(time_step * theta, G2);
344 
345  solve_p();
346 
347  system_rhs_v = G2;
348  laplace_matrix.vmult(tmp, solution_p);
349  system_rhs_v.add(-time_step * theta * wave_speed * wave_speed, tmp);
350 
351  boundary_matrix.vmult(tmp, solution_p);
352  system_rhs_v.add(-wave_speed, tmp);
353 
354  solve_v();
355 
356  output_results();
357 
358  detector_data << time;
359  for (unsigned int i = 0; i < detector_locations.size(); ++i)
360  detector_data << " "
361  << VectorTools::point_value(dof_handler,
362  solution_p,
363  detector_locations[i])
364  << " ";
365  detector_data << std::endl;
366 
367  old_solution_p = solution_p;
368  old_solution_v = solution_v;
369  }
370  }
