294  /*
295  * Initialize the state of an independent variable.
296  */
297  static void
298  independent_variable(const scalar_type &in,
299  const unsigned int index,
300  const unsigned int n_independent_variables,
301  ad_type & out)
302  {
303  // It is required that we first initialise the outer number before
304  // any of the nested ones.
305  out = ad_type(n_independent_variables, index, in);
306 
307  // Initialize potential nested directional derivatives
308  Marking<derivative_type>::independent_variable(
309  in, index, n_independent_variables, out.val());
310  }
311 
312  /*
313  * Initialize the state of a dependent variable.
314  */
315  static void
316  dependent_variable(ad_type &out, const ad_type &func)
317  {
318  out = func;
319  }
320  };

327  template <typename NumberType>
328  struct Marking<Sacado::Rad::ADvar<NumberType>>
329  {
330  using ad_type =
331  typename SacadoNumberInfo<Sacado::Rad::ADvar<NumberType>>::ad_type;
332  using derivative_type = typename SacadoNumberInfo<
333  Sacado::Rad::ADvar<NumberType>>::derivative_type;
334  using scalar_type = typename SacadoNumberInfo<
335  Sacado::Rad::ADvar<NumberType>>::scalar_type;
336 
337  /*
338  * Initialize the state of an independent variable.
339  */
340  static void
341  independent_variable(const scalar_type &in,
342  const unsigned int index,
343  const unsigned int n_independent_variables,
344  ad_type & out)
345  {
346  // For Sacado::Rad::ADvar numbers, we have to initialize the
347  // ADNumber with an already fully-configured value. This means
348  // that if this nests another ADNumber then the nested number
349  // must already be setup and ready for use.
350 
351  // Initialize potential nested directional derivatives
352  derivative_type derivative_initializer;
353  Marking<derivative_type>::independent_variable(
354  in, index, n_independent_variables, derivative_initializer);
355 
356  // Initialize the outer ad_type
357  out = derivative_initializer;
358  }

363  static void
364  dependent_variable(ad_type &out, const ad_type &func)
365  {
366  out = func;
