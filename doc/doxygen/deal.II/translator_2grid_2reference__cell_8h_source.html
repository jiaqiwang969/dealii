759  AssertIndexRange(vertex, 3);

790  {{0, 2, 3, X}},
791  {{0, 1, 4, X}},

793  {{1, 2, 3, 4}}}};
794

1628  return 1.0 - xi[std::min(0, dim - 1)] - xi[std::min(1, dim - 1)];
1629  case 1:
1630  return xi[std::min(0, dim - 1)];
1631  case 2:
1632  return xi[std::min(1, dim - 1)];
1633  }
1634  }
1635 
1636  if (*this ==
1637  ReferenceCells::Tetrahedron) // see also
1638  // BarycentricPolynomials<3>::compute_value

1640  switch (i)
1641  {
1642  case 0:

1649  case 3:

1661  i % 3) *
1662  ReferenceCell(ReferenceCells::Line)
