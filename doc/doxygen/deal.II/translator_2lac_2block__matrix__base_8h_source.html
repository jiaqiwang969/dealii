1287  Accessor<BlockMatrixType, true>::advance()
1288  {
1289  Assert(this->row_block != numbers::invalid_unsigned_int,
1290  ExcIteratorPastEnd());
1291  Assert(this->col_block != numbers::invalid_unsigned_int,
1292  ExcIteratorPastEnd());

1295  size_type local_row = base_iterator->row();

1298  // current block
1299  ++base_iterator;
1300

1302  // block (which may happen multiple

1305  // block and take the

1316  matrix->block(this->row_block, this->col_block).begin(local_row);

1321  // first block column
