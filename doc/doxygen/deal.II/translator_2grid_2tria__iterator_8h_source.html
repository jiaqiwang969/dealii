1066 template <typename Accessor>
1067 inline bool
1068 TriaRawIterator<Accessor>::
1069 operator<(const TriaRawIterator<Accessor> &other) const
1070 {
1071  Assert(state() != IteratorState::invalid,
1072  ExcDereferenceInvalidObject(accessor));
1073  Assert(other.state() != IteratorState::invalid,
1074  ExcDereferenceInvalidObject(other.accessor));
1075 
1076  Assert(&accessor.get_triangulation() == &other.accessor.get_triangulation(),
1077  ExcInvalidComparison());
1078 
1079  // Deal with iterators past end
1080  if (state() == IteratorState::past_the_end)
1081  return false;
1082  if (other.state() == IteratorState::past_the_end)
1083  return true;
1084 
1085  return ((**this) < (*other));
1086 }
