366 template <typename number>
367 inline void
368 PreconditionBlockBase<number>::reinit(unsigned int n,
369  size_type b,
370  bool compress,
371  Inversion method)
372 {
373  inversion = method;
374  var_same_diagonal = compress;
375  var_inverses_ready = false;
376  n_diagonal_blocks = n;
377 
378  if (compress)
379  {
380  switch (inversion)
381  {
382  case gauss_jordan:
383  var_inverse_full.resize(1);
384  var_inverse_full[0].reinit(b, b);
385  break;
386  case householder:
387  var_inverse_householder.resize(1);
388  break;
389  case svd:
390  var_inverse_svd.resize(1);
391  var_inverse_svd[0].reinit(b, b);
392  break;
393  default:
394  Assert(false, ExcNotImplemented());
395  }
396 
397  if (store_diagonals())
398  {
399  var_diagonal.resize(1);
400  var_diagonal[0].reinit(b, b);
401  }
402  }
403  else
404  {
405  // set the arrays to the right
406  // size. we could do it like this:
407  // var_inverse = vector<>(nblocks,FullMatrix<>())
408  // but this would involve copying many
409  // FullMatrix objects.
410  //
411  // the following is a neat trick which
412  // avoids copying
413  if (store_diagonals())
414  {
415  std::vector<FullMatrix<number>> tmp(n, FullMatrix<number>(b));
416  var_diagonal.swap(tmp);
417  }
418 
419  switch (inversion)
420  {
421  case gauss_jordan:
422  {
423  std::vector<FullMatrix<number>> tmp(n, FullMatrix<number>(b));
424  var_inverse_full.swap(tmp);
425  break;
426  }
427  case householder:
428  var_inverse_householder.resize(n);
429  break;
430  case svd:
431  {
432  std::vector<LAPACKFullMatrix<number>> tmp(
433  n, LAPACKFullMatrix<number>(b));
434  var_inverse_svd.swap(tmp);
435  break;
436  }
437  default:
438  Assert(false, ExcNotImplemented());
439  }
440  }

656  double sigmin = 1.e300, sigmax = -1.e300;
657  double kappamin = 1.e300, kappamax = -1.e300;
658 
659  for (size_type b = 0; b < size(); ++b)
660  {
661  const LAPACKFullMatrix<number> &matrix = inverse_svd(b);
662  size_type k = 1;
663  while (k <= matrix.n_cols() &&
664  matrix.singular_value(matrix.n_cols() - k) == 0)
665  ++k;
666  const double s0 = matrix.singular_value(0);
667  const double sm = matrix.singular_value(matrix.n_cols() - k);
668  const double co = sm / s0;
669 
670  if (kermin > k)
671  kermin = k - 1;
672  if (kermax < k)
673  kermax = k - 1;
674  if (s0 < sigmin)
675  sigmin = s0;
676  if (sm > sigmax)
677  sigmax = sm;
678  if (co < kappamin)
679  kappamin = co;
680  if (co > kappamax)
681  kappamax = co;
682  }
683  deallog << "dim ker [" << kermin << ':' << kermax << "] sigma [" << sigmin
684  << ':' << sigmax << "] kappa [" << kappamin << ':' << kappamax
685  << ']' << std::endl;
686  }
687  else if (inversion == householder)
688  {}
689  else if (inversion == gauss_jordan)
