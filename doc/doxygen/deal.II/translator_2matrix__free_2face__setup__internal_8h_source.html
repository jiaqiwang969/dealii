221  // who owns the element on the fine side. but we need to count
222  // the number of inner faces in order to balance the remaining
223  // faces properly

226  for (unsigned int c = 0;
227  c < (dcell->has_periodic_neighbor(f) ?
228  dcell->periodic_neighbor(f)
229  ->face(dcell->periodic_neighbor_face_no(f))
230  ->n_children() :
231  dcell->face(f)->n_children());

235  neighbor_c =
236  dcell->at_boundary(f) ?
237  dcell->periodic_neighbor_child_on_subface(f, c) :
238  dcell->neighbor_child_on_subface(f, c);
239  const types::subdomain_id neigh_domain =
240  neighbor_c->subdomain_id();

242  inner_faces_at_proc_boundary[neigh_domain]
243  .n_hanging_faces_larger_subdomain++;

245  inner_faces_at_proc_boundary[neigh_domain]
246  .n_hanging_faces_smaller_subdomain++;

251  use_active_cells ? neighbor->subdomain_id() :
252  neighbor->level_subdomain_id();

254  use_active_cells)
255  {

258  .n_hanging_faces_smaller_subdomain++;

261  .n_hanging_faces_larger_subdomain++;

273  .shared_faces.emplace_back(id_mine, id_neigh);
274  else

276  .shared_faces.emplace_back(id_neigh, id_mine);

356  // Arrange the face "ownership" such that cells that are access
357  // by more than one face (think of a cell in a corner) get
358  // ghosted. This arrangement has the advantage that we need to
359  // send less data because the same data is used twice. The
360  // strategy applied here is to ensure the same order of face
361  // pairs on both processors that share some faces, and make the
362  // same decision on both sides.
363 
364  // Create a vector with cell ids sorted over the processor with
365  // the larger rank. In the code below we need to be able to
366  // identify the same cell once for the processor with higher
367  // rank and once for the processor with the lower rank. The
368  // format for the processor with the higher rank is already
369  // contained in `shared_faces`, whereas we need a copy that we
370  // sort differently for the other way around.
371  std::vector<std::tuple<CellId, CellId, unsigned int>> other_range(
372  inner_face.second.shared_faces.size());
373  for (unsigned int i = 0; i < other_range.size(); ++i)
374  other_range[i] =
375  std::make_tuple(inner_face.second.shared_faces[i].second,
376  inner_face.second.shared_faces[i].first,
377  i);
378  std::sort(other_range.begin(), other_range.end());
379 
380  // the vector 'assignment' sets whether a particular cell
381  // appears more often and acts as a pre-selection of the rank. A
382  // value of 1 means that the process with the higher rank gets
383  // those faces, a value -1 means that the process with the lower
384  // rank gets it, whereas a value 0 means that the decision can
385  // be made in an arbitrary way.
386  unsigned int n_faces_lower_proc = 0, n_faces_higher_proc = 0;
387  std::vector<char> assignment(other_range.size(), 0);
388  if (inner_face.second.shared_faces.size() > 0)
389  {
390  // identify faces that go to the processor with the higher
391  // rank
392  unsigned int count = 0;
393  for (unsigned int i = 1;
394  i < inner_face.second.shared_faces.size();
395  ++i)
396  if (inner_face.second.shared_faces[i].first ==
397  inner_face.second.shared_faces[i - 1 - count].first)
398  ++count;
399  else
400  {
401  AssertThrow(count < 2 * dim, ExcInternalError());
402  if (count > 0)
403  {
404  for (unsigned int k = 0; k <= count; ++k)
405  assignment[i - 1 - k] = 1;
406  n_faces_higher_proc += count + 1;
407  }
408  count = 0;
409  }
410 
411  // identify faces that definitely go to the processor with
412  // the lower rank - this must use the sorting of CellId
413  // variables from the processor with the higher rank, i.e.,
414  // other_range rather than `shared_faces`.
415  count = 0;
416  for (unsigned int i = 1; i < other_range.size(); ++i)
417  if (std::get<0>(other_range[i]) ==
418  std::get<0>(other_range[i - 1 - count]))
419  ++count;
420  else
421  {
422  AssertThrow(count < 2 * dim, ExcInternalError());
423  if (count > 0)
424  {
425  for (unsigned int k = 0; k <= count; ++k)
426  {
427  Assert(inner_face.second
428  .shared_faces[std::get<2>(
429  other_range[i - 1])]
430  .second ==
431  inner_face.second
432  .shared_faces[std::get<2>(
433  other_range[i - 1 - k])]
434  .second,
435  ExcInternalError());
436  // only assign to -1 if higher rank was not
437  // yet set
438  if (assignment[std::get<2>(
439  other_range[i - 1 - k])] == 0)
440  {
441  assignment[std::get<2>(
442  other_range[i - 1 - k])] = -1;
443  ++n_faces_lower_proc;
444  }
445  }
446  }
447  count = 0;
448  }
449  }

452  // divide the faces evenly between the two processors. the
453  // processor with small rank takes the first half, the processor
454  // with larger rank the second half. Adjust for the hanging
455  // faces that always get assigned to one side, and the faces we
456  // have already assigned due to the criterion above
457  n_faces_lower_proc +=
458  inner_face.second.n_hanging_faces_smaller_subdomain;
459  n_faces_higher_proc +=
460  inner_face.second.n_hanging_faces_larger_subdomain;

462  (inner_face.second.shared_faces.size() +
463  inner_face.second.n_hanging_faces_smaller_subdomain +
464  inner_face.second.n_hanging_faces_larger_subdomain);
465  unsigned int split_index = n_total_faces_at_proc_boundary / 2;
466  if (split_index < n_faces_lower_proc)
467  split_index = 0;
468  else if (split_index <
469  n_total_faces_at_proc_boundary - n_faces_higher_proc)
470  split_index -= n_faces_lower_proc;
471  else
472  split_index = n_total_faces_at_proc_boundary -
473  n_faces_higher_proc - n_faces_lower_proc;

494  inner_face.first,

511  1,

522  std::vector<std::pair<CellId, CellId>> owned_faces_lower,

525  if (assignment[i] < 0)

527  inner_face.second.shared_faces[i]);
528  else if (assignment[i] > 0)

539  {

543  }

545  if (assignment[i] == 0)
546  {
547  owned_faces_higher.push_back(
548  inner_face.second.shared_faces[i]);

571  else

580  // cell_levels

582  for (unsigned int i = 0; i < cell_levels.size(); ++i)
583  {

585  &triangulation, cell_levels[i].first, cell_levels[i].second);
586  if (use_active_cells)
587  Assert(dcell->is_active(), ExcNotImplemented());
588  for (const auto f : dcell->face_indices())
589  {
590  if (dcell->at_boundary(f) && !dcell->has_periodic_neighbor(f))
591  face_is_owned[dcell->face(f)->index()] =
592  FaceCategory::locally_active_at_boundary;
593 
594  // treat boundaries of cells of different refinement level
595  // inside the domain in case of multigrid separately
596  else if ((dcell->at_boundary(f) == false ||
597  dcell->has_periodic_neighbor(f)) &&
598  mg_level != numbers::invalid_unsigned_int &&
599  dcell->neighbor_or_periodic_neighbor(f)->level() <
600  dcell->level())
601  {
602  face_is_owned[dcell->face(f)->index()] =
603  FaceCategory::multigrid_refinement_edge;
604  }
605  else
606  {
607  typename ::Triangulation<dim>::cell_iterator neighbor =
608  dcell->neighbor_or_periodic_neighbor(f);
609 
610  // neighbor is refined -> face will be treated by neighbor
611  if (use_active_cells && neighbor->has_children() &&
612  hold_all_faces_to_owned_cells == false)
613  continue;

615  bool add_to_ghost = false;
616  const types::subdomain_id
617  id1 = use_active_cells ? dcell->subdomain_id() :
618  dcell->level_subdomain_id(),
619  id2 = use_active_cells ?
620  (neighbor->has_children() ?
621  dcell->neighbor_child_on_subface(f, 0)
622  ->subdomain_id() :
623  neighbor->subdomain_id()) :
624  neighbor->level_subdomain_id();
625

627  // locally (instead of being processed from the other
628  // side). We process a face locally when we are more refined
629  // (in the active cell case) or when the face is listed in
630  // the `shared_faces` data structure that we built above.
631  if ((id1 == id2 &&
632  (use_active_cells == false || neighbor->is_active())) ||
633  dcell->level() > neighbor->level() ||
634  std::binary_search(
635  inner_faces_at_proc_boundary[id2].shared_faces.begin(),
636  inner_faces_at_proc_boundary[id2].shared_faces.end(),
637  std::make_pair(id1 < id2 ? dcell->id() : neighbor->id(),
638  id1 < id2 ? neighbor->id() :
639  dcell->id())))
640  {
641  face_is_owned[dcell->face(f)->index()] =
642  FaceCategory::locally_active_done_here;
643  if (dcell->level() == neighbor->level())
644  face_is_owned
645  [neighbor
646  ->face(dcell->has_periodic_neighbor(f) ?
647  dcell->periodic_neighbor_face_no(f) :
648  dcell->neighbor_face_no(f))
649  ->index()] =
650  FaceCategory::locally_active_done_here;
651

653  // dcell->subdomain_id !
654  // dcell->neighbor(f)->subdomain_id()), we need to add its
655  // index into cell level list.
656  if (use_active_cells)
657  add_to_ghost =
658  (dcell->subdomain_id() != neighbor->subdomain_id());
659  else
660  add_to_ghost = (dcell->level_subdomain_id() !=
661  neighbor->level_subdomain_id());
662  }
663  else if (hold_all_faces_to_owned_cells == true)
664  {
665  // add all cells to ghost layer...
666  face_is_owned[dcell->face(f)->index()] =
667  FaceCategory::ghosted;
668  if (use_active_cells)
669  {
670  if (neighbor->has_children())
671  for (unsigned int s = 0;
672  s < dcell->face(f)->n_children();
673  ++s)
674  if (dcell->at_boundary(f))
675  {
676  if (dcell
677  ->periodic_neighbor_child_on_subface(f,
678  s)
679  ->subdomain_id() !=
680  dcell->subdomain_id())
681  add_to_ghost = true;
682  }
683  else
684  {

1089  struct FaceComparator

1095  bool
1096  operator()(const FaceToCellTopology<length> &face1,
1097  const FaceToCellTopology<length> &face2) const
1098  {
1099  // check if active FE indices match
1100  if (face1.face_type < face2.face_type)
1101  return true;
1102  else if (face1.face_type > face2.face_type)
1103  return false;
1104 
1105  // check if active FE indices match
1106  if (active_fe_indices.size() > 0)
1107  {
1108  // ... for interior faces
1109  if (active_fe_indices[face1.cells_interior[0] / length] <
1110  active_fe_indices[face2.cells_interior[0] / length])
1111  return true;
1112  else if (active_fe_indices[face1.cells_interior[0] / length] >
1113  active_fe_indices[face2.cells_interior[0] / length])
1114  return false;
1115 
1116  // ... for exterior faces
1117  if (face2.cells_exterior[0] != numbers::invalid_unsigned_int)
1118  {
1119  if (active_fe_indices[face1.cells_exterior[0] / length] <
1120  active_fe_indices[face2.cells_exterior[0] / length])
1121  return true;
1122  else if (active_fe_indices[face1.cells_exterior[0] / length] >
1123  active_fe_indices[face2.cells_exterior[0] / length])
1124  return false;
1125  }
1126  }
1127 
1128  for (unsigned int i = 0; i < length; ++i)
1129  if (face1.cells_interior[i] < face2.cells_interior[i])
1130  return true;
1131  else if (face1.cells_interior[i] > face2.cells_interior[i])
1132  return false;
1133  for (unsigned int i = 0; i < length; ++i)
1134  if (face1.cells_exterior[i] < face2.cells_exterior[i])
1135  return true;
1136  else if (face1.cells_exterior[i] > face2.cells_exterior[i])
1137  return false;
1138  if (face1.interior_face_no < face2.interior_face_no)
1139  return true;
1140  else if (face1.interior_face_no > face2.interior_face_no)
1141  return false;
1142  if (face1.exterior_face_no < face2.exterior_face_no)
1143  return true;
1144  else if (face1.exterior_face_no > face2.exterior_face_no)
1145  return false;
1146 
1147  // we do not need to check for subface_index and orientation because
1148  // those cannot be different if when all the other values are the
1149  // same.
1150  AssertDimension(face1.subface_index, face2.subface_index);
1151  AssertDimension(face1.face_orientation, face2.face_orientation);
1152 
1153  return false;
1154  }

1172  std::vector<std::vector<unsigned int>> faces_type;
1173

1175  face_end = face_partition_data[0];
1176 
1177  face_partition_data[0] = faces_out.size();

1179  partition < face_partition_data.size() - 1;
1180  ++partition)
1181  {
1182  std::vector<std::vector<unsigned int>> new_faces_type;
1183 
1184  // start with the end point for the last partition
1185  face_start = face_end;
1186  face_end = face_partition_data[partition + 1];
1187 
1188  // set the partitioner to the new vectorized lengths
1189  face_partition_data[partition + 1] = face_partition_data[partition];
1190

1192  // to the face type
1193  for (unsigned int face = face_start; face < face_end; ++face)
1194  {
1195  for (auto &face_type : faces_type)
1196  {
1197  // Compare current face with first face of type type
1198  if (compare_faces_for_vectorization(faces_in[face],
1199  faces_in[face_type[0]],
1200  active_fe_indices,
1201  vectorization_width))
1202  {
1203  face_type.push_back(face);

1205  }
1206  }
1207  faces_type.emplace_back(1, face);
1208  face_found:
1209  {}
1210  }
1211

1213  FaceComparator<vectorization_width> face_comparator(
1214  active_fe_indices);
1215  std::set<FaceToCellTopology<vectorization_width>,
1216  FaceComparator<vectorization_width>>
1217  new_faces(face_comparator);
1218  for (const auto &face_type : faces_type)
1219  {
1220  macro_face.face_type = faces_in[face_type[0]].face_type;
1221  macro_face.interior_face_no =
1222  faces_in[face_type[0]].interior_face_no;
1223  macro_face.exterior_face_no =
1224  faces_in[face_type[0]].exterior_face_no;
1225  macro_face.subface_index = faces_in[face_type[0]].subface_index;
1226  macro_face.face_orientation =
1227  faces_in[face_type[0]].face_orientation;
1228  unsigned int no_faces = face_type.size();
1229  std::vector<unsigned char> touched(no_faces, 0);
1230 
1231  // do two passes through the data. The first is to identify
1232  // similar faces within the same index range as the cells which
1233  // will allow for vectorized read operations, the second picks up
1234  // all the rest
1235  unsigned int n_vectorized = 0;
1236  for (unsigned int f = 0; f < no_faces; ++f)
1237  if (faces_in[face_type[f]].cells_interior[0] %
1238  vectorization_width ==
1239  0)
1240  {
1241  bool is_contiguous = true;
1242  if (f + vectorization_width > no_faces)
1243  is_contiguous = false;
1244  else
1245  for (unsigned int v = 1; v < vectorization_width; ++v)
1246  if (faces_in[face_type[f + v]].cells_interior[0] !=
1247  faces_in[face_type[f]].cells_interior[0] + v)
1248  is_contiguous = false;
1249  if (is_contiguous)
1250  {
1251  AssertIndexRange(f,
1252  face_type.size() -
1253  vectorization_width + 1);
1254  for (unsigned int v = 0; v < vectorization_width; ++v)
1255  {
1256  macro_face.cells_interior[v] =
1257  faces_in[face_type[f + v]].cells_interior[0];
1258  macro_face.cells_exterior[v] =
1259  faces_in[face_type[f + v]].cells_exterior[0];
1260  touched[f + v] = 1;
1261  }
1262  new_faces.insert(macro_face);
1263  f += vectorization_width - 1;
1264  n_vectorized += vectorization_width;
1265  }
1266  }
1267 
1268  std::vector<unsigned int> untouched;
1269  untouched.reserve(no_faces - n_vectorized);
1270  for (unsigned int f = 0; f < no_faces; ++f)
1271  if (touched[f] == 0)
1272  untouched.push_back(f);
1273  unsigned int v = 0;
1274  for (const auto f : untouched)
1275  {
1276  macro_face.cells_interior[v] =
1277  faces_in[face_type[f]].cells_interior[0];
1278  macro_face.cells_exterior[v] =
1279  faces_in[face_type[f]].cells_exterior[0];
1280  ++v;
1281  if (v == vectorization_width)
1282  {
1283  new_faces.insert(macro_face);
1284  v = 0;
1285  }
1286  }
1287  if (v > 0 && v < vectorization_width)
1288  {
1289  // must add non-filled face
1290  if (hard_vectorization_boundary[partition + 1] ||
1291  partition == face_partition_data.size() - 2)
1292  {
1293  for (; v < vectorization_width; ++v)
1294  {
1295  // Dummy cell, not used
1296  macro_face.cells_interior[v] =
1297  numbers::invalid_unsigned_int;
1298  macro_face.cells_exterior[v] =
1299  numbers::invalid_unsigned_int;
1300  }
1301  new_faces.insert(macro_face);
1302  }
1303  else
1304  {

1306  std::vector<unsigned int> untreated(v);
1307  for (unsigned int f = 0; f < v; ++f)
1308  untreated[f] = face_type[*(untouched.end() - 1 - f)];

1316  faces_out.push_back(*it);

1320  faces_type = std::move(new_faces_type);
