340  const std::array<unsigned int, dim> indices = compute_index(i);
341 
342  ndarray<double, dim, order + 1> v;
343  {
344  std::vector<double> tmp(order + 1);
345  for (unsigned int d = 0; d < dim; ++d)
346  {
347  polynomials[indices[d]].value(p(d), tmp);
348  for (unsigned int j = 0; j < order + 1; ++j)
349  v[d][j] = tmp[j];
350  }
351  }
352 
353  Tensor<order, dim> derivative;
354  switch (order)
355  {
356  case 1:
357  {
358  Tensor<1, dim> &derivative_1 =
359  *reinterpret_cast<Tensor<1, dim> *>(&derivative);

361  {
362  derivative_1[d] = 1.;
363  for (unsigned int x = 0; x < dim; ++x)
364  {
365  unsigned int x_order = 0;
366  if (d == x)
367  ++x_order;
368 
369  derivative_1[d] *= v[x][x_order];
370  }
371  }
372 
373  return derivative;
374  }
375  case 2:
376  {
377  Tensor<2, dim> &derivative_2 =
378  *reinterpret_cast<Tensor<2, dim> *>(&derivative);
379  for (unsigned int d1 = 0; d1 < dim; ++d1)
