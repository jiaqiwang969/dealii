335 inline BlockMask
336 BlockMask::operator|(const BlockMask &mask) const
337 {
338  // if one of the two masks denotes the all-block mask,
339  // then return the other one
340  if (block_mask.size() == 0)
341  return mask;
342  else if (mask.block_mask.size() == 0)
343  return *this;
344  else
345  {
346  // if both masks have individual entries set, form
347  // the combination of the two
348  AssertDimension(block_mask.size(), mask.block_mask.size());
349  std::vector<bool> new_mask(block_mask.size());
350  for (unsigned int i = 0; i < block_mask.size(); ++i)
351  new_mask[i] = (block_mask[i] || mask.block_mask[i]);
352
