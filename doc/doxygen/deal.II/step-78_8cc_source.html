314  template <int dim>
315  void BlackScholes<dim>::setup_system()
316  {
317  dof_handler.distribute_dofs(fe);
318 
319  time_step = maturity_time / n_time_steps;
320 
321  constraints.clear();
322  DoFTools::make_hanging_node_constraints(dof_handler, constraints);
323  constraints.close();
324  DynamicSparsityPattern dsp(dof_handler.n_dofs());
325  DoFTools::make_sparsity_pattern(dof_handler,
326  dsp,
327  constraints,
328  /*keep_constrained_dofs = */ true);
329  sparsity_pattern.copy_from(dsp);
330 
331  mass_matrix.reinit(sparsity_pattern);
332  laplace_matrix.reinit(sparsity_pattern);
333  a_matrix.reinit(sparsity_pattern);
334  b_matrix.reinit(sparsity_pattern);
335  system_matrix.reinit(sparsity_pattern);
336 
337  MatrixCreator::create_mass_matrix(dof_handler,
338  QGauss<dim>(fe.degree + 1),
339  mass_matrix);
340 
341  const unsigned int dofs_per_cell = fe.dofs_per_cell;
342  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
343  QGauss<dim> quadrature_formula(fe.degree + 1);
344  FEValues<dim> fe_values(fe,
345  quadrature_formula,
346  update_values | update_gradients |
347  update_quadrature_points | update_JxW_values);
348  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
349  for (const auto &cell : dof_handler.active_cell_iterators())
350  {
351  cell_matrix = 0.;
352  fe_values.reinit(cell);
353  for (const unsigned int q_index : fe_values.quadrature_point_indices())
354  {
355  const double current_coefficient =
356  fe_values.quadrature_point(q_index).square();
357  for (const unsigned int i : fe_values.dof_indices())
358  {
359  for (const unsigned int j : fe_values.dof_indices())
360  cell_matrix(i, j) +=
361  (current_coefficient * // (x_q)^2
362  fe_values.shape_grad(i, q_index) * // grad phi_i(x_q)
363  fe_values.shape_grad(j, q_index) * // grad phi_j(x_q)
364  fe_values.JxW(q_index)); // dx
365  }
366  }
367  cell->get_dof_indices(local_dof_indices);
368  for (const unsigned int i : fe_values.dof_indices())
369  {
370  for (const unsigned int j : fe_values.dof_indices())
371  laplace_matrix.add(local_dof_indices[i],
372  local_dof_indices[j],
373  cell_matrix(i, j));
374  }
375  }
376 
377  for (const auto &cell : dof_handler.active_cell_iterators())
378  {
379  cell_matrix = 0.;
380  fe_values.reinit(cell);
381  for (const unsigned int q_index : fe_values.quadrature_point_indices())
382  {
383  const Tensor<1, dim> current_coefficient =
384  fe_values.quadrature_point(q_index);
385  for (const unsigned int i : fe_values.dof_indices())
386  {
387  for (const unsigned int j : fe_values.dof_indices())
388  {
389  cell_matrix(i, j) +=
390  (current_coefficient * // x_q
391  fe_values.shape_grad(i, q_index) * // grad phi_i(x_q)
392  fe_values.shape_value(j, q_index) * // phi_j(x_q)
393  fe_values.JxW(q_index)); // dx
394  }
395  }
396  }
397  cell->get_dof_indices(local_dof_indices);
398  for (const unsigned int i : fe_values.dof_indices())
399  {
400  for (const unsigned int j : fe_values.dof_indices())
401  a_matrix.add(local_dof_indices[i],
402  local_dof_indices[j],
403  cell_matrix(i, j));
404  }
405  }
406 
407  for (const auto &cell : dof_handler.active_cell_iterators())
408  {
409  cell_matrix = 0.;
410  fe_values.reinit(cell);
411  for (const unsigned int q_index : fe_values.quadrature_point_indices())
412  {
413  const Tensor<1, dim> current_coefficient =
414  fe_values.quadrature_point(q_index);
415  for (const unsigned int i : fe_values.dof_indices())
416  {
417  for (const unsigned int j : fe_values.dof_indices())
418  cell_matrix(i, j) +=
419  (current_coefficient * // x_q
420  fe_values.shape_value(i, q_index) * // phi_i(x_q)
421  fe_values.shape_grad(j, q_index) * // grad phi_j(x_q)
422  fe_values.JxW(q_index)); // dx
423  }
424  }
425  cell->get_dof_indices(local_dof_indices);
426  for (const unsigned int i : fe_values.dof_indices())
427  {
428  for (const unsigned int j : fe_values.dof_indices())
429  b_matrix.add(local_dof_indices[i],
430  local_dof_indices[j],
431  cell_matrix(i, j));
432  }
433  }
434 
435  solution.reinit(dof_handler.n_dofs());
436  system_rhs.reinit(dof_handler.n_dofs());
437  }

498  VectorTools::integrate_difference(dof_handler,
499  solution,

577  template <int dim>
578  void BlackScholes<dim>::run()
579  {
580  GridGenerator::hyper_cube(triangulation, 0.0, maximum_stock_price, true);
581  triangulation.refine_global(0);
582 
583  solution_names.emplace_back("u");
584  data_out_stack.declare_data_vector(solution_names,
585  DataOutStack<dim>::dof_vector);
586 
587  Vector<double> vmult_result;
588  Vector<double> forcing_terms;
589 
590  for (unsigned int cycle = 0; cycle < n_cycles; cycle++)
591  {
592  if (cycle != 0)
593  {
594  refine_grid();
595  time = 0.0;
596  }
597 
598  setup_system();
599 
600  std::cout << std::endl
601  << "===========================================" << std::endl
602  << "Cycle " << cycle << ':' << std::endl
603  << "Number of active cells: "
604  << triangulation.n_active_cells() << std::endl
605  << "Number of degrees of freedom: " << dof_handler.n_dofs()
606  << std::endl
607  << std::endl;
608 
609  VectorTools::interpolate(dof_handler,
610  InitialConditions<dim>(strike_price),
611  solution);
612 
613  if (cycle == (n_cycles - 1))
614  {
615  add_results_for_output();
616  }
617 
618  vmult_result.reinit(dof_handler.n_dofs());
619  forcing_terms.reinit(dof_handler.n_dofs());
620  for (unsigned int timestep_number = 0; timestep_number < n_time_steps;
621  ++timestep_number)
622  {
623  time += time_step;
624 
625  if (timestep_number % 1000 == 0)
626  std::cout << "Time step " << timestep_number << " at t=" << time
627  << std::endl;
628 
629  mass_matrix.vmult(system_rhs, solution);
630 
631  laplace_matrix.vmult(vmult_result, solution);
632  system_rhs.add(
633  (-1) * (1 - theta) * time_step *
634  Utilities::fixed_power<2, double>(asset_volatility) * 0.5,
635  vmult_result);
636  mass_matrix.vmult(vmult_result, solution);
637 
638  system_rhs.add((-1) * (1 - theta) * time_step * interest_rate * 2,
639  vmult_result);
640 
641  a_matrix.vmult(vmult_result, solution);
642  system_rhs.add((-1) * time_step * interest_rate, vmult_result);
643 
644  b_matrix.vmult(vmult_result, solution);
645  system_rhs.add(
646  (-1) * Utilities::fixed_power<2, double>(asset_volatility) *
647  time_step * 1,
648  vmult_result);
649 
650  RightHandSide<dim> rhs_function(asset_volatility, interest_rate);
651  rhs_function.set_time(time);
652  VectorTools::create_right_hand_side(dof_handler,
653  QGauss<dim>(fe.degree + 1),
654  rhs_function,
655  forcing_terms);
656  forcing_terms *= time_step * theta;
657  system_rhs -= forcing_terms;
658 
659  rhs_function.set_time(time - time_step);
660  VectorTools::create_right_hand_side(dof_handler,
661  QGauss<dim>(fe.degree + 1),
662  rhs_function,
663  forcing_terms);
664  forcing_terms *= time_step * (1 - theta);
665  system_rhs -= forcing_terms;
666 
667  system_matrix.copy_from(mass_matrix);
668  system_matrix.add(
669  (theta)*time_step *
670  Utilities::fixed_power<2, double>(asset_volatility) * 0.5,
671  laplace_matrix);
672  system_matrix.add((time_step)*interest_rate * theta * (1 + 1),
673  mass_matrix);
674 
675  constraints.condense(system_matrix, system_rhs);
676 
677  {
