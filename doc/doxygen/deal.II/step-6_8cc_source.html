144 template <int dim>
145 void Step6<dim>::assemble_system()
146 {
147  const QGauss<dim> quadrature_formula(fe.degree + 1);
148 
149  FEValues<dim> fe_values(fe,
150  quadrature_formula,
151  update_values | update_gradients |
152  update_quadrature_points | update_JxW_values);
153 
154  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
155 
156  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
157  Vector<double> cell_rhs(dofs_per_cell);
158 
159  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
160 
161  for (const auto &cell : dof_handler.active_cell_iterators())
162  {
163  cell_matrix = 0;
164  cell_rhs = 0;
165 
166  fe_values.reinit(cell);
167 
168  for (const unsigned int q_index : fe_values.quadrature_point_indices())
169  {
170  const double current_coefficient =
171  coefficient(fe_values.quadrature_point(q_index));
172  for (const unsigned int i : fe_values.dof_indices())
173  {
174  for (const unsigned int j : fe_values.dof_indices())
175  cell_matrix(i, j) +=
176  (current_coefficient * // a(x_q)
177  fe_values.shape_grad(i, q_index) * // grad phi_i(x_q)
178  fe_values.shape_grad(j, q_index) * // grad phi_j(x_q)
179  fe_values.JxW(q_index)); // dx
180 
181  cell_rhs(i) += (1.0 * // f(x)
182  fe_values.shape_value(i, q_index) * // phi_i(x_q)
183  fe_values.JxW(q_index)); // dx
184  }
185  }
186 
187  cell->get_dof_indices(local_dof_indices);
188  constraints.distribute_local_to_global(
189  cell_matrix, cell_rhs, local_dof_indices, system_matrix, system_rhs);
190  }
191 }
