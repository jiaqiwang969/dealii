496 template <typename number>
497 template <class BlockVectorType>
498 inline void
499 BlockSparseMatrix<number>::precondition_Jacobi(BlockVectorType & dst,
500  const BlockVectorType &src,
501  const number omega) const
502 {
503  Assert(this->n_block_rows() == this->n_block_cols(), ExcNotQuadratic());
504  Assert(dst.n_blocks() == this->n_block_rows(),
505  ExcDimensionMismatch(dst.n_blocks(), this->n_block_rows()));
506  Assert(src.n_blocks() == this->n_block_cols(),
507  ExcDimensionMismatch(src.n_blocks(), this->n_block_cols()));
508 
509  // do a diagonal preconditioning. uses only
510  // the diagonal blocks of the matrix
511  for (size_type i = 0; i < this->n_block_rows(); ++i)
512  this->block(i, i).precondition_Jacobi(dst.block(i), src.block(i), omega);
513 }
