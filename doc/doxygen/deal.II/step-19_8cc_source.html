155  template <int dim>
156  void CathodeRaySimulator<dim>::make_grid()
157  {
158  static_assert(dim == 2,
159  "This function is currently only implemented for 2d.");
160 
161  const double delta = 0.5;
162  const unsigned int nx = 5;
163  const unsigned int ny = 3;
164 
165  const std::vector<Point<dim>> vertices
166  = {{0, 0},
167  {1, 0},
168  {2, 0},
169  {3, 0},
170  {4, 0},
171  {delta, 1},
172  {1, 1},
173  {2, 1},
174  {3, 1},
175  {4, 1},
176  {0, 2},
177  {1, 2},
178  {2, 2},
179  {3, 2},
180  {4, 2}};
181  AssertDimension(vertices.size(), nx * ny);
182 
183  const std::vector<unsigned int> cell_vertices[(nx - 1) * (ny - 1)] = {
184  {0, 1, nx + 0, nx + 1},
185  {1, 2, nx + 1, nx + 2},
186  {2, 3, nx + 2, nx + 3},
187  {3, 4, nx + 3, nx + 4},
188 
189  {5, nx + 1, 2 * nx + 0, 2 * nx + 1},
190  {nx + 1, nx + 2, 2 * nx + 1, 2 * nx + 2},
191  {nx + 2, nx + 3, 2 * nx + 2, 2 * nx + 3},
192  {nx + 3, nx + 4, 2 * nx + 3, 2 * nx + 4}};
193 
194  std::vector<CellData<dim>> cells((nx - 1) * (ny - 1), CellData<dim>());
195  for (unsigned int i = 0; i < cells.size(); ++i)
196  {
197  cells[i].vertices = cell_vertices[i];
198  cells[i].material_id = 0;
199  }
200 
201  triangulation.create_triangulation(
202  vertices,
203  cells,
204  SubCellData()); // No boundary information
205 
206  triangulation.refine_global(2);
207 
208  for (auto &cell : triangulation.active_cell_iterators())
209  for (auto &face : cell->face_iterators())
210  if (face->at_boundary())
211  {
212  if ((face->center()[0] > 0) && (face->center()[0] < 0.5) &&
213  (face->center()[1] > 0) && (face->center()[1] < 2))
214  face->set_boundary_id(BoundaryIds::cathode);
215  else if ((face->center()[0] > 0) && (face->center()[0] < 2))
216  face->set_boundary_id(BoundaryIds::focus_element);
217  else if ((face->center()[0] > 4 - 1e-12) &&
218  ((face->center()[1] > 1.5) || (face->center()[1] < 0.5)))
219  face->set_boundary_id(BoundaryIds::anode);
220  else
221  face->set_boundary_id(BoundaryIds::open);
222  }
223 
224  triangulation.refine_global(1);
225  }

269  template <int dim>
270  void CathodeRaySimulator<dim>::assemble_system()
271  {
272  system_matrix = 0;
273  system_rhs = 0;
274 
275  const QGauss<dim> quadrature_formula(fe.degree + 1);
276 
277  FEValues<dim> fe_values(fe,
278  quadrature_formula,
279  update_values | update_gradients |
280  update_quadrature_points | update_JxW_values);
281 
282  const unsigned int dofs_per_cell = fe.dofs_per_cell;
283 
284  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
285  Vector<double> cell_rhs(dofs_per_cell);
286 
287  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
288 
289  for (const auto &cell : dof_handler.active_cell_iterators())
290  {
291  cell_matrix = 0;
292  cell_rhs = 0;
293 
294  fe_values.reinit(cell);
295 
296  for (const unsigned int q_index : fe_values.quadrature_point_indices())
297  for (const unsigned int i : fe_values.dof_indices())
298  {
299  for (const unsigned int j : fe_values.dof_indices())
300  cell_matrix(i, j) +=
301  (fe_values.shape_grad(i, q_index) * // grad phi_i(x_q)
302  fe_values.shape_grad(j, q_index) * // grad phi_j(x_q)
303  fe_values.JxW(q_index)); // dx
304  }
305 
306  if (particle_handler.n_particles_in_cell(cell) > 0)
307  for (const auto &particle : particle_handler.particles_in_cell(cell))
308  {
309  const Point<dim> &reference_location =
310  particle.get_reference_location();
311  for (const unsigned int i : fe_values.dof_indices())
312  cell_rhs(i) +=
313  (fe.shape_value(i, reference_location) * // phi_i(x_p)
314  (-Constants::electrons_per_particle * // N
315  Constants::electron_charge)); // e
316  }
317 
318  cell->get_dof_indices(local_dof_indices);
319  constraints.distribute_local_to_global(
320  cell_matrix, cell_rhs, local_dof_indices, system_matrix, system_rhs);
321  }
322  }

363  template <int dim>
364  void CathodeRaySimulator<dim>::create_particles()
365  {
366  FEFaceValues<dim> fe_face_values(fe,
367  QMidpoint<dim - 1>(),
368  update_quadrature_points |
369  update_gradients |
370  update_normal_vectors);
371 
372  std::vector<Tensor<1, dim>> solution_gradients(
373  fe_face_values.n_quadrature_points);
374 
375  for (const auto &cell : dof_handler.active_cell_iterators())
376  for (const auto &face : cell->face_iterators())
377  if (face->at_boundary() &&
378  (face->boundary_id() == BoundaryIds::cathode))
379  {
380  fe_face_values.reinit(cell, face);
381 
382  const FEValuesExtractors::Scalar electric_potential(0);
383  fe_face_values[electric_potential].get_function_gradients(
384  solution, solution_gradients);
385  for (const unsigned int q_point :
386  fe_face_values.quadrature_point_indices())
387  {
388  const Tensor<1, dim> E = solution_gradients[q_point];
389 
390  if ((E * fe_face_values.normal_vector(q_point) < 0) &&
391  (E.norm() > Constants::E_threshold))
392  {
393  const Point<dim> &location =
394  fe_face_values.quadrature_point(q_point);
395 
396  Particles::Particle<dim> new_particle;
397  new_particle.set_location(location);
398  new_particle.set_reference_location(
399  mapping.transform_real_to_unit_cell(cell, location));
400  new_particle.set_id(next_unused_particle_id);
401  particle_handler.insert_particle(new_particle, cell);
402 
403  ++next_unused_particle_id;
404  }
405  }
406  }

413  template <int dim>
414  void CathodeRaySimulator<dim>::move_particles()
415  {
416  const double dt = time.get_next_step_size();
417 
418  Vector<double> solution_values(fe.n_dofs_per_cell());
419  FEPointEvaluation<1, dim> evaluator(mapping, fe, update_gradients);
420 
421  for (const auto &cell : dof_handler.active_cell_iterators())
422  if (particle_handler.n_particles_in_cell(cell) > 0)
423  {
424  const typename Particles::ParticleHandler<
425  dim>::particle_iterator_range particles_in_cell =
426  particle_handler.particles_in_cell(cell);
427 
428  std::vector<Point<dim>> particle_positions;
429  for (const auto &particle : particles_in_cell)
430  particle_positions.push_back(particle.get_reference_location());
431 
432  cell->get_dof_values(solution, solution_values);
433 
434  evaluator.reinit(cell, particle_positions);
435  evaluator.evaluate(make_array_view(solution_values),
436  EvaluationFlags::gradients);
437 
438  {
439  typename Particles::ParticleHandler<dim>::particle_iterator
440  particle = particles_in_cell.begin();
441  for (unsigned int particle_index = 0;
442  particle != particles_in_cell.end();
443  ++particle, ++particle_index)
444  {
445  const Tensor<1, dim> &E =
446  evaluator.get_gradient(particle_index);
447 
448  const Tensor<1, dim> old_velocity(particle->get_properties());
449 
450  const Tensor<1, dim> acceleration =
451  Constants::electron_charge / Constants::electron_mass * E;
452 
453  const Tensor<1, dim> new_velocity =
454  old_velocity + acceleration * dt;
455 
456  particle->set_properties(make_array_view(new_velocity));
457 
458  const Point<dim> new_location =
459  particle->get_location() + dt * new_velocity;
460  particle->set_location(new_location);
461  }
462  }
463  }
464 
465  particle_handler.sort_particles_into_subdomains_and_cells();
466  }

470  template <int dim>
471  void CathodeRaySimulator<dim>::track_lost_particle(
472  const typename Particles::ParticleIterator<dim> & particle,
473  const typename Triangulation<dim>::active_cell_iterator &cell)
474  {
475  ++n_recently_lost_particles;
476  ++n_total_lost_particles;
477 
478  const Point<dim> current_location = particle->get_location();
479  const Point<dim> approximate_previous_location = cell->center();
480 
481  if ((approximate_previous_location[0] < 4) && (current_location[0] > 4))
482  {
483  const Tensor<1, dim> direction =
484  (current_location - approximate_previous_location) /
485  (current_location[0] - approximate_previous_location[0]);
486 
487  const double right_boundary_intercept =
488  approximate_previous_location[1] +
489  (4 - approximate_previous_location[0]) * direction[1];
490  if ((right_boundary_intercept > 0.5) &&
491  (right_boundary_intercept < 1.5))
492  ++n_particles_lost_through_anode;
493  }
494  }

499  template <int dim>
500  void CathodeRaySimulator<dim>::update_timestep_size()
501  {
502  if (time.get_step_number() > 0)
503  {
504  double min_cell_size_over_velocity = std::numeric_limits<double>::max();
505 
506  for (const auto &cell : dof_handler.active_cell_iterators())
507  if (particle_handler.n_particles_in_cell(cell) > 0)
508  {
509  const double cell_size = cell->minimum_vertex_distance();
510 
511  double max_particle_velocity(0.0);
512 
513  for (const auto &particle :
514  particle_handler.particles_in_cell(cell))
515  {
516  const Tensor<1, dim> velocity(particle.get_properties());
517  max_particle_velocity =
518  std::max(max_particle_velocity, velocity.norm());
519  }
520 
521  if (max_particle_velocity > 0)
522  min_cell_size_over_velocity =
523  std::min(min_cell_size_over_velocity,
524  cell_size / max_particle_velocity);
525  }
526 
527  constexpr double c_safety = 0.5;
528  time.set_desired_next_step_size(c_safety * 0.5 *
529  min_cell_size_over_velocity);
530  }
531  else
532  {
533  const QTrapezoid<dim> vertex_quadrature;
534  FEValues<dim> fe_values(fe, vertex_quadrature, update_gradients);
535 
536  std::vector<Tensor<1, dim>> field_gradients(vertex_quadrature.size());
537 
538  double min_timestep = std::numeric_limits<double>::max();
539 
540  for (const auto &cell : dof_handler.active_cell_iterators())
541  if (particle_handler.n_particles_in_cell(cell) > 0)
542  {
543  const double cell_size = cell->minimum_vertex_distance();
544 
545  fe_values.reinit(cell);
546  fe_values.get_function_gradients(solution, field_gradients);
547 
548  double max_E = 0;
549  for (const auto q_point : fe_values.quadrature_point_indices())
550  max_E = std::max(max_E, field_gradients[q_point].norm());
551 
552  if (max_E > 0)
553  min_timestep =
554  std::min(min_timestep,
555  std::sqrt(0.5 * cell_size *
556  Constants::electron_mass /
557  Constants::electron_charge / max_E));
558  }
559 
560  time.set_desired_next_step_size(min_timestep);
561  }
