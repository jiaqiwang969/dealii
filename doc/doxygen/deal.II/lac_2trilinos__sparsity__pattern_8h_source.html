1151  Iterator::operator++()
1152  {
1153  Assert(accessor.a_row < accessor.sparsity_pattern->n_rows(),
1154  ExcIteratorPastEnd());
1155 
1156  ++accessor.a_index;
1157 
1158  // If at end of line: do one step, then cycle until we find a row with a
1159  // nonzero number of entries that is stored locally.
1160  if (accessor.a_index >= accessor.colnum_cache->size())
1161  {
1162  accessor.a_index = 0;

1327  template <typename ForwardIterator>
1328  inline void
1329  SparsityPattern::add_entries(const size_type row,
1330  ForwardIterator begin,
1331  ForwardIterator end,
1332  const bool /*indices_are_sorted*/ )
1333  {
1334  if (begin == end)
1335  return;
1336 
1337  // verify that the size of the data type Trilinos expects matches that the
1338  // iterator points to. we allow for some slippage between signed and
1339  // unsigned and only compare that they are both either 32 or 64 bit. to
1340  // write this test properly, not that we cannot compare the size of
1341  // '*begin' because 'begin' may be an iterator and '*begin' may be an
1342  // accessor class. consequently, we need to somehow get an actual value
1343  // from it which we can by evaluating an expression such as when
1344  // multiplying the value produced by 2
1345  Assert(sizeof(TrilinosWrappers::types::int_type) == sizeof((*begin) * 2),
1346  ExcNotImplemented());
1347 
1348  TrilinosWrappers::types::int_type *col_index_ptr =
1349  reinterpret_cast<TrilinosWrappers::types::int_type *>(
1350  const_cast<typename std::decay<decltype(*begin)>::type *>(&*begin));
1351  // Check at least for the first index that the conversion actually works
1352  AssertDimension(*col_index_ptr, *begin);
1353  TrilinosWrappers::types::int_type trilinos_row_index = row;
1354  const int n_cols = static_cast<int>(end - begin);
1355 
1356  int ierr;
1357  if (row_is_stored_locally(row))
1358  ierr =
1359  graph->InsertGlobalIndices(trilinos_row_index, n_cols, col_index_ptr);
1360  else if (nonlocal_graph.get() != nullptr)
1361  {
1362  // this is the case when we have explicitly set the off-processor rows
1363  // and want to create a separate matrix object for them (to retain
1364  // thread-safety)
1365  Assert(nonlocal_graph->RowMap().LID(
1366  static_cast<TrilinosWrappers::types::int_type>(row)) != -1,
1367  ExcMessage("Attempted to write into off-processor matrix row "
1368  "that has not be specified as being writable upon "
1369  "initialization"));
1370  ierr = nonlocal_graph->InsertGlobalIndices(trilinos_row_index,
1371  n_cols,
1372  col_index_ptr);
1373  }
1374  else
1375  ierr = graph->InsertGlobalIndices(1,
1376  &trilinos_row_index,
1377  n_cols,
1378  col_index_ptr);
1379 
1380  AssertThrow(ierr >= 0, ExcTrilinosError(ierr));
1381  }
