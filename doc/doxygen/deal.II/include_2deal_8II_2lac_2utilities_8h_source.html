226  template <typename NumberType>
227  std::array<NumberType, 3>
228  hyperbolic_rotation(const NumberType &f, const NumberType &g)
229  {
230  Assert(f != 0, ExcDivideByZero());
231  const NumberType tau = g / f;
232  AssertThrow(std::abs(tau) < 1.,
233  ExcMessage(
234  "real-valued Hyperbolic rotation does not exist for (" +
235  std::to_string(f) + "," + std::to_string(g) + ")"));
236  const NumberType u =
237  std::copysign(std::sqrt((1. - tau) * (1. + tau)),
238  f); // <-- more stable than std::sqrt(1.-tau*tau)
239  std::array<NumberType, 3> csr;
240  csr[0] = 1. / u; // c
241  csr[1] = csr[0] * tau; // s
242  csr[2] = f * u; // r
243  return csr;
244  }

260  template <typename NumberType>
261  std::array<NumberType, 3>
262  givens_rotation(const NumberType &f, const NumberType &g)
263  {
264  std::array<NumberType, 3> res;
265  // naive calculation for "r" may overflow or underflow:
266  // c = x / \sqrt{x^2+y^2}
267  // s = -y / \sqrt{x^2+y^2}
268 
269  // See Golub 2013, Matrix computations, Chapter 5.1.8
270  // Algorithm 5.1.3
271  // and
272  // Anderson (2000),
273  // Discontinuous Plane Rotations and the Symmetric Eigenvalue Problem.
274  // LAPACK Working Note 150, University of Tennessee, UT-CS-00-454,
275  // December 4, 2000.
276  // Algorithm 4
277  // We implement the latter below:
278  if (g == NumberType())
279  {
280  res[0] = std::copysign(1., f);
281  res[1] = NumberType();
282  res[2] = std::abs(f);
283  }
284  else if (f == NumberType())
285  {
286  res[0] = NumberType();
287  res[1] = std::copysign(1., g);
288  res[2] = std::abs(g);
289  }
290  else if (std::abs(f) > std::abs(g))
291  {
292  const NumberType tau = g / f;
293  const NumberType u = std::copysign(std::sqrt(1. + tau * tau), f);
294  res[0] = 1. / u; // c
295  res[1] = res[0] * tau; // s
296  res[2] = f * u; // r
297  }
298  else
299  {
300  const NumberType tau = f / g;
301  const NumberType u = std::copysign(std::sqrt(1. + tau * tau), g);
302  res[1] = 1. / u; // s
303  res[0] = res[1] * tau; // c
304  res[2] = g * u; // r
305  }
306 
307  return res;
308  }

312  template <typename OperatorType, typename VectorType>
313  double
314  lanczos_largest_eigenvalue(const OperatorType & H,
315  const VectorType & v0_,
316  const unsigned int k,
317  VectorMemory<VectorType> &vector_memory,
318  std::vector<double> * eigenvalues)
319  {
320  // Do k-step Lanczos:
321 
322  typename VectorMemory<VectorType>::Pointer v(vector_memory);
323  typename VectorMemory<VectorType>::Pointer v0(vector_memory);
324  typename VectorMemory<VectorType>::Pointer f(vector_memory);
325 
326  v->reinit(v0_);
327  v0->reinit(v0_);
328  f->reinit(v0_);
329 
330  // two vectors to store diagonal and subdiagonal of the Lanczos
331  // matrix
332  std::vector<double> diagonal;
333  std::vector<double> subdiagonal;
334 
335  // 1. Normalize input vector
336  (*v) = v0_;
337  double a = v->l2_norm();
338  Assert(a != 0, ExcDivideByZero());
339  (*v) *= 1. / a;
340 
341  // 2. Compute f = Hv; a = f*v; f <- f - av; T(0,0)=a;
342  H.vmult(*f, *v);
343  a = (*f) * (*v);
344  f->add(-a, *v);
345  diagonal.push_back(a);
346 
347  // 3. Loop over steps
348  for (unsigned int i = 1; i < k; ++i)
349  {
350  // 4. L2 norm of f
351  const double b = f->l2_norm();
352  Assert(b != 0, ExcDivideByZero());
353  // 5. v0 <- v; v <- f/b
354  *v0 = *v;
355  *v = *f;
356  (*v) *= 1. / b;
357  // 6. f = Hv; f <- f - b v0;
358  H.vmult(*f, *v);
359  f->add(-b, *v0);
360  // 7. a = f*v; f <- f - a v;
361  a = (*f) * (*v);
362  f->add(-a, *v);
363  // 8. T(i,i-1) = T(i-1,i) = b; T(i,i) = a;
364  diagonal.push_back(a);
365  subdiagonal.push_back(b);
366  }
367 
368  Assert(diagonal.size() == k, ExcInternalError());
369  Assert(subdiagonal.size() == k - 1, ExcInternalError());
370 
371  // Use Lapack dstev to get ||T||_2 norm, i.e. the largest eigenvalue
372  // of T
373  const types::blas_int n = k;
374  std::vector<double> Z; // unused for eigenvalues-only ("N") job
375  const types::blas_int ldz = 1; // ^^ (>=1)
376  std::vector<double> work; // ^^
377  types::blas_int info;
378  // call lapack_templates.h wrapper:
379  internal::UtilitiesImplementation::call_stev('N',
380  n,
381  diagonal.data(),
382  subdiagonal.data(),
383  Z.data(),
384  ldz,
385  work.data(),
386  &info);
387 
388  Assert(info == 0, LAPACKSupport::ExcErrorCode("dstev", info));
389 
390  if (eigenvalues != nullptr)
391  {
392  eigenvalues->resize(diagonal.size());
393  std::copy(diagonal.begin(), diagonal.end(), eigenvalues->begin());
394  }
395 
396  // note that the largest eigenvalue of T is below the largest
397  // eigenvalue of the operator.
398  // return ||T||_2 + ||f||_2, although it is not guaranteed to be an upper
399  // bound.
400  return diagonal[k - 1] + f->l2_norm();
401  }

404  template <typename OperatorType, typename VectorType>
405  void
406  chebyshev_filter(VectorType & x,
407  const OperatorType & op,
408  const unsigned int degree,
409  const std::pair<double, double> unwanted_spectrum,
410  const double a_L,
411  VectorMemory<VectorType> & vector_memory)
412  {
413  const double a = unwanted_spectrum.first;
414  const double b = unwanted_spectrum.second;
415  Assert(degree > 0, ExcMessage("Only positive degrees make sense."));
416 
417  const bool scale = (a_L < std::numeric_limits<double>::infinity());
418  Assert(
419  a < b,
420  ExcMessage(
421  "Lower bound of the unwanted spectrum should be smaller than the upper bound."));
422 
423  Assert(a_L <= a || a_L >= b || !scale,
424  ExcMessage(
425  "Scaling point should be outside of the unwanted spectrum."));
426 
427  // Setup auxiliary vectors:
428  typename VectorMemory<VectorType>::Pointer p_y(vector_memory);
429  typename VectorMemory<VectorType>::Pointer p_yn(vector_memory);
430 
431  p_y->reinit(x);
432  p_yn->reinit(x);
433 
434  // convenience to avoid pointers
435  VectorType &y = *p_y;
436  VectorType &yn = *p_yn;
437 
438  // Below is an implementation of
439  // Algorithm 3.2 in Zhou et al, Journal of Computational Physics 274
440  // (2014) 770-782 with **a bugfix for sigma1**. Here is the original
441  // algorithm verbatim:
442  //
443  // [Y]=chebyshev_filter_scaled(X, m, a, b, aL).
444  // e=(b-a)/2; c=(a+b)/2; σ=e/(c-aL); τ=2/σ;
445  // Y=(H∗X-c∗X)∗(σ/e);
446  // for i=2 to m do
447  // σnew =1/(τ - σ);
448  // Yt =(H∗Y - c∗Y)∗(2∗σnew/e)-(σ∗σnew)∗X;
449  // X =Y; Y =Yt; σ =σnew;
450 
451  const double e = (b - a) / 2.;
452  const double c = (a + b) / 2.;
453  const double alpha = 1. / e;
454  const double beta = -c / e;
455 
456  const double sigma1 =
457  e / (a_L - c); // BUGFIX which is relevant for odd degrees
458  double sigma = scale ? sigma1 : 1.;
459  const double tau = 2. / sigma;
460  op.vmult(y, x);
461  y.sadd(alpha * sigma, beta * sigma, x);
462 
463  for (unsigned int i = 2; i <= degree; ++i)
464  {
465  const double sigma_new = scale ? 1. / (tau - sigma) : 1.;
466  op.vmult(yn, y);
467  yn.sadd(2. * alpha * sigma_new, 2. * beta * sigma_new, y);
468  yn.add(-sigma * sigma_new, x);
469  x.swap(y);
470  y.swap(yn);
471  sigma = sigma_new;
472  }
473 
474  x.swap(y);
475  }
