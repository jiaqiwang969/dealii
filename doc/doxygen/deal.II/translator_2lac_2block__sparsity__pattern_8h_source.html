848 template <typename SparsityPatternType>
849 template <typename ForwardIterator>
850 void
851 BlockSparsityPatternBase<SparsityPatternType>::add_entries(
852  const size_type row,
853  ForwardIterator begin,
854  ForwardIterator end,
855  const bool indices_are_sorted)
856 {
857  // Resize scratch arrays
858  if (block_column_indices.size() < this->n_block_cols())
859  {
860  block_column_indices.resize(this->n_block_cols());
861  counter_within_block.resize(this->n_block_cols());
862  }
863 
864  const size_type n_cols = static_cast<size_type>(end - begin);
865 
866  // Resize sub-arrays to n_cols. This
867  // is a bit wasteful, but we resize
868  // only a few times (then the maximum
869  // row length won't increase that
870  // much any more). At least we know
871  // that all arrays are going to be of
872  // the same size, so we can check
873  // whether the size of one is large
874  // enough before actually going
875  // through all of them.
876  if (block_column_indices[0].size() < n_cols)
877  for (size_type i = 0; i < this->n_block_cols(); ++i)
878  block_column_indices[i].resize(n_cols);
879 
880  // Reset the number of added elements
881  // in each block to zero.
882  for (size_type i = 0; i < this->n_block_cols(); ++i)
883  counter_within_block[i] = 0;
884 
885  // Go through the column indices to
886  // find out which portions of the
887  // values should be set in which
888  // block of the matrix. We need to
889  // touch all the data, since we can't
890  // be sure that the data of one block
891  // is stored contiguously (in fact,
892  // indices will be intermixed when it
893  // comes from an element matrix).
894  for (ForwardIterator it = begin; it != end; ++it)
895  {
896  const size_type col = *it;
897 
898  const std::pair<size_type, size_type> col_index =
899  this->column_indices.global_to_local(col);
900 
901  const size_type local_index = counter_within_block[col_index.first]++;
902 
903  block_column_indices[col_index.first][local_index] = col_index.second;
904  }
905 
906 #ifdef DEBUG
907  // If in debug mode, do a check whether
908  // the right length has been obtained.
909  size_type length = 0;
910  for (size_type i = 0; i < this->n_block_cols(); ++i)
911  length += counter_within_block[i];
912  Assert(length == n_cols, ExcInternalError());
913 #endif
914 
915  // Now we found out about where the
916  // individual columns should start and
917  // where we should start reading out
918  // data. Now let's write the data into
919  // the individual blocks!
920  const std::pair<size_type, size_type> row_index =
921  this->row_indices.global_to_local(row);
922  for (size_type block_col = 0; block_col < n_block_cols(); ++block_col)
923  {
924  if (counter_within_block[block_col] == 0)
925  continue;
926  sub_objects[row_index.first][block_col]->add_entries(
927  row_index.second,
928  block_column_indices[block_col].begin(),
929  block_column_indices[block_col].begin() +
930  counter_within_block[block_col],
931  indices_are_sorted);
932  }
933 }

990 inline BlockDynamicSparsityPattern::size_type
991 BlockDynamicSparsityPattern::column_number(const size_type row,
992  const unsigned int index) const
993 {
994  // .first= ith block, .second = jth row in that block
995  const std::pair<size_type, size_type> row_index =
996  row_indices.global_to_local(row);
997 
998  AssertIndexRange(index, row_length(row));
999 
1000  size_type c = 0;
1001  size_type block_columns = 0; // sum of n_cols for all blocks to the left
1002  for (unsigned int b = 0; b < columns; ++b)
1003  {
1004  unsigned int rowlen =
1005  sub_objects[row_index.first][b]->row_length(row_index.second);
1006  if (index < c + rowlen)
1007  return block_columns +
1008  sub_objects[row_index.first][b]->column_number(row_index.second,
1009  index - c);
1010  c += rowlen;
1011  block_columns += sub_objects[row_index.first][b]->n_cols();
1012  }
1013 
1014  Assert(false, ExcInternalError());
1015  return 0;
1016 }
