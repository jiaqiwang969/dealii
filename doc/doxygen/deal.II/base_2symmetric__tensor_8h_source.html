1268  1.0 / (t[idx_00] * t[idx_11] - t[idx_01] * t[idx_01]);
1269  tmp[idx_00] = t[idx_11];
1270  tmp[idx_01] = -t[idx_01];

1311  // (t01*t12 - t02*t11)/det_t],
1312  // [ (-t01*t22 + t02*t12)/det_t,

1360  struct Inverse<4, 2, Number>
1361  {
1362  constexpr static inline ::SymmetricTensor<4, 2, Number>
1363  value(const ::SymmetricTensor<4, 2, Number> &t)
1364  {
1365  ::SymmetricTensor<4, 2, Number> tmp;
1366 
1367  // Inverting this tensor is a little more complicated than necessary,
1368  // since we store the data of 't' as a 3x3 matrix t.data, but the
1369  // product between a rank-4 and a rank-2 tensor is really not the
1370  // product between this matrix and the 3-vector of a rhs, but rather
1371  //
1372  // B.vec = t.data * mult * A.vec
1373  //
1374  // where mult is a 3x3 matrix with entries [[1,0,0],[0,1,0],[0,0,2]] to
1375  // capture the fact that we need to add up both the c_ij12*a_12 and the
1376  // c_ij21*a_21 terms.
1377  //
1378  // In addition, in this scheme, the identity tensor has the matrix
1379  // representation mult^-1.
1380  //
1381  // The inverse of 't' therefore has the matrix representation
1382  //
1383  // inv.data = mult^-1 * t.data^-1 * mult^-1
1384  //
1385  // in order to compute it, let's first compute the inverse of t.data and
1386  // put it into tmp.data; at the end of the function we then scale the
1387  // last row and column of the inverse by 1/2, corresponding to the left
1388  // and right multiplication with mult^-1.
1389  const Number t4 = t.data[0][0] * t.data[1][1],
1390  t6 = t.data[0][0] * t.data[1][2],
1391  t8 = t.data[0][1] * t.data[1][0],
1392  t00 = t.data[0][2] * t.data[1][0],
1393  t01 = t.data[0][1] * t.data[2][0],
1394  t04 = t.data[0][2] * t.data[2][0],
1395  t07 = 1.0 / (t4 * t.data[2][2] - t6 * t.data[2][1] -
1396  t8 * t.data[2][2] + t00 * t.data[2][1] +
1397  t01 * t.data[1][2] - t04 * t.data[1][1]);
1398  tmp.data[0][0] =
1399  (t.data[1][1] * t.data[2][2] - t.data[1][2] * t.data[2][1]) * t07;
1400  tmp.data[0][1] =
1401  -(t.data[0][1] * t.data[2][2] - t.data[0][2] * t.data[2][1]) * t07;
1402  tmp.data[0][2] =
1403  -(-t.data[0][1] * t.data[1][2] + t.data[0][2] * t.data[1][1]) * t07;
1404  tmp.data[1][0] =
1405  -(t.data[1][0] * t.data[2][2] - t.data[1][2] * t.data[2][0]) * t07;
1406  tmp.data[1][1] = (t.data[0][0] * t.data[2][2] - t04) * t07;
1407  tmp.data[1][2] = -(t6 - t00) * t07;
1408  tmp.data[2][0] =
1409  -(-t.data[1][0] * t.data[2][1] + t.data[1][1] * t.data[2][0]) * t07;
1410  tmp.data[2][1] = -(t.data[0][0] * t.data[2][1] - t01) * t07;
1411  tmp.data[2][2] = (t4 - t8) * t07;
1412 
1413  // scale last row and column as mentioned
1414  // above
1415  tmp.data[2][0] /= 2;
1416  tmp.data[2][1] /= 2;
1417  tmp.data[0][2] /= 2;
1418  tmp.data[1][2] /= 2;
1419  tmp.data[2][2] /= 4;
1420 
1421  return tmp;
1422  }
1423  };

1426  template <typename Number>
1427  struct Inverse<4, 3, Number>
1428  {
1429  static ::SymmetricTensor<4, 3, Number>
1430  value(const ::SymmetricTensor<4, 3, Number> &t)
1431  {
1432  ::SymmetricTensor<4, 3, Number> tmp = t;
1433 
1434  // This function follows the exact same scheme as the 2d case, except
1435  // that hardcoding the inverse of a 6x6 matrix is pretty wasteful.
1436  // Instead, we use the Gauss-Jordan algorithm implemented for
1437  // FullMatrix. For historical reasons the following code is copied from
1438  // there, with the tangential benefit that we do not need to copy the
1439  // tensor entries to and from the FullMatrix.
1440  const unsigned int N = 6;
1441 
1442  // First get an estimate of the size of the elements of this matrix,
1443  // for later checks whether the pivot element is large enough, or
1444  // whether we have to fear that the matrix is not regular.
1445  Number diagonal_sum = internal::NumberType<Number>::value(0.0);
1446  for (unsigned int i = 0; i < N; ++i)
1447  diagonal_sum += std::fabs(tmp.data[i][i]);
1448  const Number typical_diagonal_element =
1449  diagonal_sum / static_cast<double>(N);
1450  (void)typical_diagonal_element;
1451 
1452  unsigned int p[N];
1453  for (unsigned int i = 0; i < N; ++i)
1454  p[i] = i;
1455 
1456  for (unsigned int j = 0; j < N; ++j)
1457  {
1458  // Pivot search: search that part of the line on and right of the
1459  // diagonal for the largest element.
1460  Number max = std::fabs(tmp.data[j][j]);
1461  unsigned int r = j;
1462  for (unsigned int i = j + 1; i < N; ++i)
1463  if (std::fabs(tmp.data[i][j]) > max)
1464  {
1465  max = std::fabs(tmp.data[i][j]);
1466  r = i;
1467  }
1468 
1469  // Check whether the pivot is too small
1470  Assert(max > 1.e-16 * typical_diagonal_element,
1471  ExcMessage("This tensor seems to be noninvertible"));
1472 
1473  // Row interchange
1474  if (r > j)
1475  {
1476  for (unsigned int k = 0; k < N; ++k)
1477  std::swap(tmp.data[j][k], tmp.data[r][k]);
1478 
1479  std::swap(p[j], p[r]);
1480  }
1481 
1482  // Transformation
1483  const Number hr = 1. / tmp.data[j][j];
1484  tmp.data[j][j] = hr;
1485  for (unsigned int k = 0; k < N; ++k)
1486  {
1487  if (k == j)
1488  continue;
1489  for (unsigned int i = 0; i < N; ++i)
1490  {
1491  if (i == j)
1492  continue;
1493  tmp.data[i][k] -= tmp.data[i][j] * tmp.data[j][k] * hr;
1494  }
1495  }
1496  for (unsigned int i = 0; i < N; ++i)
1497  {
1498  tmp.data[i][j] *= hr;
1499  tmp.data[j][i] *= -hr;
1500  }
1501  tmp.data[j][j] = hr;
1502  }
1503 
1504  // Column interchange
1505  Number hv[N];
1506  for (unsigned int i = 0; i < N; ++i)
1507  {
1508  for (unsigned int k = 0; k < N; ++k)
1509  hv[p[k]] = tmp.data[i][k];
1510  for (unsigned int k = 0; k < N; ++k)
1511  tmp.data[i][k] = hv[k];
1512  }
1513 
1514  // Scale rows and columns. The mult matrix
1515  // here is diag[1, 1, 1, 1/2, 1/2, 1/2].
1516  for (unsigned int i = 3; i < 6; ++i)
1517  for (unsigned int j = 0; j < 3; ++j)
1518  tmp.data[i][j] /= 2;
1519 
1520  for (unsigned int i = 0; i < 3; ++i)
1521  for (unsigned int j = 3; j < 6; ++j)
1522  tmp.data[i][j] /= 2;
1523 
1524  for (unsigned int i = 3; i < 6; ++i)
1525  for (unsigned int j = 3; j < 6; ++j)
1526  tmp.data[i][j] /= 4;
1527 
1528  return tmp;
1529  }
