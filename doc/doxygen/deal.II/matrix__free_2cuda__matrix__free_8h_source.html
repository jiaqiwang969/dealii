758  // This function determines the number of cells per block, possibly at compile
759  // time (by virtue of being 'constexpr')
760  // TODO this function should be rewritten using meta-programming
761  __host__ __device__ constexpr unsigned int
762  cells_per_block_shmem(int dim, int fe_degree)
763  {
764  /* clang-format off */
765  // We are limiting the number of threads according to the
766  // following formulas:
767  // - in 2D: `threads = cells * (k+1)^d <= 4*CUDAWrappers::warp_size`
768  // - in 3D: `threads = cells * (k+1)^d <= 2*CUDAWrappers::warp_size`
769  return dim==2 ? (fe_degree==1 ? CUDAWrappers::warp_size : // 128
770  fe_degree==2 ? CUDAWrappers::warp_size/4 : // 72
771  fe_degree==3 ? CUDAWrappers::warp_size/8 : // 64
772  fe_degree==4 ? CUDAWrappers::warp_size/8 : // 100
773  1) :
774  dim==3 ? (fe_degree==1 ? CUDAWrappers::warp_size/4 : // 64
775  fe_degree==2 ? CUDAWrappers::warp_size/16 : // 54
776  1) : 1;
777  /* clang-format on */
778  }

787  template <int dim>
788  __device__ inline unsigned int
789  q_point_id_in_cell(const unsigned int n_q_points_1d)
790  {
791  return (dim == 1 ?
792  threadIdx.x % n_q_points_1d :
793  dim == 2 ?
794  threadIdx.x % n_q_points_1d + n_q_points_1d * threadIdx.y :
795  threadIdx.x % n_q_points_1d +
796  n_q_points_1d * (threadIdx.y + n_q_points_1d * threadIdx.z));
797  }

916  template <int dim, typename Number>
917  DataHost<dim, Number>
918  copy_mf_data_to_host(

921  {

923

926  data_host.padding_length = data.padding_length;
927  data_host.row_start = data.row_start;
928  data_host.use_coloring = data.use_coloring;
929 
930  const unsigned int n_elements =
931  data_host.n_cells * data_host.padding_length;

933  {
934  data_host.q_points.resize(n_elements);
935  Utilities::CUDA::copy_to_host(data.q_points, data_host.q_points);

937

939  Utilities::CUDA::copy_to_host(data.local_to_global,
940  data_host.local_to_global);
941 
942  if (update_flags & update_gradients)
943  {
944  data_host.inv_jacobian.resize(n_elements * dim * dim);
945  Utilities::CUDA::copy_to_host(data.inv_jacobian,
946  data_host.inv_jacobian);
947  }
948 
949  if (update_flags & update_JxW_values)
950  {
951  data_host.JxW.resize(n_elements);
952  Utilities::CUDA::copy_to_host(data.JxW, data_host.JxW);
953  }
954 
955  data_host.constraint_mask.resize(data_host.n_cells);
956  Utilities::CUDA::copy_to_host(data.constraint_mask,
957  data_host.constraint_mask);
