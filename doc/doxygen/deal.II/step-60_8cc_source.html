232  template <int dim, int spacedim>
233  void DistributedLagrangeProblem<dim, spacedim>::setup_grids_and_dofs()
234  {
235  TimerOutput::Scope timer_section(monitor, "Setup grids and dofs");
236 
237  space_grid = std::make_unique<Triangulation<spacedim>>();
238 
239  GridGenerator::hyper_cube(*space_grid, 0, 1, true);
240 
241  space_grid->refine_global(parameters.initial_refinement);
242  space_grid_tools_cache =
243  std::make_unique<GridTools::Cache<spacedim, spacedim>>(*space_grid);
244 
245  embedded_grid = std::make_unique<Triangulation<dim, spacedim>>();
246  GridGenerator::hyper_cube(*embedded_grid);
247  embedded_grid->refine_global(parameters.initial_embedded_refinement);
248 
249  embedded_configuration_fe = std::make_unique<FESystem<dim, spacedim>>(
250  FE_Q<dim, spacedim>(
251  parameters.embedded_configuration_finite_element_degree),
252  spacedim);
253 
254  embedded_configuration_dh =
255  std::make_unique<DoFHandler<dim, spacedim>>(*embedded_grid);
256 
257  embedded_configuration_dh->distribute_dofs(*embedded_configuration_fe);
258  embedded_configuration.reinit(embedded_configuration_dh->n_dofs());
259 
260  VectorTools::interpolate(*embedded_configuration_dh,
261  embedded_configuration_function,
262  embedded_configuration);
263 
264 
265  if (parameters.use_displacement == true)
266  embedded_mapping =
267  std::make_unique<MappingQEulerian<dim, Vector<double>, spacedim>>(
268  parameters.embedded_configuration_finite_element_degree,
269  *embedded_configuration_dh,
270  embedded_configuration);
271  else
272  embedded_mapping =
273  std::make_unique<MappingFEField<dim, spacedim, Vector<double>>>(
274  *embedded_configuration_dh, embedded_configuration);
275 
276  setup_embedded_dofs();
277 
278  std::vector<Point<spacedim>> support_points(embedded_dh->n_dofs());
279  if (parameters.delta_refinement != 0)
280  DoFTools::map_dofs_to_support_points(*embedded_mapping,
281  *embedded_dh,
282  support_points);
283 
284  for (unsigned int i = 0; i < parameters.delta_refinement; ++i)
285  {
286  const auto point_locations =
287  GridTools::compute_point_locations(*space_grid_tools_cache,
288  support_points);
289  const auto &cells = std::get<0>(point_locations);
290  for (auto &cell : cells)
291  {
292  cell->set_refine_flag();
293  for (const auto face_no : cell->face_indices())
294  if (!cell->at_boundary(face_no))
295  cell->neighbor(face_no)->set_refine_flag();
296  }
297  space_grid->execute_coarsening_and_refinement();
298  }
299 
300  const double embedded_space_maximal_diameter =
301  GridTools::maximal_cell_diameter(*embedded_grid, *embedded_mapping);
302  double embedding_space_minimal_diameter =
303  GridTools::minimal_cell_diameter(*space_grid);
304 
305  deallog << "Embedding minimal diameter: "
306  << embedding_space_minimal_diameter
307  << ", embedded maximal diameter: "
308  << embedded_space_maximal_diameter << ", ratio: "
309  << embedded_space_maximal_diameter /
310  embedding_space_minimal_diameter
311  << std::endl;
312 
313  AssertThrow(embedded_space_maximal_diameter <
314  embedding_space_minimal_diameter,
315  ExcMessage(
316  "The embedding grid is too refined (or the embedded grid "
317  "is too coarse). Adjust the parameters so that the minimal "
318  "grid size of the embedding grid is larger "
319  "than the maximal grid size of the embedded grid."));
320 
321  setup_embedding_dofs();
322  }

324  template <int dim, int spacedim>
325  void DistributedLagrangeProblem<dim, spacedim>::setup_embedding_dofs()
326  {
327  space_dh = std::make_unique<DoFHandler<spacedim>>(*space_grid);
328  space_fe = std::make_unique<FE_Q<spacedim>>(
329  parameters.embedding_space_finite_element_degree);
330  space_dh->distribute_dofs(*space_fe);
331 
332  DoFTools::make_hanging_node_constraints(*space_dh, constraints);
333  for (auto id : parameters.homogeneous_dirichlet_ids)
334  {
335  VectorTools::interpolate_boundary_values(
336  *space_dh, id, Functions::ZeroFunction<spacedim>(), constraints);
337  }
338  constraints.close();
339 
340  DynamicSparsityPattern dsp(space_dh->n_dofs(), space_dh->n_dofs());
341  DoFTools::make_sparsity_pattern(*space_dh, dsp, constraints);
342  stiffness_sparsity.copy_from(dsp);
343  stiffness_matrix.reinit(stiffness_sparsity);
344  solution.reinit(space_dh->n_dofs());
345  rhs.reinit(space_dh->n_dofs());
346 
347  deallog << "Embedding dofs: " << space_dh->n_dofs() << std::endl;
348  }

428  template <int dim, int spacedim>
429  void DistributedLagrangeProblem<dim, spacedim>::solve()
430  {
431  TimerOutput::Scope timer_section(monitor, "Solve system");
432 
433  SparseDirectUMFPACK K_inv_umfpack;
434  K_inv_umfpack.initialize(stiffness_matrix);
435 
436  auto K = linear_operator(stiffness_matrix);
437  auto Ct = linear_operator(coupling_matrix);
438  auto C = transpose_operator(Ct);
439 
440  auto K_inv = linear_operator(K, K_inv_umfpack);
441 
442  auto S = C * K_inv * Ct;
443  SolverCG<Vector<double>> solver_cg(schur_solver_control);
444  auto S_inv = inverse_operator(S, solver_cg, PreconditionIdentity());
445 
446  lambda = S_inv * embedded_rhs;
447 
448  solution = K_inv * Ct * lambda;
449 
450  constraints.distribute(solution);
451  }
