680  const DoFHandler<dim, spacedim> &dof_handler,

683 {
684  (void)dof_handler;
685 
686  Assert(
687  initialize_dof_vector,
688  ExcMessage(
689  "To be able to use this function, a function to initialize an internal "
690  "DoF vector has to be provided in the constructor of "
691  "MGTransferGlobalCoarsening."));
692 
693  const unsigned int min_level = transfer.min_level();
694  const unsigned int max_level = transfer.max_level();
695 
696  AssertDimension(min_level, dst.min_level());
697  AssertDimension(max_level, dst.max_level());
698 
699  for (unsigned int level = min_level; level <= max_level; ++level)
700  initialize_dof_vector(level, dst[level]);
701 
702  dst[transfer.max_level()].copy_locally_owned_data_from(src);
703 
704  for (unsigned int l = max_level; l > min_level; --l)
705  this->transfer[l].interpolate(dst[l - 1], dst[l]);
706 }
