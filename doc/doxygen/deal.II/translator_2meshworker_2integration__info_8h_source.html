559  template <int dim, int sdim>
560  inline IntegrationInfo<dim, sdim>::IntegrationInfo(
561  const IntegrationInfo<dim, sdim> &other)
562  : multigrid(other.multigrid)
563  , values(other.values)
564  , gradients(other.gradients)
565  , hessians(other.hessians)
566  , global_data(other.global_data)
567  , fe_pointer(other.fe_pointer)
568  , n_components(other.n_components)
569  {
570  fevalv.resize(other.fevalv.size());
571  for (unsigned int i = 0; i < other.fevalv.size(); ++i)
572  {
573  const FEValuesBase<dim, sdim> &p = *other.fevalv[i];
574  const FEValues<dim, sdim> * pc =
575  dynamic_cast<const FEValues<dim, sdim> *>(&p);
576  const FEFaceValues<dim, sdim> *pf =
577  dynamic_cast<const FEFaceValues<dim, sdim> *>(&p);
578  const FESubfaceValues<dim, sdim> *ps =
579  dynamic_cast<const FESubfaceValues<dim, sdim> *>(&p);
580 
581  if (pc != nullptr)
582  fevalv[i] =
583  std::make_shared<FEValues<dim, sdim>>(pc->get_mapping(),
584  pc->get_fe(),
585  pc->get_quadrature(),
586  pc->get_update_flags());
587  else if (pf != nullptr)
588  fevalv[i] =
589  std::make_shared<FEFaceValues<dim, sdim>>(pf->get_mapping(),
590  pf->get_fe(),
591  pf->get_quadrature(),
592  pf->get_update_flags());
593  else if (ps != nullptr)
594  fevalv[i] = std::make_shared<FESubfaceValues<dim, sdim>>(
595  ps->get_mapping(),
596  ps->get_fe(),
597  ps->get_quadrature(),
598  ps->get_update_flags());
599  else
600  Assert(false, ExcInternalError());
601  }
602  }

606  template <int dim, int sdim>
607  template <class FEVALUES>
608  inline void
609  IntegrationInfo<dim, sdim>::initialize(
610  const FiniteElement<dim, sdim> & el,
611  const Mapping<dim, sdim> & mapping,
612  const Quadrature<FEVALUES::integral_dimension> &quadrature,
613  const UpdateFlags flags,
614  const BlockInfo * block_info)
615  {
616  fe_pointer = &el;
617  if (block_info == nullptr || block_info->local().size() == 0)
618  {
619  fevalv.resize(1);
620  fevalv[0] = std::make_shared<FEVALUES>(mapping, el, quadrature, flags);
621  }
622  else
623  {
624  fevalv.resize(el.n_base_elements());
625  for (unsigned int i = 0; i < fevalv.size(); ++i)
626  fevalv[i] = std::make_shared<FEVALUES>(mapping,
627  el.base_element(i),
628  quadrature,
629  flags);
630  }
631  n_components = el.n_components();
632  }

661  template <int dim, int spacedim>
662  template <typename number>
663  inline void
664  IntegrationInfo<dim, spacedim>::reinit(
665  const DoFInfo<dim, spacedim, number> &info)
666  {
667  for (unsigned int i = 0; i < fevalv.size(); ++i)
668  {
669  FEValuesBase<dim, spacedim> &febase = *fevalv[i];
670  if (info.sub_number != numbers::invalid_unsigned_int)
671  {
672  // This is a subface
673  FESubfaceValues<dim, spacedim> &fe =
674  dynamic_cast<FESubfaceValues<dim, spacedim> &>(febase);
675  fe.reinit(info.cell, info.face_number, info.sub_number);
676  }
677  else if (info.face_number != numbers::invalid_unsigned_int)
678  {
679  // This is a face
680  FEFaceValues<dim, spacedim> &fe =
681  dynamic_cast<FEFaceValues<dim, spacedim> &>(febase);
682  fe.reinit(info.cell, info.face_number);
683  }
684  else
685  {
686  // This is a cell
687  FEValues<dim, spacedim> &fe =
688  dynamic_cast<FEValues<dim, spacedim> &>(febase);
689  fe.reinit(info.cell);
690  }
691  }
692 
693  const bool split_fevalues = info.block_info != nullptr;
694  if (!global_data->empty())
695  fill_local_data(info, split_fevalues);
696  }
