945 template <typename MatrixType, typename inverse_type>
946 inline inverse_type
947 PreconditionBlock<MatrixType, inverse_type>::el(size_type i, size_type j) const
948 {
949  const size_type bs = blocksize;
950  const unsigned int nb = i / bs;
951 
952  const FullMatrix<inverse_type> &B = this->inverse(nb);
953 
954  const size_type ib = i % bs;
955  const size_type jb = j % bs;
956 
957  if (jb + nb * bs != j)
958  {
959  return 0.;
960  }
961 
962  return B(ib, jb);
963 }

967 template <typename MatrixType, typename inverse_type>
968 inline PreconditionBlockJacobi<MatrixType, inverse_type>::const_iterator::
969  Accessor::Accessor(
970  const PreconditionBlockJacobi<MatrixType, inverse_type> *matrix,
971  const size_type row)
972  : matrix(matrix)
973  , bs(matrix->block_size())
974  , a_block(row / bs)
975  , b_iterator(&matrix->inverse(0), 0, 0)
976  , b_end(&matrix->inverse(0), 0, 0)
977 {
978  // This is the end accessor, which
979  // does not have a valid block.
980  if (a_block == matrix->size())
981  return;
982 
983  const size_type r = row % bs;
984 
985  b_iterator = matrix->inverse(a_block).begin(r);
986  b_end = matrix->inverse(a_block).end();
987 
988  AssertIndexRange(a_block, matrix->size());
989 }
