1048 template <typename number>
1049 inline SparseMatrixEZ<number>::const_iterator::const_iterator(
1050  const SparseMatrixEZ<number> *matrix,
1051  const size_type r,
1052  const unsigned short i)
1053  : accessor(matrix, r, i)
1054 {
1055  // Finish if this is the end()
1056  if (r == accessor.matrix->m() && i == 0)
1057  return;
1058 
1059  // Make sure we never construct an
1060  // iterator pointing to a
1061  // non-existing entry
1062 
1063  // If the index points beyond the
1064  // end of the row, try the next
1065  // row.
1066  if (accessor.a_index >= accessor.matrix->row_info[accessor.a_row].length)
1067  {
1068  do
1069  {
1070  ++accessor.a_row;
1071  }
1072  // Beware! If the next row is
1073  // empty, iterate until a
1074  // non-empty row is found or we
1075  // hit the end of the matrix.
1076  while (accessor.a_row < accessor.matrix->m() &&
1077  accessor.matrix->row_info[accessor.a_row].length == 0);
1078  }

1203 template <typename number>
1204 inline typename SparseMatrixEZ<number>::Entry *
1205 SparseMatrixEZ<number>::allocate(const size_type row, const size_type col)
1206 {
1207  AssertIndexRange(row, m());
1208  AssertIndexRange(col, n());
1209 
1210  RowInfo & r = row_info[row];
1211  const size_type end = r.start + r.length;
1212 
1213  size_type i = r.start;
1214  // If diagonal exists and this
1215  // column is higher, start only
1216  // after diagonal.
1217  if (r.diagonal != RowInfo::invalid_diagonal && col >= row)
1218  i += r.diagonal;
1219  // Find position of entry
1220  while (i < end && data[i].column < col)
1221  ++i;
1222 
1223  // entry found
1224  if (i != end && data[i].column == col)
1225  return &data[i];
1226 
1227  // Now, we must insert the new
1228  // entry and move all successive
1229  // entries back.
1230 
1231  // If no more space is available
1232  // for this row, insert new
1233  // elements into the vector.
1234  // TODO:[GK] We should not extend this row if i<end
1235  if (row != row_info.size() - 1)
1236  {
1237  if (end >= row_info[row + 1].start)
1238  {
1239  // Failure if increment 0
1240  Assert(increment != 0, ExcEntryAllocationFailure(row, col));
1241 
1242  // Insert new entries
1243  data.insert(data.begin() + end, increment, Entry());
1244  // Update starts of
1245  // following rows
1246  for (size_type rn = row + 1; rn < row_info.size(); ++rn)
1247  row_info[rn].start += increment;
1248  }
1249  }
1250  else
1251  {
1252  if (end >= data.size())
1253  {
1254  // Here, appending a block
1255  // does not increase
1256  // performance.
1257  data.push_back(Entry());
1258  }
1259  }
1260 
1261  Entry *entry = &data[i];
1262  // Save original entry
1263  Entry temp = *entry;
1264  // Insert new entry here to
1265  // make sure all entries
1266  // are ordered by column
1267  // index
1268  entry->column = col;
1269  entry->value = 0;
1270  // Update row_info
1271  ++r.length;
1272  if (col == row)
1273  r.diagonal = i - r.start;
1274  else if (col < row && r.diagonal != RowInfo::invalid_diagonal)
1275  ++r.diagonal;
1276 
1277  if (i == end)
1278  return entry;
1279 
1280  // Move all entries in this
1281  // row up by one
1282  for (size_type j = i + 1; j < end; ++j)
1283  {
1284  // There should be no invalid
1285  // entry below end
1286  Assert(data[j].column != Entry::invalid, ExcInternalError());
1287 
1288  // TODO[GK]: This could be done more efficiently by moving starting at the
1289  // top rather than swapping starting at the bottom
1290  std::swap(data[j], temp);
1291  }
1292  Assert(data[end].column == Entry::invalid, ExcInternalError());
1293 
1294  data[end] = temp;
1295 
1296  return entry;
1297 }

1522 template <typename number>
1523 template <typename MatrixTypeA, typename MatrixTypeB>
1524 inline void
1525 SparseMatrixEZ<number>::conjugate_add(const MatrixTypeA &A,
1526  const MatrixTypeB &B,
1527  const bool transpose)
1528 {
1529  // Compute the result
1530  // r_ij = \sum_kl b_ik b_jl a_kl
1531 
1532  // Assert (n() == B.m(), ExcDimensionMismatch(n(), B.m()));
1533  // Assert (m() == B.m(), ExcDimensionMismatch(m(), B.m()));
1534  // Assert (A.n() == B.n(), ExcDimensionMismatch(A.n(), B.n()));
1535  // Assert (A.m() == B.n(), ExcDimensionMismatch(A.m(), B.n()));
1536 
1537  // Somehow, we have to avoid making
1538  // this an operation of complexity
1539  // n^2. For the transpose case, we
1540  // can go through the non-zero
1541  // elements of A^-1 and use the
1542  // corresponding rows of B only.
1543  // For the non-transpose case, we
1544  // must find a trick.
1545  typename MatrixTypeB::const_iterator b1 = B.begin();
1546  const typename MatrixTypeB::const_iterator b_final = B.end();
1547  if (transpose)
1548  while (b1 != b_final)
1549  {
1550  const size_type i = b1->column();
1551  const size_type k = b1->row();
1552  typename MatrixTypeB::const_iterator b2 = B.begin();
1553  while (b2 != b_final)
1554  {
1555  const size_type j = b2->column();
1556  const size_type l = b2->row();
1557 
1558  const typename MatrixTypeA::value_type a = A.el(k, l);
1559 
1560  if (a != 0.)
1561  add(i, j, a * b1->value() * b2->value());
1562  ++b2;
1563  }
1564  ++b1;
1565  }
1566  else
1567  {
1568  // Determine minimal and
1569  // maximal row for a column in
1570  // advance.
1571 
1572  std::vector<size_type> minrow(B.n(), B.m());
1573  std::vector<size_type> maxrow(B.n(), 0);
1574  while (b1 != b_final)
1575  {
1576  const size_type r = b1->row();
1577  if (r < minrow[b1->column()])
1578  minrow[b1->column()] = r;
1579  if (r > maxrow[b1->column()])
1580  maxrow[b1->column()] = r;
1581  ++b1;
1582  }
1583 
1584  typename MatrixTypeA::const_iterator ai = A.begin();
1585  const typename MatrixTypeA::const_iterator ae = A.end();
1586 
1587  while (ai != ae)
1588  {
1589  const typename MatrixTypeA::value_type a = ai->value();
1590  // Don't do anything if
1591  // this entry is zero.
1592  if (a == 0.)
1593  continue;
1594 
1595  // Now, loop over all rows
1596  // having possibly a
1597  // nonzero entry in column
1598  // ai->row()
1599  b1 = B.begin(minrow[ai->row()]);
1600  const typename MatrixTypeB::const_iterator be1 =
1601  B.end(maxrow[ai->row()]);
1602  const typename MatrixTypeB::const_iterator be2 =
1603  B.end(maxrow[ai->column()]);
1604 
1605  while (b1 != be1)
1606  {
1607  const double b1v = b1->value();
1608  // We need the product
1609  // of both. If it is
1610  // zero, we can save
1611  // the work
1612  if (b1->column() == ai->row() && (b1v != 0.))
1613  {
1614  const size_type i = b1->row();
1615 
1616  typename MatrixTypeB::const_iterator b2 =
1617  B.begin(minrow[ai->column()]);
1618  while (b2 != be2)
1619  {
1620  if (b2->column() == ai->column())
1621  {
1622  const size_type j = b2->row();
1623  add(i, j, a * b1v * b2->value());
1624  }
1625  ++b2;
1626  }
1627  }
1628  ++b1;
1629  }
1630  ++ai;
1631  }
1632  }
1633 }
