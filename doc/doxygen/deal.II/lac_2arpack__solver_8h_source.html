547 ArpackSolver::solve(const MatrixType1 & /*system_matrix*/ ,
548  const MatrixType2 & mass_matrix,
549  const INVERSE & inverse,
550  std::vector<std::complex<double>> &eigenvalues,
551  std::vector<VectorType> & eigenvectors,
552  const unsigned int n_eigenvalues)
553 {
554  // Problem size
555  unsigned int n = eigenvectors[0].size();
556 
557  // Number of eigenvalues
558  const unsigned int nev_const =
559  (n_eigenvalues == 0) ? eigenvalues.size() : n_eigenvalues;
560  // nev for arpack, which might change by plus one during dneupd
561  unsigned int nev = nev_const;
562 
563  // check input sizes
564  if (additional_data.symmetric)
565  {
566  Assert(nev <= eigenvectors.size(),
567  ArpackExcInvalidEigenvectorSize(nev, eigenvectors.size()));
568  }
569  else
570  Assert(nev + 1 <= eigenvectors.size(),
571  ArpackExcInvalidEigenvectorSizeNonsymmetric(nev,
572  eigenvectors.size()));
573 
574  Assert(nev <= eigenvalues.size(),
575  ArpackExcInvalidEigenvalueSize(nev, eigenvalues.size()));
576 
577  // check large enough problem size
578  Assert(nev < n, ArpackExcInvalidNumberofEigenvalues(nev, n));
579 
580  Assert(additional_data.number_of_arnoldi_vectors < n,
581  ArpackExcInvalidNumberofArnoldiVectors(
582  additional_data.number_of_arnoldi_vectors, n));
583 
584  // check whether we have enough Arnoldi vectors
585  Assert(additional_data.number_of_arnoldi_vectors > 2 * nev + 1,
586  ArpackExcSmallNumberofArnoldiVectors(
587  additional_data.number_of_arnoldi_vectors, nev));
588 
589  // ARPACK mode for dsaupd/dnaupd, here only mode 3, i.e. shift-invert mode
590  int mode = 3;
591 
592  // reverse communication parameter
593  int ido = 0;
594 
595  // 'G' generalized eigenvalue problem 'I' standard eigenvalue problem
596  char bmat[2] = "G";
597 
598  // Specify the eigenvalues of interest, possible parameters "LA" algebraically
599  // largest "SA" algebraically smallest "LM" largest magnitude "SM" smallest
600  // magnitude "LR" largest real part "SR" smallest real part "LI" largest
601  // imaginary part "SI" smallest imaginary part "BE" both ends of spectrum
602  // simultaneous.
603  char which[3];
604  switch (additional_data.eigenvalue_of_interest)
605  {
606  case algebraically_largest:
607  std::strcpy(which, "LA");
608  break;
609  case algebraically_smallest:
610  std::strcpy(which, "SA");
611  break;
612  case largest_magnitude:
613  std::strcpy(which, "LM");
614  break;
615  case smallest_magnitude:
616  std::strcpy(which, "SM");
617  break;
618  case largest_real_part:
619  std::strcpy(which, "LR");
620  break;
621  case smallest_real_part:
622  std::strcpy(which, "SR");
623  break;
624  case largest_imaginary_part:
625  std::strcpy(which, "LI");
626  break;
627  case smallest_imaginary_part:
628  std::strcpy(which, "SI");
629  break;
630  case both_ends:
631  std::strcpy(which, "BE");
632  break;
633  }
634 
635  // tolerance for ARPACK
636  double tol = control().tolerance();
637 
638  // if the starting vector is used it has to be in resid
639  if (!initial_vector_provided || resid.size() != n)
640  resid.resize(n, 1.);
641 
642  // number of Arnoldi basis vectors specified
643  // in additional_data
644  int ncv = additional_data.number_of_arnoldi_vectors;
645 
646  int ldv = n;
647  std::vector<double> v(ldv * ncv, 0.0);
648 
649  // information to the routines
650  std::vector<int> iparam(11, 0);
651 
652  iparam[0] = 1; // shift strategy
653 
654  // maximum number of iterations
655  iparam[2] = control().max_steps();
656 
657  // Set the mode of dsaupd. 1 is exact shifting, 2 is user-supplied shifts,
658  // 3 is shift-invert mode, 4 is buckling mode, 5 is Cayley mode.
659 
660  iparam[6] = mode;
661  std::vector<int> ipntr(14, 0);
662 
663  // work arrays for ARPACK
664  std::vector<double> workd(3 * n, 0.);
665  int lworkl =
666  additional_data.symmetric ? ncv * ncv + 8 * ncv : 3 * ncv * ncv + 6 * ncv;
667  std::vector<double> workl(lworkl, 0.);
668 
669  // information out of the iteration
670  int info = 1;
671 
672  while (ido != 99)
673  {
674  // call of ARPACK dsaupd/dnaupd routine
675  if (additional_data.symmetric)
676  dsaupd_(&ido,
677  bmat,
678  &n,
679  which,
680  &nev,
681  &tol,
682  resid.data(),
683  &ncv,
684  v.data(),
685  &ldv,
686  iparam.data(),
687  ipntr.data(),
688  workd.data(),
689  workl.data(),
690  &lworkl,
691  &info);
692  else
693  dnaupd_(&ido,
694  bmat,
695  &n,
696  which,
697  &nev,
698  &tol,
699  resid.data(),
700  &ncv,
701  v.data(),
702  &ldv,
703  iparam.data(),
704  ipntr.data(),
705  workd.data(),
706  workl.data(),
707  &lworkl,
708  &info);
709 
710  if (ido == 99)
711  break;
712 
713  switch (mode)
714  {
715  case 3:
716  {
717  switch (ido)
718  {
719  case -1:
720  {
721  VectorType src, dst, tmp;
722  src.reinit(eigenvectors[0]);
723  dst.reinit(src);
724  tmp.reinit(src);
725 
726 
727  for (size_type i = 0; i < src.size(); ++i)
728  src(i) = workd[ipntr[0] - 1 + i];
729 
730  // multiplication with mass matrix M
731  mass_matrix.vmult(tmp, src);
732  // solving linear system
733  inverse.vmult(dst, tmp);
734 
735  for (size_type i = 0; i < dst.size(); ++i)
736  workd[ipntr[1] - 1 + i] = dst(i);
737  }
738  break;
739 
740  case 1:
741  {
742  VectorType src, dst, tmp, tmp2;
743  src.reinit(eigenvectors[0]);
744  dst.reinit(src);
745  tmp.reinit(src);
746  tmp2.reinit(src);
747 
748  for (size_type i = 0; i < src.size(); ++i)
749  {
750  src(i) = workd[ipntr[2] - 1 + i];
751  tmp(i) = workd[ipntr[0] - 1 + i];
752  }
753  // solving linear system
754  inverse.vmult(dst, src);
755 
756  for (size_type i = 0; i < dst.size(); ++i)
757  workd[ipntr[1] - 1 + i] = dst(i);
758  }
759  break;
760 
761  case 2:
762  {
763  VectorType src, dst;
764  src.reinit(eigenvectors[0]);
765  dst.reinit(src);
766 
767  for (size_type i = 0; i < src.size(); ++i)
768  src(i) = workd[ipntr[0] - 1 + i];
769 
770  // Multiplication with mass matrix M
771  mass_matrix.vmult(dst, src);
772 
773  for (size_type i = 0; i < dst.size(); ++i)
774  workd[ipntr[1] - 1 + i] = dst(i);
775  }
776  break;
777 
778  default:
779  Assert(false, ArpackExcArpackIdo(ido));
780  break;
781  }
782  }
783  break;
784  default:
785  Assert(false, ArpackExcArpackMode(mode));
786  break;
787  }
788  }
789 
790  // Set number of used iterations in SolverControl
791  control().check(iparam[2], 0.);
792 
793  if (info < 0)
794  {
795  if (additional_data.symmetric)
796  {
797  Assert(false, ArpackExcArpackInfodsaupd(info));
798  }
799  else
800  Assert(false, ArpackExcArpackInfodnaupd(info));
801  }
802  else
803  {
804  // 1 - compute eigenvectors, 0 - only eigenvalues
805  int rvec = 1;
806 
807  // which eigenvectors
808  char howmany = 'A';
809 
810  std::vector<int> select(ncv, 1);
811 
812  int ldz = n;
813 
814  std::vector<double> eigenvalues_real(nev + 1, 0.);
815  std::vector<double> eigenvalues_im(nev + 1, 0.);
816 
817  // call of ARPACK dseupd/dneupd routine
818  if (additional_data.symmetric)
819  {
820  std::vector<double> z(ldz * nev, 0.);
821  dseupd_(&rvec,
822  &howmany,
823  select.data(),
824  eigenvalues_real.data(),
825  z.data(),
826  &ldz,
827  &sigmar,
828  bmat,
829  &n,
830  which,
831  &nev,
832  &tol,
833  resid.data(),

840  workl.data(),
841  &lworkl,
842  &info);
843  }
844  else
845  {
846  std::vector<double> workev(3 * ncv, 0.);
847  dneupd_(&rvec,
848  &howmany,
849  select.data(),
850  eigenvalues_real.data(),
851  eigenvalues_im.data(),
852  v.data(),
853  &ldz,

868  workd.data(),
869  workl.data(),

872  }
873 
874  if (info == 1)
875  {
876  Assert(false, ArpackExcArpackInfoMaxIt(control().max_steps()));
877  }
878  else if (info == 3)
879  {
880  Assert(false, ArpackExcArpackNoShifts());
881  }
882  else if (info != 0)
883  {
884  if (additional_data.symmetric)
885  {
886  Assert(false, ArpackExcArpackInfodseupd(info));
887  }
888  else
889  Assert(false, ArpackExcArpackInfodneupd(info));
890  }
891 
892  for (unsigned int i = 0; i < nev; ++i)
893  for (unsigned int j = 0; j < n; ++j)
894  eigenvectors[i](j) = v[i * n + j];
895 
896  for (unsigned int i = 0; i < nev_const; ++i)

898  std::complex<double>(eigenvalues_real[i], eigenvalues_im[i]);
