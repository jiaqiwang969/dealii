719 template <typename VectorType>
720 void
721 PArpackSolver<VectorType>::internal_reinit(const IndexSet &locally_owned_dofs)
722 {
723  // store local indices to write to vectors
724  locally_owned_dofs.fill_index_vector(local_indices);
725 
726  // scalars
727  nloc = locally_owned_dofs.n_elements();
728  ncv = additional_data.number_of_arnoldi_vectors;
729 
730  AssertDimension(local_indices.size(), nloc);
731 
732  // vectors
733  ldv = nloc;
734  v.resize(ldv * ncv, 0.0);
735 
736  resid.resize(nloc, 1.0);
737 
738  // work arrays for ARPACK
739  workd.resize(3 * nloc, 0.0);
740 
741  lworkl =
742  additional_data.symmetric ? ncv * ncv + 8 * ncv : 3 * ncv * ncv + 6 * ncv;
743  workl.resize(lworkl, 0.);
744 
745  ldz = nloc;
746  z.resize(ldz * ncv, 0.); // TODO we actually need only ldz*nev
747 
748  // WORKEV Double precision work array of dimension 3*NCV.
749  lworkev = additional_data.symmetric ? 0 /*not used in symmetric case*/
750  :
751  3 * ncv;
752  workev.resize(lworkev, 0.);
753 
754  select.resize(ncv, 0);
755 }

820 template <typename VectorType>
821 template <typename MatrixType1, typename MatrixType2, typename INVERSE>
822 void
823 PArpackSolver<VectorType>::solve(const MatrixType1 &system_matrix,
824  const MatrixType2 &mass_matrix,
825  const INVERSE & inverse,
826  std::vector<std::complex<double>> &eigenvalues,
827  std::vector<VectorType *> &eigenvectors,
828  const unsigned int n_eigenvalues)
829 {
830  if (additional_data.symmetric)
831  {
832  Assert(n_eigenvalues <= eigenvectors.size(),
833  PArpackExcInvalidEigenvectorSize(n_eigenvalues,
834  eigenvectors.size()));
835  }
836  else
837  Assert(n_eigenvalues + 1 <= eigenvectors.size(),
838  PArpackExcInvalidEigenvectorSizeNonsymmetric(n_eigenvalues,
839  eigenvectors.size()));
840 
841  Assert(n_eigenvalues <= eigenvalues.size(),
842  PArpackExcInvalidEigenvalueSize(n_eigenvalues, eigenvalues.size()));
843 
844 
845  // use eigenvectors to get the problem size so that it is possible to
846  // employ LinearOperator for mass_matrix.
847  Assert(n_eigenvalues < eigenvectors[0]->size(),
848  PArpackExcInvalidNumberofEigenvalues(n_eigenvalues,
849  eigenvectors[0]->size()));
850 
851  Assert(additional_data.number_of_arnoldi_vectors < eigenvectors[0]->size(),
852  PArpackExcInvalidNumberofArnoldiVectors(
853  additional_data.number_of_arnoldi_vectors, eigenvectors[0]->size()));
854 
855  Assert(additional_data.number_of_arnoldi_vectors > 2 * n_eigenvalues + 1,
856  PArpackExcSmallNumberofArnoldiVectors(
857  additional_data.number_of_arnoldi_vectors, n_eigenvalues));
858 
859  int mode = additional_data.mode;
860 
861  // reverse communication parameter
862  // must be zero on the first call to pdnaupd
863  int ido = 0;
864 
865  // 'G' generalized eigenvalue problem
866  // 'I' standard eigenvalue problem
867  char bmat[2];
868  bmat[0] = (mode == 1) ? 'I' : 'G';
869  bmat[1] = '\0';
870 
871  // Specify the eigenvalues of interest, possible parameters:
872  // "LA" algebraically largest
873  // "SA" algebraically smallest
874  // "LM" largest magnitude
875  // "SM" smallest magnitude
876  // "LR" largest real part
877  // "SR" smallest real part
878  // "LI" largest imaginary part
879  // "SI" smallest imaginary part
880  // "BE" both ends of spectrum simultaneous
881  char which[3];
882  switch (additional_data.eigenvalue_of_interest)
883  {
884  case algebraically_largest:
885  std::strcpy(which, "LA");
886  break;
887  case algebraically_smallest:
888  std::strcpy(which, "SA");
889  break;
890  case largest_magnitude:
891  std::strcpy(which, "LM");
892  break;
893  case smallest_magnitude:
894  std::strcpy(which, "SM");
895  break;
896  case largest_real_part:
897  std::strcpy(which, "LR");
898  break;
899  case smallest_real_part:
900  std::strcpy(which, "SR");
901  break;
902  case largest_imaginary_part:
903  std::strcpy(which, "LI");
904  break;
905  case smallest_imaginary_part:
906  std::strcpy(which, "SI");
907  break;
908  case both_ends:
909  std::strcpy(which, "BE");
910  break;
911  }
912 
913  // tolerance for ARPACK
914  double tol = control().tolerance();
915 
916  // information to the routines
917  std::vector<int> iparam(11, 0);
918 
919  iparam[0] = 1;
920  // shift strategy: exact shifts with respect to the current Hessenberg matrix
921  // H.
922 
923  // maximum number of iterations
924  iparam[2] = control().max_steps();
925 
926  // Parpack currently works only for NB = 1
927  iparam[3] = 1;
928 
929  // Sets the mode of dsaupd:
930  // 1 is A*x=lambda*x, OP = A, B = I
931  // 2 is A*x = lambda*M*x, OP = inv[M]*A, B = M
932  // 3 is shift-invert mode, OP = inv[A-sigma*M]*M, B = M
933  // 4 is buckling mode,
934  // 5 is Cayley mode.
935 
936  iparam[6] = mode;
937  std::vector<int> ipntr(14, 0);
938 
939  // information out of the iteration
940  // If INFO .EQ. 0, a random initial residual vector is used.
941  // If INFO .NE. 0, RESID contains the initial residual vector,
942  // possibly from a previous run.
943  // Typical choices in this situation might be to use the final value
944  // of the starting vector from the previous eigenvalue calculation
945  int info = initial_vector_provided ? 1 : 0;
946 
947  // Number of eigenvalues of OP to be computed. 0 < NEV < N.
948  int nev = n_eigenvalues;
949  int n_inside_arpack = nloc;
950 
951  // IDO = 99: done
952  while (ido != 99)
953  {
954  // call of ARPACK pdnaupd routine
955  if (additional_data.symmetric)
956  pdsaupd_(&mpi_communicator_fortran,
957  &ido,
958  bmat,
959  &n_inside_arpack,
960  which,
961  &nev,
962  &tol,
963  resid.data(),
964  &ncv,
965  v.data(),
966  &ldv,
967  iparam.data(),
968  ipntr.data(),
969  workd.data(),
970  workl.data(),
971  &lworkl,
972  &info);
973  else
974  pdnaupd_(&mpi_communicator_fortran,
975  &ido,
976  bmat,
977  &n_inside_arpack,
978  which,
979  &nev,
980  &tol,
981  resid.data(),
982  &ncv,
983  v.data(),
984  &ldv,
985  iparam.data(),
986  ipntr.data(),
987  workd.data(),
988  workl.data(),
989  &lworkl,
990  &info);
991 
992  AssertThrow(info == 0, PArpackExcInfoPdnaupd(info));
993 
994  // if we converge, we shall not modify anything in work arrays!
995  if (ido == 99)
996  break;
997 
998  // IPNTR(1) is the pointer into WORKD for X,
999  // IPNTR(2) is the pointer into WORKD for Y.
1000  const int shift_x = ipntr[0] - 1;
1001  const int shift_y = ipntr[1] - 1;
1002  Assert(shift_x >= 0, ::ExcInternalError());
1003  Assert(shift_x + nloc <= static_cast<int>(workd.size()),
1004  ::ExcInternalError());
1005  Assert(shift_y >= 0, ::ExcInternalError());
1006  Assert(shift_y + nloc <= static_cast<int>(workd.size()),
1007  ::ExcInternalError());
1008 
1009  src = 0.;
1010 
1011  // switch based on both ido and mode
1012  if ((ido == -1) || (ido == 1 && mode < 3))
1013  // compute Y = OP * X
1014  {
1015  src.add(nloc, local_indices.data(), workd.data() + shift_x);
1016  src.compress(VectorOperation::add);
1017 
1018  if (mode == 3)
