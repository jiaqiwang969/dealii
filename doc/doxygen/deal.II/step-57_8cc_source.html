185  template <class PreconditionerMp>
186  void BlockSchurPreconditioner<PreconditionerMp>::vmult(
187  BlockVector<double> & dst,
188  const BlockVector<double> &src) const
189  {
190  Vector<double> utmp(src.block(0));
191 
192  {
193  SolverControl solver_control(1000, 1e-6 * src.block(1).l2_norm());
194  SolverCG<Vector<double>> cg(solver_control);
195 
196  dst.block(1) = 0.0;
197  cg.solve(pressure_mass_matrix,
198  dst.block(1),
199  src.block(1),
200  mp_preconditioner);
201  dst.block(1) *= -(viscosity + gamma);
202  }
203 
204  {
205  stokes_matrix.block(0, 1).vmult(utmp, dst.block(1));
206  utmp *= -1.0;
207  utmp += src.block(0);
208  }
209 
210  A_inverse.vmult(dst.block(0), utmp);
211  }

288  template <int dim>
289  void StationaryNavierStokes<dim>::assemble(const bool initial_step,
290  const bool assemble_matrix)
291  {
292  if (assemble_matrix)
293  system_matrix = 0;
294 
295  system_rhs = 0;
296 
297  QGauss<dim> quadrature_formula(degree + 2);
298 
299  FEValues<dim> fe_values(fe,
300  quadrature_formula,
301  update_values | update_quadrature_points |
302  update_JxW_values | update_gradients);
303 
304  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
305  const unsigned int n_q_points = quadrature_formula.size();
306 
307  const FEValuesExtractors::Vector velocities(0);
308  const FEValuesExtractors::Scalar pressure(dim);
309 
310  FullMatrix<double> local_matrix(dofs_per_cell, dofs_per_cell);
311  Vector<double> local_rhs(dofs_per_cell);
312 
313  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
314 
315 
316  std::vector<Tensor<1, dim>> present_velocity_values(n_q_points);
317  std::vector<Tensor<2, dim>> present_velocity_gradients(n_q_points);
318  std::vector<double> present_pressure_values(n_q_points);
319 
320  std::vector<double> div_phi_u(dofs_per_cell);
321  std::vector<Tensor<1, dim>> phi_u(dofs_per_cell);
322  std::vector<Tensor<2, dim>> grad_phi_u(dofs_per_cell);
323  std::vector<double> phi_p(dofs_per_cell);
324 
325  for (const auto &cell : dof_handler.active_cell_iterators())
326  {
327  fe_values.reinit(cell);
328 
329  local_matrix = 0;
330  local_rhs = 0;
331 
332  fe_values[velocities].get_function_values(evaluation_point,
333  present_velocity_values);
334 
335  fe_values[velocities].get_function_gradients(
336  evaluation_point, present_velocity_gradients);
337 
338  fe_values[pressure].get_function_values(evaluation_point,
339  present_pressure_values);
340 
341  for (unsigned int q = 0; q < n_q_points; ++q)
342  {
343  for (unsigned int k = 0; k < dofs_per_cell; ++k)
344  {
345  div_phi_u[k] = fe_values[velocities].divergence(k, q);
346  grad_phi_u[k] = fe_values[velocities].gradient(k, q);
347  phi_u[k] = fe_values[velocities].value(k, q);
348  phi_p[k] = fe_values[pressure].value(k, q);
349  }
350 
351  for (unsigned int i = 0; i < dofs_per_cell; ++i)
352  {
353  if (assemble_matrix)
354  {
355  for (unsigned int j = 0; j < dofs_per_cell; ++j)
356  {
357  local_matrix(i, j) +=
358  (viscosity *
359  scalar_product(grad_phi_u[j], grad_phi_u[i]) +
360  present_velocity_gradients[q] * phi_u[j] * phi_u[i] +
361  grad_phi_u[j] * present_velocity_values[q] *
362  phi_u[i] -
363  div_phi_u[i] * phi_p[j] - phi_p[i] * div_phi_u[j] +
364  gamma * div_phi_u[j] * div_phi_u[i] +
365  phi_p[i] * phi_p[j]) *
366  fe_values.JxW(q);
367  }
368  }
369 
370  double present_velocity_divergence =
371  trace(present_velocity_gradients[q]);
372  local_rhs(i) +=
373  (-viscosity * scalar_product(present_velocity_gradients[q],
374  grad_phi_u[i]) -
375  present_velocity_gradients[q] * present_velocity_values[q] *
376  phi_u[i] +
377  present_pressure_values[q] * div_phi_u[i] +
378  present_velocity_divergence * phi_p[i] -
379  gamma * present_velocity_divergence * div_phi_u[i]) *
380  fe_values.JxW(q);
381  }
382  }
383 
384  cell->get_dof_indices(local_dof_indices);
385 
386  const AffineConstraints<double> &constraints_used =
387  initial_step ? nonzero_constraints : zero_constraints;
388 
389  if (assemble_matrix)
390  {
391  constraints_used.distribute_local_to_global(local_matrix,
392  local_rhs,
393  local_dof_indices,
394  system_matrix,
395  system_rhs);
396  }
397  else
398  {
399  constraints_used.distribute_local_to_global(local_rhs,
400  local_dof_indices,
401  system_rhs);
402  }
403  }
404 
405  if (assemble_matrix)
406  {
407  pressure_mass_matrix.reinit(sparsity_pattern.block(1, 1));
408  pressure_mass_matrix.copy_from(system_matrix.block(1, 1));
409 
410  system_matrix.block(1, 1) = 0;
411  }
412  }

488  template <int dim>
489  void StationaryNavierStokes<dim>::newton_iteration(
490  const double tolerance,
491  const unsigned int max_n_line_searches,
492  const unsigned int max_n_refinements,
493  const bool is_initial_step,
494  const bool output_result)
495  {
496  bool first_step = is_initial_step;
497 
498  for (unsigned int refinement_n = 0; refinement_n < max_n_refinements + 1;
499  ++refinement_n)
500  {
501  unsigned int line_search_n = 0;
502  double last_res = 1.0;
503  double current_res = 1.0;
504  std::cout << "grid refinements: " << refinement_n << std::endl
505  << "viscosity: " << viscosity << std::endl;
506 
507  while ((first_step || (current_res > tolerance)) &&
508  line_search_n < max_n_line_searches)
509  {
510  if (first_step)
511  {
512  setup_dofs();
513  initialize_system();
514  evaluation_point = present_solution;
515  assemble_system(first_step);
516  solve(first_step);
517  present_solution = newton_update;
518  nonzero_constraints.distribute(present_solution);
519  first_step = false;
520  evaluation_point = present_solution;
521  assemble_rhs(first_step);
522  current_res = system_rhs.l2_norm();
523  std::cout << "The residual of initial guess is " << current_res
524  << std::endl;
525  last_res = current_res;
526  }
527  else
528  {
529  evaluation_point = present_solution;
530  assemble_system(first_step);
531  solve(first_step);
532 
533  for (double alpha = 1.0; alpha > 1e-5; alpha *= 0.5)
534  {
535  evaluation_point = present_solution;
536  evaluation_point.add(alpha, newton_update);
537  nonzero_constraints.distribute(evaluation_point);
538  assemble_rhs(first_step);
539  current_res = system_rhs.l2_norm();
540  std::cout << " alpha: " << std::setw(10) << alpha
541  << std::setw(0) << " residual: " << current_res
542  << std::endl;
543  if (current_res < last_res)
544  break;
545  }
546  {
547  present_solution = evaluation_point;
548  std::cout << " number of line searches: " << line_search_n
549  << " residual: " << current_res << std::endl;
550  last_res = current_res;
551  }
552  ++line_search_n;
553  }
554 
555  if (output_result)
556  {
557  output_results(max_n_line_searches * refinement_n +
558  line_search_n);
559 
560  if (current_res <= tolerance)
561  process_solution(refinement_n);
562  }
563  }
564 
565  if (refinement_n < max_n_refinements)
566  {
567  refine_mesh();
568  }
569  }
