1408  (parent.block_indices.local_to_global(current_block, 0) +
1409  parent.block_indices.block_size(current_block) - 1);
1410  }
1411  else
1412  // past the end. only have one
1413  // value for this
1414  {
1415  this->global_index = parent.size();
1416  current_block = parent.n_blocks();
1417  index_within_block = 0;

1431  else
1432  {
1433  // ok, we traverse a boundary
1434  // between blocks:
1435  index_within_block = 0;
