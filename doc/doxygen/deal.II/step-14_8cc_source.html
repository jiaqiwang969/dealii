409  {

411  Threads::new_task(&Solver<dim>::assemble_rhs, *this, linear_system.rhs);

602  template <int dim>
603  void PrimalSolver<dim>::assemble_rhs(Vector<double> &rhs) const
604  {
605  FEValues<dim> fe_values(*this->fe,
606  *this->quadrature,
607  update_values | update_quadrature_points |
608  update_JxW_values);
609 
610  const unsigned int dofs_per_cell = this->fe->n_dofs_per_cell();
611  const unsigned int n_q_points = this->quadrature->size();
612 
613  Vector<double> cell_rhs(dofs_per_cell);
614  std::vector<double> rhs_values(n_q_points);
615  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
616 
617  for (const auto &cell : this->dof_handler.active_cell_iterators())
618  {
619  cell_rhs = 0;
620 
621  fe_values.reinit(cell);
622 
623  rhs_function->value_list(fe_values.get_quadrature_points(),
624  rhs_values);
625 
626  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
627  for (unsigned int i = 0; i < dofs_per_cell; ++i)
628  cell_rhs(i) += (fe_values.shape_value(i, q_point) * // phi_i(x_q)
629  rhs_values[q_point] * // f((x_q)
630  fe_values.JxW(q_point)); // dx
631 
632  cell->get_dof_indices(local_dof_indices);
633  for (unsigned int i = 0; i < dofs_per_cell; ++i)
634  rhs(local_dof_indices[i]) += cell_rhs(i);
635  }
