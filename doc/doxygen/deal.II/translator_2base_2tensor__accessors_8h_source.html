440  // OK. This is utterly brutal template magic. Therefore, we will not
441  // comment on the individual internal helper classes, because this is
442  // of not much value, but explain the general recursion procedure.
443  //
444  // (In order of appearance)
445  //
446  // Our task is to reorder access to a tensor object where a specified
447  // index is moved to the end. Thus we want to construct an object
448  // <code>reordered</code> out of a <code>tensor</code> where the
449  // following access patterns are equivalent:
450  // @code
451  // tensor [i_0]...[i_index-1][i_index][i_index+1]...[i_n]
452  // reordered [i_0]...[i_index_1][i_index+1]...[i_n][i_index]
453  // @endcode
454  //
455  // The first task is to get rid of the application of
456  // [i_0]...[i_index-1]. This is a classical recursion pattern - relay
457  // the task from <index, rank> to <index-1, rank-1> by accessing the
458  // subtensor object:
459 
460  template <int index, int rank, typename T>
461  class ReorderedIndexView
462  {
463  public:
464  constexpr ReorderedIndexView(typename ReferenceType<T>::type t)
465  : t_(t)
466  {}
467 
468  using value_type = ReorderedIndexView<index - 1,
469  rank - 1,
470  typename ValueType<T>::value_type>;
471 
472  // Recurse by applying index j directly:
473  constexpr DEAL_II_ALWAYS_INLINE value_type
474  operator[](unsigned int j) const
475  {
476  return value_type(t_[j]);
477  }
478 
479  private:
480  typename ReferenceType<T>::type t_;
481  };
482 
483  // At some point we hit the condition index == 0 and rank > 1, i.e.,
484  // the first index should be reordered to the end.
485  //
486  // At this point we cannot be lazy any more and have to start storing
487  // indices because we get them in the wrong order. The user supplies
488  // [i_0][i_1]...[i_{rank - 1}]
489  // but we have to call the subtensor object with
490  // [i_{rank - 1}[i_0][i_1]...[i_{rank-2}]
491  //
492  // So give up and relay the task to the StoreIndex class:
493 
494  template <int rank, typename T>
495  class ReorderedIndexView<0, rank, T>
496  {
497  public:
498  constexpr ReorderedIndexView(typename ReferenceType<T>::type t)
499  : t_(t)
500  {}
501 
502  using value_type = StoreIndex<rank - 1, internal::Identity<T>>;
503 
504  constexpr DEAL_II_ALWAYS_INLINE value_type
505  operator[](unsigned int j) const
506  {
507  return value_type(Identity<T>(t_), j);
508  }
509 
510  private:
511  typename ReferenceType<T>::type t_;
512  };

669  // Straightforward recursive pattern:
670  //
671  // As long as rank_1 > no_contr, assign indices from the left tensor to
672  // result. This builds up the first part of the nested outer loops:
673  //
674  // for(unsigned int i_0; i_0 < dim; ++i_0)
675  // ...
676  // for(i_; i_ < dim; ++i_)
677  // [...]
678  // result[i_0]..[i_] ... left[i_0]..[i_] ...
679 
680  template <int no_contr, int rank_1, int rank_2, int dim>
681  class Contract
682  {
683  public:
684  template <typename T1, typename T2, typename T3>
685  constexpr inline DEAL_II_ALWAYS_INLINE static void
686  contract(T1 &result, const T2 &left, const T3 &right)
687  {
688  for (unsigned int i = 0; i < dim; ++i)
689  Contract<no_contr, rank_1 - 1, rank_2, dim>::contract(result[i],
690  left[i],
691  right);
692  }
693  };
694 
695  // If rank_1 == no_contr leave out the remaining no_contr indices for
696  // the contraction and assign indices from the right tensor to the
697  // result. This builds up the second part of the nested loops:
698  //
699  // for(unsigned int i_0 = 0; i_0 < dim; ++i_0)
700  // ...
701  // for(unsigned int i_ = 0; i_ < dim; ++i_)
702  // for(unsigned int j_0 = 0; j_0 < dim; ++j_0)
703  // ...
704  // for(unsigned int j_ = 0; j_ < dim; ++j_)
705  // [...]
706  // result[i_0]..[i_][j_0]..[j_] ... left[i_0]..[i_] ...
707  // right[j_0]..[j_]
708  //
709 
710  template <int no_contr, int rank_2, int dim>
711  class Contract<no_contr, no_contr, rank_2, dim>
712  {
713  public:
714  template <typename T1, typename T2, typename T3>
715  constexpr inline DEAL_II_ALWAYS_INLINE static void
716  contract(T1 &result, const T2 &left, const T3 &right)
717  {
718  for (unsigned int i = 0; i < dim; ++i)
719  Contract<no_contr, no_contr, rank_2 - 1, dim>::contract(result[i],
720  left,
721  right[i]);
722  }
723  };
724 
725  // If rank_1 == rank_2 == no_contr we have built up all of the outer
726  // loop. Now, it is time to do the actual contraction:
727  //
728  // [...]
729  // {
730  // result[i_0]..[i_][j_0]..[j_] = 0.;
731  // for(unsigned int k_0 = 0; k_0 < dim; ++k_0)
732  // ...
733  // for(unsigned int k_ = 0; k_ < dim; ++k_)
734  // result[i_0]..[i_][j_0]..[j_] += left[i_0]..[i_][k_0]..[k_] *
735  // right[j_0]..[j_][k_0]..[k_];
736  // }
737  //
738  // Relay this summation to another helper class.
739 
740  template <int no_contr, int dim>
741  class Contract<no_contr, no_contr, no_contr, dim>
742  {
743  public:
744  template <typename T1, typename T2, typename T3>
745  constexpr inline DEAL_II_ALWAYS_INLINE static void
746  contract(T1 &result, const T2 &left, const T3 &right)
747  {
748  result = Contract2<no_contr, dim>::template contract2<T1>(left, right);
749  }
750  };

752  // Straightforward recursion:
753  //
754  // Contract leftmost index and recurse one down.
755 
756  template <int no_contr, int dim>
757  class Contract2
758  {
759  public:
760  template <typename T1, typename T2, typename T3>
761  constexpr inline DEAL_II_ALWAYS_INLINE static T1
762  contract2(const T2 &left, const T3 &right)
763  {
764  // Some auto-differentiable numbers need explicit
765  // zero initialization.
766  if (dim == 0)
767  {
768  T1 result = ::internal::NumberType<T1>::value(0.0);
769  return result;
770  }
771  else
772  {
773  T1 result =
774  Contract2<no_contr - 1, dim>::template contract2<T1>(left[0],
775  right[0]);
776  for (unsigned int i = 1; i < dim; ++i)
777  result +=
778  Contract2<no_contr - 1, dim>::template contract2<T1>(left[i],
779  right[i]);
780  return result;
781  }
782  }
783  };

805  // Fully contract three tensorial objects
806  //
807  // As long as rank_1 > 0, recurse over left and middle:
808  //
809  // for(unsigned int i_0; i_0 < dim; ++i_0)
810  // ...
811  // for(i_; i_ < dim; ++i_)
812  // [...]
813  // left[i_0]..[i_] ... middle[i_0]..[i_] ... right
814 
815  template <int rank_1, int rank_2, int dim>
816  class Contract3
817  {
818  public:
819  template <typename T1, typename T2, typename T3, typename T4>
820  constexpr static inline T1
821  contract3(const T2 &left, const T3 &middle, const T4 &right)
822  {
823  // Some auto-differentiable numbers need explicit
824  // zero initialization.
825  T1 result = ::internal::NumberType<T1>::value(0.0);
826  for (unsigned int i = 0; i < dim; ++i)
827  result += Contract3<rank_1 - 1, rank_2, dim>::template contract3<T1>(
828  left[i], middle[i], right);

834  //
835  // for(unsigned int i_0; i_0 < dim; ++i_0)
836  // ...
837  // for(i_; i_ < dim; ++i_)
838  // for(unsigned int j_0; j_0 < dim; ++j_0)
839  // ...
840  // for(j_; j_ < dim; ++j_)
841  // [...]
