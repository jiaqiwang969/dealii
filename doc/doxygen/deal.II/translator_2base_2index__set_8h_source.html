1782 template <typename ForwardIterator>
1783 inline void
1784 IndexSet::add_indices(const ForwardIterator &begin, const ForwardIterator &end)
1785 {
1786  if (begin == end)
1787  return;
1788 
1789  // identify ranges in the given iterator range by checking whether some
1790  // indices happen to be consecutive. to avoid quadratic complexity when
1791  // calling add_range many times (as add_range() going into the middle of an
1792  // already existing range must shift entries around), we first collect a
1793  // vector of ranges.
1794  std::vector<std::pair<size_type, size_type>> tmp_ranges;
1795  bool ranges_are_sorted = true;
1796  for (ForwardIterator p = begin; p != end;)
1797  {
1798  const size_type begin_index = *p;
1799  size_type end_index = begin_index + 1;
1800  ForwardIterator q = p;
1801  ++q;
1802  while ((q != end) && (*q == end_index))
1803  {
1804  ++end_index;
1805  ++q;
1806  }
1807 
1808  tmp_ranges.emplace_back(begin_index, end_index);
1809  p = q;
1810 
1811  // if the starting index of the next go-around of the for loop is less
1812  // than the end index of the one just identified, then we will have at
1813  // least one pair of ranges that are not sorted, and consequently the
1814  // whole collection of ranges is not sorted.
1815  if (p != end && *p < end_index)
1816  ranges_are_sorted = false;
1817  }
1818 
1819  if (!ranges_are_sorted)
1820  std::sort(tmp_ranges.begin(), tmp_ranges.end());
1821 
1822  // if we have many ranges, we first construct a temporary index set (where
1823  // we add ranges in a consecutive way, so fast), otherwise, we work with
1824  // add_range(). the number 9 is chosen heuristically given the fact that
1825  // there are typically up to 8 independent ranges when adding the degrees of
1826  // freedom on a 3D cell or 9 when adding degrees of freedom of faces. if
1827  // doing cell-by-cell additions, we want to avoid repeated calls to
1828  // IndexSet::compress() which gets called upon merging two index sets, so we
1829  // want to be in the other branch then.
1830  if (tmp_ranges.size() > 9)
1831  {
1832  IndexSet tmp_set(size());
1833  tmp_set.ranges.reserve(tmp_ranges.size());
1834  for (const auto &i : tmp_ranges)
1835  tmp_set.add_range(i.first, i.second);
1836  this->add_indices(tmp_set);
1837  }
1838  else

1840  add_range(i.first, i.second);
