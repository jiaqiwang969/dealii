731  template <typename OutputVector>
732  void
733  SolverBase::solve(const PETScWrappers::MatrixBase &A,
734  std::vector<PetscScalar> & eigenvalues,
735  std::vector<OutputVector> & eigenvectors,
736  const unsigned int n_eigenpairs)
737  {
738  // Panic if the number of eigenpairs wanted is out of bounds.
739  AssertThrow((n_eigenpairs > 0) && (n_eigenpairs <= A.m()),
740  ExcSLEPcWrappersUsageError());
741 
742  // Set the matrices of the problem
743  set_matrices(A);
744 
745  // and solve
746  unsigned int n_converged = 0;
747  solve(n_eigenpairs, &n_converged);
748 
749  if (n_converged > n_eigenpairs)
750  n_converged = n_eigenpairs;
751  AssertThrow(n_converged == n_eigenpairs,
752  ExcSLEPcEigenvectorConvergenceMismatchError(n_converged,
753  n_eigenpairs));
754 
755  AssertThrow(eigenvectors.size() != 0, ExcSLEPcWrappersUsageError());
756  eigenvectors.resize(n_converged, eigenvectors.front());
757  eigenvalues.resize(n_converged);
758 
759  for (unsigned int index = 0; index < n_converged; ++index)
760  get_eigenpair(index, eigenvalues[index], eigenvectors[index]);
761  }

802  void
803  SolverBase::solve(const PETScWrappers::MatrixBase &A,
804  const PETScWrappers::MatrixBase &B,
805  std::vector<double> & real_eigenvalues,
806  std::vector<double> & imag_eigenvalues,
807  std::vector<OutputVector> & real_eigenvectors,
808  std::vector<OutputVector> & imag_eigenvectors,
809  const unsigned int n_eigenpairs)
810  {
811  // Guard against incompatible matrix sizes:
812  AssertThrow(A.m() == B.m(), ExcDimensionMismatch(A.m(), B.m()));
813  AssertThrow(A.n() == B.n(), ExcDimensionMismatch(A.n(), B.n()));
814 
815  // and incompatible eigenvalue/eigenvector sizes
816  AssertThrow(real_eigenvalues.size() == imag_eigenvalues.size(),
817  ExcDimensionMismatch(real_eigenvalues.size(),
818  imag_eigenvalues.size()));
819  AssertThrow(real_eigenvectors.size() == imag_eigenvectors.size(),
820  ExcDimensionMismatch(real_eigenvectors.size(),
821  imag_eigenvectors.size()));
822 
823  // Panic if the number of eigenpairs wanted is out of bounds.
824  AssertThrow((n_eigenpairs > 0) && (n_eigenpairs <= A.m()),
825  ExcSLEPcWrappersUsageError());
826 
827  // Set the matrices of the problem
828  set_matrices(A, B);
829 
830  // and solve
831  unsigned int n_converged = 0;
832  solve(n_eigenpairs, &n_converged);
833 
834  if (n_converged >= n_eigenpairs)
835  n_converged = n_eigenpairs;
836 
837  AssertThrow(n_converged == n_eigenpairs,
838  ExcSLEPcEigenvectorConvergenceMismatchError(n_converged,
839  n_eigenpairs));
840  AssertThrow((real_eigenvectors.size() != 0) &&
841  (imag_eigenvectors.size() != 0),
842  ExcSLEPcWrappersUsageError());
843 
844  real_eigenvectors.resize(n_converged, real_eigenvectors.front());
845  imag_eigenvectors.resize(n_converged, imag_eigenvectors.front());
846  real_eigenvalues.resize(n_converged);
847  imag_eigenvalues.resize(n_converged);
848 
849  for (unsigned int index = 0; index < n_converged; ++index)
850  get_eigenpair(index,
851  real_eigenvalues[index],
852  imag_eigenvalues[index],
853  real_eigenvectors[index],
854  imag_eigenvectors[index]);
855  }

857  template <typename Vector>
858  void
859  SolverBase::set_initial_space(const std::vector<Vector> &this_initial_space)
860  {
861  std::vector<Vec> vecs(this_initial_space.size());
862 
863  for (unsigned int i = 0; i < this_initial_space.size(); i++)
864  {
865  Assert(this_initial_space[i].l2_norm() > 0.0,
866  ExcMessage("Initial vectors should be nonzero."));
867  vecs[i] = this_initial_space[i];
868  }
869 
870  // if the eigensolver supports only a single initial vector, but several
871  // guesses are provided, then all except the first one will be discarded.
872  // One could still build a vector that is rich in the directions of all
873  // guesses, by taking a linear combination of them. (TODO: make function
874  // virtual?)
875 
876  const PetscErrorCode ierr =
877  EPSSetInitialSpace(eps, vecs.size(), vecs.data());
878  AssertThrow(ierr == 0, ExcSLEPcError(ierr));
879  }
