438  mg_solve(SolverControl & solver_control,
439  VectorType & dst,
440  const VectorType & src,
441  const MultigridParameters &mg_data,
442  const DoFHandler<dim> & dof,
443  const SystemMatrixType & fine_matrix,
444  const MGLevelObject<std::unique_ptr<LevelMatrixType>> &mg_matrices,
445  const MGTransferType & mg_transfer)
446  {
447  AssertThrow(mg_data.coarse_solver.type == "cg_with_amg",

449  AssertThrow(mg_data.smoother.type == "chebyshev", ExcNotImplemented());

451  const unsigned int min_level = mg_matrices.min_level();
452  const unsigned int max_level = mg_matrices.max_level();

454  using SmootherPreconditionerType = DiagonalMatrix<VectorType>;
455  using SmootherType = PreconditionChebyshev<LevelMatrixType,

461 
462  MGLevelObject<typename SmootherType::AdditionalData> smoother_data(
463  min_level, max_level);
464 
465  for (unsigned int level = min_level; level <= max_level; level++)
466  {
467  smoother_data[level].preconditioner =
468  std::make_shared<SmootherPreconditionerType>();
469  mg_matrices[level]->compute_inverse_diagonal(
470  smoother_data[level].preconditioner->get_vector());
471  smoother_data[level].smoothing_range = mg_data.smoother.smoothing_range;
472  smoother_data[level].degree = mg_data.smoother.degree;

481  ReductionControl coarse_grid_solver_control(mg_data.coarse_solver.maxiter,
482  mg_data.coarse_solver.abstol,
483  mg_data.coarse_solver.reltol,
484  false,
485  false);

488  std::unique_ptr<MGCoarseGridBase<VectorType>> mg_coarse;

493  amg_data.n_cycles = mg_data.coarse_solver.n_cycles;
494  amg_data.smoother_type = mg_data.coarse_solver.smoother_type.c_str();

518  template <typename VectorType, typename OperatorType, int dim>
519  void solve_with_gmg(SolverControl & solver_control,
520  const OperatorType & system_matrix,
521  VectorType & dst,
522  const VectorType & src,
523  const MultigridParameters & mg_data,
524  const hp::MappingCollection<dim> mapping_collection,
525  const DoFHandler<dim> & dof_handler,
526  const hp::QCollection<dim> & quadrature_collection)
527  {
528  MGLevelObject<DoFHandler<dim>> dof_handlers;
529  MGLevelObject<std::unique_ptr<OperatorType>> operators;
530  MGLevelObject<MGTwoLevelTransfer<dim, VectorType>> transfers;
531 
532  std::vector<std::shared_ptr<const Triangulation<dim>>>
533  coarse_grid_triangulations;
534  if (mg_data.transfer.perform_h_transfer)
535  coarse_grid_triangulations =
536  MGTransferGlobalCoarseningTools::create_geometric_coarsening_sequence(
537  dof_handler.get_triangulation());
538  else
539  coarse_grid_triangulations.emplace_back(
540  const_cast<Triangulation<dim> *>(&(dof_handler.get_triangulation())),
541  [](auto &) {});
542 
543  const unsigned int n_h_levels = coarse_grid_triangulations.size() - 1;
544 
545  const auto get_max_active_fe_degree = [&](const auto &dof_handler) {
546  unsigned int max = 0;
547 
548  for (auto &cell : dof_handler.active_cell_iterators())
549  if (cell->is_locally_owned())
550  max =
551  std::max(max, dof_handler.get_fe(cell->active_fe_index()).degree);
552 
553  return Utilities::MPI::max(max, MPI_COMM_WORLD);
554  };
555 
556  const unsigned int n_p_levels =
557  MGTransferGlobalCoarseningTools::create_polynomial_coarsening_sequence(
558  get_max_active_fe_degree(dof_handler), mg_data.transfer.p_sequence)
559  .size();
560 
561  std::map<unsigned int, unsigned int> fe_index_for_degree;
562  for (unsigned int i = 0; i < dof_handler.get_fe_collection().size(); ++i)
563  {
564  const unsigned int degree = dof_handler.get_fe(i).degree;
565  Assert(fe_index_for_degree.find(degree) == fe_index_for_degree.end(),
566  ExcMessage("FECollection does not contain unique degrees."));
567  fe_index_for_degree[degree] = i;
568  }
569 
570  unsigned int minlevel = 0;
571  unsigned int minlevel_p = n_h_levels;
572  unsigned int maxlevel = n_h_levels + n_p_levels - 1;
573 
574  dof_handlers.resize(minlevel, maxlevel);
575  operators.resize(minlevel, maxlevel);
576  transfers.resize(minlevel, maxlevel);
577 
578  for (unsigned int l = 0; l < n_h_levels; ++l)
579  {
580  dof_handlers[l].reinit(*coarse_grid_triangulations[l]);
581  dof_handlers[l].distribute_dofs(dof_handler.get_fe_collection());
582  }
583 
584  for (unsigned int i = 0, l = maxlevel; i < n_p_levels; ++i, --l)
585  {
586  dof_handlers[l].reinit(dof_handler.get_triangulation());
587 
588  if (l == maxlevel) // finest level
589  {
590  auto &dof_handler_mg = dof_handlers[l];
591 
592  auto cell_other = dof_handler.begin_active();
593  for (auto &cell : dof_handler_mg.active_cell_iterators())
594  {
595  if (cell->is_locally_owned())
596  cell->set_active_fe_index(cell_other->active_fe_index());
597  cell_other++;
598  }
599  }
600  else // coarse level
601  {
602  auto &dof_handler_fine = dof_handlers[l + 1];
603  auto &dof_handler_coarse = dof_handlers[l + 0];
604 
605  auto cell_other = dof_handler_fine.begin_active();
606  for (auto &cell : dof_handler_coarse.active_cell_iterators())
607  {
608  if (cell->is_locally_owned())
609  {
610  const unsigned int next_degree =
611  MGTransferGlobalCoarseningTools::
612  create_next_polynomial_coarsening_degree(
613  cell_other->get_fe().degree,
614  mg_data.transfer.p_sequence);
615  Assert(fe_index_for_degree.find(next_degree) !=
616  fe_index_for_degree.end(),
617  ExcMessage("Next polynomial degree in sequence "
618  "does not exist in FECollection."));
619 
620  cell->set_active_fe_index(fe_index_for_degree[next_degree]);
621  }
622  cell_other++;
623  }
624  }
625 
626  dof_handlers[l].distribute_dofs(dof_handler.get_fe_collection());
627  }
628 
629  MGLevelObject<AffineConstraints<typename VectorType::value_type>>
630  constraints(minlevel, maxlevel);
631 
632  for (unsigned int level = minlevel; level <= maxlevel; ++level)
633  {
634  const auto &dof_handler = dof_handlers[level];
635  auto & constraint = constraints[level];
636 
637  IndexSet locally_relevant_dofs;
638  DoFTools::extract_locally_relevant_dofs(dof_handler,
639  locally_relevant_dofs);
640  constraint.reinit(locally_relevant_dofs);
641 
642  DoFTools::make_hanging_node_constraints(dof_handler, constraint);
643  VectorTools::interpolate_boundary_values(mapping_collection,
644  dof_handler,
645  0,
646  Functions::ZeroFunction<dim>(),
647  constraint);
648  constraint.close();
649 
650  VectorType dummy;
651 
652  operators[level] = std::make_unique<OperatorType>(mapping_collection,
653  dof_handler,
654  quadrature_collection,
655  constraint,
656  dummy);
657  }
658 
659  for (unsigned int level = minlevel; level < minlevel_p; ++level)
660  transfers[level + 1].reinit_geometric_transfer(dof_handlers[level + 1],
661  dof_handlers[level],
662  constraints[level + 1],
663  constraints[level]);
664 
665  for (unsigned int level = minlevel_p; level < maxlevel; ++level)
666  transfers[level + 1].reinit_polynomial_transfer(dof_handlers[level + 1],
667  dof_handlers[level],
668  constraints[level + 1],
669  constraints[level]);
670 
671  MGTransferGlobalCoarsening<dim, VectorType> transfer(
672  transfers, [&](const auto l, auto &vec) {
673  operators[l]->initialize_dof_vector(vec);
674  });
675 
676  mg_solve(solver_control,
677  dst,
678  src,
679  mg_data,
680  dof_handler,
681  system_matrix,
682  operators,
683  transfer);
684  }
