302 inline number
303 TridiagonalMatrix<number>::operator()(size_type i, size_type j) const
304 {
305  AssertIndexRange(i, n());
306  AssertIndexRange(j, n());
307  Assert(i <= j + 1, ExcIndexRange(i, j - 1, j + 2));
308  Assert(j <= i + 1, ExcIndexRange(j, i - 1, i + 2));
309 
310  if (j == i)
311  return diagonal[i];
312  if (j == i - 1)
313  {
314  if (is_symmetric)
315  return right[i - 1];
316  else
317  return left[i];
318  }
319 
320  if (j == i + 1)
321  return right[i];
322 
323  Assert(false, ExcInternalError());
324  return 0;
325 }
