424  static std::string ic_type_str = "use_y_diff";
425  prm.add_parameter(
426  "Correction type at initial time",
427  ic_type_str,
428  "This is one of the following three options for the "
429  "initial condition calculation. \n"
430  " none: do not try to make initial conditions consistent. \n"
431  " use_y_diff: compute the algebraic components of y and differential\n"
432  " components of y_dot, given the differential components of y. \n"
433  " This option requires that the user specifies differential and \n"
434  " algebraic components in the function get_differential_components.\n"
435  " use_y_dot: compute all components of y, given y_dot.",
436  Patterns::Selection("none|use_y_diff|use_y_dot"));
437  prm.add_action("Correction type at initial time",
438  [&](const std::string &value) {
439  if (value == "use_y_diff")
440  ic_type = use_y_diff;
441  else if (value == "use_y_dot")
442  ic_type = use_y_dot;
443  else if (value == "none")
444  ic_type = none;
445  else
446  AssertThrow(false, ExcInternalError());
447  });

454  "initial condition calculation. \n"
