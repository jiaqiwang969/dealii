1160 FullMatrix<number>::copy_from(const MatrixType &M)
1161 {
1162  this->reinit(M.m(), M.n());
1163 
1164  // loop over the elements of the argument matrix row by row, as suggested
1165  // in the documentation of the sparse matrix iterator class, and
1166  // copy them into the current object
1167  for (size_type row = 0; row < M.m(); ++row)
1168  {
1169  const typename MatrixType::const_iterator end_row = M.end(row);
1170  for (typename MatrixType::const_iterator entry = M.begin(row);
1171  entry != end_row;
1172  ++entry)
1173  this->el(row, entry->column()) = entry->value();
1174  }
1175 }
