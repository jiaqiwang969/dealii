2098  // For the implementation of the <code>vmult</code> function with deal.II
2099  // data structures we note that invoking a call of the Trilinos
2100  // preconditioner requires us to use Epetra vectors as well. We do this by
2101  // providing a view, i.e., feed Trilinos with a pointer to the data, so we
2102  // avoid copying the content of the vectors during the iteration (this
2103  // function is only useful when used in serial anyway). In the declaration
2104  // of the right hand side, we need to cast the source vector (that is
2105  // <code>const</code> in all deal.II calls) to non-constant value, as this
2106  // is the way Trilinos wants to have them.
2107  inline void
2108  PreconditionBase::vmult(::Vector<double> & dst,
2109  const ::Vector<double> &src) const
2110  {
2111  AssertDimension(dst.size(),
2112  preconditioner->OperatorDomainMap().NumMyElements());
2113  AssertDimension(src.size(),
2114  preconditioner->OperatorRangeMap().NumMyElements());
2115  Epetra_Vector tril_dst(View,
2116  preconditioner->OperatorDomainMap(),
2117  dst.begin());
2118  Epetra_Vector tril_src(View,
2119  preconditioner->OperatorRangeMap(),
2120  const_cast<double *>(src.begin()));
2121 
2122  const int ierr = preconditioner->ApplyInverse(tril_src, tril_dst);
2123  AssertThrow(ierr == 0, ExcTrilinosError(ierr));
2124  }
