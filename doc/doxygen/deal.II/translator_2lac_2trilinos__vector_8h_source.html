1484  vector->Map().LID(row);

1531  inline void
1532  Vector::add(const size_type n_elements,
1533  const size_type * indices,
1534  const TrilinosScalar *values)
1535  {
1536  // if we have ghost values, do not allow
1537  // writing to this vector at all.
1538  Assert(!has_ghost_elements(), ExcGhostsPresent());
1539 
1540  if (last_action != Add)
1541  {
1542  if (last_action == Insert)
1543  {
1544  const int ierr = vector->GlobalAssemble(Insert);
1545  AssertThrow(ierr == 0, ExcTrilinosError(ierr));
1546  }
1547  last_action = Add;
1548  }
1549 
1550  for (size_type i = 0; i < n_elements; ++i)
1551  {
1552  const size_type row = indices[i];
1553  const TrilinosWrappers::types::int_type local_row = vector->Map().LID(
1554  static_cast<TrilinosWrappers::types::int_type>(row));
1555  if (local_row != -1)
1556  (*vector)[0][local_row] += values[i];
1557  else if (nonlocal_vector.get() == nullptr)
1558  {
1559  const int ierr = vector->SumIntoGlobalValues(
1560  1,
1561  reinterpret_cast<const TrilinosWrappers::types::int_type *>(
1562  &row),
1563  &values[i]);
1564  AssertThrow(ierr == 0, ExcTrilinosError(ierr));
1565  compressed = false;
1566  }
1567  else
1568  {
1569  // use pre-allocated vector for non-local entries if it exists for
1570  // addition operation
1571  const TrilinosWrappers::types::int_type my_row =
1572  nonlocal_vector->Map().LID(
