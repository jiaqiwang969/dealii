435  class Material_Compressible_Neo_Hook_Three_Field
436  {
437  public:
438  Material_Compressible_Neo_Hook_Three_Field(const double mu, const double nu)
439  : kappa((2.0 * mu * (1.0 + nu)) / (3.0 * (1.0 - 2.0 * nu)))
440  , c_1(mu / 2.0)
441  , det_F(1.0)
442  , p_tilde(0.0)
443  , J_tilde(1.0)
444  , b_bar(Physics::Elasticity::StandardTensors<dim>::I)
445  {
446  Assert(kappa > 0, ExcInternalError());
447  }
448 
449  void update_material_data(const Tensor<2, dim> &F,
450  const double p_tilde_in,
451  const double J_tilde_in)
452  {
453  det_F = determinant(F);
454  const Tensor<2, dim> F_bar = Physics::Elasticity::Kinematics::F_iso(F);
455  b_bar = Physics::Elasticity::Kinematics::b(F_bar);
456  p_tilde = p_tilde_in;
457  J_tilde = J_tilde_in;
458 
459  Assert(det_F > 0, ExcInternalError());
460  }
461 
462  SymmetricTensor<2, dim> get_tau()
463  {
464  return get_tau_iso() + get_tau_vol();
465  }
466 
467  SymmetricTensor<4, dim> get_Jc() const
468  {
469  return get_Jc_vol() + get_Jc_iso();
470  }
471 
472  double get_dPsi_vol_dJ() const
473  {
474  return (kappa / 2.0) * (J_tilde - 1.0 / J_tilde);
475  }
476 
477  double get_d2Psi_vol_dJ2() const
478  {
479  return ((kappa / 2.0) * (1.0 + 1.0 / (J_tilde * J_tilde)));
480  }
481 
482  double get_det_F() const
483  {
484  return det_F;
485  }
486 
487  double get_p_tilde() const
488  {
489  return p_tilde;
490  }
491 
492  double get_J_tilde() const
493  {
494  return J_tilde;
495  }
496 
497  protected:
498  const double kappa;
499  const double c_1;
500 
501  double det_F;
502  double p_tilde;
503  double J_tilde;
504  SymmetricTensor<2, dim> b_bar;
505 
506  SymmetricTensor<2, dim> get_tau_vol() const
507  {
508  return p_tilde * det_F * Physics::Elasticity::StandardTensors<dim>::I;
509  }
510 
511  SymmetricTensor<2, dim> get_tau_iso() const
512  {
513  return Physics::Elasticity::StandardTensors<dim>::dev_P * get_tau_bar();
514  }
515 
516  SymmetricTensor<2, dim> get_tau_bar() const
517  {
518  return 2.0 * c_1 * b_bar;
519  }
520 
521  SymmetricTensor<4, dim> get_Jc_vol() const
522  {
523  return p_tilde * det_F *
524  (Physics::Elasticity::StandardTensors<dim>::IxI -
525  (2.0 * Physics::Elasticity::StandardTensors<dim>::S));
526  }
527 
528  SymmetricTensor<4, dim> get_Jc_iso() const
529  {
530  const SymmetricTensor<2, dim> tau_bar = get_tau_bar();
531  const SymmetricTensor<2, dim> tau_iso = get_tau_iso();
532  const SymmetricTensor<4, dim> tau_iso_x_I =
533  outer_product(tau_iso, Physics::Elasticity::StandardTensors<dim>::I);
534  const SymmetricTensor<4, dim> I_x_tau_iso =
535  outer_product(Physics::Elasticity::StandardTensors<dim>::I, tau_iso);
536  const SymmetricTensor<4, dim> c_bar = get_c_bar();
537 
538  return (2.0 / dim) * trace(tau_bar) *
539  Physics::Elasticity::StandardTensors<dim>::dev_P -
540  (2.0 / dim) * (tau_iso_x_I + I_x_tau_iso) +
541  Physics::Elasticity::StandardTensors<dim>::dev_P * c_bar *
542  Physics::Elasticity::StandardTensors<dim>::dev_P;
543  }
544 
545  SymmetricTensor<4, dim> get_c_bar() const
546  {
547  return SymmetricTensor<4, dim>();
548  }
549  };

1529  PerTaskData_ASM & data) const

1537  quadrature_point_history.get_data(cell);
1538  Assert(lqph.size() == n_q_points, ExcInternalError());
1539 
1540  for (const unsigned int q_point :
1541  scratch.fe_values.quadrature_point_indices())
1542  {
1543  const Tensor<2, dim> F_inv = lqph[q_point]->get_F_inv();
1544  for (const unsigned int k : scratch.fe_values.dof_indices())
1545  {
1546  const unsigned int k_group = fe.system_to_base_index(k).first.first;
1547 
1548  if (k_group == u_dof)
1549  {
1550  scratch.grad_Nx[q_point][k] =
1551  scratch.fe_values[u_fe].gradient(k, q_point) * F_inv;
1552  scratch.symm_grad_Nx[q_point][k] =
1553  symmetrize(scratch.grad_Nx[q_point][k]);
1554  }
1555  else if (k_group == p_dof)
1556  scratch.Nx[q_point][k] =
1557  scratch.fe_values[p_fe].value(k, q_point);
1558  else if (k_group == J_dof)
1559  scratch.Nx[q_point][k] =
1560  scratch.fe_values[J_fe].value(k, q_point);
1561  else
1562  Assert(k_group <= J_dof, ExcInternalError());

1564  }
1565 
1566  for (const unsigned int q_point :
1567  scratch.fe_values.quadrature_point_indices())
1568  {
1569  const SymmetricTensor<2, dim> tau = lqph[q_point]->get_tau();
1570  const Tensor<2, dim> tau_ns = lqph[q_point]->get_tau();
1571  const SymmetricTensor<4, dim> Jc = lqph[q_point]->get_Jc();
1572  const double det_F = lqph[q_point]->get_det_F();
1573  const double p_tilde = lqph[q_point]->get_p_tilde();
1574  const double J_tilde = lqph[q_point]->get_J_tilde();
1575  const double dPsi_vol_dJ = lqph[q_point]->get_dPsi_vol_dJ();
1576  const double d2Psi_vol_dJ2 = lqph[q_point]->get_d2Psi_vol_dJ2();

1578  Physics::Elasticity::StandardTensors<dim>::I;
1579 
1580  SymmetricTensor<2, dim> symm_grad_Nx_i_x_Jc;
1581  Tensor<1, dim> grad_Nx_i_comp_i_x_tau;
1582 
1583  const std::vector<double> & N = scratch.Nx[q_point];
1584  const std::vector<SymmetricTensor<2, dim>> &symm_grad_Nx =
1585  scratch.symm_grad_Nx[q_point];
1586  const std::vector<Tensor<2, dim>> &grad_Nx = scratch.grad_Nx[q_point];
1587  const double JxW = scratch.fe_values.JxW(q_point);
1588 
1589  for (const unsigned int i : scratch.fe_values.dof_indices())
1590  {
1591  const unsigned int component_i =
1592  fe.system_to_component_index(i).first;
1593  const unsigned int i_group = fe.system_to_base_index(i).first.first;
1594 
1595  if (i_group == u_dof)
1596  data.cell_rhs(i) -= (symm_grad_Nx[i] * tau) * JxW;
1597  else if (i_group == p_dof)
1598  data.cell_rhs(i) -= N[i] * (det_F - J_tilde) * JxW;
1599  else if (i_group == J_dof)
1600  data.cell_rhs(i) -= N[i] * (dPsi_vol_dJ - p_tilde) * JxW;
1601  else
1602  Assert(i_group <= J_dof, ExcInternalError());
1603 
1604  if (i_group == u_dof)
1605  {
1606  symm_grad_Nx_i_x_Jc = symm_grad_Nx[i] * Jc;
1607  grad_Nx_i_comp_i_x_tau = grad_Nx[i][component_i] * tau_ns;
1608  }
1609 
1610  for (const unsigned int j :
1611  scratch.fe_values.dof_indices_ending_at(i))
1612  {
1613  const unsigned int component_j =
1614  fe.system_to_component_index(j).first;
1615  const unsigned int j_group =
1616  fe.system_to_base_index(j).first.first;
1617 
1618  if ((i_group == j_group) && (i_group == u_dof))
1619  {
1620  data.cell_matrix(i, j) += symm_grad_Nx_i_x_Jc *
1621  symm_grad_Nx[j] * JxW;
1622 
1623  if (component_i == component_j)
1624  data.cell_matrix(i, j) +=
1625  grad_Nx_i_comp_i_x_tau * grad_Nx[j][component_j] * JxW;
1626  }
1627  else if ((i_group == p_dof) && (j_group == u_dof))
1628  {
1629  data.cell_matrix(i, j) += N[i] * det_F *
1630  (symm_grad_Nx[j] * I) * JxW;
1631  }
1632  else if ((i_group == J_dof) && (j_group == p_dof))
1633  data.cell_matrix(i, j) -= N[i] * N[j] * JxW;
1634  else if ((i_group == j_group) && (i_group == J_dof))
1635  data.cell_matrix(i, j) += N[i] * d2Psi_vol_dJ2 * N[j] * JxW;
1636  else
1637  Assert((i_group <= J_dof) && (j_group <= J_dof),
1638  ExcInternalError());
1639  }
1640  }
1641  }
1642 
1643  for (const auto &face : cell->face_iterators())
1644  if (face->at_boundary() && face->boundary_id() == 6)
1645  {
1646  scratch.fe_face_values.reinit(cell, face);
1647 
1648  for (const unsigned int f_q_point :
1649  scratch.fe_face_values.quadrature_point_indices())
1650  {
1651  const Tensor<1, dim> &N =
1652  scratch.fe_face_values.normal_vector(f_q_point);
1653 
1654  static const double p0 =
1655  -4.0 / (parameters.scale * parameters.scale);
1656  const double time_ramp = (time.current() / time.end());
1657  const double pressure = p0 * parameters.p_p0 * time_ramp;
1658  const Tensor<1, dim> traction = pressure * N;
1659 
1660  for (const unsigned int i : scratch.fe_values.dof_indices())
1661  {
1662  const unsigned int i_group =
1663  fe.system_to_base_index(i).first.first;
1664 
1665  if (i_group == u_dof)
1666  {
1667  const unsigned int component_i =
1668  fe.system_to_component_index(i).first;
1669  const double Ni =
1670  scratch.fe_face_values.shape_value(i, f_q_point);
1671  const double JxW = scratch.fe_face_values.JxW(f_q_point);
1672 
1673  data.cell_rhs(i) += (Ni * traction[component_i]) * JxW;
1674  }
1675  }
1676  }
1677  }
1678 
1679  for (const unsigned int i : scratch.fe_values.dof_indices())
1680  for (const unsigned int j :
1681  scratch.fe_values.dof_indices_starting_at(i + 1))
1682  data.cell_matrix(i, j) = data.cell_matrix(j, i);
