399 template <typename Range>
400 PackagedOperation<Range> operator*(const PackagedOperation<Range> &comp,
401  typename Range::value_type number)
402 {
403  PackagedOperation<Range> return_comp;
404 
405  return_comp.reinit_vector = comp.reinit_vector;
406 
407  // the trivial case: number is zero
408  if (number == 0.)
409  {
410  return_comp.apply = [](Range &v) { v = 0.; };
411 
412  return_comp.apply_add = [](Range &) {};
413  }
414  else
415  {
416  return_comp.apply = [comp, number](Range &v) {
417  comp.apply(v);
418  v *= number;
419  };
420 
421  return_comp.apply_add = [comp, number](Range &v) {
422  v /= number;
423  comp.apply_add(v);
424  v *= number;
425  };
426  }
427 
428  return return_comp;
429 }
