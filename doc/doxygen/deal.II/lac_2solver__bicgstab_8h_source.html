379 template <typename VectorType>
380 template <typename MatrixType, typename PreconditionerType>
381 typename SolverBicgstab<VectorType>::IterationResult
382 SolverBicgstab<VectorType>::iterate(const MatrixType & A,
383  const PreconditionerType &preconditioner)
384 {
385  A.vmult(*Vr, *Vx);
386  Vr->sadd(-1., 1., *Vb);
387  res = Vr->l2_norm();
388 
389  SolverControl::State state = this->iteration_status(step, res, *Vx);
390  if (state == SolverControl::State::success)
391  return IterationResult(false, state, step, res);
392 
393  alpha = omega = rho = 1.;
394 
395  VectorType &r = *Vr;
396  VectorType &rbar = *Vrbar;
397  VectorType &p = *Vp;
398  VectorType &y = *Vy;
399  VectorType &z = *Vz;
400  VectorType &t = *Vt;
401  VectorType &v = *Vv;
402 
403  rbar = r;
404  bool startup = true;
405 
406  do
407  {
408  ++step;
409 
410  rhobar = r * rbar;
411  if (std::fabs(rhobar) < additional_data.breakdown)
412  {
413  return IterationResult(true, state, step, res);
414  }
415  beta = rhobar * alpha / (rho * omega);
416  rho = rhobar;
417  if (startup == true)
418  {
419  p = r;
420  startup = false;
421  }
422  else
423  {
424  p.sadd(beta, 1., r);
425  p.add(-beta * omega, v);
426  }
427 
428  preconditioner.vmult(y, p);
429  A.vmult(v, y);
430  rhobar = rbar * v;
431  if (std::fabs(rhobar) < additional_data.breakdown)
432  {
433  return IterationResult(true, state, step, res);
434  }
435 
436  alpha = rho / rhobar;
437 
438  res = std::sqrt(r.add_and_dot(-alpha, v, r));
439 
440  // check for early success, see the lac/bicgstab_early testcase as to
441  // why this is necessary
442  //
443  // note: the vector *Vx we pass to the iteration_status signal here is
444  // only the current approximation, not the one we will return with, which
445  // will be x=*Vx + alpha*y
446  if (this->iteration_status(step, res, *Vx) == SolverControl::success)
447  {
448  Vx->add(alpha, y);
449  print_vectors(step, *Vx, r, y);
450  return IterationResult(false, SolverControl::success, step, res);
451  }
452 
453  preconditioner.vmult(z, r);
454  A.vmult(t, z);
455  rhobar = t * r;
456  auto t_squared = t * t;
457  if (t_squared < additional_data.breakdown)
458  {
459  return IterationResult(true, state, step, res);
460  }
461  omega = rhobar / (t * t);
462  Vx->add(alpha, y, omega, z);
463 
464  if (additional_data.exact_residual)
465  {
466  r.add(-omega, t);
467  res = criterion(A, *Vx, *Vb);
468  }
469  else
470  res = std::sqrt(r.add_and_dot(-omega, t, r));
471 
472  state = this->iteration_status(step, res, *Vx);
473  print_vectors(step, *Vx, r, y);
474  }
475  while (state == SolverControl::iterate);
476 
477  return IterationResult(false, state, step, res);
478 }
