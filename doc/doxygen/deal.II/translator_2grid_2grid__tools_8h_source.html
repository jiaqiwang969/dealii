3472  Point<spacedim>
3473  project_to_d_linear_object(const Iterator & object,
3474  const Point<spacedim> &trial_point)
3475  {
3476  // let's look at this for simplicity for a quad (structdim==2) in a
3477  // space with spacedim>2 (notate trial_point by y): all points on the
3478  // surface are given by
3479  // x(\xi) = sum_i v_i phi_x(\xi)
3480  // where v_i are the vertices of the quad, and \xi=(\xi_1,\xi_2) are the
3481  // reference coordinates of the quad. so what we are trying to do is
3482  // find a point x on the surface that is closest to the point y. there
3483  // are different ways to solve this problem, but in the end it's a
3484  // nonlinear problem and we have to find reference coordinates \xi so
3485  // that J(\xi) = 1/2 || x(\xi)-y ||^2 is minimal. x(\xi) is a function
3486  // that is structdim-linear in \xi, so J(\xi) is a polynomial of degree
3487  // 2*structdim that we'd like to minimize. unless structdim==1, we'll
3488  // have to use a Newton method to find the answer. This leads to the
3489  // following formulation of Newton steps:
3490  //
3491  // Given \xi_k, find \delta\xi_k so that
3492  // H_k \delta\xi_k = - F_k
3493  // where H_k is an approximation to the second derivatives of J at
3494  // \xi_k, and F_k is the first derivative of J. We'll iterate this a
3495  // number of times until the right hand side is small enough. As a
3496  // stopping criterion, we terminate if ||\delta\xi||<eps.
3497  //
3498  // As for the Hessian, the best choice would be
3499  // H_k = J''(\xi_k)
3500  // but we'll opt for the simpler Gauss-Newton form
3501  // H_k = A^T A
3502  // i.e.
3503  // (H_k)_{nm} = \sum_{i,j} v_i*v_j *
3504  // \partial_n phi_i *
3505  // \partial_m phi_j
3506  // we start at xi=(0.5, 0.5).
3507  Point<structdim> xi;
3508  for (unsigned int d = 0; d < structdim; ++d)
3509  xi[d] = 0.5;
3510 
3511  Point<spacedim> x_k;
3512  for (const unsigned int i : GeometryInfo<structdim>::vertex_indices())
3513  x_k += object->vertex(i) *
3514  GeometryInfo<structdim>::d_linear_shape_function(xi, i);
3515 
3516  do
3517  {
3518  Tensor<1, structdim> F_k;
3519  for (const unsigned int i :
3520  GeometryInfo<structdim>::vertex_indices())
3521  F_k +=
3522  (x_k - trial_point) * object->vertex(i) *
3523  GeometryInfo<structdim>::d_linear_shape_function_gradient(xi,
3524  i);
3525 
3526  Tensor<2, structdim> H_k;
3527  for (const unsigned int i :
3528  GeometryInfo<structdim>::vertex_indices())
3529  for (const unsigned int j :
3530  GeometryInfo<structdim>::vertex_indices())
3531  {

3533  GeometryInfo<structdim>::d_linear_shape_function_gradient(
3534  xi, i),
3535  GeometryInfo<structdim>::d_linear_shape_function_gradient(
3536  xi, j));
3537  H_k += (object->vertex(i) * object->vertex(j)) * tmp;
3538  }

3541  xi += delta_xi;

3637  //

3644  // the other weights and get, for one coordinate direction
3645  //
