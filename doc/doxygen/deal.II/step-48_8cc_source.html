82  template <int dim, int fe_degree>
83  SineGordonOperation<dim, fe_degree>::SineGordonOperation(
84  const MatrixFree<dim, double> &data_in,
85  const double time_step)
86  : data(data_in)
87  , delta_t_sqr(make_vectorized_array(time_step * time_step))
88  {
89  data.initialize_dof_vector(inv_mass_matrix);
90 
91  FEEvaluation<dim, fe_degree> fe_eval(data);
92  const unsigned int n_q_points = fe_eval.n_q_points;
93 
94  for (unsigned int cell = 0; cell < data.n_cell_batches(); ++cell)
95  {
96  fe_eval.reinit(cell);
97  for (unsigned int q = 0; q < n_q_points; ++q)
98  fe_eval.submit_value(make_vectorized_array(1.), q);
99  fe_eval.integrate(EvaluationFlags::values);
100  fe_eval.distribute_local_to_global(inv_mass_matrix);
101  }
102 
103  inv_mass_matrix.compress(VectorOperation::add);
104  for (unsigned int k = 0; k < inv_mass_matrix.locally_owned_size(); ++k)
105  if (inv_mass_matrix.local_element(k) > 1e-15)
106  inv_mass_matrix.local_element(k) =
107  1. / inv_mass_matrix.local_element(k);
108  else
109  inv_mass_matrix.local_element(k) = 1;
110  }

116  template <int dim, int fe_degree>
117  void SineGordonOperation<dim, fe_degree>::local_apply(
118  const MatrixFree<dim> & data,
119  LinearAlgebra::distributed::Vector<double> & dst,
120  const std::vector<LinearAlgebra::distributed::Vector<double> *> &src,
121  const std::pair<unsigned int, unsigned int> &cell_range) const
122  {
123  AssertDimension(src.size(), 2);
124  FEEvaluation<dim, fe_degree> current(data), old(data);
125  for (unsigned int cell = cell_range.first; cell < cell_range.second; ++cell)
126  {
127  current.reinit(cell);
128  old.reinit(cell);
129 
130  current.read_dof_values(*src[0]);
131  old.read_dof_values(*src[1]);
132 
133  current.evaluate(EvaluationFlags::values | EvaluationFlags::gradients);
134  old.evaluate(EvaluationFlags::values);
135 
136  for (unsigned int q = 0; q < current.n_q_points; ++q)
137  {
138  const VectorizedArray<double> current_value = current.get_value(q);
139  const VectorizedArray<double> old_value = old.get_value(q);
140 
141  current.submit_value(2. * current_value - old_value -
142  delta_t_sqr * std::sin(current_value),
143  q);
144  current.submit_gradient(-delta_t_sqr * current.get_gradient(q), q);
145  }
146 
147  current.integrate(EvaluationFlags::values | EvaluationFlags::gradients);
148  current.distribute_local_to_global(dst);
149  }
150  }

168  template <int dim>
169  class InitialCondition : public Function<dim>
170  {
171  public:
172  InitialCondition(const unsigned int n_components = 1,
173  const double time = 0.)
174  : Function<dim>(n_components, time)
175  {}
176  virtual double value(const Point<dim> &p,
177  const unsigned int /*component*/) const override
178  {
179  double t = this->get_time();
180 
181  const double m = 0.5;
182  const double c1 = 0.;
183  const double c2 = 0.;
184  const double factor =
185  (m / std::sqrt(1. - m * m) * std::sin(std::sqrt(1. - m * m) * t + c2));
186  double result = 1.;
187  for (unsigned int d = 0; d < dim; ++d)
188  result *= -4. * std::atan(factor / std::cosh(m * p[d] + c1));
189  return result;
190  }

355  template <int dim>
356  void SineGordonProblem<dim>::run()
357  {
358  {
359  pcout << "Number of MPI ranks: "
360  << Utilities::MPI::n_mpi_processes(MPI_COMM_WORLD) << std::endl;
361  pcout << "Number of threads on each rank: "
362  << MultithreadInfo::n_threads() << std::endl;
363  const unsigned int n_vect_doubles = VectorizedArray<double>::size();
364  const unsigned int n_vect_bits = 8 * sizeof(double) * n_vect_doubles;
365  pcout << "Vectorization over " << n_vect_doubles
366  << " doubles = " << n_vect_bits << " bits ("
367  << Utilities::System::get_current_vectorization_level() << ")"
368  << std::endl
369  << std::endl;
370  }
371  make_grid_and_dofs();
372 
373  const double local_min_cell_diameter =
374  triangulation.last()->diameter() / std::sqrt(dim);
375  const double global_min_cell_diameter =
376  -Utilities::MPI::max(-local_min_cell_diameter, MPI_COMM_WORLD);
377  time_step = cfl_number * global_min_cell_diameter;
378  time_step = (final_time - time) / (int((final_time - time) / time_step));
379  pcout << " Time step size: " << time_step
380  << ", finest cell: " << global_min_cell_diameter << std::endl
381  << std::endl;
382 
383 
384  VectorTools::interpolate(mapping,
385  dof_handler,
386  InitialCondition<dim>(1, time),
387  solution);
388  VectorTools::interpolate(mapping,
389  dof_handler,
390  InitialCondition<dim>(1, time - time_step),
391  old_solution);
392  output_results(0);
393 
394  std::vector<LinearAlgebra::distributed::Vector<double> *>
395  previous_solutions({&old_solution, &old_old_solution});
396 
397  SineGordonOperation<dim, fe_degree> sine_gordon_op(matrix_free_data,
398  time_step);
399 
400  unsigned int timestep_number = 1;
401 
402  Timer timer;
403  double wtime = 0;
404  double output_time = 0;
405  for (time += time_step; time <= final_time;
406  time += time_step, ++timestep_number)
407  {
408  timer.restart();
409  old_old_solution.swap(old_solution);
410  old_solution.swap(solution);
411  sine_gordon_op.apply(solution, previous_solutions);
412  wtime += timer.wall_time();
413 
414  timer.restart();
415  if (timestep_number % output_timestep_skip == 0)
416  output_results(timestep_number / output_timestep_skip);
417 
418  output_time += timer.wall_time();
419  }
420  timer.restart();
421  output_results(timestep_number / output_timestep_skip + 1);
422  output_time += timer.wall_time();
423 
424  pcout << std::endl
425  << " Performed " << timestep_number << " time steps." << std::endl;
426 
427  pcout << " Average wallclock time per time step: "
428  << wtime / timestep_number << "s" << std::endl;
429 
430  pcout << " Spent " << output_time << "s on output and " << wtime
431  << "s on computations." << std::endl;
432  }
