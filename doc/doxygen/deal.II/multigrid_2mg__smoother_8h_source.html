719  template <class RelaxationType, typename VectorType>
720  inline void
721  SmootherRelaxation<RelaxationType, VectorType>::smooth(
722  const unsigned int level,
723  VectorType & u,
724  const VectorType & rhs) const
725  {
726  unsigned int maxlevel = this->max_level();
727  unsigned int steps2 = this->steps;
728 
729  if (this->variable)
730  steps2 *= (1 << (maxlevel - level));
731 
732  bool T = this->transpose;
733  if (this->symmetric && (steps2 % 2 == 0))
734  T = false;
735  if (this->debug > 0)
736  deallog << 'S' << level << ' ';
737 
738  for (unsigned int i = 0; i < steps2; ++i)
739  {
740  if (T)
741  (*this)[level].Tstep(u, rhs);
742  else
743  (*this)[level].step(u, rhs);
744  if (this->symmetric)
745  T = !T;
746  }
747  }
