599 template <typename VectorType>
600 bool
601 ImplicitQR<VectorType>::append_column(const VectorType &column)
602 {
603  if (this->current_size == 0)
604  {
605  this->R.grow_or_shrink(this->current_size + 1);
606  this->columns.push_back(std::make_unique<VectorType>(column));
607  this->R(0, 0) = column.l2_norm();
608  ++this->current_size;
609  }
610  else
611  {
612  // first get scalar products with A^T
613  Vector<Number> u(this->current_size);
614  this->multiply_with_AT(u, column);
615 
616  // now solve R^T x = (A^T * column)
617  const int lda = this->current_size;
618  const int ldb = this->current_size;
619  const int N = this->current_size;
620  const int n_rhs = 1;
621  int info = 0;
622  internal::QRImplementation::call_trtrs(
623  'U', 'T', 'N', N, n_rhs, &this->R(0, 0), lda, &u(0), ldb, &info);
624 
625  // finally get the diagonal element:
626  // rho2 = |column|^2 - |u|^2
627  const Number column_norm_sqr = column.norm_sqr();
628  const Number rho2 = column_norm_sqr - u.norm_sqr();
629  const bool linearly_independent =
630  column_signal.empty() ? rho2 > 0 :
631  column_signal(u, rho2, column_norm_sqr).get();
632 
633  // bail out if it turns out to be linearly dependent
634  if (!linearly_independent)
635  return false;
636 
637  // at this point we update is successful and we can enlarge R
638  // and store the column:
639  this->columns.push_back(std::make_unique<VectorType>(column));
640  this->R.grow_or_shrink(this->current_size + 1);
641  this->R(this->current_size, this->current_size) = std::sqrt(rho2);
642  for (unsigned int i = 0; i < this->current_size; ++i)
643  this->R(i, this->current_size) = u(i);
644 
645  this->current_size++;
646  }
647 
648  return true;
649 }

653 template <typename VectorType>
654 void
655 ImplicitQR<VectorType>::apply_givens_rotation(const unsigned int i,
656  const unsigned int k)
657 {
658  AssertIndexRange(i, k);
659  AssertIndexRange(k, this->current_size);
660  const std::array<Number, 3> csr =
661  ::Utilities::LinearAlgebra::givens_rotation<Number>(this->R(i, k),
662  this->R(k, k));
663 
664  // first, set k'th column:
665  this->R(i, k) = csr[2];
666  this->R(k, k) = 0.;
667  // now do the rest:
668  for (unsigned int j = 0; j < this->R.n(); ++j)
669  if (j != k)
670  {
671  const Number t = this->R(i, j);
672  this->R(i, j) = csr[0] * this->R(i, j) + csr[1] * this->R(k, j);
673  this->R(k, j) = -csr[1] * t + csr[0] * this->R(k, j);
674  }
675 
676  if (!this->givens_signal.empty())
677  this->givens_signal(i, k, csr);
678 }

682 template <typename VectorType>
683 void
684 ImplicitQR<VectorType>::remove_column(const unsigned int k)
685 {
686  // before actually removing a column from Q and resizing R,
687  // apply givens rotations to bring H into upper triangular form:
688  for (unsigned int j = k + 1; j < this->R.n(); ++j)
689  {
690  const unsigned int i = j - 1;
691  apply_givens_rotation(i, j);
692  }
693 
694  // remove last row and k-th column
695  --this->current_size;
696  this->R.remove_row_and_column(this->current_size, k);
697 
698  // Finally remove the column from A
699  this->columns.erase(this->columns.begin() + k);
700 }

759 template <typename VectorType>
760 bool
761 QR<VectorType>::append_column(const VectorType &column)
762 {
763  // resize R:
764  this->R.grow_or_shrink(this->current_size + 1);
765  this->columns.push_back(std::make_unique<VectorType>(column));
766 
767  // now a Gram-Schmidt part: orthonormalize the new column
768  // against everything we have so far:
769  auto &last_col = *this->columns.back();
770  for (unsigned int i = 0; i < this->current_size; ++i)
771  {
772  const auto &i_col = *this->columns[i];
773  this->R(i, this->current_size) = i_col * last_col;
774  last_col.add(-this->R(i, this->current_size), i_col);
775  }
776 
777  this->R(this->current_size, this->current_size) = last_col.l2_norm();
778 
779  Assert(this->R(this->current_size, this->current_size) > 0.,
780  ExcDivideByZero());
781  last_col *= 1. / this->R(this->current_size, this->current_size);
782 
783  ++this->current_size;
784  return true;
785 }

789 template <typename VectorType>
790 void
791 QR<VectorType>::apply_givens_rotation(const unsigned int i,
792  const unsigned int k)
793 {
794  AssertIndexRange(i, k);
795  AssertIndexRange(k, this->current_size);
796  const std::array<Number, 3> csr =
797  ::Utilities::LinearAlgebra::givens_rotation<Number>(this->R(i, k),
798  this->R(k, k));
799 
800  // first, set k'th column:
801  this->R(i, k) = csr[2];
802  this->R(k, k) = 0.;
803  // now do the rest:
804  for (unsigned int j = 0; j < this->R.n(); ++j)
805  if (j != k)
806  {
807  const Number t = this->R(i, j);
808  this->R(i, j) = csr[0] * this->R(i, j) + csr[1] * this->R(k, j);
809  this->R(k, j) = -csr[1] * t + csr[0] * this->R(k, j);
810  }
811 
812  // now adjust i,k columns due to multiplication with the
813  // transpose Givens matrix from right:
814  auto &col_i = *this->columns[i];
815  auto &col_k = *this->columns[k];
816  // save column i:
817  tmp = col_i;
818  col_i.sadd(csr[0], csr[1], col_k);
819  col_k.sadd(csr[0], -csr[1], tmp);
820 
821  if (!this->givens_signal.empty())
822  this->givens_signal(i, k, csr);
823 }

827 template <typename VectorType>
828 void
829 QR<VectorType>::remove_column(const unsigned int k)
830 {
831  AssertIndexRange(k, this->current_size);
832  Assert(this->current_size > 0,
833  ExcMessage("Can not remove a column if QR is empty"));
834  // apply a sequence of Givens rotations
835  // see section 6.5 "Updating matrix factorizations" in Golub 2013, Matrix
836  // computations
837 
838  // So we want to have QR for \tilde A \in R^{m*(n-1)}
839  // if we remove the column k, we end up with upper Hessenberg matrix
840  // x x x x x
841  // x x x x
842  // H = x x x
843  // x x x
844  // x x
845  // x
846  // where k = 2 (3rd column), m = 7, n = 6
847  //
848  // before actually removing a column from Q and resizing R,
849  // apply givens rotations to bring H into upper triangular form:
850  for (unsigned int j = k + 1; j < this->R.n(); ++j)
851  {
852  const unsigned int i = j - 1;
853  apply_givens_rotation(i, j);
854  }
855 
856  // now we can throw away the column from Q and adjust R
857  // since we do thin-QR, after Givens rotations we need to throw
858  // away the last column:
859  const unsigned int size_minus_1 = this->columns.size() - 1;
860  this->columns.erase(this->columns.begin() + size_minus_1);
861 
862  // remove last row and k-th column
863  --this->current_size;
864  this->R.remove_row_and_column(this->current_size, k);
