68 template <int dim, int spacedim>
69 DataOut<dim, spacedim>::DataOut()
70 {
71  set_cell_selection(
72  [this](const Triangulation<dim, spacedim> &) {
73  typename Triangulation<dim, spacedim>::active_cell_iterator cell =
74  this->triangulation->begin_active();
75 
76  // skip cells if the current one has no children (is active) and is a
77  // ghost or artificial cell
78  while ((cell != this->triangulation->end()) && !cell->is_locally_owned())
79  ++cell;
80  return cell;
81  },
82  [this](const Triangulation<dim, spacedim> &,
83  const cell_iterator &old_cell) {

85  old_cell;
86  ++cell;

88  ++cell;
89  return cell;
90  });

95 template <int dim, int spacedim>
96 void
97 DataOut<dim, spacedim>::build_one_patch(
98  const std::pair<cell_iterator, unsigned int> * cell_and_index,
99  internal::DataOutImplementation::ParallelData<dim, spacedim> &scratch_data,
100  const unsigned int n_subdivisions,
101  const CurvedCellRegion curved_cell_region)
102 {
103  // first create the output object that we will write into
104 
105  ::DataOutBase::Patch<dim, spacedim> patch;
106  patch.n_subdivisions = n_subdivisions;
107  patch.reference_cell = cell_and_index->first->reference_cell();
108 
109  // initialize FEValues
110  scratch_data.reinit_all_fe_values(this->dof_data, cell_and_index->first);
111 
112  const FEValuesBase<dim, spacedim> &fe_patch_values =
113  scratch_data.get_present_fe_values(0);
114 
115  // set the vertices of the patch. if the mapping does not preserve locations
116  // (e.g. MappingQEulerian), we need to compute the offset of the vertex for
117  // the graphical output. Otherwise, we can just use the vertex info.
118  for (const unsigned int vertex : cell_and_index->first->vertex_indices())
119  if (scratch_data.mapping_collection[0].preserves_vertex_locations())
120  patch.vertices[vertex] = cell_and_index->first->vertex(vertex);
121  else
122  patch.vertices[vertex] =
123  fe_patch_values.get_mapping().transform_unit_to_real_cell(
124  cell_and_index->first, GeometryInfo<dim>::unit_cell_vertex(vertex));
125 
126  const unsigned int n_q_points = fe_patch_values.n_quadrature_points;
127 
128  scratch_data.patch_values_scalar.solution_values.resize(n_q_points);
129  scratch_data.patch_values_scalar.solution_gradients.resize(n_q_points);
130  scratch_data.patch_values_scalar.solution_hessians.resize(n_q_points);
131  scratch_data.patch_values_system.solution_values.resize(n_q_points);
132  scratch_data.patch_values_system.solution_gradients.resize(n_q_points);
133  scratch_data.patch_values_system.solution_hessians.resize(n_q_points);
134 
135  for (unsigned int dataset = 0;
136  dataset < scratch_data.postprocessed_values.size();
137  ++dataset)
138  if (scratch_data.postprocessed_values[dataset].size() != 0)
139  scratch_data.postprocessed_values[dataset].resize(n_q_points);
140 
141  // First fill the geometric information for the patch: Where are the
142  // nodes in question located.
143  //
144  // Depending on the requested output of curved cells, if necessary
145  // append the quadrature points to the last rows of the patch.data
146  // member. This is the case if we want to produce curved cells at the
147  // boundary and this cell actually is at the boundary, or else if we
148  // want to produce curved cells everywhere
149  //
150  // note: a cell is *always* at the boundary if dim<spacedim
151  if (curved_cell_region == curved_inner_cells ||
152  (curved_cell_region == curved_boundary &&
153  (cell_and_index->first->at_boundary() || (dim != spacedim))) ||
154  (cell_and_index->first->reference_cell() !=
155  ReferenceCells::get_hypercube<dim>()))
156  {
157  Assert(patch.space_dim == spacedim, ExcInternalError());
158 
159  // set the flag indicating that for this cell the points are
160  // explicitly given
161  patch.points_are_available = true;
162 
163  // then resize the patch.data member in order to have enough memory for
164  // the quadrature points as well, and copy the quadrature points there
165  const std::vector<Point<spacedim>> &q_points =
166  fe_patch_values.get_quadrature_points();
167 
168  patch.data.reinit(scratch_data.n_datasets + spacedim, n_q_points);
169  for (unsigned int i = 0; i < spacedim; ++i)
170  for (unsigned int q = 0; q < n_q_points; ++q)
171  patch.data(patch.data.size(0) - spacedim + i, q) = q_points[q][i];
172  }
173  else
174  {
175  patch.data.reinit(scratch_data.n_datasets, n_q_points);
176  patch.points_are_available = false;
177  }
178 
179 
180  // Next fill the information we get from DoF data
181  if (scratch_data.n_datasets > 0)
182  {
183  // counter for data records
184  unsigned int offset = 0;
185 
186  // first fill dof_data
187  unsigned int dataset_number = 0;
188  for (const auto &dataset : this->dof_data)
189  {
190  const FEValuesBase<dim, spacedim> &this_fe_patch_values =
191  scratch_data.get_present_fe_values(dataset_number);
192  const unsigned int n_components =
193  this_fe_patch_values.get_fe().n_components();
194 
195  const DataPostprocessor<spacedim> *postprocessor =
196  dataset->postprocessor;
197 
198  if (postprocessor != nullptr)
199  {
200  // We have to postprocess the data, so determine, which fields
201  // have to be updated
202  const UpdateFlags update_flags =
203  postprocessor->get_needed_update_flags();
204 
205  if ((n_components == 1) &&
206  (dataset->is_complex_valued() == false))
207  {
208  // At each point there is only one component of value,
209  // gradient etc. Based on the 'if' statement above, we
210  // know that the solution is scalar and real-valued, so
211  // we do not need to worry about getting any imaginary
212  // components to the postprocessor, and we can safely
213  // call the function that evaluates a scalar field
214  if (update_flags & update_values)
215  dataset->get_function_values(
216  this_fe_patch_values,
217  internal::DataOutImplementation::ComponentExtractor::
218  real_part,
219  scratch_data.patch_values_scalar.solution_values);
220 
221  if (update_flags & update_gradients)
222  dataset->get_function_gradients(
223  this_fe_patch_values,
224  internal::DataOutImplementation::ComponentExtractor::
225  real_part,
226  scratch_data.patch_values_scalar.solution_gradients);
227 
228  if (update_flags & update_hessians)
229  dataset->get_function_hessians(
230  this_fe_patch_values,
231  internal::DataOutImplementation::ComponentExtractor::
232  real_part,
233  scratch_data.patch_values_scalar.solution_hessians);
234 
235  // Also fill some of the other fields postprocessors may
236  // want to access.
237  if (update_flags & update_quadrature_points)
238  scratch_data.patch_values_scalar.evaluation_points =
239  this_fe_patch_values.get_quadrature_points();
240

242  dh_cell(&cell_and_index->first->get_triangulation(),
243  cell_and_index->first->level(),
244  cell_and_index->first->index(),
245  dataset->dof_handler);
246  scratch_data.patch_values_scalar.template set_cell<dim>(
247  dh_cell);
248 
249  // Finally call the postprocessor's function that
250  // deals with scalar inputs.
251  postprocessor->evaluate_scalar_field(
252  scratch_data.patch_values_scalar,
253  scratch_data.postprocessed_values[dataset_number]);
254  }
255  else
256  {
257  // At each point we now have to evaluate a vector valued
258  // function and its derivatives. It may be that the solution
259  // is scalar and complex-valued, but we treat this as a vector
260  // field with two components.
261 
262  // At this point, we need to ask how we fill the fields that
263  // we want to pass on to the postprocessor. If the field in
264  // question is real-valued, we'll just extract the (only)
265  // real component from the solution fields
266  if (dataset->is_complex_valued() == false)
267  {
268  scratch_data.resize_system_vectors(n_components);
269 
270  if (update_flags & update_values)
271  dataset->get_function_values(
272  this_fe_patch_values,
273  internal::DataOutImplementation::ComponentExtractor::
274  real_part,
275  scratch_data.patch_values_system.solution_values);
276 
277  if (update_flags & update_gradients)
278  dataset->get_function_gradients(
279  this_fe_patch_values,
280  internal::DataOutImplementation::ComponentExtractor::
281  real_part,
282  scratch_data.patch_values_system.solution_gradients);
283 
284  if (update_flags & update_hessians)
285  dataset->get_function_hessians(
286  this_fe_patch_values,
287  internal::DataOutImplementation::ComponentExtractor::
288  real_part,
289  scratch_data.patch_values_system.solution_hessians);
290  }
291  else
292  {
293  // The solution is complex-valued. Let's cover the scalar
294  // case first (i.e., one scalar but complex-valued field,
295  // which we will have to split into its real and imaginar
296  // parts).
297  if (n_components == 1)
298  {
299  scratch_data.resize_system_vectors(2);
300 
301  // First get the real component of the scalar solution
302  // and copy the data into the
303  // scratch_data.patch_values_system output fields
304  if (update_flags & update_values)
305  {
306  dataset->get_function_values(
307  this_fe_patch_values,
308  internal::DataOutImplementation::
309  ComponentExtractor::real_part,
310  scratch_data.patch_values_scalar
311  .solution_values);
312 
313  for (unsigned int i = 0;
314  i < scratch_data.patch_values_scalar
315  .solution_values.size();
316  ++i)
317  {
318  AssertDimension(
319  scratch_data.patch_values_system
320  .solution_values[i]
321  .size(),
322  2);
323  scratch_data.patch_values_system
324  .solution_values[i][0] =
325  scratch_data.patch_values_scalar
326  .solution_values[i];
327  }
328  }
329

331  {
332  dataset->get_function_gradients(
333  this_fe_patch_values,
334  internal::DataOutImplementation::
335  ComponentExtractor::real_part,
336  scratch_data.patch_values_scalar
337  .solution_gradients);
338

340  i < scratch_data.patch_values_scalar
341  .solution_gradients.size();
342  ++i)
343  {
344  AssertDimension(
345  scratch_data.patch_values_system
346  .solution_values[i]
347  .size(),
348  2);
349  scratch_data.patch_values_system
350  .solution_gradients[i][0] =
351  scratch_data.patch_values_scalar
352  .solution_gradients[i];
353  }
354  }
355 
356  if (update_flags & update_hessians)
357  {
358  dataset->get_function_hessians(
359  this_fe_patch_values,
360  internal::DataOutImplementation::
361  ComponentExtractor::real_part,
362  scratch_data.patch_values_scalar
363  .solution_hessians);
364 
365  for (unsigned int i = 0;
366  i < scratch_data.patch_values_scalar
367  .solution_hessians.size();
368  ++i)
369  {
370  AssertDimension(
371  scratch_data.patch_values_system
372  .solution_hessians[i]
373  .size(),
374  2);

376  .solution_hessians[i][0] =
377  scratch_data.patch_values_scalar
378  .solution_hessians[i];
379  }
380  }

383  // component of the scalar solution
384  // and copy the data into the
385  // scratch_data.patch_values_system output fields
386  // that follow the real one
387  if (update_flags & update_values)
388  {
389  dataset->get_function_values(
390  this_fe_patch_values,
391  internal::DataOutImplementation::
392  ComponentExtractor::imaginary_part,
393  scratch_data.patch_values_scalar
394  .solution_values);
395 
396  for (unsigned int i = 0;
397  i < scratch_data.patch_values_scalar
398  .solution_values.size();

401  AssertDimension(
402  scratch_data.patch_values_system
403  .solution_values[i]
404  .size(),
405  2);
406  scratch_data.patch_values_system
407  .solution_values[i][1] =
408  scratch_data.patch_values_scalar
409  .solution_values[i];
410  }
411  }

414  {

416  this_fe_patch_values,
417  internal::DataOutImplementation::
418  ComponentExtractor::imaginary_part,
419  scratch_data.patch_values_scalar
420  .solution_gradients);

431  2);

434  scratch_data.patch_values_scalar

436  }
437  }
438

440  {
441  dataset->get_function_hessians(
442  this_fe_patch_values,
443  internal::DataOutImplementation::
444  ComponentExtractor::imaginary_part,
445  scratch_data.patch_values_scalar
446  .solution_hessians);
447

449  i < scratch_data.patch_values_scalar

452  {

455  .solution_hessians[i]

461  .solution_hessians[i];
462  }
463  }
464  }

466  {
467  scratch_data.resize_system_vectors(2 * n_components);
468 
469  // This is the vector-valued, complex-valued case. In
470  // essence, we just need to do the same as above,
471  // i.e., call the functions in dataset
472  // to retrieve first the real and then the imaginary
473  // part of the solution, then copy them to the
474  // scratch_data.patch_values_system. The difference to
475  // the scalar case is that there, we could (ab)use the
476  // scratch_data.patch_values_scalar members for first
477  // the real part and then the imaginary part, copying
478  // them into the scratch_data.patch_values_system
479  // variable one after the other. We can't do this
480  // here because the solution is vector-valued, and
481  // so using the single
482  // scratch_data.patch_values_system object doesn't
483  // work.
484  //
485  // Rather, we need to come up with a temporary object
486  // for this (one for each the values, the gradients,
487  // and the hessians).
488  //
489  // Compared to the previous code path, we here
490  // first get the real and imaginary parts of the
491  // values, then the real and imaginary parts of the
492  // gradients, etc. This allows us to scope the
493  // temporary objects better
494  if (update_flags & update_values)
495  {
496  std::vector<Vector<double>> tmp(
497  scratch_data.patch_values_system.solution_values
498  .size(),

500 
501  // First get the real part into the tmp object
502  dataset->get_function_values(
503  this_fe_patch_values,
504  internal::DataOutImplementation::
505  ComponentExtractor::real_part,
506  tmp);
507 
508  // Then copy these values into the first
509  // n_components slots of the output object.
510  for (unsigned int i = 0;
511  i < scratch_data.patch_values_system
512  .solution_values.size();
513  ++i)

516  scratch_data.patch_values_system
517  .solution_values[i]
518  .size(),
519  2 * n_components);

521  ++j)
522  scratch_data.patch_values_system
523  .solution_values[i][j] = tmp[i][j];
524  }
525

527  // copying past the end of the previous set of
528  // values.
529  dataset->get_function_values(
530  this_fe_patch_values,
531  internal::DataOutImplementation::
532  ComponentExtractor::imaginary_part,
533  tmp);
534 
535  for (unsigned int i = 0;
536  i < scratch_data.patch_values_system
537  .solution_values.size();
538  ++i)
539  {
540  for (unsigned int j = 0; j < n_components;
541  ++j)
542  scratch_data.patch_values_system
543  .solution_values[i][j + n_components] =
544  tmp[i][j];
545  }
546  }
547

550  {
551  std::vector<std::vector<Tensor<1, spacedim>>> tmp(

557  dataset->get_function_gradients(
558  this_fe_patch_values,
559  internal::DataOutImplementation::
560  ComponentExtractor::real_part,
561  tmp);
562 
563  for (unsigned int i = 0;
564  i < scratch_data.patch_values_system
565  .solution_gradients.size();
566  ++i)
567  {
568  AssertDimension(
569  scratch_data.patch_values_system
570  .solution_gradients[i]
571  .size(),
572  2 * n_components);
573  for (unsigned int j = 0; j < n_components;
574  ++j)
575  scratch_data.patch_values_system
576  .solution_gradients[i][j] = tmp[i][j];
577  }
578 
579  // Then the imaginary part
580  dataset->get_function_gradients(
581  this_fe_patch_values,
582  internal::DataOutImplementation::
583  ComponentExtractor::imaginary_part,
584  tmp);
585 
586  for (unsigned int i = 0;
587  i < scratch_data.patch_values_system
588  .solution_gradients.size();
589  ++i)
590  {
591  for (unsigned int j = 0; j < n_components;
592  ++j)
593  scratch_data.patch_values_system
594  .solution_gradients[i][j + n_components] =
595  tmp[i][j];
596  }
597  }
598 
599  // And finally the Hessians. Same scheme as above.
600  if (update_flags & update_hessians)
601  {
602  std::vector<std::vector<Tensor<2, spacedim>>> tmp(
603  scratch_data.patch_values_system
604  .solution_gradients.size(),
605  std::vector<Tensor<2, spacedim>>(n_components));
606 
607  // First the real part
608  dataset->get_function_hessians(
609  this_fe_patch_values,
610  internal::DataOutImplementation::
611  ComponentExtractor::real_part,
612  tmp);
613 
614  for (unsigned int i = 0;
615  i < scratch_data.patch_values_system
616  .solution_hessians.size();
617  ++i)
618  {
619  AssertDimension(
620  scratch_data.patch_values_system
621  .solution_hessians[i]
622  .size(),
623  2 * n_components);
624  for (unsigned int j = 0; j < n_components;
625  ++j)
626  scratch_data.patch_values_system
627  .solution_hessians[i][j] = tmp[i][j];
628  }
629 
630  // Then the imaginary part
631  dataset->get_function_hessians(
632  this_fe_patch_values,
633  internal::DataOutImplementation::
634  ComponentExtractor::imaginary_part,
635  tmp);
636 
637  for (unsigned int i = 0;
638  i < scratch_data.patch_values_system
639  .solution_hessians.size();
640  ++i)
641  {
642  for (unsigned int j = 0; j < n_components;
643  ++j)
644  scratch_data.patch_values_system
645  .solution_hessians[i][j + n_components] =
646  tmp[i][j];
647  }
648  }
649  }
650  }
651 
652  // Now set other fields we may need
653  if (update_flags & update_quadrature_points)
654  scratch_data.patch_values_system.evaluation_points =
655  this_fe_patch_values.get_quadrature_points();
656 
657  const typename DoFHandler<dim, spacedim>::cell_iterator
658  dh_cell(&cell_and_index->first->get_triangulation(),
659  cell_and_index->first->level(),
660  cell_and_index->first->index(),
661  dataset->dof_handler);
662  scratch_data.patch_values_system.template set_cell<dim>(
663  dh_cell);
664

666  // complex-vector-valued doesn't matter -- we took it apart
667  // into several fields and so we have to call the
668  // evaluate_vector_field() function.
669  postprocessor->evaluate_vector_field(
670  scratch_data.patch_values_system,
671  scratch_data.postprocessed_values[dataset_number]);
672  }
673

675  // the Patch object where it can then be further processed
676  // by the functions in DataOutBase

688  }
689  else
690  {
691  // use the given data vector directly, without a postprocessor.
692  // again, we treat single component functions separately for
693  // efficiency reasons.
694  if (n_components == 1)
695  {
696  Assert(dataset->n_output_variables == 1, ExcInternalError());
697 
698  // First output the real part of the solution vector
699  dataset->get_function_values(
700  this_fe_patch_values,
701  internal::DataOutImplementation::ComponentExtractor::
702  real_part,
703  scratch_data.patch_values_scalar.solution_values);
704  for (unsigned int q = 0; q < n_q_points; ++q)
705  patch.data(offset, q) =
706  scratch_data.patch_values_scalar.solution_values[q];
707  offset += 1;
708 
709  // And if there is one, also output the imaginary part. Note
710  // that the problem is scalar-valued, so we can freely add the
711  // imaginary part after the real part without having to worry
712  // that we are interleaving the real components of a vector
713  // with the imaginary components of the same vector.
714  if (dataset->is_complex_valued() == true)
715  {
716  dataset->get_function_values(
717  this_fe_patch_values,
718  internal::DataOutImplementation::ComponentExtractor::
719  imaginary_part,
720  scratch_data.patch_values_scalar.solution_values);
721  for (unsigned int q = 0; q < n_q_points; ++q)
722  patch.data(offset, q) =
723  scratch_data.patch_values_scalar.solution_values[q];
724  offset += 1;
725  }
726  }
727  else
728  {
729  scratch_data.resize_system_vectors(n_components);
730 
731  // So we have a multi-component DoFHandler here. That's more
732  // complicated. If the vector is real-valued, then we can just
733  // get everything at all quadrature points and copy them into
734  // the output array. In fact, we don't have to worry at all
735  // about the interpretation of the components.
736  if (dataset->is_complex_valued() == false)
737  {
738  dataset->get_function_values(
739  this_fe_patch_values,
740  internal::DataOutImplementation::ComponentExtractor::
741  real_part,
742  scratch_data.patch_values_system.solution_values);
743  for (unsigned int component = 0; component < n_components;
744  ++component)
745  for (unsigned int q = 0; q < n_q_points; ++q)
746  patch.data(offset + component, q) =
747  scratch_data.patch_values_system.solution_values[q](
748  component);
749 
750  // Increment the counter for the actual data record.
751  offset += dataset->n_output_variables;
752  }
753  else
754  // The situation is more complicated if the input vector is
755  // complex-valued. The easiest approach would have been to
756  // just have all real and then all imaginary components.
757  // This would have been conceptually easy, but it has the
758  // annoying downside that if you have a vector-valued
759  // problem (say, [u v]) then the output order would have
760  // been [u_re, v_re, u_im, v_im]. That's tolerable, but not
761  // quite so nice because one typically thinks of real and
762  // imaginary parts as belonging together. We would really
763  // like the output order to be [u_re, u_im, v_re, v_im].
764  // That, too, would have been easy to implement because one
765  // just has to interleave real and imaginary parts.
766  //
767  // But that's also not what we want. That's because if one
768  // were, for example, to solve a complex-valued Stokes
769  // problem (e.g., computing eigenfunctions of the Stokes
770  // operator), then one has solution components
771  // [[u v] p] and the proper output order is
772  // [[u_re v_re] [u_im v_im] p_re p_im].
773  // In other words, the order in which we want to output
774  // data depends on the *interpretation* of components.
775  //
776  // Doing this requires a bit more code, and also needs to
777  // be in sync with what we do in
778  // DataOut_DoFData::get_dataset_names() and
779  // DataOut_DoFData::get_nonscalar_data_ranges().
780  {
781  // Given this description, first get the real parts of
782  // all components:
783  dataset->get_function_values(
784  this_fe_patch_values,
785  internal::DataOutImplementation::ComponentExtractor::
786  real_part,
787  scratch_data.patch_values_system.solution_values);
788 
789  // Then we need to distribute them to the correct
790  // location. This requires knowledge of the interpretation
791  // of components as discussed above.
792  {
793  Assert(dataset->data_component_interpretation.size() ==
794  n_components,
795  ExcInternalError());
796 
797  unsigned int destination = offset;
798  for (unsigned int component = 0;
799  component < n_components;
800  /* component is updated below */)
801  {
802  switch (
803  dataset->data_component_interpretation[component])
804  {
805  case DataComponentInterpretation::
806  component_is_scalar:
807  {
808  // OK, a scalar component. Put all of the
809  // values into the current row
810  // ('destination'); then move 'component'
811  // forward by one (so we treat the next
812  // component) and 'destination' forward by
813  // two (because we're going to put the
814  // imaginary part of the current component
815  // into the next slot).
816  for (unsigned int q = 0; q < n_q_points;
817  ++q)
818  patch.data(destination, q) =
819  scratch_data.patch_values_system
820  .solution_values[q](component);
821 
822  ++component;
823  destination += 2;
824 
825  break;
826  }
827 
828  case DataComponentInterpretation::
829  component_is_part_of_vector:
830  {
831  // A vector component. Put the
832  // spacedim
833  // components into the next set of
834  // contiguous rows
835  // ('destination+c'); then move 'component'
836  // forward by spacedim (so we get to the
837  // next component after the current vector)
838  // and 'destination' forward by two*spacedim
839  // (because we're going to put the imaginary
840  // part of the vector into the subsequent
841  // spacedim slots).
842  const unsigned int size = spacedim;
843  for (unsigned int c = 0; c < size; ++c)
844  for (unsigned int q = 0; q < n_q_points;
845  ++q)
846  patch.data(destination + c, q) =
847  scratch_data.patch_values_system
848  .solution_values[q](component + c);
849 
850  component += size;
851  destination += 2 * size;
852 
853  break;
854  }
855 
856  case DataComponentInterpretation::
857  component_is_part_of_tensor:
858  {
859  // Same approach as for vectors above.
860  const unsigned int size =
861  spacedim * spacedim;
862  for (unsigned int c = 0; c < size; ++c)
863  for (unsigned int q = 0; q < n_q_points;
864  ++q)
865  patch.data(destination + c, q) =
866  scratch_data.patch_values_system
867  .solution_values[q](component + c);
868 
869  component += size;
870  destination += 2 * size;
871 
872  break;
873  }
874 
875  default:
876  Assert(false, ExcNotImplemented());
877  }
878  }
879  }
880 
881  // And now we need to do the same thing again for the
882  // imaginary parts, starting at the top of the list of
883  // components/destinations again.
884  dataset->get_function_values(
885  this_fe_patch_values,
886  internal::DataOutImplementation::ComponentExtractor::
887  imaginary_part,
888  scratch_data.patch_values_system.solution_values);
889  {
890  unsigned int destination = offset;
891  for (unsigned int component = 0;
892  component < n_components;
893  /* component is updated below */)
894  {
895  switch (
896  dataset->data_component_interpretation[component])
897  {
898  case DataComponentInterpretation::
899  component_is_scalar:
900  {
901  // OK, a scalar component. Put all of the
902  // values into the row past the current one
903  // ('destination+1') since 'destination' is
904  // occupied by the real part.
905  for (unsigned int q = 0; q < n_q_points;
906  ++q)
907  patch.data(destination + 1, q) =
908  scratch_data.patch_values_system
909  .solution_values[q](component);
910 
911  ++component;
912  destination += 2;
913 
914  break;
915  }
916 
917  case DataComponentInterpretation::
918  component_is_part_of_vector:
919  {
920  // A vector component. Put the
921  // spacedim
922  // components into the set of contiguous
923  // rows that follow the real parts
924  // ('destination+spacedim+c').
925  const unsigned int size = spacedim;
926  for (unsigned int c = 0; c < size; ++c)
927  for (unsigned int q = 0; q < n_q_points;
928  ++q)
929  patch.data(destination + size + c, q) =
930  scratch_data.patch_values_system
931  .solution_values[q](component + c);
932 
933  component += size;
934  destination += 2 * size;
935 
936  break;
937  }
938 
939  case DataComponentInterpretation::
940  component_is_part_of_tensor:
941  {
942  // Same as for vectors.
943  const unsigned int size =
944  spacedim * spacedim;
945  for (unsigned int c = 0; c < size; ++c)
946  for (unsigned int q = 0; q < n_q_points;
947  ++q)
948  patch.data(destination + size + c, q) =
949  scratch_data.patch_values_system
950  .solution_values[q](component + c);
951 
952  component += size;
953  destination += 2 * size;
954 
955  break;
956  }
957 
958  default:
959  Assert(false, ExcNotImplemented());
960  }
961  }
962  }
963 
964  // Increment the counter for the actual data record. We
965  // need to move it forward a number of positions equal to
966  // the number of components of this data set, times two
967  // because we dealt with a complex-valued input vector
968  offset += dataset->n_output_variables * 2;
969  }
970  }
971  }
972 
973  // Also update the dataset_number index that we carry along with the
974  // for-loop over all data sets.
975  ++dataset_number;
976  }
977 
978  // Then do the cell data. At least, we don't have to worry about
979  // complex-valued vectors/tensors since cell data is always scalar.
980  if (this->cell_data.size() != 0)
981  {
982  Assert(!cell_and_index->first->has_children(), ExcNotImplemented());
983 
984  for (const auto &dataset : this->cell_data)
985  {
986  // as above, first output the real part
987  {
988  const double value =
989  dataset->get_cell_data_value(cell_and_index->second,
990  internal::DataOutImplementation::
991  ComponentExtractor::real_part);
992  for (unsigned int q = 0; q < n_q_points; ++q)
993  patch.data(offset, q) = value;
994  }
995 
996  // and if there is one, also output the imaginary part
997  if (dataset->is_complex_valued() == true)
998  {
999  const double value = dataset->get_cell_data_value(
1000  cell_and_index->second,
1001  internal::DataOutImplementation::ComponentExtractor::
1002  imaginary_part);
1003  for (unsigned int q = 0; q < n_q_points; ++q)
1004  patch.data(offset + 1, q) = value;
1005  }
1006 
1007  offset += (dataset->is_complex_valued() ? 2 : 1);
1008  }
1009  }
1010  }
1011 
1012 
1013  for (const unsigned int f : cell_and_index->first->face_indices())
1014  {
1015  // let's look up whether the neighbor behind that face is noted in the
1016  // table of cells which we treat. this can only happen if the neighbor
1017  // exists, and is on the same level as this cell, but it may also happen
1018  // that the neighbor is not a member of the range of cells over which we
1019  // loop, in which case the respective entry in the
1020  // cell_to_patch_index_map will have the value no_neighbor. (note that
1021  // since we allocated only as much space in this array as the maximum
1022  // index of the cells we loop over, not every neighbor may have its
1023  // space in it, so we have to assume that it is extended by values
1024  // no_neighbor)
1025  if (cell_and_index->first->at_boundary(f) ||
1026  (cell_and_index->first->neighbor(f)->level() !=
1027  cell_and_index->first->level()))
1028  {
1029  patch.neighbors[f] = numbers::invalid_unsigned_int;
1030  continue;
1031  }
1032 
1033  const cell_iterator neighbor = cell_and_index->first->neighbor(f);
1034  Assert(static_cast<unsigned int>(neighbor->level()) <
1035  scratch_data.cell_to_patch_index_map->size(),
1036  ExcInternalError());
1037  if ((static_cast<unsigned int>(neighbor->index()) >=
1038  (*scratch_data.cell_to_patch_index_map)[neighbor->level()].size()) ||
1039  ((*scratch_data.cell_to_patch_index_map)[neighbor->level()]
1040  [neighbor->index()] ==
1041  ::DataOutBase::Patch<dim>::no_neighbor))
1042  {
1043  patch.neighbors[f] = numbers::invalid_unsigned_int;
1044  continue;
1045  }
1046 
1047  // now, there is a neighbor, so get its patch number and set it for the
1048  // neighbor index
1049  patch.neighbors[f] =
1050  (*scratch_data
1051  .cell_to_patch_index_map)[neighbor->level()][neighbor->index()];
1052  }
1053 
1054  const unsigned int patch_idx =
1055  (*scratch_data.cell_to_patch_index_map)[cell_and_index->first->level()]
1056  [cell_and_index->first->index()];
1057  // did we mess up the indices?
1058  Assert(patch_idx < this->patches.size(), ExcInternalError());
1059  patch.patch_index = patch_idx;
1060 
1061  // Put the patch into the patches vector. instead of copying the data,
1062  // simply swap the contents to avoid the penalty of writing into another
1063  // processor's memory
1064  this->patches[patch_idx].swap(patch);
1065 }

1096 template <int dim, int spacedim>
1097 void
1098 DataOut<dim, spacedim>::build_patches(
1099  const hp::MappingCollection<dim, spacedim> &mapping,
1100  const unsigned int n_subdivisions_,
1101  const CurvedCellRegion curved_region)
1102 {
1103  // Check consistency of redundant template parameter
1104  Assert(dim == dim, ExcDimensionMismatch(dim, dim));
1105 
1106  Assert(this->triangulation != nullptr,
1107  Exceptions::DataOutImplementation::ExcNoTriangulationSelected());
1108 
1109  const unsigned int n_subdivisions =
1110  (n_subdivisions_ != 0) ? n_subdivisions_ : this->default_subdivisions;
1111  Assert(n_subdivisions >= 1,
1112  Exceptions::DataOutImplementation::ExcInvalidNumberOfSubdivisions(
1113  n_subdivisions));
1114 
1115  this->validate_dataset_names();
1116 
1117  // First count the cells we want to create patches of. Also fill the object
1118  // that maps the cell indices to the patch numbers, as this will be needed
1119  // for generation of neighborship information.
1120  // Note, there is a confusing mess of different indices here at play:
1121  // - patch_index: the index of a patch in all_cells
1122  // - cell->index: only unique on each level, used in cell_to_patch_index_map
1123  // - active_index: index for a cell when counting from begin_active() using
1124  // ++cell (identical to cell->active_cell_index())
1125  // - cell_index: unique index of a cell counted using
1126  // next_cell_function() starting from first_cell_function()
1127  //
1128  // It turns out that we create one patch for each selected cell, so
1129  // patch_index==cell_index.
1130  //
1131  // Now construct the map such that
1132  // cell_to_patch_index_map[cell->level][cell->index] = patch_index
1133  std::vector<std::vector<unsigned int>> cell_to_patch_index_map;
1134  cell_to_patch_index_map.resize(this->triangulation->n_levels());
1135  for (unsigned int l = 0; l < this->triangulation->n_levels(); ++l)
1136  {
1137  // max_index is the largest cell->index on level l
1138  unsigned int max_index = 0;
1139  for (cell_iterator cell = first_cell_function(*this->triangulation);
1140  cell != this->triangulation->end();
1141  cell = next_cell_function(*this->triangulation, cell))
1142  if (static_cast<unsigned int>(cell->level()) == l)
1143  max_index =
1144  std::max(max_index, static_cast<unsigned int>(cell->index()));
1145 
1146  cell_to_patch_index_map[l].resize(
1147  max_index + 1, ::DataOutBase::Patch<dim, spacedim>::no_neighbor);
1148  }
1149 
1150  // will be all_cells[patch_index] = pair(cell, active_index)
1151  std::vector<std::pair<cell_iterator, unsigned int>> all_cells;
1152  {
1153  // important: we need to compute the active_index of the cell in the range
1154  // 0..n_active_cells() because this is where we need to look up cell
1155  // data from (cell data vectors do not have the length distance computed by
1156  // first_cell_function/next_cell_function because this might skip
1157  // some values (FilteredIterator).
1158  auto active_cell = this->triangulation->begin_active();
1159  unsigned int active_index = 0;
1160  cell_iterator cell = first_cell_function(*this->triangulation);
1161  for (; cell != this->triangulation->end();
1162  cell = next_cell_function(*this->triangulation, cell))
1163  {
1164  // move forward until active_cell points at the cell (cell) we are
1165  // looking at to compute the current active_index
1166  while (active_cell != this->triangulation->end() && cell->is_active() &&
1167  decltype(active_cell)(cell) != active_cell)
1168  {
1169  ++active_cell;
1170  ++active_index;
1171  }
1172 
1173  Assert(static_cast<unsigned int>(cell->level()) <
1174  cell_to_patch_index_map.size(),
1175  ExcInternalError());
1176  Assert(static_cast<unsigned int>(cell->index()) <
1177  cell_to_patch_index_map[cell->level()].size(),
1178  ExcInternalError());
1179  Assert(active_index < this->triangulation->n_active_cells(),
1180  ExcInternalError());
1181  cell_to_patch_index_map[cell->level()][cell->index()] =
1182  all_cells.size();
1183 
1184  all_cells.emplace_back(cell, active_index);
1185  }
1186  }
1187 
1188  this->patches.clear();
1189  this->patches.resize(all_cells.size());
1190 
1191  // Now create a default object for the WorkStream object to work with. The
1192  // first step is to count how many output data sets there will be. This is,
1193  // in principle, just the number of components of each data set, but we
1194  // need to allocate two entries per component if there are
1195  // complex-valued input data (unless we use a postprocessor on this
1196  // output -- all postprocessor outputs are real-valued)
1197  unsigned int n_datasets = 0;
1198  for (unsigned int i = 0; i < this->cell_data.size(); ++i)
1199  n_datasets += (this->cell_data[i]->is_complex_valued() &&
1200  (this->cell_data[i]->postprocessor == nullptr) ?
1201  2 :
1202  1);
1203  for (unsigned int i = 0; i < this->dof_data.size(); ++i)
1204  n_datasets += (this->dof_data[i]->n_output_variables *
1205  (this->dof_data[i]->is_complex_valued() &&
1206  (this->dof_data[i]->postprocessor == nullptr) ?
1207  2 :
1208  1));
1209 
1210  std::vector<unsigned int> n_postprocessor_outputs(this->dof_data.size());
1211  for (unsigned int dataset = 0; dataset < this->dof_data.size(); ++dataset)
1212  if (this->dof_data[dataset]->postprocessor)
1213  n_postprocessor_outputs[dataset] =
1214  this->dof_data[dataset]->n_output_variables;
1215  else
1216  n_postprocessor_outputs[dataset] = 0;
1217 
1218  const CurvedCellRegion curved_cell_region =
1219  (n_subdivisions < 2 ? no_curved_cells : curved_region);
1220 
1221  UpdateFlags update_flags = update_values;
1222  if (curved_cell_region != no_curved_cells)
1223  update_flags |= update_quadrature_points;
1224 
1225  for (unsigned int i = 0; i < this->dof_data.size(); ++i)
1226  if (this->dof_data[i]->postprocessor)
1227  update_flags |=
1228  this->dof_data[i]->postprocessor->get_needed_update_flags();
1229  // perhaps update_normal_vectors is present, which would only be useful on
1230  // faces, but we may not use it here.
1231  Assert(
1232  !(update_flags & update_normal_vectors),
1233  ExcMessage(
1234  "The update of normal vectors may not be requested for evaluation of "
1235  "data on cells via DataPostprocessor."));
1236 
1237  internal::DataOutImplementation::ParallelData<dim, spacedim> thread_data(
1238  n_datasets,
1239  n_subdivisions,
1240  n_postprocessor_outputs,
1241  mapping,
1242  this->get_fes(),
1243  update_flags,
1244  cell_to_patch_index_map);
1245 
1246  auto worker = [this, n_subdivisions, curved_cell_region](
1247  const std::pair<cell_iterator, unsigned int> *cell_and_index,
1248  internal::DataOutImplementation::ParallelData<dim, spacedim>
1249  &scratch_data,
1250  // this function doesn't actually need a copy data object --
1251  // it just writes everything right into the output array
1252  int) {
1253  this->build_one_patch(cell_and_index,
1254  scratch_data,
1255  n_subdivisions,
1256  curved_cell_region);
1257  };
1258 
1259  // now build the patches in parallel
1260  if (all_cells.size() > 0)
1261  WorkStream::run(all_cells.data(),
1262  all_cells.data() + all_cells.size(),
1263  worker,
1264  // no copy-local-to-global function needed here
1265  std::function<void(const int)>(),
1266  thread_data,
1267  /* dummy CopyData object = */ 0,
1268  // experimenting shows that we can make things run a bit
1269  // faster if we increase the number of cells we work on
1270  // per item (i.e., WorkStream's chunk_size argument,
1271  // about 10% improvement) and the items in flight at any
1272  // given time (another 5% on the testcase discussed in
1273  // @ref workstream_paper, on 32 cores) and if
1274  8 * MultithreadInfo::n_threads(),
1275  64);
1276 }
