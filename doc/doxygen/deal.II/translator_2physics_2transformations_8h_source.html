732 template <typename Number>
733 Tensor<2, 3, Number>
734 Physics::Transformations::Rotations::rotation_matrix_3d(
735  const Point<3, Number> &axis,
736  const Number & angle)
737 {
738  Assert(std::abs(axis.norm() - 1.0) < 1e-9,
739  ExcMessage("The supplied axial vector is not a unit vector."));
740  const Number c = std::cos(angle);
741  const Number s = std::sin(angle);
742  const Number t = 1. - c;
743  const Number rotation[3][3] = {{t * axis[0] * axis[0] + c,
744  t * axis[0] * axis[1] - s * axis[2],
745  t * axis[0] * axis[2] + s * axis[1]},
746  {t * axis[0] * axis[1] + s * axis[2],
747  t * axis[1] * axis[1] + c,
748  t * axis[1] * axis[2] - s * axis[0]},
749  {t * axis[0] * axis[2] - s * axis[1],
750  t * axis[1] * axis[2] + s * axis[0],
751  t * axis[2] * axis[2] + c}};
752  return Tensor<2, 3, Number>(rotation);
753 }

1112 template <int dim, typename Number>
1113 inline SymmetricTensor<2, dim, Number>
1114 Physics::Transformations::basis_transformation(
1115  const SymmetricTensor<2, dim, Number> &T,
1116  const Tensor<2, dim, Number> & B)
1117 {
1118  Tensor<2, dim, Number> tmp_1;
1119  for (unsigned int i = 0; i < dim; ++i)
1120  for (unsigned int J = 0; J < dim; ++J)
1121  // Loop over I but complex.h defines a macro I, so use I_ instead
1122  for (unsigned int I_ = 0; I_ < dim; ++I_)
1123  tmp_1[i][J] += B[i][I_] * T[I_][J];
1124 
1125  SymmetricTensor<2, dim, Number> out;
1126  for (unsigned int i = 0; i < dim; ++i)
1127  for (unsigned int j = i; j < dim; ++j)
1128  for (unsigned int J = 0; J < dim; ++J)
1129  out[i][j] += B[j][J] * tmp_1[i][J];
1130 
1131  return out;
1132 }

1136 template <int dim, typename Number>
1137 inline Tensor<4, dim, Number>
1138 Physics::Transformations::basis_transformation(const Tensor<4, dim, Number> &H,
1139  const Tensor<2, dim, Number> &B)
1140 {
1141  // This contraction order and indexing might look a bit dubious, so a
1142  // quick explanation as to what's going on is probably in order:
1143  //
1144  // When the contract() function operates on the inner indices, the
1145  // result has the inner index and outer index transposed, i.e.
1146  // contract<2,1>(H,F) implies
1147  // T_{IJLk} = (H_{IJMN} F_{mM}) \delta_{mL} \delta_{Nk}
1148  // rather than T_{IJkL} (the desired result).

1150  // second argument results in its transposition with respect to its

1152  // leading to the free index being on the right hand side of the result.

1156  // sets of contractions.
1157  //
1158  // Note: Its significantly quicker (in 3d) to push forward
1159  // each index individually
1160  return contract<1, 1>(
