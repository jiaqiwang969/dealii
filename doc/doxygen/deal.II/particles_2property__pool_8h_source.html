284  template <int dim, int spacedim>
285  inline const Point<spacedim> &
286  PropertyPool<dim, spacedim>::get_location(const Handle handle) const
287  {
288  const std::vector<double>::size_type data_index =
289  (handle != invalid_handle) ? handle : 0;
290 
291  // Ideally we would need to assert that 'handle' has not been deallocated
292  // by searching through 'currently_available_handles'. However, this
293  // is expensive and this function is performance critical, so instead
294  // just check against the array range, and rely on the fact
295  // that handles are invalidated when handed over to
296  // deallocate_properties_array().
297  Assert(data_index <= locations.size() - 1,
298  ExcMessage("Invalid location handle. This can happen if the "
299  "handle was duplicated and then one copy was deallocated "
300  "before trying to access the properties."));
301 
302  return locations[data_index];
303  }
