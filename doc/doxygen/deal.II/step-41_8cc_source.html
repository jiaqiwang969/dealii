211  template <int dim>
212  void ObstacleProblem<dim>::assemble_system()
213  {
214  std::cout << " Assembling system..." << std::endl;
215 
216  system_matrix = 0;
217  system_rhs = 0;
218 
219  const QGauss<dim> quadrature_formula(fe.degree + 1);
220  RightHandSide<dim> right_hand_side;
221 
222  FEValues<dim> fe_values(fe,
223  quadrature_formula,
224  update_values | update_gradients |
225  update_quadrature_points | update_JxW_values);
226 
227  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
228  const unsigned int n_q_points = quadrature_formula.size();
229 
230  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
231  Vector<double> cell_rhs(dofs_per_cell);
232 
233  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
234 
235  for (const auto &cell : dof_handler.active_cell_iterators())
236  {
237  fe_values.reinit(cell);
238  cell_matrix = 0;
239  cell_rhs = 0;
240 
241  for (unsigned int q_point = 0; q_point < n_q_points; ++q_point)
242  for (unsigned int i = 0; i < dofs_per_cell; ++i)
243  {
244  for (unsigned int j = 0; j < dofs_per_cell; ++j)
245  cell_matrix(i, j) +=
246  (fe_values.shape_grad(i, q_point) *
247  fe_values.shape_grad(j, q_point) * fe_values.JxW(q_point));
248 
249  cell_rhs(i) +=
250  (fe_values.shape_value(i, q_point) *
251  right_hand_side.value(fe_values.quadrature_point(q_point)) *
252  fe_values.JxW(q_point));
253  }
254 
255  cell->get_dof_indices(local_dof_indices);
256 
257  constraints.distribute_local_to_global(cell_matrix,
258  cell_rhs,
259  local_dof_indices,
260  system_matrix,
261  system_rhs,
262  true);
263  }

307  template <int dim>
308  void ObstacleProblem<dim>::update_solution_and_constraints()
309  {
310  std::cout << " Updating active set..." << std::endl;
311 
312  const double penalty_parameter = 100.0;
313 
314  TrilinosWrappers::MPI::Vector lambda(
315  complete_index_set(dof_handler.n_dofs()));
316  complete_system_matrix.residual(lambda, solution, complete_system_rhs);
317 
318  contact_force = lambda;
319  contact_force.scale(diagonal_of_mass_matrix);
320  contact_force *= -1;
321 
322  constraints.clear();
323  active_set.clear();
324 
325  const Obstacle<dim> obstacle;
326  std::vector<bool> dof_touched(dof_handler.n_dofs(), false);
327 
328  for (const auto &cell : dof_handler.active_cell_iterators())
329  for (const auto v : cell->vertex_indices())
330  {
331  Assert(dof_handler.get_fe().n_dofs_per_cell() == cell->n_vertices(),
332  ExcNotImplemented());
333 
334  const unsigned int dof_index = cell->vertex_dof_index(v, 0);
335 
336  if (dof_touched[dof_index] == false)
337  dof_touched[dof_index] = true;
338  else
339  continue;
340 
341  const double obstacle_value = obstacle.value(cell->vertex(v));
342  const double solution_value = solution(dof_index);
343 
344  if (lambda(dof_index) + penalty_parameter *
345  diagonal_of_mass_matrix(dof_index) *
346  (solution_value - obstacle_value) <
347  0)
348  {
349  active_set.add_index(dof_index);
350  constraints.add_line(dof_index);
351  constraints.set_inhomogeneity(dof_index, obstacle_value);
352 
353  solution(dof_index) = obstacle_value;
354 
355  lambda(dof_index) = 0;
356  }
357  }
358  std::cout << " Size of active set: " << active_set.n_elements()
359  << std::endl;
360 
361  std::cout << " Residual of the non-contact part of the system: "
362  << lambda.l2_norm() << std::endl;
363 
364  VectorTools::interpolate_boundary_values(dof_handler,
365  0,
366  BoundaryValues<dim>(),
367  constraints);
368  constraints.close();
369  }

420  template <int dim>
421  void ObstacleProblem<dim>::run()
422  {
423  make_grid();
424  setup_system();
425 
426  IndexSet active_set_old(active_set);
427  for (unsigned int iteration = 0; iteration <= solution.size(); ++iteration)
428  {
429  std::cout << "Newton iteration " << iteration << std::endl;
430 
431  assemble_system();
432 
433  if (iteration == 0)
434  {
435  complete_system_matrix.copy_from(system_matrix);
436  complete_system_rhs = system_rhs;
437  }
438 
439  solve();
440  update_solution_and_constraints();
441  output_results(iteration);
442 
443  if (active_set == active_set_old)
444  break;
445 
446  active_set_old = active_set;
447 
448  std::cout << std::endl;
449  }
