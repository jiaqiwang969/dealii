448  auto &&transpose_fill_nm = [](Number * out,
449  const Number * in,
450  const unsigned int n,
451  const unsigned int m) {
452  for (unsigned int mm = 0; mm < m; ++mm)
453  for (unsigned int nn = 0; nn < n; ++nn)
454  out[mm + nn * m] = *(in++);

465  eigenvecs);

467  for (unsigned int i = 0; i < n_rows; ++i)
468  for (unsigned int j = 0; j < n_cols; ++j, ++eigenvectors)
469  *eigenvectors = eigenvecs[j][i];
470

502 template <int dim, typename Number, int n_rows_1d>
503 inline void
504 TensorProductMatrixSymmetricSumBase<dim, Number, n_rows_1d>::vmult(
505  const ArrayView<Number> & dst_view,
506  const ArrayView<const Number> &src_view) const
507 {
508  AssertDimension(dst_view.size(), this->m());
509  AssertDimension(src_view.size(), this->n());
510  std::lock_guard<std::mutex> lock(this->mutex);
511  const unsigned int n = Utilities::fixed_power<dim>(
512  n_rows_1d > 0 ? n_rows_1d : eigenvalues[0].size());
513  tmp_array.resize_fast(n * 2);
514  constexpr int kernel_size = n_rows_1d > 0 ? n_rows_1d : 0;
515  internal::EvaluatorTensorProduct<internal::evaluate_general,
516  dim,
517  kernel_size,
518  kernel_size,
519  Number>
520  eval(AlignedVector<Number>{},
521  AlignedVector<Number>{},
522  AlignedVector<Number>{},
523  mass_matrix[0].n_rows(),
524  mass_matrix[0].n_rows());
525  Number * t = tmp_array.begin();
526  const Number *src = src_view.begin();
527  Number * dst = dst_view.data();
528 
529  if (dim == 1)
530  {
531  const Number *A = &derivative_matrix[0](0, 0);
532  eval.template apply<0, false, false>(A, src, dst);
533  }
534 
535  else if (dim == 2)
536  {
537  const Number *A0 = &derivative_matrix[0](0, 0);
538  const Number *M0 = &mass_matrix[0](0, 0);
539  const Number *A1 = &derivative_matrix[1](0, 0);
540  const Number *M1 = &mass_matrix[1](0, 0);
541  eval.template apply<0, false, false>(M0, src, t);
542  eval.template apply<1, false, false>(A1, t, dst);
543  eval.template apply<0, false, false>(A0, src, t);
544  eval.template apply<1, false, true>(M1, t, dst);
545  }
546 
547  else if (dim == 3)
548  {
549  const Number *A0 = &derivative_matrix[0](0, 0);
550  const Number *M0 = &mass_matrix[0](0, 0);
551  const Number *A1 = &derivative_matrix[1](0, 0);
552  const Number *M1 = &mass_matrix[1](0, 0);
553  const Number *A2 = &derivative_matrix[2](0, 0);
554  const Number *M2 = &mass_matrix[2](0, 0);
555  eval.template apply<0, false, false>(M0, src, t + n);
556  eval.template apply<1, false, false>(M1, t + n, t);
557  eval.template apply<2, false, false>(A2, t, dst);
558  eval.template apply<1, false, false>(A1, t + n, t);
559  eval.template apply<0, false, false>(A0, src, t + n);
560  eval.template apply<1, false, true>(M1, t + n, t);
561  eval.template apply<2, false, true>(M2, t, dst);
562  }
563 
564  else
565  AssertThrow(false, ExcNotImplemented());
566 }

570 template <int dim, typename Number, int n_rows_1d>
571 inline void
572 TensorProductMatrixSymmetricSumBase<dim, Number, n_rows_1d>::apply_inverse(
573  const ArrayView<Number> & dst_view,
574  const ArrayView<const Number> &src_view) const
575 {
576  AssertDimension(dst_view.size(), this->n());
577  AssertDimension(src_view.size(), this->m());
578  std::lock_guard<std::mutex> lock(this->mutex);
579  const unsigned int n = n_rows_1d > 0 ? n_rows_1d : eigenvalues[0].size();
580  tmp_array.resize_fast(Utilities::fixed_power<dim>(n));
581  constexpr int kernel_size = n_rows_1d > 0 ? n_rows_1d : 0;
582  internal::EvaluatorTensorProduct<internal::evaluate_general,
583  dim,
584  kernel_size,
585  kernel_size,
586  Number>
587  eval(AlignedVector<Number>(),
588  AlignedVector<Number>(),
589  AlignedVector<Number>(),
590  mass_matrix[0].n_rows(),
591  mass_matrix[0].n_rows());
592  Number * t = tmp_array.begin();
593  const Number *src = src_view.data();
594  Number * dst = dst_view.data();
595 
596  // NOTE: dof_to_quad has to be interpreted as 'dof to eigenvalue index'
597  // --> apply<.,true,.> (S,src,dst) calculates dst = S^T * src,
598  // --> apply<.,false,.> (S,src,dst) calculates dst = S * src,
599  // while the eigenvectors are stored column-wise in S, i.e.
600  // rows correspond to dofs whereas columns to eigenvalue indices!
601  if (dim == 1)
602  {
603  const Number *S = &eigenvectors[0](0, 0);
604  eval.template apply<0, true, false>(S, src, t);
605  for (unsigned int i = 0; i < n; ++i)
606  t[i] /= eigenvalues[0][i];
607  eval.template apply<0, false, false>(S, t, dst);
608  }
609 
610  else if (dim == 2)
611  {
612  const Number *S0 = &(eigenvectors[0](0, 0));
613  const Number *S1 = &(eigenvectors[1](0, 0));
614  eval.template apply<0, true, false>(S0, src, t);
615  eval.template apply<1, true, false>(S1, t, dst);
616  for (unsigned int i1 = 0, c = 0; i1 < n; ++i1)
617  for (unsigned int i0 = 0; i0 < n; ++i0, ++c)
618  dst[c] /= (eigenvalues[1][i1] + eigenvalues[0][i0]);
619  eval.template apply<0, false, false>(S0, dst, t);
620  eval.template apply<1, false, false>(S1, t, dst);
621  }
622 
623  else if (dim == 3)
624  {
625  const Number *S0 = &eigenvectors[0](0, 0);
626  const Number *S1 = &eigenvectors[1](0, 0);
627  const Number *S2 = &eigenvectors[2](0, 0);
628  eval.template apply<0, true, false>(S0, src, t);
629  eval.template apply<1, true, false>(S1, t, dst);
630  eval.template apply<2, true, false>(S2, dst, t);
631  for (unsigned int i2 = 0, c = 0; i2 < n; ++i2)
632  for (unsigned int i1 = 0; i1 < n; ++i1)
633  for (unsigned int i0 = 0; i0 < n; ++i0, ++c)
634  t[c] /=
635  (eigenvalues[2][i2] + eigenvalues[1][i1] + eigenvalues[0][i0]);
636  eval.template apply<0, false, false>(S0, t, dst);
637  eval.template apply<1, false, false>(S1, dst, t);
638  eval.template apply<2, false, false>(S2, t, dst);
639  }
640 
641  else
642  Assert(false, ExcNotImplemented());
643 }

805 template <typename MatrixArray>
806 inline void
807 TensorProductMatrixSymmetricSum<dim, VectorizedArray<Number>, n_rows_1d>::
808  reinit_impl(MatrixArray &&mass_matrices_, MatrixArray &&derivative_matrices_)
809 {
810  auto &&mass_matrix = std::forward<MatrixArray>(mass_matrices_);
811  auto &&derivative_matrix = std::forward<MatrixArray>(derivative_matrices_);
812  this->mass_matrix = mass_matrix;
813  this->derivative_matrix = derivative_matrix;
814 
815  constexpr unsigned int macro_size = VectorizedArray<Number>::size();
816  std::size_t n_rows_max = (n_rows_1d > 0) ? n_rows_1d : 0;
817  if (n_rows_1d == -1)
818  for (unsigned int d = 0; d < dim; ++d)
819  n_rows_max = std::max(n_rows_max, mass_matrix[d].n_rows());
820  const std::size_t nm_flat_size_max = n_rows_max * n_rows_max * macro_size;
821  const std::size_t n_flat_size_max = n_rows_max * macro_size;
822 
823  std::vector<Number> mass_matrix_flat;
824  std::vector<Number> deriv_matrix_flat;
825  std::vector<Number> eigenvalues_flat;
826  std::vector<Number> eigenvectors_flat;
827  mass_matrix_flat.resize(nm_flat_size_max);
828  deriv_matrix_flat.resize(nm_flat_size_max);
829  eigenvalues_flat.resize(n_flat_size_max);
830  eigenvectors_flat.resize(nm_flat_size_max);
831  std::array<unsigned int, macro_size> offsets_nm;
832  std::array<unsigned int, macro_size> offsets_n;
833  for (int dir = 0; dir < dim; ++dir)
834  {
835  Assert(n_rows_1d == -1 ||
836  (n_rows_1d > 0 && static_cast<unsigned int>(n_rows_1d) ==
837  mass_matrix[dir].n_rows()),
838  ExcDimensionMismatch(n_rows_1d, mass_matrix[dir].n_rows()));
839  AssertDimension(mass_matrix[dir].n_rows(), mass_matrix[dir].n_cols());
840  AssertDimension(mass_matrix[dir].n_rows(),
841  derivative_matrix[dir].n_rows());
842  AssertDimension(mass_matrix[dir].n_rows(),
843  derivative_matrix[dir].n_cols());
844 
845  const unsigned int n_rows = mass_matrix[dir].n_rows();
846  const unsigned int n_cols = mass_matrix[dir].n_cols();
847  const unsigned int nm = n_rows * n_cols;
848  for (unsigned int vv = 0; vv < macro_size; ++vv)
849  offsets_nm[vv] = nm * vv;
850 
851  vectorized_transpose_and_store(false,
852  nm,
853  &(mass_matrix[dir](0, 0)),
854  offsets_nm.cbegin(),
855  mass_matrix_flat.data());
856  vectorized_transpose_and_store(false,
857  nm,
858  &(derivative_matrix[dir](0, 0)),
859  offsets_nm.cbegin(),
860  deriv_matrix_flat.data());
861 
862  const Number *mass_cbegin = mass_matrix_flat.data();
863  const Number *deriv_cbegin = deriv_matrix_flat.data();
864  Number * eigenvec_begin = eigenvectors_flat.data();
865  Number * eigenval_begin = eigenvalues_flat.data();
866  for (unsigned int lane = 0; lane < macro_size; ++lane)
867  internal::TensorProductMatrix::spectral_assembly<Number>(
868  mass_cbegin + nm * lane,
869  deriv_cbegin + nm * lane,
870  n_rows,
871  n_cols,
872  eigenval_begin + n_rows * lane,
873  eigenvec_begin + nm * lane);
874 
875  this->eigenvalues[dir].resize(n_rows);
876  this->eigenvectors[dir].reinit(n_rows, n_cols);
877  for (unsigned int vv = 0; vv < macro_size; ++vv)
878  offsets_n[vv] = n_rows * vv;
